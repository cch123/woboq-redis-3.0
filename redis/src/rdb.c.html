79/* Saves an encoded length. The first two bits in the first byte are used to
80 * hold the encoding type. See the REDIS_RDB_* definitions for more information
81 * on the types of encoding. */
82int rdbSaveLen(rio *rdb, uint32_t len) {
83 unsigned char buf[2];
84 size_t nwritten;
85
86 if (len < (1<<6)) {
87 /* Save a 6 bit len */
88 buf[0] = (len&0xFF)|(REDIS_RDB_6BITLEN<<6);
89 if (rdbWriteRaw(rdb,buf,1) == -1) return -1;
90 nwritten = 1;
91 } else if (len < (1<<14)) {
92 /* Save a 14 bit len */
93 buf[0] = ((len>>8)&0xFF)|(REDIS_RDB_14BITLEN<<6);
94 buf[1] = len&0xFF;
95 if (rdbWriteRaw(rdb,buf,2) == -1) return -1;
96 nwritten = 2;
97 } else {
98 /* Save a 32 bit len */
99 buf[0] = (REDIS_RDB_32BITLEN<<6);
100 if (rdbWriteRaw(rdb,buf,1) == -1) return -1;
101 len = htonl(len);
102 if (rdbWriteRaw(rdb,&len,4) == -1) return -1;
103 nwritten = 1+4;
104 }
105 return nwritten;
106}

108/* Load an encoded length. The "isencoded" argument is set to 1 if the length
109 * is not actually a length but an "encoding type". See the REDIS_RDB_ENC_*
110 * definitions in rdb.h for more information. */
111uint32_t rdbLoadLen(rio *rdb, int *isencoded) {
112 unsigned char buf[2];
113 uint32_t len;
114 int type;
115
116 if (isencoded) *isencoded = 0;
117 if (rioRead(rdb,buf,1) == 0) return REDIS_RDB_LENERR;
118 type = (buf[0]&0xC0)>>6;
119 if (type == REDIS_RDB_ENCVAL) {
120 /* Read a 6 bit encoding type. */
121 if (isencoded) *isencoded = 1;
122 return buf[0]&0x3F;
123 } else if (type == REDIS_RDB_6BITLEN) {
124 /* Read a 6 bit len. */
125 return buf[0]&0x3F;
126 } else if (type == REDIS_RDB_14BITLEN) {
127 /* Read a 14 bit len. */
128 if (rioRead(rdb,buf+1,1) == 0) return REDIS_RDB_LENERR;
129 return ((buf[0]&0x3F)<<8)|buf[1];
130 } else {
131 /* Read a 32 bit len. */
132 if (rioRead(rdb,&len,4) == 0) return REDIS_RDB_LENERR;
133 return ntohl(len);
134 }
135}

137/* Encodes the "value" argument as integer when it fits in the supported ranges
138 * for encoded types. If the function successfully encodes the integer, the
139 * representation is stored in the buffer pointer to by "enc" and the string
140 * length is returned. Otherwise 0 is returned. */
141int rdbEncodeInteger(long long value, unsigned char *enc) {
142 if (value >= -(1<<7) && value <= (1<<7)-1) {
143 enc[0] = (REDIS_RDB_ENCVAL<<6)|REDIS_RDB_ENC_INT8;
144 enc[1] = value&0xFF;
145 return 2;
146 } else if (value >= -(1<<15) && value <= (1<<15)-1) {
147 enc[0] = (REDIS_RDB_ENCVAL<<6)|REDIS_RDB_ENC_INT16;
148 enc[1] = value&0xFF;
149 enc[2] = (value>>8)&0xFF;
150 return 3;
151 } else if (value >= -((long long)1<<31) && value <= ((long long)1<<31)-1) {
152 enc[0] = (REDIS_RDB_ENCVAL<<6)|REDIS_RDB_ENC_INT32;
153 enc[1] = value&0xFF;
154 enc[2] = (value>>8)&0xFF;
155 enc[3] = (value>>16)&0xFF;
156 enc[4] = (value>>24)&0xFF;
157 return 5;
158 } else {
159 return 0;
160 }
161}

163/* Loads an integer-encoded object with the specified encoding type "enctype".
164 * If the "encode" argument is set the function may return an integer-encoded
165 * string object, otherwise it always returns a raw string object. */
166robj *rdbLoadIntegerObject(rio *rdb, int enctype, int encode) {
167 unsigned char enc[4];
168 long long val;
169
170 if (enctype == REDIS_RDB_ENC_INT8) {
171 if (rioRead(rdb,enc,1) == 0) return NULL;
172 val = (signed char)enc[0];
173 } else if (enctype == REDIS_RDB_ENC_INT16) {
174 uint16_t v;
175 if (rioRead(rdb,enc,2) == 0) return NULL;
176 v = enc[0]|(enc[1]<<8);
177 val = (int16_t)v;
178 } else if (enctype == REDIS_RDB_ENC_INT32) {
179 uint32_t v;
180 if (rioRead(rdb,enc,4) == 0) return NULL;
181 v = enc[0]|(enc[1]<<8)|(enc[2]<<16)|(enc[3]<<24);
182 val = (int32_t)v;
183 } else {
184 val = 0; /* anti-warning */
185 redisPanic("Unknown RDB integer encoding type");
186 }
187 if (encode)
188 return createStringObjectFromLongLong(val);
189 else
190 return createObject(REDIS_STRING,sdsfromlonglong(val));
191}

193/* String objects in the form "2391" "-100" without any space and with a
194 * range of values that can fit in an 8, 16 or 32 bit signed value can be
195 * encoded as integers to save space */
196int rdbTryIntegerEncoding(char *s, size_t len, unsigned char *enc) {
197 long long value;
198 char *endptr, buf[32];
199
200 /* Check if it's possible to encode this value as a number */
201 value = strtoll(s, &endptr, 10);
202 if (endptr[0] != '\0') return 0;
203 ll2string(buf,32,value);
204
205 /* If the number converted back into a string is not identical
206 * then it's not possible to encode the string as integer */
207 if (strlen(buf) != len || memcmp(buf,s,len)) return 0;
208
209 return rdbEncodeInteger(value,enc);
210}

212int rdbSaveLzfStringObject(rio *rdb, unsigned char *s, size_t len) {
213 size_t comprlen, outlen;
214 unsigned char byte;
215 int n, nwritten = 0;
216 void *out;
217
218 /* We require at least four bytes compression for this to be worth it */
219 if (len <= 4) return 0;
220 outlen = len-4;
221 if ((out = zmalloc(outlen+1)) == NULL) return 0;
222 comprlen = lzf_compress(s, len, out, outlen);
223 if (comprlen == 0) {
224 zfree(out);
225 return 0;
226 }
227 /* Data compressed! Let's save it on disk */
228 byte = (REDIS_RDB_ENCVAL<<6)|REDIS_RDB_ENC_LZF;
229 if ((n = rdbWriteRaw(rdb,&byte,1)) == -1) goto writeerr;
230 nwritten += n;
231
232 if ((n = rdbSaveLen(rdb,comprlen)) == -1) goto writeerr;
233 nwritten += n;
234
235 if ((n = rdbSaveLen(rdb,len)) == -1) goto writeerr;
236 nwritten += n;
237
238 if ((n = rdbWriteRaw(rdb,out,comprlen)) == -1) goto writeerr;
239 nwritten += n;
240
241 zfree(out);
242 return nwritten;
243
244writeerr:
245 zfree(out);
246 return -1;
247}

249robj *rdbLoadLzfStringObject(rio *rdb) {
250 unsigned int len, clen;
251 unsigned char *c = NULL;
252 sds val = NULL;
253
254 if ((clen = rdbLoadLen(rdb,NULL)) == REDIS_RDB_LENERR) return NULL;
255 if ((len = rdbLoadLen(rdb,NULL)) == REDIS_RDB_LENERR) return NULL;
256 if ((c = zmalloc(clen)) == NULL) goto err;
257 if ((val = sdsnewlen(NULL,len)) == NULL) goto err;
258 if (rioRead(rdb,c,clen) == 0) goto err;
259 if (lzf_decompress(c,clen,val,len) == 0) goto err;
260 zfree(c);
261 return createObject(REDIS_STRING,val);
262err:
263 zfree(c);
264 sdsfree(val);
265 return NULL;
266}

268/* Save a string object as [len][data] on disk. If the object is a string
269 * representation of an integer value we try to save it in a special form */
270int rdbSaveRawString(rio *rdb, unsigned char *s, size_t len) {
271 int enclen;
272 int n, nwritten = 0;
273
274 /* Try integer encoding */
275 if (len <= 11) {
276 unsigned char buf[5];
277 if ((enclen = rdbTryIntegerEncoding((char*)s,len,buf)) > 0) {
278 if (rdbWriteRaw(rdb,buf,enclen) == -1) return -1;
279 return enclen;
280 }
281 }
282
283 /* Try LZF compression - under 20 bytes it's unable to compress even
284 * aaaaaaaaaaaaaaaaaa so skip it */
285 if (server.rdb_compression && len > 20) {
286 n = rdbSaveLzfStringObject(rdb,s,len);
287 if (n == -1) return -1;
288 if (n > 0) return n;
289 /* Return value of 0 means data can't be compressed, save the old way */
290 }
291
292 /* Store verbatim */
293 if ((n = rdbSaveLen(rdb,len)) == -1) return -1;
294 nwritten += n;
295 if (len > 0) {
296 if (rdbWriteRaw(rdb,s,len) == -1) return -1;
297 nwritten += len;
298 }
299 return nwritten;
300}

302/* Save a long long value as either an encoded string or a string. */
303int rdbSaveLongLongAsStringObject(rio *rdb, long long value) {
304 unsigned char buf[32];
305 int n, nwritten = 0;
306 int enclen = rdbEncodeInteger(value,buf);
307 if (enclen > 0) {
308 return rdbWriteRaw(rdb,buf,enclen);
309 } else {
310 /* Encode as string */
311 enclen = ll2string((char*)buf,32,value);
312 redisAssert(enclen < 32);
313 if ((n = rdbSaveLen(rdb,enclen)) == -1) return -1;
314 nwritten += n;
315 if ((n = rdbWriteRaw(rdb,buf,enclen)) == -1) return -1;
316 nwritten += n;
317 }
318 return nwritten;
319}

321/* Like rdbSaveStringObjectRaw() but handle encoded objects */
322int rdbSaveStringObject(rio *rdb, robj *obj) {
323 /* Avoid to decode the object, then encode it again, if the
324 * object is already integer encoded. */
325 if (obj->encoding == REDIS_ENCODING_INT) {
326 return rdbSaveLongLongAsStringObject(rdb,(long)obj->ptr);
327 } else {
328 redisAssertWithInfo(NULL,obj,sdsEncodedObject(obj));
329 return rdbSaveRawString(rdb,obj->ptr,sdslen(obj->ptr));
330 }
331}

333robj *rdbGenericLoadStringObject(rio *rdb, int encode) {
334 int isencoded;
335 uint32_t len;
336 robj *o;
337
338 len = rdbLoadLen(rdb,&isencoded);
339 if (isencoded) {
340 switch(len) {
341 case REDIS_RDB_ENC_INT8:
342 case REDIS_RDB_ENC_INT16:
343 case REDIS_RDB_ENC_INT32:
344 return rdbLoadIntegerObject(rdb,len,encode);
345 case REDIS_RDB_ENC_LZF:
346 return rdbLoadLzfStringObject(rdb);
347 default:
348 redisPanic("Unknown RDB encoding type");
349 }
350 }
351
352 if (len == REDIS_RDB_LENERR) return NULL;
353 o = encode ? createStringObject(NULL,len) :
354 createRawStringObject(NULL,len);
355 if (len && rioRead(rdb,o->ptr,len) == 0) {
356 decrRefCount(o);
357 return NULL;
358 }
359 return o;
360}

370/* Save a double value. Doubles are saved as strings prefixed by an unsigned
371 * 8 bit integer specifying the length of the representation.
372 * This 8 bit integer has special values in order to specify the following
373 * conditions:
374 * 253: not a number
375 * 254: + inf
376 * 255: - inf
377 */
378int rdbSaveDoubleValue(rio *rdb, double val) {
379 unsigned char buf[128];
380 int len;
381
382 if (isnan(val)) {
383 buf[0] = 253;
384 len = 1;
385 } else if (!isfinite(val)) {
386 len = 1;
387 buf[0] = (val < 0) ? 255 : 254;
388 } else {
389#if (DBL_MANT_DIG >= 52) && (LLONG_MAX == 0x7fffffffffffffffLL)
390 /* Check if the float is in a safe range to be casted into a
391 * long long. We are assuming that long long is 64 bit here.
392 * Also we are assuming that there are no implementations around where
393 * double has precision < 52 bit.
394 *
395 * Under this assumptions we test if a double is inside an interval
396 * where casting to long long is safe. Then using two castings we
397 * make sure the decimal part is zero. If all this is true we use
398 * integer printing function that is much faster. */
399 double min = -4503599627370495; /* (2^52)-1 */
400 double max = 4503599627370496; /* -(2^52) */
401 if (val > min && val < max && val == ((double)((long long)val)))
402 ll2string((char*)buf+1,sizeof(buf)-1,(long long)val);
403 else
404#endif
405 snprintf((char*)buf+1,sizeof(buf)-1,"%.17g",val);
406 buf[0] = strlen((char*)buf+1);
407 len = buf[0]+1;
408 }
409 return rdbWriteRaw(rdb,buf,len);
410}

412/* For information about double serialization check rdbSaveDoubleValue() */
413int rdbLoadDoubleValue(rio *rdb, double *val) {
414 char buf[256];
415 unsigned char len;
416
417 if (rioRead(rdb,&len,1) == 0) return -1;
418 switch(len) {
419 case 255: *val = R_NegInf; return 0;
420 case 254: *val = R_PosInf; return 0;
421 case 253: *val = R_Nan; return 0;
422 default:
423 if (rioRead(rdb,buf,len) == 0) return -1;
424 buf[len] = '\0';
425 sscanf(buf, "%lg", val);
426 return 0;
427 }
428}

430/* Save the object type of object "o". */
431int rdbSaveObjectType(rio *rdb, robj *o) {
432 switch (o->type) {
433 case REDIS_STRING:
434 return rdbSaveType(rdb,REDIS_RDB_TYPE_STRING);
435 case REDIS_LIST:
436 if (o->encoding == REDIS_ENCODING_ZIPLIST)
437 return rdbSaveType(rdb,REDIS_RDB_TYPE_LIST_ZIPLIST);
438 else if (o->encoding == REDIS_ENCODING_LINKEDLIST)
439 return rdbSaveType(rdb,REDIS_RDB_TYPE_LIST);
440 else
441 redisPanic("Unknown list encoding");
442 case REDIS_SET:
443 if (o->encoding == REDIS_ENCODING_INTSET)
444 return rdbSaveType(rdb,REDIS_RDB_TYPE_SET_INTSET);
445 else if (o->encoding == REDIS_ENCODING_HT)
446 return rdbSaveType(rdb,REDIS_RDB_TYPE_SET);
447 else
448 redisPanic("Unknown set encoding");
449 case REDIS_ZSET:
450 if (o->encoding == REDIS_ENCODING_ZIPLIST)
451 return rdbSaveType(rdb,REDIS_RDB_TYPE_ZSET_ZIPLIST);
452 else if (o->encoding == REDIS_ENCODING_SKIPLIST)
453 return rdbSaveType(rdb,REDIS_RDB_TYPE_ZSET);
454 else
455 redisPanic("Unknown sorted set encoding");
456 case REDIS_HASH:
457 if (o->encoding == REDIS_ENCODING_ZIPLIST)
458 return rdbSaveType(rdb,REDIS_RDB_TYPE_HASH_ZIPLIST);
459 else if (o->encoding == REDIS_ENCODING_HT)
460 return rdbSaveType(rdb,REDIS_RDB_TYPE_HASH);
461 else
462 redisPanic("Unknown hash encoding");
463 default:
464 redisPanic("Unknown object type");
465 }
466 return -1; /* avoid warning */
467}

478/* Save a Redis object. Returns -1 on error, number of bytes written on success. */
479int rdbSaveObject(rio *rdb, robj *o) {
480 int n, nwritten = 0;
481
482 if (o->type == REDIS_STRING) {
483 /* Save a string value */
484 if ((n = rdbSaveStringObject(rdb,o)) == -1) return -1;
485 nwritten += n;
486 } else if (o->type == REDIS_LIST) {
487 /* Save a list value */
488 if (o->encoding == REDIS_ENCODING_ZIPLIST) {
489 size_t l = ziplistBlobLen((unsigned char*)o->ptr);
490
491 if ((n = rdbSaveRawString(rdb,o->ptr,l)) == -1) return -1;
492 nwritten += n;
493 } else if (o->encoding == REDIS_ENCODING_LINKEDLIST) {
494 list *list = o->ptr;
495 listIter li;
496 listNode *ln;
497
498 if ((n = rdbSaveLen(rdb,listLength(list))) == -1) return -1;
499 nwritten += n;
500
501 listRewind(list,&li);
502 while((ln = listNext(&li))) {
503 robj *eleobj = listNodeValue(ln);
504 if ((n = rdbSaveStringObject(rdb,eleobj)) == -1) return -1;
505 nwritten += n;
506 }
507 } else {
508 redisPanic("Unknown list encoding");
509 }
510 } else if (o->type == REDIS_SET) {
511 /* Save a set value */
512 if (o->encoding == REDIS_ENCODING_HT) {
513 dict *set = o->ptr;
514 dictIterator *di = dictGetIterator(set);
515 dictEntry *de;
516
517 if ((n = rdbSaveLen(rdb,dictSize(set))) == -1) return -1;
518 nwritten += n;
519
520 while((de = dictNext(di)) != NULL) {
521 robj *eleobj = dictGetKey(de);
522 if ((n = rdbSaveStringObject(rdb,eleobj)) == -1) return -1;
523 nwritten += n;
524 }
525 dictReleaseIterator(di);
526 } else if (o->encoding == REDIS_ENCODING_INTSET) {
527 size_t l = intsetBlobLen((intset*)o->ptr);
528
529 if ((n = rdbSaveRawString(rdb,o->ptr,l)) == -1) return -1;
530 nwritten += n;
531 } else {
532 redisPanic("Unknown set encoding");
533 }
534 } else if (o->type == REDIS_ZSET) {
535 /* Save a sorted set value */
536 if (o->encoding == REDIS_ENCODING_ZIPLIST) {
537 size_t l = ziplistBlobLen((unsigned char*)o->ptr);
538
539 if ((n = rdbSaveRawString(rdb,o->ptr,l)) == -1) return -1;
540 nwritten += n;
541 } else if (o->encoding == REDIS_ENCODING_SKIPLIST) {
542 zset *zs = o->ptr;
543 dictIterator *di = dictGetIterator(zs->dict);
544 dictEntry *de;
545
546 if ((n = rdbSaveLen(rdb,dictSize(zs->dict))) == -1) return -1;
547 nwritten += n;
548
549 while((de = dictNext(di)) != NULL) {
550 robj *eleobj = dictGetKey(de);
551 double *score = dictGetVal(de);
552
553 if ((n = rdbSaveStringObject(rdb,eleobj)) == -1) return -1;
554 nwritten += n;
555 if ((n = rdbSaveDoubleValue(rdb,*score)) == -1) return -1;
556 nwritten += n;
557 }
558 dictReleaseIterator(di);
559 } else {
560 redisPanic("Unknown sorted set encoding");
561 }
562 } else if (o->type == REDIS_HASH) {
563 /* Save a hash value */
564 if (o->encoding == REDIS_ENCODING_ZIPLIST) {
565 size_t l = ziplistBlobLen((unsigned char*)o->ptr);
566
567 if ((n = rdbSaveRawString(rdb,o->ptr,l)) == -1) return -1;
568 nwritten += n;
569
570 } else if (o->encoding == REDIS_ENCODING_HT) {
571 dictIterator *di = dictGetIterator(o->ptr);
572 dictEntry *de;
573
574 if ((n = rdbSaveLen(rdb,dictSize((dict*)o->ptr))) == -1) return -1;
575 nwritten += n;
576
577 while((de = dictNext(di)) != NULL) {
578 robj *key = dictGetKey(de);
579 robj *val = dictGetVal(de);
580
581 if ((n = rdbSaveStringObject(rdb,key)) == -1) return -1;
582 nwritten += n;
583 if ((n = rdbSaveStringObject(rdb,val)) == -1) return -1;
584 nwritten += n;
585 }
586 dictReleaseIterator(di);
587
588 } else {
589 redisPanic("Unknown hash encoding");
590 }
591
592 } else {
593 redisPanic("Unknown object type");
594 }
595 return nwritten;
596}

608/* Save a key-value pair, with expire time, type, key, value.
609 * On error -1 is returned.
610 * On success if the key was actually saved 1 is returned, otherwise 0
611 * is returned (the key was already expired). */
612int rdbSaveKeyValuePair(rio *rdb, robj *key, robj *val,
613 long long expiretime, long long now)
614{
615 /* Save the expire time */
616 if (expiretime != -1) {
617 /* If this key is already expired skip it */
618 if (expiretime < now) return 0;
619 if (rdbSaveType(rdb,REDIS_RDB_OPCODE_EXPIRETIME_MS) == -1) return -1;
620 if (rdbSaveMillisecondTime(rdb,expiretime) == -1) return -1;
621 }
622
623 /* Save type, key, value */
624 if (rdbSaveObjectType(rdb,val) == -1) return -1;
625 if (rdbSaveStringObject(rdb,key) == -1) return -1;
626 if (rdbSaveObject(rdb,val) == -1) return -1;
627 return 1;
628}

630/* Produces a dump of the database in RDB format sending it to the specified
631 * Redis I/O channel. On success REDIS_OK is returned, otherwise REDIS_ERR
632 * is returned and part of the output, or all the output, can be
633 * missing because of I/O errors.
634 *
635 * When the function returns REDIS_ERR and if 'error' is not NULL, the
636 * integer pointed by 'error' is set to the value of errno just after the I/O
637 * error. */
638int rdbSaveRio(rio *rdb, int *error) {
639 dictIterator *di = NULL;
640 dictEntry *de;
641 char magic[10];
642 int j;
643 long long now = mstime();
644 uint64_t cksum;
645
646 if (server.rdb_checksum)
647 rdb->update_cksum = rioGenericUpdateChecksum;
648 snprintf(magic,sizeof(magic),"REDIS%04d",REDIS_RDB_VERSION);
649 if (rdbWriteRaw(rdb,magic,9) == -1) goto werr;
650
651 for (j = 0; j < server.dbnum; j++) {
652 redisDb *db = server.db+j;
653 dict *d = db->dict;
654 if (dictSize(d) == 0) continue;
655 di = dictGetSafeIterator(d);
656 if (!di) return REDIS_ERR;
657
658 /* Write the SELECT DB opcode */
659 if (rdbSaveType(rdb,REDIS_RDB_OPCODE_SELECTDB) == -1) goto werr;
660 if (rdbSaveLen(rdb,j) == -1) goto werr;
661
662 /* Iterate this DB writing every entry */
663 while((de = dictNext(di)) != NULL) {
664 sds keystr = dictGetKey(de);
665 robj key, *o = dictGetVal(de);
666 long long expire;
667
668 initStaticStringObject(key,keystr);
669 expire = getExpire(db,&key);
670 if (rdbSaveKeyValuePair(rdb,&key,o,expire,now) == -1) goto werr;
671 }
672 dictReleaseIterator(di);
673 }
674 di = NULL; /* So that we don't release it again on error. */
675
676 /* EOF opcode */
677 if (rdbSaveType(rdb,REDIS_RDB_OPCODE_EOF) == -1) goto werr;
678
679 /* CRC64 checksum. It will be zero if checksum computation is disabled, the
680 * loading code skips the check in this case. */
681 cksum = rdb->cksum;
682 memrev64ifbe(&cksum);
683 if (rioWrite(rdb,&cksum,8) == 0) goto werr;
684 return REDIS_OK;
685
686werr:
687 if (error) *error = errno;
688 if (di) dictReleaseIterator(di);
689 return REDIS_ERR;
690}
691
692/* This is just a wrapper to rdbSaveRio() that additionally adds a prefix
693 * and a suffix to the generated RDB dump. The prefix is:
694 *
695 * $EOF:<40 bytes unguessable hex string>\r\n
696 *
697 * While the suffix is the 40 bytes hex string we announced in the prefix.
698 * This way processes receiving the payload can understand when it ends
699 * without doing any processing of the content. */
700int rdbSaveRioWithEOFMark(rio *rdb, int *error) {
701 char eofmark[REDIS_EOF_MARK_SIZE];
702
703 getRandomHexChars(eofmark,REDIS_EOF_MARK_SIZE);
704 if (error) *error = 0;
705 if (rioWrite(rdb,"$EOF:",5) == 0) goto werr;
706 if (rioWrite(rdb,eofmark,REDIS_EOF_MARK_SIZE) == 0) goto werr;
707 if (rioWrite(rdb,"\r\n",2) == 0) goto werr;
708 if (rdbSaveRio(rdb,error) == REDIS_ERR) goto werr;
709 if (rioWrite(rdb,eofmark,REDIS_EOF_MARK_SIZE) == 0) goto werr;
710 return REDIS_OK;
711
712werr: /* Write error. */
713 /* Set 'error' only if not already set by rdbSaveRio() call. */
714 if (error && *error == 0) *error = errno;
715 return REDIS_ERR;
716}

718/* Save the DB on disk. Return REDIS_ERR on error, REDIS_OK on success. */
719int rdbSave(char *filename) {
720 char tmpfile[256];
721 FILE *fp;
722 rio rdb;
723 int error;
724
725 snprintf(tmpfile,256,"temp-%d.rdb", (int) getpid());
726 fp = fopen(tmpfile,"w");
727 if (!fp) {
728 redisLog(REDIS_WARNING, "Failed opening .rdb for saving: %s",
729 strerror(errno));
730 return REDIS_ERR;
731 }
732
733 rioInitWithFile(&rdb,fp);
734 if (rdbSaveRio(&rdb,&error) == REDIS_ERR) {
735 errno = error;
736 goto werr;
737 }
738
739 /* Make sure data will not remain on the OS's output buffers */
740 if (fflush(fp) == EOF) goto werr;
741 if (fsync(fileno(fp)) == -1) goto werr;
742 if (fclose(fp) == EOF) goto werr;
743
744 /* Use RENAME to make sure the DB file is changed atomically only
745 * if the generate DB file is ok. */
746 if (rename(tmpfile,filename) == -1) {
747 redisLog(REDIS_WARNING,"Error moving temp DB file on the final destination: %s", strerror(errno));
748 unlink(tmpfile);
749 return REDIS_ERR;
750 }
751 redisLog(REDIS_NOTICE,"DB saved on disk");
752 server.dirty = 0;
753 server.lastsave = time(NULL);
754 server.lastbgsave_status = REDIS_OK;
755 return REDIS_OK;
756
757werr:
758 fclose(fp);
759 unlink(tmpfile);
760 redisLog(REDIS_WARNING,"Write error saving DB on disk: %s", strerror(errno));
761 return REDIS_ERR;
762}

819/* Load a Redis object of the specified type from the specified file.
820 * On success a newly allocated object is returned, otherwise NULL. */
821robj *rdbLoadObject(int rdbtype, rio *rdb) {
822 robj *o, *ele, *dec;
823 size_t len;
824 unsigned int i;
825
826 if (rdbtype == REDIS_RDB_TYPE_STRING) {
827 /* Read string value */
828 if ((o = rdbLoadEncodedStringObject(rdb)) == NULL) return NULL;
829 o = tryObjectEncoding(o);
830 } else if (rdbtype == REDIS_RDB_TYPE_LIST) {
831 /* Read list value */
832 if ((len = rdbLoadLen(rdb,NULL)) == REDIS_RDB_LENERR) return NULL;
833
834 /* Use a real list when there are too many entries */
835 if (len > server.list_max_ziplist_entries) {
836 o = createListObject();
837 } else {
838 o = createZiplistObject();
839 }
840
841 /* Load every single element of the list */
842 while(len--) {
843 if ((ele = rdbLoadEncodedStringObject(rdb)) == NULL) return NULL;
844
845 /* If we are using a ziplist and the value is too big, convert
846 * the object to a real list. */
847 if (o->encoding == REDIS_ENCODING_ZIPLIST &&
848 sdsEncodedObject(ele) &&
849 sdslen(ele->ptr) > server.list_max_ziplist_value)
850 listTypeConvert(o,REDIS_ENCODING_LINKEDLIST);
851
852 if (o->encoding == REDIS_ENCODING_ZIPLIST) {
853 dec = getDecodedObject(ele);
854 o->ptr = ziplistPush(o->ptr,dec->ptr,sdslen(dec->ptr),REDIS_TAIL);
855 decrRefCount(dec);
856 decrRefCount(ele);
857 } else {
858 ele = tryObjectEncoding(ele);
859 listAddNodeTail(o->ptr,ele);
860 }
861 }
862 } else if (rdbtype == REDIS_RDB_TYPE_SET) {
863 /* Read list/set value */
864 if ((len = rdbLoadLen(rdb,NULL)) == REDIS_RDB_LENERR) return NULL;
865
866 /* Use a regular set when there are too many entries. */
867 if (len > server.set_max_intset_entries) {
868 o = createSetObject();
869 /* It's faster to expand the dict to the right size asap in order
870 * to avoid rehashing */
871 if (len > DICT_HT_INITIAL_SIZE)
872 dictExpand(o->ptr,len);
873 } else {
874 o = createIntsetObject();
875 }
876
877 /* Load every single element of the list/set */
878 for (i = 0; i < len; i++) {
879 long long llval;
880 if ((ele = rdbLoadEncodedStringObject(rdb)) == NULL) return NULL;
881 ele = tryObjectEncoding(ele);
882
883 if (o->encoding == REDIS_ENCODING_INTSET) {
884 /* Fetch integer value from element */
885 if (isObjectRepresentableAsLongLong(ele,&llval) == REDIS_OK) {
886 o->ptr = intsetAdd(o->ptr,llval,NULL);
887 } else {
888 setTypeConvert(o,REDIS_ENCODING_HT);
889 dictExpand(o->ptr,len);
890 }
891 }
892
893 /* This will also be called when the set was just converted
894 * to a regular hash table encoded set */
895 if (o->encoding == REDIS_ENCODING_HT) {
896 dictAdd((dict*)o->ptr,ele,NULL);
897 } else {
898 decrRefCount(ele);
899 }
900 }
901 } else if (rdbtype == REDIS_RDB_TYPE_ZSET) {
902 /* Read list/set value */
903 size_t zsetlen;
904 size_t maxelelen = 0;
905 zset *zs;
906
907 if ((zsetlen = rdbLoadLen(rdb,NULL)) == REDIS_RDB_LENERR) return NULL;
908 o = createZsetObject();
909 zs = o->ptr;
910
911 /* Load every single element of the list/set */
912 while(zsetlen--) {
913 robj *ele;
914 double score;
915 zskiplistNode *znode;
916
917 if ((ele = rdbLoadEncodedStringObject(rdb)) == NULL) return NULL;
918 ele = tryObjectEncoding(ele);
919 if (rdbLoadDoubleValue(rdb,&score) == -1) return NULL;
920
921 /* Don't care about integer-encoded strings. */
922 if (sdsEncodedObject(ele) && sdslen(ele->ptr) > maxelelen)
923 maxelelen = sdslen(ele->ptr);
924
925 znode = zslInsert(zs->zsl,score,ele);
926 dictAdd(zs->dict,ele,&znode->score);
927 incrRefCount(ele); /* added to skiplist */
928 }
929
930 /* Convert *after* loading, since sorted sets are not stored ordered. */
931 if (zsetLength(o) <= server.zset_max_ziplist_entries &&
932 maxelelen <= server.zset_max_ziplist_value)
933 zsetConvert(o,REDIS_ENCODING_ZIPLIST);
934 } else if (rdbtype == REDIS_RDB_TYPE_HASH) {
935 size_t len;
936 int ret;
937
938 len = rdbLoadLen(rdb, NULL);
939 if (len == REDIS_RDB_LENERR) return NULL;
940
941 o = createHashObject();
942
943 /* Too many entries? Use a hash table. */
944 if (len > server.hash_max_ziplist_entries)
945 hashTypeConvert(o, REDIS_ENCODING_HT);
946
947 /* Load every field and value into the ziplist */
948 while (o->encoding == REDIS_ENCODING_ZIPLIST && len > 0) {
949 robj *field, *value;
950
951 len--;
952 /* Load raw strings */
953 field = rdbLoadStringObject(rdb);
954 if (field == NULL) return NULL;
955 redisAssert(sdsEncodedObject(field));
956 value = rdbLoadStringObject(rdb);
957 if (value == NULL) return NULL;
958 redisAssert(sdsEncodedObject(value));
959
960 /* Add pair to ziplist */
961 o->ptr = ziplistPush(o->ptr, field->ptr, sdslen(field->ptr), ZIPLIST_TAIL);
962 o->ptr = ziplistPush(o->ptr, value->ptr, sdslen(value->ptr), ZIPLIST_TAIL);
963 /* Convert to hash table if size threshold is exceeded */
964 if (sdslen(field->ptr) > server.hash_max_ziplist_value ||
965 sdslen(value->ptr) > server.hash_max_ziplist_value)
966 {
967 decrRefCount(field);
968 decrRefCount(value);
969 hashTypeConvert(o, REDIS_ENCODING_HT);
970 break;
971 }
972 decrRefCount(field);
973 decrRefCount(value);
974 }
975
976 /* Load remaining fields and values into the hash table */
977 while (o->encoding == REDIS_ENCODING_HT && len > 0) {
978 robj *field, *value;
979
980 len--;
981 /* Load encoded strings */
982 field = rdbLoadEncodedStringObject(rdb);
983 if (field == NULL) return NULL;
984 value = rdbLoadEncodedStringObject(rdb);
985 if (value == NULL) return NULL;
986
987 field = tryObjectEncoding(field);
988 value = tryObjectEncoding(value);
989
990 /* Add pair to hash table */
991 ret = dictAdd((dict*)o->ptr, field, value);
992 redisAssert(ret == DICT_OK);
993 }
994
995 /* All pairs should be read by now */
996 redisAssert(len == 0);
997
998 } else if (rdbtype == REDIS_RDB_TYPE_HASH_ZIPMAP ||
999 rdbtype == REDIS_RDB_TYPE_LIST_ZIPLIST ||
1000 rdbtype == REDIS_RDB_TYPE_SET_INTSET ||
1001 rdbtype == REDIS_RDB_TYPE_ZSET_ZIPLIST ||
1002 rdbtype == REDIS_RDB_TYPE_HASH_ZIPLIST)
1003 {
1004 robj *aux = rdbLoadStringObject(rdb);
1005
1006 if (aux == NULL) return NULL;
1007 o = createObject(REDIS_STRING,NULL); /* string is just placeholder */
1008 o->ptr = zmalloc(sdslen(aux->ptr));
1009 memcpy(o->ptr,aux->ptr,sdslen(aux->ptr));
1010 decrRefCount(aux);
1011
1012 /* Fix the object encoding, and make sure to convert the encoded
1013 * data type into the base type if accordingly to the current
1014 * configuration there are too many elements in the encoded data
1015 * type. Note that we only check the length and not max element
1016 * size as this is an O(N) scan. Eventually everything will get
1017 * converted. */
1018 switch(rdbtype) {
1019 case REDIS_RDB_TYPE_HASH_ZIPMAP:
1020 /* Convert to ziplist encoded hash. This must be deprecated
1021 * when loading dumps created by Redis 2.4 gets deprecated. */
1022 {
1023 unsigned char *zl = ziplistNew();
1024 unsigned char *zi = zipmapRewind(o->ptr);
1025 unsigned char *fstr, *vstr;
1026 unsigned int flen, vlen;
1027 unsigned int maxlen = 0;
1028
1029 while ((zi = zipmapNext(zi, &fstr, &flen, &vstr, &vlen)) != NULL) {
1030 if (flen > maxlen) maxlen = flen;
1031 if (vlen > maxlen) maxlen = vlen;
1032 zl = ziplistPush(zl, fstr, flen, ZIPLIST_TAIL);
1033 zl = ziplistPush(zl, vstr, vlen, ZIPLIST_TAIL);
1034 }
1035
1036 zfree(o->ptr);
1037 o->ptr = zl;
1038 o->type = REDIS_HASH;
1039 o->encoding = REDIS_ENCODING_ZIPLIST;
1040
1041 if (hashTypeLength(o) > server.hash_max_ziplist_entries ||
1042 maxlen > server.hash_max_ziplist_value)
1043 {
1044 hashTypeConvert(o, REDIS_ENCODING_HT);
1045 }
1046 }
1047 break;
1048 case REDIS_RDB_TYPE_LIST_ZIPLIST:
1049 o->type = REDIS_LIST;
1050 o->encoding = REDIS_ENCODING_ZIPLIST;
1051 if (ziplistLen(o->ptr) > server.list_max_ziplist_entries)
1052 listTypeConvert(o,REDIS_ENCODING_LINKEDLIST);
1053 break;
1054 case REDIS_RDB_TYPE_SET_INTSET:
1055 o->type = REDIS_SET;
1056 o->encoding = REDIS_ENCODING_INTSET;
1057 if (intsetLen(o->ptr) > server.set_max_intset_entries)
1058 setTypeConvert(o,REDIS_ENCODING_HT);
1059 break;
1060 case REDIS_RDB_TYPE_ZSET_ZIPLIST:
1061 o->type = REDIS_ZSET;
1062 o->encoding = REDIS_ENCODING_ZIPLIST;
1063 if (zsetLength(o) > server.zset_max_ziplist_entries)
1064 zsetConvert(o,REDIS_ENCODING_SKIPLIST);
1065 break;
1066 case REDIS_RDB_TYPE_HASH_ZIPLIST:
1067 o->type = REDIS_HASH;
1068 o->encoding = REDIS_ENCODING_ZIPLIST;
1069 if (hashTypeLength(o) > server.hash_max_ziplist_entries)
1070 hashTypeConvert(o, REDIS_ENCODING_HT);
1071 break;
1072 default:
1073 redisPanic("Unknown encoding");
1074 break;
1075 }
1076 } else {
1077 redisPanic("Unknown object type");
1078 }
1079 return o;
1080}

1129int rdbLoad(char *filename) {
1130 uint32_t dbid;
1131 int type, rdbver;
1132 redisDb *db = server.db+0;
1133 char buf[1024];
1134 long long expiretime, now = mstime();
1135 FILE *fp;
1136 rio rdb;
1137
1138 if ((fp = fopen(filename,"r")) == NULL) return REDIS_ERR;
1139
1140 rioInitWithFile(&rdb,fp);
1141 rdb.update_cksum = rdbLoadProgressCallback;
1142 rdb.max_processing_chunk = server.loading_process_events_interval_bytes;
1143 if (rioRead(&rdb,buf,9) == 0) goto eoferr;
1144 buf[9] = '\0';
1145 if (memcmp(buf,"REDIS",5) != 0) {
1146 fclose(fp);
1147 redisLog(REDIS_WARNING,"Wrong signature trying to load DB from file");
1148 errno = EINVAL;
1149 return REDIS_ERR;
1150 }
1151 rdbver = atoi(buf+5);
1152 if (rdbver < 1 || rdbver > REDIS_RDB_VERSION) {
1153 fclose(fp);
1154 redisLog(REDIS_WARNING,"Can't handle RDB format version %d",rdbver);
1155 errno = EINVAL;
1156 return REDIS_ERR;
1157 }
1158
1159 startLoading(fp);
1160 while(1) {
1161 robj *key, *val;
1162 expiretime = -1;
1163
1164 /* Read type. */
1165 if ((type = rdbLoadType(&rdb)) == -1) goto eoferr;
1166 if (type == REDIS_RDB_OPCODE_EXPIRETIME) {
1167 if ((expiretime = rdbLoadTime(&rdb)) == -1) goto eoferr;
1168 /* We read the time so we need to read the object type again. */
1169 if ((type = rdbLoadType(&rdb)) == -1) goto eoferr;
1170 /* the EXPIRETIME opcode specifies time in seconds, so convert
1171 * into milliseconds. */
1172 expiretime *= 1000;
1173 } else if (type == REDIS_RDB_OPCODE_EXPIRETIME_MS) {
1174 /* Milliseconds precision expire times introduced with RDB
1175 * version 3. */
1176 if ((expiretime = rdbLoadMillisecondTime(&rdb)) == -1) goto eoferr;
1177 /* We read the time so we need to read the object type again. */
1178 if ((type = rdbLoadType(&rdb)) == -1) goto eoferr;
1179 }
1180
1181 if (type == REDIS_RDB_OPCODE_EOF)
1182 break;
1183
1184 /* Handle SELECT DB opcode as a special case */
1185 if (type == REDIS_RDB_OPCODE_SELECTDB) {
1186 if ((dbid = rdbLoadLen(&rdb,NULL)) == REDIS_RDB_LENERR)
1187 goto eoferr;
1188 if (dbid >= (unsigned)server.dbnum) {
1189 redisLog(REDIS_WARNING,"FATAL: Data file was created with a Redis server configured to handle more than %d databases. Exiting\n", server.dbnum);
1190 exit(1);
1191 }
1192 db = server.db+dbid;
1193 continue;
1194 }
1195 /* Read key */
1196 if ((key = rdbLoadStringObject(&rdb)) == NULL) goto eoferr;
1197 /* Read value */
1198 if ((val = rdbLoadObject(type,&rdb)) == NULL) goto eoferr;
1199 /* Check if the key already expired. This function is used when loading
1200 * an RDB file from disk, either at startup, or when an RDB was
1201 * received from the master. In the latter case, the master is
1202 * responsible for key expiry. If we would expire keys here, the
1203 * snapshot taken by the master may not be reflected on the slave. */
1204 if (server.masterhost == NULL && expiretime != -1 && expiretime < now) {
1205 decrRefCount(key);
1206 decrRefCount(val);
1207 continue;
1208 }
1209 /* Add the new object in the hash table */
1210 dbAdd(db,key,val);
1211
1212 /* Set the expire time if needed */
1213 if (expiretime != -1) setExpire(db,key,expiretime);
1214
1215 decrRefCount(key);
1216 }
1217 /* Verify the checksum if RDB version is >= 5 */
1218 if (rdbver >= 5 && server.rdb_checksum) {
1219 uint64_t cksum, expected = rdb.cksum;
1220
1221 if (rioRead(&rdb,&cksum,8) == 0) goto eoferr;
1222 memrev64ifbe(&cksum);
1223 if (cksum == 0) {
1224 redisLog(REDIS_WARNING,"RDB file was saved with checksum disabled: no check performed.");
1225 } else if (cksum != expected) {
1226 redisLog(REDIS_WARNING,"Wrong RDB checksum. Aborting now.");
1227 exit(1);
1228 }
1229 }
1230
1231 fclose(fp);
1232 stopLoading();
1233 return REDIS_OK;
1234
1235eoferr: /* unexpected end of file is handled here with a fatal exit */
1236 redisLog(REDIS_WARNING,"Short read or OOM loading DB. Unrecoverable error, aborting now.");
1237 exit(1);
1238 return REDIS_ERR; /* Just to avoid warning */
1239}

1241/* A background saving child (BGSAVE) terminated its work. Handle this.
1242 * This function covers the case of actual BGSAVEs. */
1243void backgroundSaveDoneHandlerDisk(int exitcode, int bysignal) {
1244 if (!bysignal && exitcode == 0) {
1245 redisLog(REDIS_NOTICE,
1246 "Background saving terminated with success");
1247 server.dirty = server.dirty - server.dirty_before_bgsave;
1248 server.lastsave = time(NULL);
1249 server.lastbgsave_status = REDIS_OK;
1250 } else if (!bysignal && exitcode != 0) {
1251 redisLog(REDIS_WARNING, "Background saving error");
1252 server.lastbgsave_status = REDIS_ERR;
1253 } else {
1254 mstime_t latency;
1255
1256 redisLog(REDIS_WARNING,
1257 "Background saving terminated by signal %d", bysignal);
1258 latencyStartMonitor(latency);
1259 rdbRemoveTempFile(server.rdb_child_pid);
1260 latencyEndMonitor(latency);
1261 latencyAddSampleIfNeeded("rdb-unlink-temp-file",latency);
1262 /* SIGUSR1 is whitelisted, so we have a way to kill a child without
1263 * tirggering an error conditon. */
1264 if (bysignal != SIGUSR1)
1265 server.lastbgsave_status = REDIS_ERR;
1266 }
1267 server.rdb_child_pid = -1;
1268 server.rdb_child_type = REDIS_RDB_CHILD_TYPE_NONE;
1269 server.rdb_save_time_last = time(NULL)-server.rdb_save_time_start;
1270 server.rdb_save_time_start = -1;
1271 /* Possibly there are slaves waiting for a BGSAVE in order to be served
1272 * (the first stage of SYNC is a bulk transfer of dump.rdb) */
1273 updateSlavesWaitingBgsave((!bysignal && exitcode == 0) ? REDIS_OK : REDIS_ERR, REDIS_RDB_CHILD_TYPE_DISK);
1274}

1276/* A background saving child (BGSAVE) terminated its work. Handle this.
1277 * This function covers the case of RDB -> Salves socket transfers for
1278 * diskless replication. */
1279void backgroundSaveDoneHandlerSocket(int exitcode, int bysignal) {
1280 uint64_t *ok_slaves;
1281
1282 if (!bysignal && exitcode == 0) {
1283 redisLog(REDIS_NOTICE,
1284 "Background RDB transfer terminated with success");
1285 } else if (!bysignal && exitcode != 0) {
1286 redisLog(REDIS_WARNING, "Background transfer error");
1287 } else {
1288 redisLog(REDIS_WARNING,
1289 "Background transfer terminated by signal %d", bysignal);
1290 }
1291 server.rdb_child_pid = -1;
1292 server.rdb_child_type = REDIS_RDB_CHILD_TYPE_NONE;
1293 server.rdb_save_time_start = -1;
1294
1295 /* If the child returns an OK exit code, read the set of slave client
1296 * IDs and the associated status code. We'll terminate all the slaves
1297 * in error state.
1298 *
1299 * If the process returned an error, consider the list of slaves that
1300 * can continue to be emtpy, so that it's just a special case of the
1301 * normal code path. */
1302 ok_slaves = zmalloc(sizeof(uint64_t)); /* Make space for the count. */
1303 ok_slaves[0] = 0;
1304 if (!bysignal && exitcode == 0) {
1305 int readlen = sizeof(uint64_t);
1306
1307 if (read(server.rdb_pipe_read_result_from_child, ok_slaves, readlen) ==
1308 readlen)
1309 {
1310 readlen = ok_slaves[0]*sizeof(uint64_t)*2;
1311
1312 /* Make space for enough elements as specified by the first
1313 * uint64_t element in the array. */
1314 ok_slaves = zrealloc(ok_slaves,sizeof(uint64_t)+readlen);
1315 if (readlen &&
1316 read(server.rdb_pipe_read_result_from_child, ok_slaves+1,
1317 readlen) != readlen)
1318 {
1319 ok_slaves[0] = 0;
1320 }
1321 }
1322 }
1323
1324 close(server.rdb_pipe_read_result_from_child);
1325 close(server.rdb_pipe_write_result_to_parent);
1326
1327 /* We can continue the replication process with all the slaves that
1328 * correctly received the full payload. Others are terminated. */
1329 listNode *ln;
1330 listIter li;
1331
1332 listRewind(server.slaves,&li);
1333 while((ln = listNext(&li))) {
1334 redisClient *slave = ln->value;
1335
1336 if (slave->replstate == REDIS_REPL_WAIT_BGSAVE_END) {
1337 uint64_t j;
1338 int errorcode = 0;
1339
1340 /* Search for the slave ID in the reply. In order for a slave to
1341 * continue the replication process, we need to find it in the list,
1342 * and it must have an error code set to 0 (which means success). */
1343 for (j = 0; j < ok_slaves[0]; j++) {
1344 if (slave->id == ok_slaves[2*j+1]) {
1345 errorcode = ok_slaves[2*j+2];
1346 break; /* Found in slaves list. */
1347 }
1348 }
1349 if (j == ok_slaves[0] || errorcode != 0) {
1350 redisLog(REDIS_WARNING,
1351 "Closing slave %s: child->slave RDB transfer failed: %s",
1352 replicationGetSlaveName(slave),
1353 (errorcode == 0) ? "RDB transfer child aborted"
1354 : strerror(errorcode));
1355 freeClient(slave);
1356 } else {
1357 redisLog(REDIS_WARNING,
1358 "Slave %s correctly received the streamed RDB file.",
1359 replicationGetSlaveName(slave));
1360 /* Restore the socket as non-blocking. */
1361 anetNonBlock(NULL,slave->fd);
1362 anetSendTimeout(NULL,slave->fd,0);
1363 }
1364 }
1365 }
1366 zfree(ok_slaves);
1367
1368 updateSlavesWaitingBgsave((!bysignal && exitcode == 0) ? REDIS_OK : REDIS_ERR, REDIS_RDB_CHILD_TYPE_SOCKET);
1369}

1371/* When a background RDB saving/transfer terminates, call the right handler. */
1372void backgroundSaveDoneHandler(int exitcode, int bysignal) {
1373 switch(server.rdb_child_type) {
1374 case REDIS_RDB_CHILD_TYPE_DISK:
1375 backgroundSaveDoneHandlerDisk(exitcode,bysignal);
1376 break;
1377 case REDIS_RDB_CHILD_TYPE_SOCKET:
1378 backgroundSaveDoneHandlerSocket(exitcode,bysignal);
1379 break;
1380 default:
1381 redisPanic("Unknown RDB child type.");
1382 break;
1383 }
1384}

1386/* Spawn an RDB child that writes the RDB to the sockets of the slaves
1387 * that are currently in REDIS_REPL_WAIT_BGSAVE_START state. */
1388int rdbSaveToSlavesSockets(void) {
1389 int *fds;
1390 uint64_t *clientids;
1391 int numfds;
1392 listNode *ln;
1393 listIter li;
1394 pid_t childpid;
1395 long long start;
1396 int pipefds[2];
1397
1398 if (server.rdb_child_pid != -1) return REDIS_ERR;
1399
1400 /* Before to fork, create a pipe that will be used in order to
1401 * send back to the parent the IDs of the slaves that successfully
1402 * received all the writes. */
1403 if (pipe(pipefds) == -1) return REDIS_ERR;
1404 server.rdb_pipe_read_result_from_child = pipefds[0];
1405 server.rdb_pipe_write_result_to_parent = pipefds[1];
1406
1407 /* Collect the file descriptors of the slaves we want to transfer
1408 * the RDB to, which are i WAIT_BGSAVE_START state. */
1409 fds = zmalloc(sizeof(int)*listLength(server.slaves));
1410 /* We also allocate an array of corresponding client IDs. This will
1411 * be useful for the child process in order to build the report
1412 * (sent via unix pipe) that will be sent to the parent. */
1413 clientids = zmalloc(sizeof(uint64_t)*listLength(server.slaves));
1414 numfds = 0;
1415
1416 listRewind(server.slaves,&li);
1417 while((ln = listNext(&li))) {
1418 redisClient *slave = ln->value;
1419
1420 if (slave->replstate == REDIS_REPL_WAIT_BGSAVE_START) {
1421 clientids[numfds] = slave->id;
1422 fds[numfds++] = slave->fd;
1423 slave->replstate = REDIS_REPL_WAIT_BGSAVE_END;
1424 /* Put the socket in non-blocking mode to simplify RDB transfer.
1425 * We'll restore it when the children returns (since duped socket
1426 * will share the O_NONBLOCK attribute with the parent). */
1427 anetBlock(NULL,slave->fd);
1428 anetSendTimeout(NULL,slave->fd,server.repl_timeout*1000);
1429 }
1430 }
1431
1432 /* Create the child process. */
1433 start = ustime();
1434 if ((childpid = fork()) == 0) {
1435 /* Child */
1436 int retval;
1437 rio slave_sockets;
1438
1439 rioInitWithFdset(&slave_sockets,fds,numfds);
1440 zfree(fds);
1441
1442 closeListeningSockets(0);
1443 redisSetProcTitle("redis-rdb-to-slaves");
1444
1445 retval = rdbSaveRioWithEOFMark(&slave_sockets,NULL);
1446 if (retval == REDIS_OK && rioFlush(&slave_sockets) == 0)
1447 retval = REDIS_ERR;
1448
1449 if (retval == REDIS_OK) {
1450 size_t private_dirty = zmalloc_get_private_dirty();
1451
1452 if (private_dirty) {
1453 redisLog(REDIS_NOTICE,
1454 "RDB: %zu MB of memory used by copy-on-write",
1455 private_dirty/(1024*1024));
1456 }
1457
1458 /* If we are returning OK, at least one slave was served
1459 * with the RDB file as expected, so we need to send a report
1460 * to the parent via the pipe. The format of the message is:
1461 *
1462 * <len> <slave[0].id> <slave[0].error> ...
1463 *
1464 * len, slave IDs, and slave errors, are all uint64_t integers,
1465 * so basically the reply is composed of 64 bits for the len field
1466 * plus 2 additional 64 bit integers for each entry, for a total
1467 * of 'len' entries.
1468 *
1469 * The 'id' represents the slave's client ID, so that the master
1470 * can match the report with a specific slave, and 'error' is
1471 * set to 0 if the replication process terminated with a success
1472 * or the error code if an error occurred. */
1473 void *msg = zmalloc(sizeof(uint64_t)*(1+2*numfds));
1474 uint64_t *len = msg;
1475 uint64_t *ids = len+1;
1476 int j, msglen;
1477
1478 *len = numfds;
1479 for (j = 0; j < numfds; j++) {
1480 *ids++ = clientids[j];
1481 *ids++ = slave_sockets.io.fdset.state[j];
1482 }
1483
1484 /* Write the message to the parent. If we have no good slaves or
1485 * we are unable to transfer the message to the parent, we exit
1486 * with an error so that the parent will abort the replication
1487 * process with all the childre that were waiting. */
1488 msglen = sizeof(uint64_t)*(1+2*numfds);
1489 if (*len == 0 ||
1490 write(server.rdb_pipe_write_result_to_parent,msg,msglen)
1491 != msglen)
1492 {
1493 retval = REDIS_ERR;
1494 }
1495 zfree(msg);
1496 }
1497 zfree(clientids);
1498 exitFromChild((retval == REDIS_OK) ? 0 : 1);
1499 } else {
1500 /* Parent */
1501 zfree(clientids); /* Not used by parent. Free ASAP. */
1502 server.stat_fork_time = ustime()-start;
1503 server.stat_fork_rate = (double) zmalloc_used_memory() * 1000000 / server.stat_fork_time / (1024*1024*1024); /* GB per second. */
1504 latencyAddSampleIfNeeded("fork",server.stat_fork_time/1000);
1505 if (childpid == -1) {
1506 redisLog(REDIS_WARNING,"Can't save in background: fork: %s",
1507 strerror(errno));
1508 zfree(fds);
1509 close(pipefds[0]);
1510 close(pipefds[1]);
1511 return REDIS_ERR;
1512 }
1513 redisLog(REDIS_NOTICE,"Background RDB transfer started by pid %d",childpid);
1514 server.rdb_save_time_start = time(NULL);
1515 server.rdb_child_pid = childpid;
1516 server.rdb_child_type = REDIS_RDB_CHILD_TYPE_SOCKET;
1517 updateDictResizePolicy();
1518 zfree(fds);
1519 return REDIS_OK;
1520 }
1521 return REDIS_OK; /* unreached */
1522}
