199/* Defines related to the dump file format. To store 32 bits lengths for short
200 * keys requires a lot of space, so we check the most significant 2 bits of
201 * the first byte to interpreter the length:
202 *
203 * 00|000000 => if the two MSB are 00 the len is the 6 bits of this byte
204 * 01|000000 00000000 => 01, the len is 14 byes, 6 bits + 8 bits of next byte
205 * 10|000000 [32 bit integer] => if it's 10, a full 32 bit len will follow
206 * 11|000000 this means: specially encoded object will follow. The six bits
207 * number specify the kind of object that follows.
208 * See the REDIS_RDB_ENC_* defines.
209 *
210 * Lengths up to 63 are stored using a single byte, most DB keys, and may
211 * values, will fit inside. */
212#define REDIS_RDB_6BITLEN 0
213#define REDIS_RDB_14BITLEN 1
214#define REDIS_RDB_32BITLEN 2
215#define REDIS_RDB_ENCVAL 3
216#define REDIS_RDB_LENERR UINT_MAX
