35/* ZSETs are ordered sets using two data structures to hold the same elements
36 * in order to get O(log(N)) INSERT and REMOVE operations into a sorted
37 * data structure.
38 *
39 * The elements are added to a hash table mapping Redis objects to scores.
40 * At the same time the elements are added to a skip list mapping scores
41 * to Redis objects (so objects are sorted by scores in this "view"). */
42
43/* This skiplist implementation is almost a C translation of the original
44 * algorithm described by William Pugh in "Skip Lists: A Probabilistic
45 * Alternative to Balanced Trees", modified in three ways:
46 * a) this implementation allows for repeated scores.
47 * b) the comparison is not just by key (our 'score') but by satellite data.
48 * c) there is a back pointer, so it's a doubly linked list with the back
49 * pointers being only at "level 1". This allows to traverse the list
50 * from tail to head, useful for ZREVRANGE. */
51

58zskiplistNode *zslCreateNode(int level, double score, robj *obj) {
59 zskiplistNode *zn = zmalloc(sizeof(*zn)+level*sizeof(struct zskiplistLevel));
60 zn->score = score;
61 zn->obj = obj;
62 return zn;
63}

65zskiplist *zslCreate(void) {
66 int j;
67 zskiplist *zsl;
68
69 zsl = zmalloc(sizeof(*zsl));
70 zsl->level = 1;
71 zsl->length = 0;
72 zsl->header = zslCreateNode(ZSKIPLIST_MAXLEVEL,0,NULL);
73 for (j = 0; j < ZSKIPLIST_MAXLEVEL; j++) {
74 zsl->header->level[j].forward = NULL;
75 zsl->header->level[j].span = 0;
76 }
77 zsl->header->backward = NULL;
78 zsl->tail = NULL;
79 return zsl;
80}

87void zslFree(zskiplist *zsl) {
88 zskiplistNode *node = zsl->header->level[0].forward, *next;
89
90 zfree(zsl->header);
91 while(node) {
92 next = node->level[0].forward;
93 zslFreeNode(node);
94 node = next;
95 }
96 zfree(zsl);
97}

99/* Returns a random level for the new skiplist node we are going to create.
100 * The return value of this function is between 1 and ZSKIPLIST_MAXLEVEL
101 * (both inclusive), with a powerlaw-alike distribution where higher
102 * levels are less likely to be returned. */
103int zslRandomLevel(void) {
104 int level = 1;
105 while ((random()&0xFFFF) < (ZSKIPLIST_P * 0xFFFF))
106 level += 1;
107 return (level<ZSKIPLIST_MAXLEVEL) ? level : ZSKIPLIST_MAXLEVEL;
108}

110zskiplistNode *zslInsert(zskiplist *zsl, double score, robj *obj) {
111 zskiplistNode *update[ZSKIPLIST_MAXLEVEL], *x;
112 unsigned int rank[ZSKIPLIST_MAXLEVEL];
113 int i, level;
114
115 redisAssert(!isnan(score));
116 x = zsl->header;
117 for (i = zsl->level-1; i >= 0; i--) {
118 /* store rank that is crossed to reach the insert position */
119 rank[i] = i == (zsl->level-1) ? 0 : rank[i+1];
120 while (x->level[i].forward &&
121 (x->level[i].forward->score < score ||
122 (x->level[i].forward->score == score &&
123 compareStringObjects(x->level[i].forward->obj,obj) < 0))) {
124 rank[i] += x->level[i].span;
125 x = x->level[i].forward;
126 }
127 update[i] = x;
128 }
129 /* we assume the key is not already inside, since we allow duplicated
130 * scores, and the re-insertion of score and redis object should never
131 * happen since the caller of zslInsert() should test in the hash table
132 * if the element is already inside or not. */
133 level = zslRandomLevel();
134 if (level > zsl->level) {
135 for (i = zsl->level; i < level; i++) {
136 rank[i] = 0;
137 update[i] = zsl->header;
138 update[i]->level[i].span = zsl->length;
139 }
140 zsl->level = level;
141 }
142 x = zslCreateNode(level,score,obj);
143 for (i = 0; i < level; i++) {
144 x->level[i].forward = update[i]->level[i].forward;
145 update[i]->level[i].forward = x;
146
147 /* update span covered by update[i] as x is inserted here */
148 x->level[i].span = update[i]->level[i].span - (rank[0] - rank[i]);
149 update[i]->level[i].span = (rank[0] - rank[i]) + 1;
150 }
151
152 /* increment span for untouched levels */
153 for (i = level; i < zsl->level; i++) {
154 update[i]->level[i].span++;
155 }
156
157 x->backward = (update[0] == zsl->header) ? NULL : update[0];
158 if (x->level[0].forward)
159 x->level[0].forward->backward = x;
160 else
161 zsl->tail = x;
162 zsl->length++;
163 return x;
164}

166/* Internal function used by zslDelete, zslDeleteByScore and zslDeleteByRank */
167void zslDeleteNode(zskiplist *zsl, zskiplistNode *x, zskiplistNode **update) {
168 int i;
169 for (i = 0; i < zsl->level; i++) {
170 if (update[i]->level[i].forward == x) {
171 update[i]->level[i].span += x->level[i].span - 1;
172 update[i]->level[i].forward = x->level[i].forward;
173 } else {
174 update[i]->level[i].span -= 1;
175 }
176 }
177 if (x->level[0].forward) {
178 x->level[0].forward->backward = x->backward;
179 } else {
180 zsl->tail = x->backward;
181 }
182 while(zsl->level > 1 && zsl->header->level[zsl->level-1].forward == NULL)
183 zsl->level--;
184 zsl->length--;
185}

187/* Delete an element with matching score/object from the skiplist. */
188int zslDelete(zskiplist *zsl, double score, robj *obj) {
189 zskiplistNode *update[ZSKIPLIST_MAXLEVEL], *x;
190 int i;
191
192 x = zsl->header;
193 for (i = zsl->level-1; i >= 0; i--) {
194 while (x->level[i].forward &&
195 (x->level[i].forward->score < score ||
196 (x->level[i].forward->score == score &&
197 compareStringObjects(x->level[i].forward->obj,obj) < 0)))
198 x = x->level[i].forward;
199 update[i] = x;
200 }
201 /* We may have multiple elements with the same score, what we need
202 * is to find the element with both the right score and object. */
203 x = x->level[0].forward;
204 if (x && score == x->score && equalStringObjects(x->obj,obj)) {
205 zslDeleteNode(zsl, x, update);
206 zslFreeNode(x);
207 return 1;
208 }
209 return 0; /* not found */
210}

212static int zslValueGteMin(double value, zrangespec *spec) {
213 return spec->minex ? (value > spec->min) : (value >= spec->min);

217 return spec->maxex ? (value < spec->max) : (value <= spec->max);

220/* Returns if there is a part of the zset is in range. */
221int zslIsInRange(zskiplist *zsl, zrangespec *range) {
222 zskiplistNode *x;
223
224 /* Test for ranges that will always be empty. */
225 if (range->min > range->max ||
226 (range->min == range->max && (range->minex || range->maxex)))
227 return 0;
228 x = zsl->tail;
229 if (x == NULL || !zslValueGteMin(x->score,range))
230 return 0;
231 x = zsl->header->level[0].forward;
232 if (x == NULL || !zslValueLteMax(x->score,range))
233 return 0;
234 return 1;
235}

237/* Find the first node that is contained in the specified range.
238 * Returns NULL when no element is contained in the range. */
239zskiplistNode *zslFirstInRange(zskiplist *zsl, zrangespec *range) {
240 zskiplistNode *x;
241 int i;
242
243 /* If everything is out of range, return early. */
244 if (!zslIsInRange(zsl,range)) return NULL;
245
246 x = zsl->header;
247 for (i = zsl->level-1; i >= 0; i--) {
248 /* Go forward while *OUT* of range. */
249 while (x->level[i].forward &&
250 !zslValueGteMin(x->level[i].forward->score,range))
251 x = x->level[i].forward;
252 }
253
254 /* This is an inner range, so the next node cannot be NULL. */
255 x = x->level[0].forward;
256 redisAssert(x != NULL);
257
258 /* Check if score <= max. */
259 if (!zslValueLteMax(x->score,range)) return NULL;
260 return x;
261}

265zskiplistNode *zslLastInRange(zskiplist *zsl, zrangespec *range) {

267 int i;

269 /* If everything is out of range, return early. */

288/* Delete all the elements with score between min and max from the skiplist.
289 * Min and max are inclusive, so a score >= min || score <= max is deleted.
290 * Note that this function takes the reference to the hash table view of the
291 * sorted set, in order to remove the elements from the hash table too. */
292unsigned long zslDeleteRangeByScore(zskiplist *zsl, zrangespec *range, dict *dict) {
293 zskiplistNode *update[ZSKIPLIST_MAXLEVEL], *x;
294 unsigned long removed = 0;
295 int i;
296
297 x = zsl->header;
298 for (i = zsl->level-1; i >= 0; i--) {
299 while (x->level[i].forward && (range->minex ?
300 x->level[i].forward->score <= range->min :
301 x->level[i].forward->score < range->min))
302 x = x->level[i].forward;
303 update[i] = x;
304 }
305
306 /* Current node is the last with score < or <= min. */
307 x = x->level[0].forward;
308
309 /* Delete nodes while in range. */
310 while (x &&
311 (range->maxex ? x->score < range->max : x->score <= range->max))
312 {
313 zskiplistNode *next = x->level[0].forward;
314 zslDeleteNode(zsl,x,update);
315 dictDelete(dict,x->obj);
316 zslFreeNode(x);
317 removed++;
318 x = next;
319 }
320 return removed;
321}

323unsigned long zslDeleteRangeByLex(zskiplist *zsl, zlexrangespec *range, dict *dict) {
324 zskiplistNode *update[ZSKIPLIST_MAXLEVEL], *x;
325 unsigned long removed = 0;
326 int i;
327
328
329 x = zsl->header;
330 for (i = zsl->level-1; i >= 0; i--) {
331 while (x->level[i].forward &&

352/* Delete all the elements with rank between start and end from the skiplist.
353 * Start and end are inclusive. Note that start and end need to be 1-based */
354unsigned long zslDeleteRangeByRank(zskiplist *zsl, unsigned int start, unsigned int end, dict *dict) {
355 zskiplistNode *update[ZSKIPLIST_MAXLEVEL], *x;
356 unsigned long traversed = 0, removed = 0;
357 int i;
358
359 x = zsl->header;
360 for (i = zsl->level-1; i >= 0; i--) {
361 while (x->level[i].forward && (traversed + x->level[i].span) < start) {
362 traversed += x->level[i].span;
363 x = x->level[i].forward;
364 }
365 update[i] = x;
366 }
367
368 traversed++;
369 x = x->level[0].forward;
370 while (x && traversed <= end) {
371 zskiplistNode *next = x->level[0].forward;
372 zslDeleteNode(zsl,x,update);
373 dictDelete(dict,x->obj);
374 zslFreeNode(x);
375 removed++;
376 traversed++;
377 x = next;
378 }
379 return removed;
380}

382/* Find the rank for an element by both score and key.
383 * Returns 0 when the element cannot be found, rank otherwise.
384 * Note that the rank is 1-based due to the span of zsl->header to the
385 * first element. */
386unsigned long zslGetRank(zskiplist *zsl, double score, robj *o) {
387 zskiplistNode *x;
388 unsigned long rank = 0;
389 int i;
390
391 x = zsl->header;
392 for (i = zsl->level-1; i >= 0; i--) {
393 while (x->level[i].forward &&
394 (x->level[i].forward->score < score ||
395 (x->level[i].forward->score == score &&
396 compareStringObjects(x->level[i].forward->obj,o) <= 0))) {
397 rank += x->level[i].span;
398 x = x->level[i].forward;
399 }
400
401 /* x might be equal to zsl->header, so test if obj is non-NULL */
402 if (x->obj && equalStringObjects(x->obj,o)) {
403 return rank;
404 }
405 }
406 return 0;
407}

409/* Finds an element by its rank. The rank argument needs to be 1-based. */
410zskiplistNode* zslGetElementByRank(zskiplist *zsl, unsigned long rank) {
411 zskiplistNode *x;
412 unsigned long traversed = 0;
413 int i;
414
415 x = zsl->header;
416 for (i = zsl->level-1; i >= 0; i--) {
417 while (x->level[i].forward && (traversed + x->level[i].span) <= rank)
418 {
419 traversed += x->level[i].span;
420 x = x->level[i].forward;
421 }
422 if (traversed == rank) {
423 return x;
424 }
425 }
426 return NULL;
427}

429/* Populate the rangespec according to the objects min and max. */
430static int zslParseRange(robj *min, robj *max, zrangespec *spec) {
431 char *eptr;
432 spec->minex = spec->maxex = 0;
433
434 /* Parse the min-max interval. If one of the values is prefixed
435 * by the "(" character, it's considered "open". For instance
436 * ZRANGEBYSCORE zset (1.5 (2.5 will match min < x < max
437 * ZRANGEBYSCORE zset 1.5 2.5 will instead match min <= x <= max */
438 if (min->encoding == REDIS_ENCODING_INT) {
439 spec->min = (long)min->ptr;
440 } else {
441 if (((char*)min->ptr)[0] == '(') {
442 spec->min = strtod((char*)min->ptr+1,&eptr);
443 if (eptr[0] != '\0' || isnan(spec->min)) return REDIS_ERR;
444 spec->minex = 1;
445 } else {
446 spec->min = strtod((char*)min->ptr,&eptr);
447 if (eptr[0] != '\0' || isnan(spec->min)) return REDIS_ERR;
448 }
449 }
450 if (max->encoding == REDIS_ENCODING_INT) {
451 spec->max = (long)max->ptr;
452 } else {
453 if (((char*)max->ptr)[0] == '(') {
454 spec->max = strtod((char*)max->ptr+1,&eptr);
455 if (eptr[0] != '\0' || isnan(spec->max)) return REDIS_ERR;
456 spec->maxex = 1;
457 } else {
458 spec->max = strtod((char*)max->ptr,&eptr);
459 if (eptr[0] != '\0' || isnan(spec->max)) return REDIS_ERR;
460 }
461 }
462
463 return REDIS_OK;
464}

468/* Parse max or min argument of ZRANGEBYLEX.
469 * (foo means foo (open interval)
470 * [foo means foo (closed interval)
471 * - means the min string possible
472 * + means the max string possible
473 *
474 * If the string is valid the *dest pointer is set to the redis object
475 * that will be used for the comparision, and ex will be set to 0 or 1
476 * respectively if the item is exclusive or inclusive. REDIS_OK will be
477 * returned.
478 *
479 * If the string is not a valid range REDIS_ERR is returned, and the value
480 * of *dest and *ex is undefined. */
481int zslParseLexRangeItem(robj *item, robj **dest, int *ex) {
482 char *c = item->ptr;
483
484 switch(c[0]) {
485 case '+':
486 if (c[1] != '\0') return REDIS_ERR;
487 *ex = 0;
488 *dest = shared.maxstring;
489 incrRefCount(shared.maxstring);
490 return REDIS_OK;
491 case '-':
492 if (c[1] != '\0') return REDIS_ERR;
493 *ex = 0;
494 *dest = shared.minstring;
495 incrRefCount(shared.minstring);
496 return REDIS_OK;
497 case '(':
498 *ex = 1;
499 *dest = createStringObject(c+1,sdslen(c)-1);
500 return REDIS_OK;
501 case '[':
502 *ex = 0;
503 *dest = createStringObject(c+1,sdslen(c)-1);
504 return REDIS_OK;
505 default:
506 return REDIS_ERR;
507 }
508}

510/* Populate the rangespec according to the objects min and max.
511 *
512 * Return REDIS_OK on success. On error REDIS_ERR is returned.
513 * When OK is returned the structure must be freed with zslFreeLexRange(),
514 * otherwise no release is needed. */
515static int zslParseLexRange(robj *min, robj *max, zlexrangespec *spec) {
516 /* The range can't be valid if objects are integer encoded.
517 * Every item must start with ( or [. */
518 if (min->encoding == REDIS_ENCODING_INT ||
519 max->encoding == REDIS_ENCODING_INT) return REDIS_ERR;
520
521 spec->min = spec->max = NULL;
522 if (zslParseLexRangeItem(min, &spec->min, &spec->minex) == REDIS_ERR ||
523 zslParseLexRangeItem(max, &spec->max, &spec->maxex) == REDIS_ERR) {
524 if (spec->min) decrRefCount(spec->min);
525 if (spec->max) decrRefCount(spec->max);
526 return REDIS_ERR;
527 } else {
528 return REDIS_OK;
529 }
530}

539/* This is just a wrapper to compareStringObjects() that is able to
540 * handle shared.minstring and shared.maxstring as the equivalent of
541 * -inf and +inf for strings */
542int compareStringObjectsForLexRange(robj *a, robj *b) {
543 if (a == b) return 0; /* This makes sure that we handle inf,inf and
544 -inf,-inf ASAP. One special case less. */
545 if (a == shared.minstring || b == shared.maxstring) return -1;
546 if (a == shared.maxstring || b == shared.minstring) return 1;
547 return compareStringObjects(a,b);
548}

550static int zslLexValueGteMin(robj *value, zlexrangespec *spec) {
551 return spec->minex ?
552 (compareStringObjectsForLexRange(value,spec->min) > 0) :
553 (compareStringObjectsForLexRange(value,spec->min) >= 0);
554}
555
556static int zslLexValueLteMax(robj *value, zlexrangespec *spec) {
557 return spec->maxex ?
558 (compareStringObjectsForLexRange(value,spec->max) < 0) :
559 (compareStringObjectsForLexRange(value,spec->max) <= 0);
560}

562/* Returns if there is a part of the zset is in the lex range. */
563int zslIsInLexRange(zskiplist *zsl, zlexrangespec *range) {
564 zskiplistNode *x;
565
566 /* Test for ranges that will always be empty. */
567 if (compareStringObjectsForLexRange(range->min,range->max) > 1 ||
568 (compareStringObjects(range->min,range->max) == 0 &&
569 (range->minex || range->maxex)))
570 return 0;
571 x = zsl->tail;
572 if (x == NULL || !zslLexValueGteMin(x->obj,range))
573 return 0;
574 x = zsl->header->level[0].forward;
575 if (x == NULL || !zslLexValueLteMax(x->obj,range))
576 return 0;
577 return 1;
578}

635double zzlGetScore(unsigned char *sptr) {
636 unsigned char *vstr;
637 unsigned int vlen;
638 long long vlong;
639 char buf[128];
640 double score;
641
642 redisAssert(sptr != NULL);
643 redisAssert(ziplistGet(sptr,&vstr,&vlen,&vlong));
644
645 if (vstr) {
646 memcpy(buf,vstr,vlen);
647 buf[vlen] = '\0';
648 score = strtod(buf,NULL);
649 } else {
650 score = vlong;
651 }
652
653 return score;

737/* Returns if there is a part of the zset is in range. Should only be used
738 * internally by zzlFirstInRange and zzlLastInRange. */
739int zzlIsInRange(unsigned char *zl, zrangespec *range) {
740 unsigned char *p;
741 double score;
742
743 /* Test for ranges that will always be empty. */
744 if (range->min > range->max ||
745 (range->min == range->max && (range->minex || range->maxex)))
746 return 0;
747
748 p = ziplistIndex(zl,-1); /* Last score. */
749 if (p == NULL) return 0; /* Empty sorted set */
750 score = zzlGetScore(p);
751 if (!zslValueGteMin(score,range))
752 return 0;
753
754 p = ziplistIndex(zl,1); /* First score. */
755 redisAssert(p != NULL);
756 score = zzlGetScore(p);
757 if (!zslValueLteMax(score,range))
758 return 0;
759
760 return 1;
761}

763/* Find pointer to the first element contained in the specified range.
764 * Returns NULL when no element is contained in the range. */
765unsigned char *zzlFirstInRange(unsigned char *zl, zrangespec *range) {
766 unsigned char *eptr = ziplistIndex(zl,0), *sptr;
767 double score;
768
769 /* If everything is out of range, return early. */
770 if (!zzlIsInRange(zl,range)) return NULL;

779 if (zslValueLteMax(score,range))
780 return eptr;
781 return NULL;
782 }
783
784 /* Move to next element. */
785 eptr = ziplistNext(zl,sptr);
786 }
787
788 return NULL;

841 unsigned char *p;
842

916 unsigned char *eptr = ziplistIndex(zl,0), *sptr;
917
918 ele = getDecodedObject(ele);

930 /* Move to next element. */

932 }

949unsigned char *zzlInsertAt(unsigned char *zl, unsigned char *eptr, robj *ele, double score) {
950 unsigned char *sptr;
951 char scorebuf[128];
952 int scorelen;
953 size_t offset;
954
955 redisAssertWithInfo(NULL,ele,sdsEncodedObject(ele));
956 scorelen = d2string(scorebuf,sizeof(scorebuf),score);
957 if (eptr == NULL) {
958 zl = ziplistPush(zl,ele->ptr,sdslen(ele->ptr),ZIPLIST_TAIL);
959 zl = ziplistPush(zl,(unsigned char*)scorebuf,scorelen,ZIPLIST_TAIL);
960 } else {
961 /* Keep offset relative to zl, as it might be re-allocated. */
962 offset = eptr-zl;
963 zl = ziplistInsert(zl,eptr,ele->ptr,sdslen(ele->ptr));
964 eptr = zl+offset;
965
966 /* Insert score after the element. */
967 redisAssertWithInfo(NULL,ele,(sptr = ziplistNext(zl,eptr)) != NULL);
968 zl = ziplistInsert(zl,sptr,(unsigned char*)scorebuf,scorelen);
969 }
970
971 return zl;
972}

974/* Insert (element,score) pair in ziplist. This function assumes the element is
975 * not yet present in the list. */
976unsigned char *zzlInsert(unsigned char *zl, robj *ele, double score) {
977 unsigned char *eptr = ziplistIndex(zl,0), *sptr;
978 double s;
979
980 ele = getDecodedObject(ele);
981 while (eptr != NULL) {
982 sptr = ziplistNext(zl,eptr);
983 redisAssertWithInfo(NULL,ele,sptr != NULL);
984 s = zzlGetScore(sptr);
985
986 if (s > score) {
987 /* First element with score larger than score for element to be
988 * inserted. This means we should take its spot in the list to
989 * maintain ordering. */
990 zl = zzlInsertAt(zl,eptr,ele,score);
991 break;
992 } else if (s == score) {
993 /* Ensure lexicographical ordering for elements. */
994 if (zzlCompareElements(eptr,ele->ptr,sdslen(ele->ptr)) > 0) {
995 zl = zzlInsertAt(zl,eptr,ele,score);
996 break;
997 }
998 }
999
1000 /* Move to next element. */
1001 eptr = ziplistNext(zl,sptr);
1002 }
1003
1004 /* Push on tail of list when it was not yet inserted. */
1005 if (eptr == NULL)
1006 zl = zzlInsertAt(zl,NULL,ele,score);
1007
1008 decrRefCount(ele);
1009 return zl;
1010}

1093void zsetConvert(robj *zobj, int encoding) {
1094 zset *zs;
1095 zskiplistNode *node, *next;
1096 robj *ele;
1097 double score;
1098
1099 if (zobj->encoding == encoding) return;
1100 if (zobj->encoding == REDIS_ENCODING_ZIPLIST) {
1101 unsigned char *zl = zobj->ptr;
1102 unsigned char *eptr, *sptr;
1103 unsigned char *vstr;
1104 unsigned int vlen;
1105 long long vlong;
1106
1107 if (encoding != REDIS_ENCODING_SKIPLIST)
1108 redisPanic("Unknown target encoding");
1109
1110 zs = zmalloc(sizeof(*zs));
1111 zs->dict = dictCreate(&zsetDictType,NULL);
1112 zs->zsl = zslCreate();
1113
1114 eptr = ziplistIndex(zl,0);
1115 redisAssertWithInfo(NULL,zobj,eptr != NULL);
1116 sptr = ziplistNext(zl,eptr);
1117 redisAssertWithInfo(NULL,zobj,sptr != NULL);
1118
1119 while (eptr != NULL) {
1120 score = zzlGetScore(sptr);
1121 redisAssertWithInfo(NULL,zobj,ziplistGet(eptr,&vstr,&vlen,&vlong));
1122 if (vstr == NULL)
1123 ele = createStringObjectFromLongLong(vlong);
1124 else
1125 ele = createStringObject((char*)vstr,vlen);
1126
1127 /* Has incremented refcount since it was just created. */
1128 node = zslInsert(zs->zsl,score,ele);
1129 redisAssertWithInfo(NULL,zobj,dictAdd(zs->dict,ele,&node->score) == DICT_OK);
1130 incrRefCount(ele); /* Added to dictionary. */
1131 zzlNext(zl,&eptr,&sptr);
1132 }
1133
1134 zfree(zobj->ptr);
1135 zobj->ptr = zs;
1136 zobj->encoding = REDIS_ENCODING_SKIPLIST;
1137 } else if (zobj->encoding == REDIS_ENCODING_SKIPLIST) {
1138 unsigned char *zl = ziplistNew();
1139
1140 if (encoding != REDIS_ENCODING_ZIPLIST)
1141 redisPanic("Unknown target encoding");
1142
1143 /* Approach similar to zslFree(), since we want to free the skiplist at
1144 * the same time as creating the ziplist. */
1145 zs = zobj->ptr;
1146 dictRelease(zs->dict);
1147 node = zs->zsl->header->level[0].forward;
1148 zfree(zs->zsl->header);
1149 zfree(zs->zsl);
1150
1151 while (node) {
1152 ele = getDecodedObject(node->obj);
1153 zl = zzlInsertAt(zl,NULL,ele,node->score);
1154 decrRefCount(ele);
1155
1156 next = node->level[0].forward;
1157 zslFreeNode(node);
1158 node = next;
1159 }
1160
1161 zfree(zs);
1162 zobj->ptr = zl;
1163 zobj->encoding = REDIS_ENCODING_ZIPLIST;
1164 } else {
1165 redisPanic("Unknown sorted set encoding");
1166 }
1167}

1173/* This generic command implements both ZADD and ZINCRBY. */
1174void zaddGenericCommand(redisClient *c, int incr) {
1175 static char *nanerr = "resulting score is not a number (NaN)";
1176 robj *key = c->argv[1];
1177 robj *ele;
1178 robj *zobj;
1179 robj *curobj;
1180 double score = 0, *scores = NULL, curscore = 0.0;
1181 int j, elements = (c->argc-2)/2;
1182 int added = 0, updated = 0;
1183
1184 if (c->argc % 2) {
1185 addReply(c,shared.syntaxerr);
1186 return;
1187 }
1188
1189 /* Start parsing all the scores, we need to emit any syntax error
1190 * before executing additions to the sorted set, as the command should
1191 * either execute fully or nothing at all. */
1192 scores = zmalloc(sizeof(double)*elements);
1193 for (j = 0; j < elements; j++) {
1194 if (getDoubleFromObjectOrReply(c,c->argv[2+j*2],&scores[j],NULL)
1195 != REDIS_OK) goto cleanup;
1196 }
1197
1198 /* Lookup the key and create the sorted set if does not exist. */
1199 zobj = lookupKeyWrite(c->db,key);
1200 if (zobj == NULL) {
1201 if (server.zset_max_ziplist_entries == 0 ||
1202 server.zset_max_ziplist_value < sdslen(c->argv[3]->ptr))
1203 {
1204 zobj = createZsetObject();
1205 } else {
1206 zobj = createZsetZiplistObject();
1207 }
1208 dbAdd(c->db,key,zobj);
1209 } else {
1210 if (zobj->type != REDIS_ZSET) {
1211 addReply(c,shared.wrongtypeerr);
1212 goto cleanup;
1213 }
1214 }
1215
1216 for (j = 0; j < elements; j++) {
1217 score = scores[j];
1218
1219 if (zobj->encoding == REDIS_ENCODING_ZIPLIST) {
1220 unsigned char *eptr;
1221
1222 /* Prefer non-encoded element when dealing with ziplists. */
1223 ele = c->argv[3+j*2];
1224 if ((eptr = zzlFind(zobj->ptr,ele,&curscore)) != NULL) {
1225 if (incr) {
1226 score += curscore;
1227 if (isnan(score)) {
1228 addReplyError(c,nanerr);
1229 goto cleanup;
1230 }
1231 }
1232
1233 /* Remove and re-insert when score changed. */
1234 if (score != curscore) {
1235 zobj->ptr = zzlDelete(zobj->ptr,eptr);
1236 zobj->ptr = zzlInsert(zobj->ptr,ele,score);
1237 server.dirty++;
1238 updated++;
1239 }
1240 } else {
1241 /* Optimize: check if the element is too large or the list
1242 * becomes too long *before* executing zzlInsert. */
1243 zobj->ptr = zzlInsert(zobj->ptr,ele,score);
1244 if (zzlLength(zobj->ptr) > server.zset_max_ziplist_entries)
1245 zsetConvert(zobj,REDIS_ENCODING_SKIPLIST);
1246 if (sdslen(ele->ptr) > server.zset_max_ziplist_value)
1247 zsetConvert(zobj,REDIS_ENCODING_SKIPLIST);
1248 server.dirty++;
1249 added++;
1250 }
1251 } else if (zobj->encoding == REDIS_ENCODING_SKIPLIST) {
1252 zset *zs = zobj->ptr;
1253 zskiplistNode *znode;
1254 dictEntry *de;
1255
1256 ele = c->argv[3+j*2] = tryObjectEncoding(c->argv[3+j*2]);
1257 de = dictFind(zs->dict,ele);
1258 if (de != NULL) {
1259 curobj = dictGetKey(de);
1260 curscore = *(double*)dictGetVal(de);
1261
1262 if (incr) {
1263 score += curscore;
1264 if (isnan(score)) {
1265 addReplyError(c,nanerr);
1266 /* Don't need to check if the sorted set is empty
1267 * because we know it has at least one element. */
1268 goto cleanup;
1269 }
1270 }
1271
1272 /* Remove and re-insert when score changed. We can safely
1273 * delete the key object from the skiplist, since the
1274 * dictionary still has a reference to it. */
1275 if (score != curscore) {
1276 redisAssertWithInfo(c,curobj,zslDelete(zs->zsl,curscore,curobj));
1277 znode = zslInsert(zs->zsl,score,curobj);
1278 incrRefCount(curobj); /* Re-inserted in skiplist. */
1279 dictGetVal(de) = &znode->score; /* Update score ptr. */
1280 server.dirty++;
1281 updated++;
1282 }
1283 } else {
1284 znode = zslInsert(zs->zsl,score,ele);
1285 incrRefCount(ele); /* Inserted in skiplist. */
1286 redisAssertWithInfo(c,NULL,dictAdd(zs->dict,ele,&znode->score) == DICT_OK);
1287 incrRefCount(ele); /* Added to dictionary. */
1288 server.dirty++;
1289 added++;
1290 }
1291 } else {
1292 redisPanic("Unknown sorted set encoding");
1293 }
1294 }
1295 if (incr) /* ZINCRBY */
1296 addReplyDouble(c,score);
1297 else /* ZADD */
1298 addReplyLongLong(c,added);
1299
1300cleanup:
1301 zfree(scores);
1302 if (added || updated) {
1303 signalModifiedKey(c->db,key);
1304 notifyKeyspaceEvent(REDIS_NOTIFY_ZSET,
1305 incr ? "zincr" : "zadd", key, c->db->id);
1306 }
1307}

1377/* Implements ZREMRANGEBYRANK, ZREMRANGEBYSCORE, ZREMRANGEBYLEX commands. */
1378#define ZRANGE_RANK 0
1379#define ZRANGE_SCORE 1
1380#define ZRANGE_LEX 2
1381void zremrangeGenericCommand(redisClient *c, int rangetype) {
1382 robj *key = c->argv[1];
1383 robj *zobj;
1384 int keyremoved = 0;
1385 unsigned long deleted;
1386 zrangespec range;
1387 zlexrangespec lexrange;
1388 long start, end, llen;
1389
1390 /* Step 1: Parse the range. */
1391 if (rangetype == ZRANGE_RANK) {
1392 if ((getLongFromObjectOrReply(c,c->argv[2],&start,NULL) != REDIS_OK) ||
1393 (getLongFromObjectOrReply(c,c->argv[3],&end,NULL) != REDIS_OK))
1394 return;
1395 } else if (rangetype == ZRANGE_SCORE) {
1396 if (zslParseRange(c->argv[2],c->argv[3],&range) != REDIS_OK) {
1397 addReplyError(c,"min or max is not a float");
1398 return;
1399 }
1400 } else if (rangetype == ZRANGE_LEX) {
1401 if (zslParseLexRange(c->argv[2],c->argv[3],&lexrange) != REDIS_OK) {
1402 addReplyError(c,"min or max not valid string range item");
1403 return;
1404 }
1405 }
1406
1407 /* Step 2: Lookup & range sanity checks if needed. */
1408 if ((zobj = lookupKeyWriteOrReply(c,key,shared.czero)) == NULL ||
1409 checkType(c,zobj,REDIS_ZSET)) goto cleanup;
1410
1411 if (rangetype == ZRANGE_RANK) {
1412 /* Sanitize indexes. */
1413 llen = zsetLength(zobj);
1414 if (start < 0) start = llen+start;
1415 if (end < 0) end = llen+end;
1416 if (start < 0) start = 0;
1417

1419 * The range is empty when start > end or start >= length. */
1420 if (start > end || start >= llen) {
1421 addReply(c,shared.czero);
1422 goto cleanup;
1423 }
1424 if (end >= llen) end = llen-1;
1425 }
1426
1427 /* Step 3: Perform the range deletion operation. */
1428 if (zobj->encoding == REDIS_ENCODING_ZIPLIST) {
1429 switch(rangetype) {
1430 case ZRANGE_RANK:
1431 zobj->ptr = zzlDeleteRangeByRank(zobj->ptr,start+1,end+1,&deleted);
1432 break;
1433 case ZRANGE_SCORE:
1434 zobj->ptr = zzlDeleteRangeByScore(zobj->ptr,&range,&deleted);
1435 break;
1436 case ZRANGE_LEX:
1437 zobj->ptr = zzlDeleteRangeByLex(zobj->ptr,&lexrange,&deleted);
1438 break;
1439 }
1440 if (zzlLength(zobj->ptr) == 0) {
1441 dbDelete(c->db,key);
1442 keyremoved = 1;
1443 }
1444 } else if (zobj->encoding == REDIS_ENCODING_SKIPLIST) {
1445 zset *zs = zobj->ptr;
1446 switch(rangetype) {
1447 case ZRANGE_RANK:
1448 deleted = zslDeleteRangeByRank(zs->zsl,start+1,end+1,zs->dict);
1449 break;
1450 case ZRANGE_SCORE:
1451 deleted = zslDeleteRangeByScore(zs->zsl,&range,zs->dict);
1452 break;
1453 case ZRANGE_LEX:
1454 deleted = zslDeleteRangeByLex(zs->zsl,&lexrange,zs->dict);
1455 break;
1456 }
1457 if (htNeedsResize(zs->dict)) dictResize(zs->dict);
1458 if (dictSize(zs->dict) == 0) {
1459 dbDelete(c->db,key);
1460 keyremoved = 1;
1461 }
1462 } else {
1463 redisPanic("Unknown sorted set encoding");
1464 }
1465
1466 /* Step 4: Notifications and reply. */
1467 if (deleted) {
1468 char *event[3] = {"zremrangebyrank","zremrangebyscore","zremrangebylex"};
1469 signalModifiedKey(c->db,key);
1470 notifyKeyspaceEvent(REDIS_NOTIFY_ZSET,event[rangetype],key,c->db->id);
1471 if (keyremoved)
1472 notifyKeyspaceEvent(REDIS_NOTIFY_GENERIC,"del",key,c->db->id);
1473 }
1474 server.dirty += deleted;
1475 addReplyLongLong(c,deleted);
1476
1477cleanup:
1478 if (rangetype == ZRANGE_LEX) zslFreeLexRange(&lexrange);
1479}

1642/* Check if the current value is valid. If so, store it in the passed structure
1643 * and move to the next element. If not valid, this means we have reached the
1644 * end of the structure and can abort. */
1645int zuiNext(zsetopsrc *op, zsetopval *val) {
1646 if (op->subject == NULL)
1647 return 0;
1648
1649 if (val->flags & OPVAL_DIRTY_ROBJ)
1650 decrRefCount(val->ele);
1651
1652 memset(val,0,sizeof(zsetopval));
1653
1654 if (op->type == REDIS_SET) {
1655 iterset *it = &op->iter.set;
1656 if (op->encoding == REDIS_ENCODING_INTSET) {
1657 int64_t ell;
1658
1659 if (!intsetGet(it->is.is,it->is.ii,&ell))
1660 return 0;
1661 val->ell = ell;
1662 val->score = 1.0;
1663
1664 /* Move to next element. */
1665 it->is.ii++;
1666 } else if (op->encoding == REDIS_ENCODING_HT) {
1667 if (it->ht.de == NULL)
1668 return 0;
1669 val->ele = dictGetKey(it->ht.de);
1670 val->score = 1.0;
1671
1672 /* Move to next element. */
1673 it->ht.de = dictNext(it->ht.di);
1674 } else {
1675 redisPanic("Unknown set encoding");
1676 }
1677 } else if (op->type == REDIS_ZSET) {
1678 iterzset *it = &op->iter.zset;
1679 if (op->encoding == REDIS_ENCODING_ZIPLIST) {
1680 /* No need to check both, but better be explicit. */
1681 if (it->zl.eptr == NULL || it->zl.sptr == NULL)
1682 return 0;
1683 redisAssert(ziplistGet(it->zl.eptr,&val->estr,&val->elen,&val->ell));
1684 val->score = zzlGetScore(it->zl.sptr);
1685
1686 /* Move to next element. */
1687 zzlNext(it->zl.zl,&it->zl.eptr,&it->zl.sptr);
1688 } else if (op->encoding == REDIS_ENCODING_SKIPLIST) {
1689 if (it->sl.node == NULL)
1690 return 0;
1691 val->ele = it->sl.node->obj;
1692 val->score = it->sl.node->score;
1693
1694 /* Move to next element. */
1695 it->sl.node = it->sl.node->level[0].forward;
1696 } else {
1697 redisPanic("Unknown sorted set encoding");
1698 }
1699 } else {
1700 redisPanic("Unsupported type");
1701 }
1702 return 1;
1703}

1821#define REDIS_AGGR_SUM 1
1822#define REDIS_AGGR_MIN 2
1823#define REDIS_AGGR_MAX 3
1824#define zunionInterDictValue(_e) (dictGetVal(_e) == NULL ? 1.0 : *(double*)dictGetVal(_e))

1826inline static void zunionInterAggregate(double *target, double val, int aggregate) {
1827 if (aggregate == REDIS_AGGR_SUM) {
1828 *target = *target + val;
1829 /* The result of adding two doubles is NaN when one variable
1830 * is +inf and the other is -inf. When these numbers are added,
1831 * we maintain the convention of the result being 0.0. */
1832 if (isnan(*target)) *target = 0.0;
1833 } else if (aggregate == REDIS_AGGR_MIN) {
1834 *target = val < *target ? val : *target;
1835 } else if (aggregate == REDIS_AGGR_MAX) {
1836 *target = val > *target ? val : *target;
1837 } else {
1838 /* safety net */
1839 redisPanic("Unknown ZUNION/INTER aggregate type");
1840 }
1841}

1843void zunionInterGenericCommand(redisClient *c, robj *dstkey, int op) {
1844 int i, j;
1845 long setnum;
1846 int aggregate = REDIS_AGGR_SUM;
1847 zsetopsrc *src;
1848 zsetopval zval;
1849 robj *tmp;
1850 unsigned int maxelelen = 0;
1851 robj *dstobj;
1852 zset *dstzset;
1853 zskiplistNode *znode;
1854 int touched = 0;
1855
1856 /* expect setnum input keys to be given */
1857 if ((getLongFromObjectOrReply(c, c->argv[2], &setnum, NULL) != REDIS_OK))
1858 return;
1859
1860 if (setnum < 1) {
1861 addReplyError(c,
1862 "at least 1 input key is needed for ZUNIONSTORE/ZINTERSTORE");
1863 return;
1864 }
1865
1866 /* test if the expected number of keys would overflow */
1867 if (setnum > c->argc-3) {
1868 addReply(c,shared.syntaxerr);
1869 return;
1870 }
1871
1872 /* read keys to be used for input */
1873 src = zcalloc(sizeof(zsetopsrc) * setnum);
1874 for (i = 0, j = 3; i < setnum; i++, j++) {
1875 robj *obj = lookupKeyWrite(c->db,c->argv[j]);
1876 if (obj != NULL) {
1877 if (obj->type != REDIS_ZSET && obj->type != REDIS_SET) {
1878 zfree(src);
1879 addReply(c,shared.wrongtypeerr);
1880 return;
1881 }
1882
1883 src[i].subject = obj;
1884 src[i].type = obj->type;
1885 src[i].encoding = obj->encoding;
1886 } else {
1887 src[i].subject = NULL;
1888 }
1889
1890 /* Default all weights to 1. */
1891 src[i].weight = 1.0;
1892 }
1893
1894 /* parse optional extra arguments */
1895 if (j < c->argc) {
1896 int remaining = c->argc - j;
1897
1898 while (remaining) {
1899 if (remaining >= (setnum + 1) && !strcasecmp(c->argv[j]->ptr,"weights")) {
1900 j++; remaining--;
1901 for (i = 0; i < setnum; i++, j++, remaining--) {
1902 if (getDoubleFromObjectOrReply(c,c->argv[j],&src[i].weight,
1903 "weight value is not a float") != REDIS_OK)
1904 {
1905 zfree(src);
1906 return;
1907 }
1908 }
1909 } else if (remaining >= 2 && !strcasecmp(c->argv[j]->ptr,"aggregate")) {
1910 j++; remaining--;
1911 if (!strcasecmp(c->argv[j]->ptr,"sum")) {
1912 aggregate = REDIS_AGGR_SUM;
1913 } else if (!strcasecmp(c->argv[j]->ptr,"min")) {
1914 aggregate = REDIS_AGGR_MIN;
1915 } else if (!strcasecmp(c->argv[j]->ptr,"max")) {
1916 aggregate = REDIS_AGGR_MAX;
1917 } else {
1918 zfree(src);
1919 addReply(c,shared.syntaxerr);
1920 return;
1921 }
1922 j++; remaining--;
1923 } else {
1924 zfree(src);
1925 addReply(c,shared.syntaxerr);
1926 return;
1927 }
1928 }
1929 }
1930
1931 /* sort sets from the smallest to largest, this will improve our
1932 * algorithm's performance */
1933 qsort(src,setnum,sizeof(zsetopsrc),zuiCompareByCardinality);
1934
1935 dstobj = createZsetObject();
1936 dstzset = dstobj->ptr;
1937 memset(&zval, 0, sizeof(zval));
1938
1939 if (op == REDIS_OP_INTER) {
1940 /* Skip everything if the smallest input is empty. */
1941 if (zuiLength(&src[0]) > 0) {
1942 /* Precondition: as src[0] is non-empty and the inputs are ordered
1943 * by size, all src[i > 0] are non-empty too. */
1944 zuiInitIterator(&src[0]);
1945 while (zuiNext(&src[0],&zval)) {
1946 double score, value;
1947
1948 score = src[0].weight * zval.score;
1949 if (isnan(score)) score = 0;
1950
1951 for (j = 1; j < setnum; j++) {
1952 /* It is not safe to access the zset we are
1953 * iterating, so explicitly check for equal object. */
1954 if (src[j].subject == src[0].subject) {
1955 value = zval.score*src[j].weight;
1956 zunionInterAggregate(&score,value,aggregate);
1957 } else if (zuiFind(&src[j],&zval,&value)) {
1958 value *= src[j].weight;
1959 zunionInterAggregate(&score,value,aggregate);
1960 } else {
1961 break;
1962 }
1963 }
1964
1965 /* Only continue when present in every input. */

2069 addReplyLongLong(c,zsetLength(dstobj));

2072 (op == REDIS_OP_UNION) ? "zunionstore" : "zinterstore",
2073 dstkey,c->db->id);
2074 server.dirty++;
2075 } else {
2076 decrRefCount(dstobj);
2077 addReply(c,shared.czero);
2078 if (touched)
2079 notifyKeyspaceEvent(REDIS_NOTIFY_GENERIC,"del",dstkey,c->db->id);
2080 }
2081 zfree(src);

2640 zzlNext(zl,&eptr,&sptr);
2641 }
2642 }
2643
2644 while (eptr && limit--) {
2645 /* Abort when the node is no longer in range. */
2646 if (reverse) {
2647 if (!zzlLexValueGteMin(eptr,&range)) break;
2648 } else {
2649 if (!zzlLexValueLteMax(eptr,&range)) break;
2650 }
2651
2652 /* We know the element exists, so ziplistGet should always
2653 * succeed. */
2654 redisAssertWithInfo(c,zobj,ziplistGet(eptr,&vstr,&vlen,&vlong));
2655
2656 rangelen++;
2657 if (vstr == NULL) {
2658 addReplyBulkLongLong(c,vlong);
2659 } else {
2660 addReplyBulkCBuffer(c,vstr,vlen);
2661 }
2662
2663 /* Move to next node */
2664 if (reverse) {
2665 zzlPrev(zl,&eptr,&sptr);
2666 } else {
2667 zzlNext(zl,&eptr,&sptr);
2668 }
2669 }
2670 } else if (zobj->encoding == REDIS_ENCODING_SKIPLIST) {
2671 zset *zs = zobj->ptr;
2672 zskiplist *zsl = zs->zsl;
2673 zskiplistNode *ln;
2674
2675 /* If reversed, get the last node in range as starting point. */
2676 if (reverse) {
2677 ln = zslLastInLexRange(zsl,&range);
2678 } else {
2679 ln = zslFirstInLexRange(zsl,&range);
2680 }
2681
2682 /* No "first" element in the specified interval. */
2683 if (ln == NULL) {
2684 addReply(c, shared.emptymultibulk);
2685 zslFreeLexRange(&range);
2686 return;
2687 }
2688
2689 /* We don't know in advance how many matching elements there are in the
2690 * list, so we push this object that will represent the multi-bulk
2691 * length in the output buffer, and will "fix" it later */
2692 replylen = addDeferredMultiBulkLength(c);
2693
2694 /* If there is an offset, just traverse the number of elements without
2695 * checking the score because that is done in the next loop. */
2696 while (ln && offset--) {
2697 if (reverse) {
2698 ln = ln->backward;
2699 } else {
2700 ln = ln->level[0].forward;
2701 }
2702 }
2703
2704 while (ln && limit--) {
2705 /* Abort when the node is no longer in range. */
2706 if (reverse) {
2707 if (!zslLexValueGteMin(ln->obj,&range)) break;
2708 } else {
2709 if (!zslLexValueLteMax(ln->obj,&range)) break;
2710 }
2711
2712 rangelen++;
2713 addReplyBulk(c,ln->obj);
2714
2715 /* Move to next node */
2716 if (reverse) {
2717 ln = ln->backward;
2718 } else {
2719 ln = ln->level[0].forward;
2720 }
2721 }
2722 } else {
2723 redisPanic("Unknown sorted set encoding");
2724 }
2725
2726 zslFreeLexRange(&range);
2727 setDeferredMultiBulkLength(c, replylen, rangelen);
2728}

2804 rank++;

2810 addReplyLongLong(c,llen-rank);

2813 } else {

2815 }

2817 zset *zs = zobj->ptr;
2818 zskiplist *zsl = zs->zsl;
2819 dictEntry *de;
