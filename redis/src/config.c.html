63int yesnotoi(char *s) {
64 if (!strcasecmp(s,"yes")) return 1;
65 else if (!strcasecmp(s,"no")) return 0;
66 else return -1;

69void appendServerSaveParams(time_t seconds, int changes) {
70 server.saveparams = zrealloc(server.saveparams,sizeof(struct saveparam)*(server.saveparamslen+1));
71 server.saveparams[server.saveparamslen].seconds = seconds;
72 server.saveparams[server.saveparamslen].changes = changes;
73 server.saveparamslen++;
74}
75
76void resetServerSaveParams(void) {
77 zfree(server.saveparams);
78 server.saveparams = NULL;
79 server.saveparamslen = 0;
80}

82void loadServerConfigFromString(char *config) {
83 char *err = NULL;
84 int linenum = 0, totlines, i;
85 int slaveof_linenum = 0;
86 sds *lines;
87
88 lines = sdssplitlen(config,strlen(config),"\n",1,&totlines);
89
90 for (i = 0; i < totlines; i++) {
91 sds *argv;
92 int argc;
93
94 linenum = i+1;
95 lines[i] = sdstrim(lines[i]," \t\r\n");
96
97 /* Skip comments and blank lines */
98 if (lines[i][0] == '#' || lines[i][0] == '\0') continue;
99
100 /* Split into arguments */
101 argv = sdssplitargs(lines[i],&argc);
102 if (argv == NULL) {
103 err = "Unbalanced quotes in configuration line";
104 goto loaderr;
105 }
106
107 /* Skip this line if the resulting command vector is empty. */
108 if (argc == 0) {
109 sdsfreesplitres(argv,argc);
110 continue;
111 }
112 sdstolower(argv[0]);
113
114 /* Execute config directives */
115 if (!strcasecmp(argv[0],"timeout") && argc == 2) {
116 server.maxidletime = atoi(argv[1]);
117 if (server.maxidletime < 0) {
118 err = "Invalid timeout value"; goto loaderr;
119 }
120 } else if (!strcasecmp(argv[0],"tcp-keepalive") && argc == 2) {

122 if (server.tcpkeepalive < 0) {
123 err = "Invalid tcp-keepalive value"; goto loaderr;
124 }

568/* Load the server configuration from the specified filename.
569 * The function appends the additional configuration directives stored
570 * in the 'options' string to the config file before loading.
571 *
572 * Both filename and options can be NULL, in such a case are considered
573 * empty. This way loadServerConfig can be used to just load a file or
574 * just load a string. */
575void loadServerConfig(char *filename, char *options) {
576 sds config = sdsempty();
577 char buf[REDIS_CONFIGLINE_MAX+1];
578
579 /* Load the file content */
580 if (filename) {
581 FILE *fp;
582
583 if (filename[0] == '-' && filename[1] == '\0') {
584 fp = stdin;
585 } else {
586 if ((fp = fopen(filename,"r")) == NULL) {
587 redisLog(REDIS_WARNING,
588 "Fatal error, can't open config file '%s'", filename);
589 exit(1);
590 }
591 }
592 while(fgets(buf,REDIS_CONFIGLINE_MAX+1,fp) != NULL)
593 config = sdscat(config,buf);
594 if (fp != stdin) fclose(fp);
595 }
596 /* Append the additional options */
597 if (options) {
598 config = sdscat(config,"\n");
599 config = sdscat(config,options);
600 }
601 loadServerConfigFromString(config);
602 sdsfree(config);
603}

1309/* Read the old file, split it into lines to populate a newly created
1310 * config rewrite state, and return it to the caller.
1311 *
1312 * If it is impossible to read the old file, NULL is returned.
1313 * If the old file does not exist at all, an empty state is returned. */
1314struct rewriteConfigState *rewriteConfigReadOldFile(char *path) {
1315 FILE *fp = fopen(path,"r");
1316 struct rewriteConfigState *state = zmalloc(sizeof(*state));
1317 char buf[REDIS_CONFIGLINE_MAX+1];
1318 int linenum = -1;
1319
1320 if (fp == NULL && errno != ENOENT) return NULL;
1321
1322 state->option_to_line = dictCreate(&optionToLineDictType,NULL);
1323 state->rewritten = dictCreate(&optionSetDictType,NULL);
1324 state->numlines = 0;
1325 state->lines = NULL;
1326 state->has_tail = 0;
1327 if (fp == NULL) return state;
1328
1329 /* Read the old file line by line, populate the state. */
1330 while(fgets(buf,REDIS_CONFIGLINE_MAX+1,fp) != NULL) {
1331 int argc;
1332 sds *argv;
1333 sds line = sdstrim(sdsnew(buf),"\r\n\t ");
1334
1335 linenum++; /* Zero based, so we init at -1 */
1336
1337 /* Handle comments and empty lines. */
1338 if (line[0] == '#' || line[0] == '\0') {
1339 if (!state->has_tail && !strcmp(line,REDIS_CONFIG_REWRITE_SIGNATURE))
1340 state->has_tail = 1;
1341 rewriteConfigAppendLine(state,line);
1342 continue;
1343 }
1344
1345 /* Not a comment, split into arguments. */
1346 argv = sdssplitargs(line,&argc);
1347 if (argv == NULL) {
1348 /* Apparently the line is unparsable for some reason, for
1349 * instance it may have unbalanced quotes. Load it as a
1350 * comment. */
1351 sds aux = sdsnew("# ??? ");
1352 aux = sdscatsds(aux,line);
1353 sdsfree(line);
1354 rewriteConfigAppendLine(state,aux);
1355 continue;
1356 }
1357
1358 sdstolower(argv[0]); /* We only want lowercase config directives. */
1359
1360 /* Now we populate the state according to the content of this line.
1361 * Append the line and populate the option -> line numbers map. */
1362 rewriteConfigAppendLine(state,line);
1363 rewriteConfigAddLineNumberToOption(state,argv[0],linenum);
1364
1365 sdsfreesplitres(argv,argc);
1366 }
1367 fclose(fp);
1368 return state;
1369}

1371/* Rewrite the specified configuration option with the new "line".
1372 * It progressively uses lines of the file that were already used for the same
1373 * configuration option in the old version of the file, removing that line from
1374 * the map of options -> line numbers.
1375 *
1376 * If there are lines associated with a given configuration option and
1377 * "force" is non-zero, the line is appended to the configuration file.
1378 * Usually "force" is true when an option has not its default value, so it
1379 * must be rewritten even if not present previously.
1380 *
1381 * The first time a line is appended into a configuration file, a comment
1382 * is added to show that starting from that point the config file was generated
1383 * by CONFIG REWRITE.
1384 *
1385 * "line" is either used, or freed, so the caller does not need to free it
1386 * in any way. */
1387void rewriteConfigRewriteLine(struct rewriteConfigState *state, char *option, sds line, int force) {
1388 sds o = sdsnew(option);
1389 list *l = dictFetchValue(state->option_to_line,o);
1390
1391 rewriteConfigMarkAsProcessed(state,option);
1392
1393 if (!l && !force) {
1394 /* Option not used previously, and we are not forced to use it. */
1395 sdsfree(line);
1396 sdsfree(o);
1397 return;
1398 }
1399
1400 if (l) {
1401 listNode *ln = listFirst(l);
1402 int linenum = (long) ln->value;
1403
1404 /* There are still lines in the old configuration file we can reuse
1405 * for this option. Replace the line with the new one. */
1406 listDelNode(l,ln);
1407 if (listLength(l) == 0) dictDelete(state->option_to_line,o);
1408 sdsfree(state->lines[linenum]);
1409 state->lines[linenum] = line;
1410 } else {
1411 /* Append a new line. */
1412 if (!state->has_tail) {
1413 rewriteConfigAppendLine(state,
1414 sdsnew(REDIS_CONFIG_REWRITE_SIGNATURE));
1415 state->has_tail = 1;
1416 }
1417 rewriteConfigAppendLine(state,line);
1418 }
1419 sdsfree(o);
1420}

1499 * is specified, terminated by NULL. After NULL the default value is
1500 * specified. See how the function is used for more information. */

1504 int enum_val, def_val, force;

1508 while(1) {

1510 enum_val = va_arg(ap,int);

1516 }
1517 va_end(ap);
1518
1519 force = value != def_val;
1520 line = sdscatprintf(sdsempty(),"%s %s",option,matching_name);
1521 rewriteConfigRewriteLine(state,option,line,force);

1650 * rewrite state into a single string, stripping multiple empty lines. */
1651sds rewriteConfigGetContentFromState(struct rewriteConfigState *state) {
1652 sds content = sdsempty();
1653 int j, was_empty = 0;
1654
1655 for (j = 0; j < state->numlines; j++) {
1656 /* Every cluster of empty lines is turned into a single empty line. */
1657 if (sdslen(state->lines[j]) == 0) {
1658 if (was_empty) continue;
1659 was_empty = 1;
1660 } else {
1661 was_empty = 0;
1662 }
1663 content = sdscatsds(content,state->lines[j]);
1664 content = sdscatlen(content,"\n",1);
1665 }
1666 return content;
1667}

1677/* At the end of the rewrite process the state contains the remaining
1678 * map between "option name" => "lines in the original config file".
1679 * Lines used by the rewrite process were removed by the function
1680 * rewriteConfigRewriteLine(), all the other lines are "orphaned" and
1681 * should be replaced by empty lines.
1682 *
1683 * This function does just this, iterating all the option names and
1684 * blanking all the lines still associated. */
1685void rewriteConfigRemoveOrphaned(struct rewriteConfigState *state) {
1686 dictIterator *di = dictGetIterator(state->option_to_line);
1687 dictEntry *de;
1688
1689 while((de = dictNext(di)) != NULL) {
1690 list *l = dictGetVal(de);
1691 sds option = dictGetKey(de);
1692
1693 /* Don't blank lines about options the rewrite process
1694 * don't understand. */
1695 if (dictFind(state->rewritten,option) == NULL) {
1696 redisLog(REDIS_DEBUG,"Not rewritten option: %s", option);
1697 continue;
1698 }
1699
1700 while(listLength(l)) {
1701 listNode *ln = listFirst(l);
1702 int linenum = (long) ln->value;
1703
1704 sdsfree(state->lines[linenum]);
1705 state->lines[linenum] = sdsempty();
1706 listDelNode(l,ln);
1707 }
1708 }
1709 dictReleaseIterator(di);
1710}

1712/* This function overwrites the old configuration file with the new content.
1713 *
1714 * 1) The old file length is obtained.
1715 * 2) If the new content is smaller, padding is added.
1716 * 3) A single write(2) call is used to replace the content of the file.
1717 * 4) Later the file is truncated to the length of the new content.
1718 *
1719 * This way we are sure the file is left in a consistent state even if the
1720 * process is stopped between any of the four operations.
1721 *
1722 * The function returns 0 on success, otherwise -1 is returned and errno
1723 * set accordingly. */
1724int rewriteConfigOverwriteFile(char *configfile, sds content) {
1725 int retval = 0;
1726 int fd = open(configfile,O_RDWR|O_CREAT,0644);
1727 int content_size = sdslen(content), padding = 0;
1728 struct stat sb;
1729 sds content_padded;
1730
1731 /* 1) Open the old file (or create a new one if it does not
1732 * exist), get the size. */
1733 if (fd == -1) return -1; /* errno set by open(). */
1734 if (fstat(fd,&sb) == -1) {
1735 close(fd);
1736 return -1; /* errno set by fstat(). */
1737 }
1738
1739 /* 2) Pad the content at least match the old file size. */
1740 content_padded = sdsdup(content);
1741 if (content_size < sb.st_size) {
1742 /* If the old file was bigger, pad the content with
1743 * a newline plus as many "#" chars as required. */
1744 padding = sb.st_size - content_size;
1745 content_padded = sdsgrowzero(content_padded,sb.st_size);
1746 content_padded[content_size] = '\n';
1747 memset(content_padded+content_size+1,'#',padding-1);
1748 }
1749
1750 /* 3) Write the new content using a single write(2). */
1751 if (write(fd,content_padded,strlen(content_padded)) == -1) {
1752 retval = -1;
1753 goto cleanup;
1754 }
1755
1756 /* 4) Truncate the file to the right length if we used padding. */
1757 if (padding) {
1758 if (ftruncate(fd,content_size) == -1) {
1759 /* Non critical error... */
1760 }
1761 }
1762
1763cleanup:
1764 sdsfree(content_padded);
1765 close(fd);
1766 return retval;
1767}

1895void configCommand(redisClient *c) {
1896 if (!strcasecmp(c->argv[1]->ptr,"set")) {
1897 if (c->argc != 4) goto badarity;
1898 configSetCommand(c);
1899 } else if (!strcasecmp(c->argv[1]->ptr,"get")) {
1900 if (c->argc != 3) goto badarity;
1901 configGetCommand(c);
1902 } else if (!strcasecmp(c->argv[1]->ptr,"resetstat")) {
1903 if (c->argc != 2) goto badarity;
1904 resetServerStats();
1905 resetCommandTableStats();
1906 addReply(c,shared.ok);
1907 } else if (!strcasecmp(c->argv[1]->ptr,"rewrite")) {
1908 if (c->argc != 2) goto badarity;
1909 if (server.configfile == NULL) {
1910 addReplyError(c,"The server is running without a config file");
1911 return;
1912 }
1913 if (rewriteConfig(server.configfile) == -1) {
1914 redisLog(REDIS_WARNING,"CONFIG REWRITE failed: %s", strerror(errno));
1915 addReplyErrorFormat(c,"Rewriting config file: %s", strerror(errno));
1916 } else {
1917 redisLog(REDIS_WARNING,"CONFIG REWRITE executed with success.");
1918 addReply(c,shared.ok);
1919 }
1920 } else {
1921 addReplyError(c,
1922 "CONFIG subcommand must be one of GET, SET, RESETSTAT, REWRITE");
1923 }
1924 return;
1925
1926badarity:
1927 addReplyErrorFormat(c,"Wrong number of arguments for CONFIG %s",
1928 (char*) c->argv[1]->ptr);
1929}
