127static long long ustime(void) {
128 struct timeval tv;
129 long long ust;
130
131 gettimeofday(&tv, NULL);
132 ust = ((long long)tv.tv_sec)*1000000;
133 ust += tv.tv_usec;
134 return ust;
135}

158static sds getHistoryPath() {
159 char *path = NULL;
160 sds historyPath = NULL;
161
162 /* check the env for a histfile override */
163 path = getenv(REDIS_CLI_HISTFILE_ENV);
164 if (path != NULL && *path != '\0') {
165 if (!strcmp("/dev/null", path)) {
166 return NULL;
167 }
168
169 /* if the env is set, return it */
170 historyPath = sdscatprintf(sdsempty(), "%s", path);
171 } else {
172 char *home = getenv("HOME");
173 if (home != NULL && *home != '\0') {
174 /* otherwise, return the default */
175 historyPath = sdscatprintf(sdsempty(), "%s/%s", home, REDIS_CLI_HISTFILE_DEFAULT);
176 }
177 }
178
179 return historyPath;
180}

216static void cliInitHelp(void) {
217 int commandslen = sizeof(commandHelp)/sizeof(struct commandHelp);
218 int groupslen = sizeof(commandGroups)/sizeof(char*);
219 int i, len, pos = 0;
220 helpEntry tmp;
221
222 helpEntriesLen = len = commandslen+groupslen;
223 helpEntries = malloc(sizeof(helpEntry)*len);
224
225 for (i = 0; i < groupslen; i++) {
226 tmp.argc = 1;
227 tmp.argv = malloc(sizeof(sds));
228 tmp.argv[0] = sdscatprintf(sdsempty(),"@%s",commandGroups[i]);
229 tmp.full = tmp.argv[0];
230 tmp.type = CLI_HELP_GROUP;
231 tmp.org = NULL;
232 helpEntries[pos++] = tmp;
233 }
234
235 for (i = 0; i < commandslen; i++) {
236 tmp.argv = sdssplitargs(commandHelp[i].name,&tmp.argc);
237 tmp.full = sdsnew(commandHelp[i].name);
238 tmp.type = CLI_HELP_COMMAND;
239 tmp.org = &commandHelp[i];
240 helpEntries[pos++] = tmp;
241 }
242}

268/* Output all command help, filtering by group or command name. */
269static void cliOutputHelp(int argc, char **argv) {
270 int i, j, len;
271 int group = -1;
272 helpEntry *entry;
273 struct commandHelp *help;
274
275 if (argc == 0) {
276 cliOutputGenericHelp();
277 return;
278 } else if (argc > 0 && argv[0][0] == '@') {
279 len = sizeof(commandGroups)/sizeof(char*);
280 for (i = 0; i < len; i++) {
281 if (strcasecmp(argv[0]+1,commandGroups[i]) == 0) {
282 group = i;
283 break;
284 }
285 }
286 }
287
288 assert(argc > 0);
289 for (i = 0; i < helpEntriesLen; i++) {
290 entry = &helpEntries[i];
291 if (entry->type != CLI_HELP_COMMAND) continue;
292
293 help = entry->org;
294 if (group == -1) {
295 /* Compare all arguments */
296 if (argc == entry->argc) {
297 for (j = 0; j < argc; j++) {
298 if (strcasecmp(argv[j],entry->argv[j]) != 0) break;
299 }
300 if (j == argc) {
301 cliOutputCommandHelp(help,1);
302 }
303 }
304 } else {
305 if (group == help->group) {
306 cliOutputCommandHelp(help,0);
307 }
308 }
309 }
310 printf("\r\n");
311}

313static void completionCallback(const char *buf, linenoiseCompletions *lc) {
314 size_t startpos = 0;
315 int mask;
316 int i;
317 size_t matchlen;
318 sds tmp;
319
320 if (strncasecmp(buf,"help ",5) == 0) {
321 startpos = 5;
322 while (isspace(buf[startpos])) startpos++;
323 mask = CLI_HELP_COMMAND | CLI_HELP_GROUP;
324 } else {
325 mask = CLI_HELP_COMMAND;
326 }
327
328 for (i = 0; i < helpEntriesLen; i++) {
329 if (!(helpEntries[i].type & mask)) continue;
330
331 matchlen = strlen(buf+startpos);
332 if (strncasecmp(buf+startpos,helpEntries[i].full,matchlen) == 0) {
333 tmp = sdsnewlen(buf,startpos);
334 tmp = sdscat(tmp,helpEntries[i].full);
335 linenoiseAddCompletion(lc,tmp);
336 sdsfree(tmp);
337 }
338 }
339}

373/* Connect to the server. If force is not zero the connection is performed
374 * even if there is already a connected socket. */
375static int cliConnect(int force) {
376 if (context == NULL || force) {
377 if (context != NULL)
378 redisFree(context);
379
380 if (config.hostsocket == NULL) {
381 context = redisConnect(config.hostip,config.hostport);
382 } else {
383 context = redisConnectUnix(config.hostsocket);
384 }
385
386 if (context->err) {
387 fprintf(stderr,"Could not connect to Redis at ");
388 if (config.hostsocket == NULL)
389 fprintf(stderr,"%s:%d: %s\n",config.hostip,config.hostport,context->errstr);
390 else
391 fprintf(stderr,"%s: %s\n",config.hostsocket,context->errstr);
392 redisFree(context);
393 context = NULL;
394 return REDIS_ERR;
395 }
396
397 /* Set aggressive KEEP_ALIVE socket option in the Redis context socket
398 * in order to prevent timeouts caused by the execution of long
399 * commands. At the same time this improves the detection of real
400 * errors. */
401 anetKeepAlive(NULL, context->fd, REDIS_CLI_KEEPALIVE_INTERVAL);
402
403 /* Do AUTH and select the right DB. */
404 if (cliAuth() != REDIS_OK)
405 return REDIS_ERR;
406 if (cliSelect() != REDIS_OK)
407 return REDIS_ERR;
408 }
409 return REDIS_OK;
410}

417static sds cliFormatReplyTTY(redisReply *r, char *prefix) {
418 sds out = sdsempty();
419 switch (r->type) {
420 case REDIS_REPLY_ERROR:
421 out = sdscatprintf(out,"(error) %s\n", r->str);
422 break;
423 case REDIS_REPLY_STATUS:
424 out = sdscat(out,r->str);
425 out = sdscat(out,"\n");
426 break;
427 case REDIS_REPLY_INTEGER:
428 out = sdscatprintf(out,"(integer) %lld\n",r->integer);
429 break;
430 case REDIS_REPLY_STRING:
431 /* If you are producing output for the standard output we want
432 * a more interesting output with quoted characters and so forth */
433 out = sdscatrepr(out,r->str,r->len);
434 out = sdscat(out,"\n");
435 break;
436 case REDIS_REPLY_NIL:
437 out = sdscat(out,"(nil)\n");
438 break;
439 case REDIS_REPLY_ARRAY:
440 if (r->elements == 0) {
441 out = sdscat(out,"(empty list or set)\n");
442 } else {
443 unsigned int i, idxlen = 0;
444 char _prefixlen[16];
445 char _prefixfmt[16];
446 sds _prefix;
447 sds tmp;
448
449 /* Calculate chars needed to represent the largest index */
450 i = r->elements;
451 do {
452 idxlen++;
453 i /= 10;
454 } while(i);
455
456 /* Prefix for nested multi bulks should grow with idxlen+2 spaces */
457 memset(_prefixlen,' ',idxlen+2);
458 _prefixlen[idxlen+2] = '\0';
459 _prefix = sdscat(sdsnew(prefix),_prefixlen);
460
461 /* Setup prefix format for every entry */
462 snprintf(_prefixfmt,sizeof(_prefixfmt),"%%s%%%dd) ",idxlen);
463
464 for (i = 0; i < r->elements; i++) {
465 /* Don't use the prefix for the first element, as the parent
466 * caller already prepended the index number. */
467 out = sdscatprintf(out,_prefixfmt,i == 0 ? "" : prefix,i+1);
468
469 /* Format the multi bulk entry */
470 tmp = cliFormatReplyTTY(r->element[i],_prefix);
471 out = sdscatlen(out,tmp,sdslen(tmp));
472 sdsfree(tmp);
473 }
474 sdsfree(_prefix);
475 }
476 break;
477 default:
478 fprintf(stderr,"Unknown reply type: %d\n", r->type);
479 exit(1);
480 }
481 return out;
482}

484static sds cliFormatReplyRaw(redisReply *r) {
485 sds out = sdsempty(), tmp;
486 size_t i;
487
488 switch (r->type) {
489 case REDIS_REPLY_NIL:
490 /* Nothing... */
491 break;
492 case REDIS_REPLY_ERROR:
493 out = sdscatlen(out,r->str,r->len);
494 out = sdscatlen(out,"\n",1);
495 break;
496 case REDIS_REPLY_STATUS:
497 case REDIS_REPLY_STRING:
498 out = sdscatlen(out,r->str,r->len);
499 break;
500 case REDIS_REPLY_INTEGER:
501 out = sdscatprintf(out,"%lld",r->integer);
502 break;
503 case REDIS_REPLY_ARRAY:
504 for (i = 0; i < r->elements; i++) {
505 if (i > 0) out = sdscat(out,config.mb_delim);
506 tmp = cliFormatReplyRaw(r->element[i]);
507 out = sdscatlen(out,tmp,sdslen(tmp));
508 sdsfree(tmp);
509 }
510 break;
511 default:
512 fprintf(stderr,"Unknown reply type: %d\n", r->type);
513 exit(1);
514 }
515 return out;
516}

554static int cliReadReply(int output_raw_strings) {
555 void *_reply;
556 redisReply *reply;
557 sds out = NULL;
558 int output = 1;
559
560 if (redisGetReply(context,&_reply) != REDIS_OK) {
561 if (config.shutdown) {
562 redisFree(context);
563 context = NULL;
564 return REDIS_OK;
565 }
566 if (config.interactive) {
567 /* Filter cases where we should reconnect */
568 if (context->err == REDIS_ERR_IO &&
569 (errno == ECONNRESET || errno == EPIPE))
570 return REDIS_ERR;
571 if (context->err == REDIS_ERR_EOF)
572 return REDIS_ERR;
573 }
574 cliPrintContextError();
575 exit(1);
576 return REDIS_ERR; /* avoid compiler warning */
577 }
578
579 reply = (redisReply*)_reply;
580
581 config.last_cmd_type = reply->type;
582
583 /* Check if we need to connect to a different node and reissue the
584 * request. */
585 if (config.cluster_mode && reply->type == REDIS_REPLY_ERROR &&
586 (!strncmp(reply->str,"MOVED",5) || !strcmp(reply->str,"ASK")))
587 {
588 char *p = reply->str, *s;
589 int slot;
590
591 output = 0;
592 /* Comments show the position of the pointer as:
593 *
594 * [S] for pointer 's'
595 * [P] for pointer 'p'
596 */
597 s = strchr(p,' '); /* MOVED[S]3999 127.0.0.1:6381 */
598 p = strchr(s+1,' '); /* MOVED[S]3999[P]127.0.0.1:6381 */
599 *p = '\0';
600 slot = atoi(s+1);
601 s = strchr(p+1,':'); /* MOVED 3999[P]127.0.0.1[S]6381 */
602 *s = '\0';
603 sdsfree(config.hostip);
604 config.hostip = sdsnew(p+1);
605 config.hostport = atoi(s+1);
606 if (config.interactive)
607 printf("-> Redirected to slot [%d] located at %s:%d\n",
608 slot, config.hostip, config.hostport);
609 config.cluster_reissue_command = 1;
610 cliRefreshPrompt();
611 }
612
613 if (output) {
614 if (output_raw_strings) {
615 out = cliFormatReplyRaw(reply);
616 } else {
617 if (config.output == OUTPUT_RAW) {
618 out = cliFormatReplyRaw(reply);
619 out = sdscat(out,"\n");
620 } else if (config.output == OUTPUT_STANDARD) {
621 out = cliFormatReplyTTY(reply,"");
622 } else if (config.output == OUTPUT_CSV) {
623 out = cliFormatReplyCSV(reply);
624 out = sdscat(out,"\n");
625 }
626 }
627 fwrite(out,sdslen(out),1,stdout);
628 sdsfree(out);
629 }
630 freeReplyObject(reply);
631 return REDIS_OK;
632}

634static int cliSendCommand(int argc, char **argv, int repeat) {
635 char *command = argv[0];
636 size_t *argvlen;
637 int j, output_raw;
638
639 if (!strcasecmp(command,"help") || !strcasecmp(command,"?")) {
640 cliOutputHelp(--argc, ++argv);
641 return REDIS_OK;
642 }
643
644 if (context == NULL) return REDIS_ERR;
645
646 output_raw = 0;
647 if (!strcasecmp(command,"info") ||
648 (argc == 2 && !strcasecmp(command,"cluster") &&
649 (!strcasecmp(argv[1],"nodes") ||
650 !strcasecmp(argv[1],"info"))) ||
651 (argc == 2 && !strcasecmp(command,"client") &&
652 !strcasecmp(argv[1],"list")) ||
653 (argc == 3 && !strcasecmp(command,"latency") &&
654 !strcasecmp(argv[1],"graph")) ||
655 (argc == 2 && !strcasecmp(command,"latency") &&
656 !strcasecmp(argv[1],"doctor")))
657 {
658 output_raw = 1;
659 }
660
661 if (!strcasecmp(command,"shutdown")) config.shutdown = 1;
662 if (!strcasecmp(command,"monitor")) config.monitor_mode = 1;
663 if (!strcasecmp(command,"subscribe") ||
664 !strcasecmp(command,"psubscribe")) config.pubsub_mode = 1;
665 if (!strcasecmp(command,"sync") ||
666 !strcasecmp(command,"psync")) config.slave_mode = 1;
667
668 /* Setup argument length */
669 argvlen = malloc(argc*sizeof(size_t));
670 for (j = 0; j < argc; j++)
671 argvlen[j] = sdslen(argv[j]);
672
673 while(repeat--) {
674 redisAppendCommandArgv(context,argc,(const char**)argv,argvlen);
675 while (config.monitor_mode) {
676 if (cliReadReply(output_raw) != REDIS_OK) exit(1);
677 fflush(stdout);
678 }
679
680 if (config.pubsub_mode) {
681 if (config.output != OUTPUT_RAW)
682 printf("Reading messages... (press Ctrl-C to quit)\n");
683 while (1) {
684 if (cliReadReply(output_raw) != REDIS_OK) exit(1);
685 }
686 }
687
688 if (config.slave_mode) {
689 printf("Entering slave output mode... (press Ctrl-C to quit)\n");
690 slaveMode();
691 config.slave_mode = 0;
692 free(argvlen);
693 return REDIS_ERR; /* Error = slaveMode lost connection to master */
694 }
695
696 if (cliReadReply(output_raw) != REDIS_OK) {
697 free(argvlen);
698 return REDIS_ERR;
699 } else {
700 /* Store database number when SELECT was successfully executed. */
701 if (!strcasecmp(command,"select") && argc == 2) {
702 config.dbnum = atoi(argv[1]);
703 cliRefreshPrompt();
704 } else if (!strcasecmp(command,"auth") && argc == 2) {
705 cliSelect();
706 }
707 }
708 if (config.interval) usleep(config.interval);
709 fflush(stdout); /* Make it grep friendly */
710 }
711
712 free(argvlen);
713 return REDIS_OK;
714}

716/* Send a command reconnecting the link if needed. */
717static redisReply *reconnectingRedisCommand(redisContext *c, const char *fmt, ...) {
718 redisReply *reply = NULL;
719 int tries = 0;
720 va_list ap;
721
722 assert(!c->err);
723 while(reply == NULL) {
724 while (c->err & (REDIS_ERR_IO | REDIS_ERR_EOF)) {
725 printf("\r\x1b[0K"); /* Cursor to left edge + clear line. */
726 printf("Reconnecting... %d\r", ++tries);
727 fflush(stdout);
728
729 redisFree(c);
730 c = redisConnect(config.hostip,config.hostport);
731 usleep(1000000);
732 }
733
734 va_start(ap,fmt);
735 reply = redisvCommand(c,fmt,ap);
736 va_end(ap);
737
738 if (c->err && !(c->err & (REDIS_ERR_IO | REDIS_ERR_EOF))) {
739 fprintf(stderr, "Error: %s\n", c->errstr);
740 exit(1);
741 } else if (tries > 0) {
742 printf("\r\x1b[0K"); /* Cursor to left edge + clear line. */
743 }
744 }
745
746 context = c;
747 return reply;
748}

1284/* Sends SYNC and reads the number of bytes in the payload. Used both by
1285 * slaveMode() and getRDB(). */
1286unsigned long long sendSync(int fd) {
1287 /* To start we need to send the SYNC command and return the payload.
1288 * The hiredis client lib does not understand this part of the protocol
1289 * and we don't want to mess with its buffers, so everything is performed
1290 * using direct low-level I/O. */
1291 char buf[4096], *p;
1292 ssize_t nread;
1293
1294 /* Send the SYNC command. */
1295 if (write(fd,"SYNC\r\n",6) != 6) {
1296 fprintf(stderr,"Error writing to master\n");
1297 exit(1);
1298 }
1299
1300 /* Read $<payload>\r\n, making sure to read just up to "\n" */
1301 p = buf;
1302 while(1) {
1303 nread = read(fd,p,1);
1304 if (nread <= 0) {
1305 fprintf(stderr,"Error reading bulk length while SYNCing\n");
1306 exit(1);
1307 }
1308 if (*p == '\n' && p != buf) break;
1309 if (*p != '\n') p++;
1310 }
1311 *p = '\0';
1312 if (buf[0] == '-') {
1313 printf("SYNC with master failed: %s\n", buf);
1314 exit(1);
1315 }
1316 return strtoull(buf+1,NULL,10);
1317}

1319static void slaveMode(void) {
1320 int fd = context->fd;
1321 unsigned long long payload = sendSync(fd);
1322 char buf[1024];
1323 int original_output = config.output;
1324
1325 fprintf(stderr,"SYNC with master, discarding %llu "
1326 "bytes of bulk transfer...\n", payload);
1327
1328 /* Discard the payload. */
1329 while(payload) {
1330 ssize_t nread;
1331
1332 nread = read(fd,buf,(payload > sizeof(buf)) ? sizeof(buf) : payload);
1333 if (nread <= 0) {
1334 fprintf(stderr,"Error reading RDB payload while SYNCing\n");
1335 exit(1);
1336 }
1337 payload -= nread;
1338 }
1339 fprintf(stderr,"SYNC done. Logging commands from master.\n");
1340
1341 /* Now we can use hiredis to read the incoming protocol. */
1342 config.output = OUTPUT_CSV;
1343 while (cliReadReply(0) == REDIS_OK);
1344 config.output = original_output;
1345}

1354 int s = context->fd;
1355 int fd;
1356 unsigned long long payload = sendSync(s);
1357 char buf[4096];
1358
1359 fprintf(stderr,"SYNC sent to master, writing %llu bytes to '%s'\n",
1360 payload, config.rdb_filename);
1361
1362 /* Write to file. */
1363 if (!strcmp(config.rdb_filename,"-")) {
1364 fd = STDOUT_FILENO;
1365 } else {
1366 fd = open(config.rdb_filename, O_CREAT|O_WRONLY, 0644);
1367 if (fd == -1) {
1368 fprintf(stderr, "Error opening '%s': %s\n", config.rdb_filename,
1369 strerror(errno));
1370 exit(1);
1371 }
1372 }
1373
1374 while(payload) {
1375 ssize_t nread, nwritten;
1376
1377 nread = read(s,buf,(payload > sizeof(buf)) ? sizeof(buf) : payload);
1378 if (nread <= 0) {
1379 fprintf(stderr,"I/O Error reading RDB payload from socket\n");
1380 exit(1);
1381 }
1382 nwritten = write(fd, buf, nread);
1383 if (nwritten != nread) {
1384 fprintf(stderr,"Error writing data to file: %s\n",
1385 strerror(errno));
1386 exit(1);
1387 }
1388 payload -= nread;
1389 }
1390 close(s); /* Close the file descriptor ASAP as fsync() may take time. */
1391 fsync(fd);
1392 fprintf(stderr,"Transfer finished with success.\n");
1393 exit(0);
1394}

1400static void pipeMode(void) {
1401 int fd = context->fd;
1402 long long errors = 0, replies = 0, obuf_len = 0, obuf_pos = 0;
1403 char ibuf[1024*16], obuf[1024*16]; /* Input and output buffers */
1404 char aneterr[ANET_ERR_LEN];
1405 redisReader *reader = redisReaderCreate();
1406 redisReply *reply;
1407 int eof = 0; /* True once we consumed all the standard input. */
1408 int done = 0;
1409 char magic[20]; /* Special reply we recognize. */
1410 time_t last_read_time = time(NULL);
1411
1412 srand(time(NULL));
1413
1414 /* Use non blocking I/O. */
1415 if (anetNonBlock(aneterr,fd) == ANET_ERR) {
1416 fprintf(stderr, "Can't set the socket in non blocking mode: %s\n",
1417 aneterr);
1418 exit(1);
1419 }
1420
1421 /* Transfer raw protocol and read replies from the server at the same
1422 * time. */
1423 while(!done) {
1424 int mask = AE_READABLE;
1425
1426 if (!eof || obuf_len != 0) mask |= AE_WRITABLE;
1427 mask = aeWait(fd,mask,1000);
1428
1429 /* Handle the readable state: we can read replies from the server. */
1430 if (mask & AE_READABLE) {
1431 ssize_t nread;
1432
1433 /* Read from socket and feed the hiredis reader. */
1434 do {
1435 nread = read(fd,ibuf,sizeof(ibuf));
1436 if (nread == -1 && errno != EAGAIN && errno != EINTR) {
1437 fprintf(stderr, "Error reading from the server: %s\n",
1438 strerror(errno));
1439 exit(1);
1440 }
1441 if (nread > 0) {
1442 redisReaderFeed(reader,ibuf,nread);
1443 last_read_time = time(NULL);
1444 }
1445 } while(nread > 0);
1446
1447 /* Consume replies. */
1448 do {
1449 if (redisReaderGetReply(reader,(void**)&reply) == REDIS_ERR) {
1450 fprintf(stderr, "Error reading replies from server\n");
1451 exit(1);
1452 }
1453 if (reply) {
1454 if (reply->type == REDIS_REPLY_ERROR) {
1455 fprintf(stderr,"%s\n", reply->str);
1456 errors++;
1457 } else if (eof && reply->type == REDIS_REPLY_STRING &&
1458 reply->len == 20) {
1459 /* Check if this is the reply to our final ECHO
1460 * command. If so everything was received
1461 * from the server. */
1462 if (memcmp(reply->str,magic,20) == 0) {
1463 printf("Last reply received from server.\n");
1464 done = 1;
1465 replies--;
1466 }
1467 }
1468 replies++;
1469 freeReplyObject(reply);
1470 }
1471 } while(reply);
1472 }
1473
1474 /* Handle the writable state: we can send protocol to the server. */
1475 if (mask & AE_WRITABLE) {
1476 while(1) {
1477 /* Transfer current buffer to server. */
1478 if (obuf_len != 0) {
1479 ssize_t nwritten = write(fd,obuf+obuf_pos,obuf_len);
1480
1481 if (nwritten == -1) {
1482 if (errno != EAGAIN && errno != EINTR) {
1483 fprintf(stderr, "Error writing to the server: %s\n",
1484 strerror(errno));
1485 exit(1);
1486 } else {
1487 nwritten = 0;
1488 }
1489 }
1490 obuf_len -= nwritten;
1491 obuf_pos += nwritten;
1492 if (obuf_len != 0) break; /* Can't accept more data. */
1493 }
1494 /* If buffer is empty, load from stdin. */
1495 if (obuf_len == 0 && !eof) {
1496 ssize_t nread = read(STDIN_FILENO,obuf,sizeof(obuf));
1497
1498 if (nread == 0) {
1499 /* The ECHO sequence starts with a "\r\n" so that if there
1500 * is garbage in the protocol we read from stdin, the ECHO
1501 * will likely still be properly formatted.
1502 * CRLF is ignored by Redis, so it has no effects. */
1503 char echo[] =
1504 "\r\n*2\r\n$4\r\nECHO\r\n$20\r\n01234567890123456789\r\n";
1505 int j;
1506
1507 eof = 1;
1508 /* Everything transferred, so we queue a special
1509 * ECHO command that we can match in the replies
1510 * to make sure everything was read from the server. */
1511 for (j = 0; j < 20; j++)
1512 magic[j] = rand() & 0xff;
1513 memcpy(echo+21,magic,20);
1514 memcpy(obuf,echo,sizeof(echo)-1);
1515 obuf_len = sizeof(echo)-1;
1516 obuf_pos = 0;
1517 printf("All data transferred. Waiting for the last reply...\n");
1518 } else if (nread == -1) {
1519 fprintf(stderr, "Error reading from stdin: %s\n",
1520 strerror(errno));
1521 exit(1);
1522 } else {
1523 obuf_len = nread;
1524 obuf_pos = 0;
1525 }
1526 }
1527 if (obuf_len == 0 && eof) break;
1528 }
1529 }
1530
1531 /* Handle timeout, that is, we reached EOF, and we are not getting
1532 * replies from the server for a few seconds, nor the final ECHO is
1533 * received. */
1534 if (eof && config.pipe_timeout > 0 &&
1535 time(NULL)-last_read_time > config.pipe_timeout)
1536 {
1537 fprintf(stderr,"No replies for %d seconds: exiting.\n",
1538 config.pipe_timeout);
1539 errors++;
1540 break;
1541 }
1542 }
1543 redisReaderFree(reader);
1544 printf("errors: %lld, replies: %lld\n", errors, replies);
1545 if (errors)
1546 exit(1);
1547 else
1548 exit(0);
1549}

1698static void findBigKeys(void) {
1699 unsigned long long biggest[5] = {0}, counts[5] = {0}, totalsize[5] = {0};
1700 unsigned long long sampled = 0, total_keys, totlen=0, *sizes=NULL, it=0;
1701 sds maxkeys[5] = {0};
1702 char *typename[] = {"string","list","set","hash","zset"};
1703 char *typeunit[] = {"bytes","items","members","fields","members"};
1704 redisReply *reply, *keys;
1705 unsigned int arrsize=0, i;
1706 int type, *types=NULL;
1707 double pct;
1708
1709 /* Total keys pre scanning */
1710 total_keys = getDbSize();
1711
1712 /* Status message */
1713 printf("\n# Scanning the entire keyspace to find biggest keys as well as\n");
1714 printf("# average sizes per key type. You can use -i 0.1 to sleep 0.1 sec\n");
1715 printf("# per 100 SCAN commands (not usually needed).\n\n");
1716
1717 /* New up sds strings to keep track of overall biggest per type */
1718 for(i=0;i<TYPE_NONE; i++) {
1719 maxkeys[i] = sdsempty();
1720 if(!maxkeys[i]) {
1721 fprintf(stderr, "Failed to allocate memory for largest key names!\n");
1722 exit(1);
1723 }
1724 }
1725
1726 /* SCAN loop */
1727 do {
1728 /* Calculate approximate percentage completion */
1729 pct = 100 * (double)sampled/total_keys;
1730
1731 /* Grab some keys and point to the keys array */
1732 reply = sendScan(&it);
1733 keys = reply->element[1];
1734
1735 /* Reallocate our type and size array if we need to */
1736 if(keys->elements > arrsize) {
1737 types = zrealloc(types, sizeof(int)*keys->elements);
1738 sizes = zrealloc(sizes, sizeof(unsigned long long)*keys->elements);
1739
1740 if(!types || !sizes) {
1741 fprintf(stderr, "Failed to allocate storage for keys!\n");
1742 exit(1);
1743 }
1744
1745 arrsize = keys->elements;
1746 }
1747
1748 /* Retreive types and then sizes */
1749 getKeyTypes(keys, types);
1750 getKeySizes(keys, types, sizes);
1751
1752 /* Now update our stats */
1753 for(i=0;i<keys->elements;i++) {
1754 if((type = types[i]) == TYPE_NONE)
1755 continue;
1756
1757 totalsize[type] += sizes[i];
1758 counts[type]++;
1759 totlen += keys->element[i]->len;
1760 sampled++;
1761
1762 if(biggest[type]<sizes[i]) {
1763 printf(
1764 "[%05.2f%%] Biggest %-6s found so far '%s' with %llu %s\n",
1765 pct, typename[type], keys->element[i]->str, sizes[i],
1766 typeunit[type]);
1767
1768 /* Keep track of biggest key name for this type */
1769 maxkeys[type] = sdscpy(maxkeys[type], keys->element[i]->str);
1770 if(!maxkeys[type]) {
1771 fprintf(stderr, "Failed to allocate memory for key!\n");
1772 exit(1);
1773 }
1774
1775 /* Keep track of the biggest size for this type */
1776 biggest[type] = sizes[i];
1777 }
1778
1779 /* Update overall progress */
1780 if(sampled % 1000000 == 0) {
1781 printf("[%05.2f%%] Sampled %llu keys so far\n", pct, sampled);
1782 }
1783 }
1784
1785 /* Sleep if we've been directed to do so */
1786 if(sampled && (sampled %100) == 0 && config.interval) {
1787 usleep(config.interval);
1788 }
1789
1790 freeReplyObject(reply);
1791 } while(it != 0);
1792
1793 if(types) zfree(types);
1794 if(sizes) zfree(sizes);
1795
1796 /* We're done */
1797 printf("\n-------- summary -------\n\n");
1798
1799 printf("Sampled %llu keys in the keyspace!\n", sampled);
1800 printf("Total key length in bytes is %llu (avg len %.2f)\n\n",
1801 totlen, totlen ? (double)totlen/sampled : 0);
1802
1803 /* Output the biggest keys we found, for types we did find */
1804 for(i=0;i<TYPE_NONE;i++) {
1805 if(sdslen(maxkeys[i])>0) {
1806 printf("Biggest %6s found '%s' has %llu %s\n", typename[i], maxkeys[i],
1807 biggest[i], typeunit[i]);
1808 }
1809 }
1810
1811 printf("\n");
1812
1813 for(i=0;i<TYPE_NONE;i++) {
1814 printf("%llu %ss with %llu %s (%05.2f%% of keys, avg size %.2f)\n",
1815 counts[i], typename[i], totalsize[i], typeunit[i],
1816 sampled ? 100 * (double)counts[i]/sampled : 0,
1817 counts[i] ? (double)totalsize[i]/counts[i] : 0);
1818 }
1819
1820 /* Free sds strings containing max keys */
1821 for(i=0;i<TYPE_NONE;i++) {
1822 sdsfree(maxkeys[i]);
1823 }
1824
1825 /* Success! */
1826 exit(0);
1827}

2009/*------------------------------------------------------------------------------
2010 * LRU test mode
2011 *--------------------------------------------------------------------------- */
2012
2013/* Return an integer from min to max (both inclusive) using a power-law
2014 * distribution, depending on the value of alpha: the greater the alpha
2015 * the more bias towards lower values.
2016 *
2017 * With alpha = 6.2 the output follows the 80-20 rule where 20% of
2018 * the returned numbers will account for 80% of the frequency. */
2019long long powerLawRand(long long min, long long max, double alpha) {
2020 double pl, r;
2021
2022 max += 1;
2023 r = ((double)rand()) / RAND_MAX;
2024 pl = pow(
2025 ((pow(max,alpha+1) - pow(min,alpha+1))*r + pow(min,alpha+1)),
2026 (1.0/(alpha+1)));
2027 return (max-1-(long long)pl)+min;
2028}

2050 start_cycle = mstime();
2051 long long hits = 0, misses = 0;
2052 while(mstime() - start_cycle < 1000) {
2053 /* Write cycle. */
2054 for (j = 0; j < LRU_CYCLE_PIPELINE_SIZE; j++) {
2055 LRUTestGenKey(key,sizeof(key));
2056 redisAppendCommand(context, "SET %s val",key);
2057 }
2058 for (j = 0; j < LRU_CYCLE_PIPELINE_SIZE; j++)
2059 redisGetReply(context, (void**)&reply);
2060
2061 /* Read cycle. */
2062 for (j = 0; j < LRU_CYCLE_PIPELINE_SIZE; j++) {
2063 LRUTestGenKey(key,sizeof(key));
2064 redisAppendCommand(context, "GET %s",key);
2065 }
2066 for (j = 0; j < LRU_CYCLE_PIPELINE_SIZE; j++) {
2067 if (redisGetReply(context, (void**)&reply) == REDIS_OK) {
2068 switch(reply->type) {
2069 case REDIS_REPLY_ERROR:
2070 printf("%s\n", reply->str);
2071 break;
2072 case REDIS_REPLY_NIL:
2073 misses++;
2074 break;
2075 default:
2076 hits++;
2077 break;
2078 }
2079 }
2080 }
2081
2082 if (context->err) {
2083 fprintf(stderr,"I/O error during LRU test\n");
2084 exit(1);
2085 }
2086 }
2087 /* Print stats. */
2088 printf(
2089 "%lld Gets/sec | Hits: %lld (%.2f%%) | Misses: %lld (%.2f%%)\n",
2090 hits+misses,
2091 hits, (double)hits/(hits+misses)*100,
2092 misses, (double)misses/(hits+misses)*100);
2093 }

2097/*------------------------------------------------------------------------------
2098 * Intrisic latency mode.
2099 *
2100 * Measure max latency of a running process that does not result from
2101 * syscalls. Basically this software should provide an hint about how much
2102 * time the kernel leaves the process without a chance to run.
2103 *--------------------------------------------------------------------------- */
2104
2105/* This is just some computation the compiler can't optimize out.
2106 * Should run in less than 100-200 microseconds even using very
2107 * slow hardware. Runs in less than 10 microseconds in modern HW. */
2108unsigned long compute_something_fast(void) {
2109 unsigned char s[256], i, j, t;
2110 int count = 1000, k;
2111 unsigned long output = 0;
2112
2113 for (k = 0; k < 256; k++) s[k] = k;
2114
2115 i = 0;
2116 j = 0;
2117 while(count--) {
2118 i++;
2119 j = j + s[i];
2120 t = s[i];
2121 s[i] = s[j];
2122 s[j] = t;
2123 output += s[(s[i]+s[j])&255];
2124 }
2125 return output;
2126}

2133static void intrinsicLatencyMode(void) {
2134 long long test_end, run_time, max_latency = 0, runs = 0;
2135
2136 run_time = config.intrinsic_latency_duration*1000000;
2137 test_end = ustime() + run_time;

2141 long long start, end, latency;
2142
2143 start = ustime();
2144 compute_something_fast();

2147 runs++;
2148 if (latency <= 0) continue;

2150 /* Reporting */
2151 if (latency > max_latency) {
2152 max_latency = latency;

2154 }
2155
2156 double avg_us = (double)run_time/runs;
2157 double avg_ns = avg_us * 10e3;
2158 if (force_cancel_loop || end > test_end) {
2159 printf("\n%lld total runs "
2160 "(avg latency: "
2161 "%.4f microseconds / %.2f nanoseconds per run).\n",
2162 runs, avg_us, avg_ns);
2163 printf("Worst run took %.0fx longer than the average latency.\n",
2164 max_latency / avg_us);
2165 exit(0);
2166 }
2167 }
