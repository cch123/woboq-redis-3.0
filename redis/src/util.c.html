46/* Glob-style pattern matching. */
47int stringmatchlen(const char *pattern, int patternLen,
48 const char *string, int stringLen, int nocase)
49{
50 while(patternLen) {
51 switch(pattern[0]) {
52 case '*':
53 while (pattern[1] == '*') {
54 pattern++;
55 patternLen--;
56 }
57 if (patternLen == 1)
58 return 1; /* match */
59 while(stringLen) {
60 if (stringmatchlen(pattern+1, patternLen-1,
61 string, stringLen, nocase))
62 return 1; /* match */
63 string++;
64 stringLen--;
65 }
66 return 0; /* no match */
67 break;
68 case '?':
69 if (stringLen == 0)
70 return 0; /* no match */
71 string++;
72 stringLen--;
73 break;
74 case '[':
75 {
76 int not, match;
77
78 pattern++;
79 patternLen--;
80 not = pattern[0] == '^';
81 if (not) {
82 pattern++;
83 patternLen--;
84 }
85 match = 0;
86 while(1) {
87 if (pattern[0] == '\\') {
88 pattern++;
89 patternLen--;
90 if (pattern[0] == string[0])
91 match = 1;
92 } else if (pattern[0] == ']') {
93 break;
94 } else if (patternLen == 0) {
95 pattern--;
96 patternLen++;
97 break;
98 } else if (pattern[1] == '-' && patternLen >= 3) {
99 int start = pattern[0];
100 int end = pattern[2];
101 int c = string[0];
102 if (start > end) {
103 int t = start;
104 start = end;
105 end = t;
106 }
107 if (nocase) {
108 start = tolower(start);
109 end = tolower(end);
110 c = tolower(c);
111 }
112 pattern += 2;
113 patternLen -= 2;
114 if (c >= start && c <= end)
115 match = 1;
116 } else {
117 if (!nocase) {
118 if (pattern[0] == string[0])
119 match = 1;
120 } else {
121 if (tolower((int)pattern[0]) == tolower((int)string[0]))
122 match = 1;
123 }
124 }
125 pattern++;
126 patternLen--;
127 }
128 if (not)
129 match = !match;
130 if (!match)
131 return 0; /* no match */
132 string++;
133 stringLen--;
134 break;
135 }
136 case '\\':
137 if (patternLen >= 2) {
138 pattern++;
139 patternLen--;
140 }
141 /* fall through */
142 default:
143 if (!nocase) {
144 if (pattern[0] != string[0])
145 return 0; /* no match */
146 } else {
147 if (tolower((int)pattern[0]) != tolower((int)string[0]))
148 return 0; /* no match */
149 }
150 string++;
151 stringLen--;
152 break;
153 }
154 pattern++;
155 patternLen--;
156 if (stringLen == 0) {
157 while(*pattern == '*') {
158 pattern++;
159 patternLen--;
160 }
161 break;
162 }
163 }
164 if (patternLen == 0 && stringLen == 0)
165 return 1;
166 return 0;
167}

173/* Convert a string representing an amount of memory into the number of
174 * bytes, so for instance memtoll("1Gb") will return 1073741824 that is
175 * (1024*1024*1024).
176 *
177 * On parsing error, if *err is not NULL, it's set to 1, otherwise it's
178 * set to 0. On error the function return value is 0, regardless of the
179 * fact 'err' is NULL or not. */
180long long memtoll(const char *p, int *err) {
181 const char *u;
182 char buf[128];
183 long mul; /* unit multiplier */
184 long long val;
185 unsigned int digits;
186
187 if (err) *err = 0;
188
189 /* Search the first non digit character. */
190 u = p;
191 if (*u == '-') u++;
192 while(*u && isdigit(*u)) u++;
193 if (*u == '\0' || !strcasecmp(u,"b")) {
194 mul = 1;
195 } else if (!strcasecmp(u,"k")) {
196 mul = 1000;
197 } else if (!strcasecmp(u,"kb")) {
198 mul = 1024;
199 } else if (!strcasecmp(u,"m")) {
200 mul = 1000*1000;
201 } else if (!strcasecmp(u,"mb")) {
202 mul = 1024*1024;
203 } else if (!strcasecmp(u,"g")) {
204 mul = 1000L*1000*1000;
205 } else if (!strcasecmp(u,"gb")) {
206 mul = 1024L*1024*1024;
207 } else {
208 if (err) *err = 1;
209 return 0;
210 }
211
212 /* Copy the digits into a buffer, we'll use strtoll() to convert
213 * the digit (without the unit) into a number. */
214 digits = u-p;
215 if (digits >= sizeof(buf)) {
216 if (err) *err = 1;
217 return 0;
218 }
219 memcpy(buf,p,digits);
220 buf[digits] = '\0';
221
222 char *endptr;
223 errno = 0;
224 val = strtoll(buf,&endptr,10);
225 if ((val == 0 && errno == EINVAL) || *endptr != '\0') {
226 if (err) *err = 1;
227 return 0;
228 }
229 return val*mul;
230}

232/* Return the number of digits of 'v' when converted to string in radix 10.
233 * See ll2string() for more information. */
234uint32_t digits10(uint64_t v) {
235 if (v < 10) return 1;
236 if (v < 100) return 2;
237 if (v < 1000) return 3;
238 if (v < 1000000000000UL) {
239 if (v < 100000000UL) {
240 if (v < 1000000) {
241 if (v < 10000) return 4;
242 return 5 + (v >= 100000);
243 }
244 return 7 + (v >= 10000000UL);
245 }
246 if (v < 10000000000UL) {
247 return 9 + (v >= 1000000000UL);
248 }
249 return 11 + (v >= 100000000000UL);
250 }
251 return 12 + digits10(v / 1000000000000UL);
252}

254/* Convert a long long into a string. Returns the number of
255 * characters needed to represent the number.
256 * If the buffer is not big enough to store the string, 0 is returned.
257 *
258 * Based on the following article (that apparently does not provide a
259 * novel approach but only publicizes an already used technique):
260 *
261 * https://www.facebook.com/notes/facebook-engineering/three-optimization-tips-for-c/10151361643253920
262 *
263 * Modified in order to handle signed integers since the original code was
264 * designed for unsigned integers. */
265int ll2string(char* dst, size_t dstlen, long long svalue) {
266 static const char digits[201] =
267 "0001020304050607080910111213141516171819"
268 "2021222324252627282930313233343536373839"
269 "4041424344454647484950515253545556575859"
270 "6061626364656667686970717273747576777879"
271 "8081828384858687888990919293949596979899";
272 int negative;
273 unsigned long long value;
274
275 /* The main loop works with 64bit unsigned integers for simplicity, so
276 * we convert the number here and remember if it is negative. */
277 if (svalue < 0) {
278 if (svalue != LLONG_MIN) {
279 value = -svalue;
280 } else {
281 value = ((unsigned long long) LLONG_MAX)+1;
282 }
283 negative = 1;
284 } else {
285 value = svalue;
286 negative = 0;
287 }
288
289 /* Check length. */
290 uint32_t const length = digits10(value)+negative;
291 if (length >= dstlen) return 0;
292
293 /* Null term. */
294 uint32_t next = length;
295 dst[next] = '\0';
296 next--;
297 while (value >= 100) {
298 int const i = (value % 100) * 2;
299 value /= 100;
300 dst[next] = digits[i + 1];
301 dst[next - 1] = digits[i];
302 next -= 2;
303 }
304
305 /* Handle last 1-2 digits. */
306 if (value < 10) {
307 dst[next] = '0' + (uint32_t) value;
308 } else {
309 int i = (uint32_t) value * 2;
310 dst[next] = digits[i + 1];
311 dst[next - 1] = digits[i];
312 }
313
314 /* Add sign. */
315 if (negative) dst[0] = '-';
316 return length;
317}

319/* Convert a string into a long long. Returns 1 if the string could be parsed
320 * into a (non-overflowing) long long, 0 otherwise. The value will be set to
321 * the parsed value when appropriate. */
322int string2ll(const char *s, size_t slen, long long *value) {
323 const char *p = s;
324 size_t plen = 0;
325 int negative = 0;
326 unsigned long long v;
327
328 if (plen == slen)
329 return 0;
330
331 /* Special case: first and only digit is 0. */
332 if (slen == 1 && p[0] == '0') {
333 if (value != NULL) *value = 0;
334 return 1;
335 }
336
337 if (p[0] == '-') {
338 negative = 1;
339 p++; plen++;
340
341 /* Abort on only a negative sign. */
342 if (plen == slen)
343 return 0;
344 }
345
346 /* First digit should be 1-9, otherwise the string should just be 0. */
347 if (p[0] >= '1' && p[0] <= '9') {
348 v = p[0]-'0';
349 p++; plen++;
350 } else if (p[0] == '0' && slen == 1) {
351 *value = 0;
352 return 1;
353 } else {
354 return 0;
355 }
356
357 while (plen < slen && p[0] >= '0' && p[0] <= '9') {
358 if (v > (ULLONG_MAX / 10)) /* Overflow. */
359 return 0;
360 v *= 10;
361
362 if (v > (ULLONG_MAX - (p[0]-'0'))) /* Overflow. */
363 return 0;
364 v += p[0]-'0';
365
366 p++; plen++;
367 }
368
369 /* Return if not all bytes were used. */
370 if (plen < slen)
371 return 0;
372
373 if (negative) {
374 if (v > ((unsigned long long)(-(LLONG_MIN+1))+1)) /* Overflow. */
375 return 0;
376 if (value != NULL) *value = -v;
377 } else {
378 if (v > LLONG_MAX) /* Overflow. */
379 return 0;
380 if (value != NULL) *value = v;
381 }
382 return 1;
383}

385/* Convert a string into a long. Returns 1 if the string could be parsed into a
386 * (non-overflowing) long, 0 otherwise. The value will be set to the parsed
387 * value when appropriate. */
388int string2l(const char *s, size_t slen, long *lval) {
389 long long llval;
390
391 if (!string2ll(s,slen,&llval))
392 return 0;
393
394 if (llval < LONG_MIN || llval > LONG_MAX)
395 return 0;
396
397 *lval = (long)llval;
398 return 1;
399}

401/* Convert a double to a string representation. Returns the number of bytes
402 * required. The representation should always be parsable by strtod(3). */
403int d2string(char *buf, size_t len, double value) {
404 if (isnan(value)) {
405 len = snprintf(buf,len,"nan");
406 } else if (isinf(value)) {
407 if (value < 0)
408 len = snprintf(buf,len,"-inf");
409 else
410 len = snprintf(buf,len,"inf");
411 } else if (value == 0) {
412 /* See: http://en.wikipedia.org/wiki/Signed_zero, "Comparisons". */
413 if (1.0/value < 0)
414 len = snprintf(buf,len,"-0");
415 else
416 len = snprintf(buf,len,"0");
417 } else {
418#if (DBL_MANT_DIG >= 52) && (LLONG_MAX == 0x7fffffffffffffffLL)
419 /* Check if the float is in a safe range to be casted into a
420 * long long. We are assuming that long long is 64 bit here.
421 * Also we are assuming that there are no implementations around where
422 * double has precision < 52 bit.
423 *
424 * Under this assumptions we test if a double is inside an interval
425 * where casting to long long is safe. Then using two castings we
426 * make sure the decimal part is zero. If all this is true we use
427 * integer printing function that is much faster. */
428 double min = -4503599627370495; /* (2^52)-1 */
429 double max = 4503599627370496; /* -(2^52) */
430 if (value > min && value < max && value == ((double)((long long)value)))
431 len = ll2string(buf,len,(long long)value);
432 else
433#endif
434 len = snprintf(buf,len,"%.17g",value);
435 }
436
437 return len;
438}

440/* Generate the Redis "Run ID", a SHA1-sized random number that identifies a
441 * given execution of Redis, so that if you are talking with an instance
442 * having run_id == A, and you reconnect and it has run_id == B, you can be
443 * sure that it is either a different instance or it was restarted. */
444void getRandomHexChars(char *p, unsigned int len) {
445 char *charset = "0123456789abcdef";
446 unsigned int j;
447
448 /* Global state. */
449 static int seed_initialized = 0;
450 static unsigned char seed[20]; /* The SHA1 seed, from /dev/urandom. */
451 static uint64_t counter = 0; /* The counter we hash with the seed. */
452
453 if (!seed_initialized) {
454 /* Initialize a seed and use SHA1 in counter mode, where we hash
455 * the same seed with a progressive counter. For the goals of this
456 * function we just need non-colliding strings, there are no
457 * cryptographic security needs. */
458 FILE *fp = fopen("/dev/urandom","r");
459 if (fp && fread(seed,sizeof(seed),1,fp) == 1)
460 seed_initialized = 1;
461 if (fp) fclose(fp);
462 }
463
464 if (seed_initialized) {
465 while(len) {
466 unsigned char digest[20];
467 SHA1_CTX ctx;
468 unsigned int copylen = len > 20 ? 20 : len;
469
470 SHA1Init(&ctx);
471 SHA1Update(&ctx, seed, sizeof(seed));
472 SHA1Update(&ctx, (unsigned char*)&counter,sizeof(counter));
473 SHA1Final(digest, &ctx);
474 counter++;
475
476 memcpy(p,digest,copylen);
477 /* Convert to hex digits. */
478 for (j = 0; j < copylen; j++) p[j] = charset[p[j] & 0x0F];
479 len -= copylen;
480 p += copylen;
481 }
482 } else {
483 /* If we can't read from /dev/urandom, do some reasonable effort
484 * in order to create some entropy, since this function is used to
485 * generate run_id and cluster instance IDs */
486 char *x = p;
487 unsigned int l = len;
488 struct timeval tv;
489 pid_t pid = getpid();
490
491 /* Use time and PID to fill the initial array. */
492 gettimeofday(&tv,NULL);
493 if (l >= sizeof(tv.tv_usec)) {
494 memcpy(x,&tv.tv_usec,sizeof(tv.tv_usec));
495 l -= sizeof(tv.tv_usec);
496 x += sizeof(tv.tv_usec);
497 }
498 if (l >= sizeof(tv.tv_sec)) {
499 memcpy(x,&tv.tv_sec,sizeof(tv.tv_sec));
500 l -= sizeof(tv.tv_sec);
501 x += sizeof(tv.tv_sec);
502 }
503 if (l >= sizeof(pid)) {
504 memcpy(x,&pid,sizeof(pid));
505 l -= sizeof(pid);
506 x += sizeof(pid);
507 }
508 /* Finally xor it with rand() output, that was already seeded with
509 * time() at startup, and convert to hex digits. */
510 for (j = 0; j < len; j++) {
511 p[j] ^= rand();
512 p[j] = charset[p[j] & 0x0F];
513 }
514 }
515}

517/* Given the filename, return the absolute path as an SDS string, or NULL
518 * if it fails for some reason. Note that "filename" may be an absolute path
519 * already, this will be detected and handled correctly.
520 *
521 * The function does not try to normalize everything, but only the obvious
522 * case of one or more "../" appearning at the start of "filename"
523 * relative path. */
524sds getAbsolutePath(char *filename) {
525 char cwd[1024];
526 sds abspath;
527 sds relpath = sdsnew(filename);
528
529 relpath = sdstrim(relpath," \r\n\t");
530 if (relpath[0] == '/') return relpath; /* Path is already absolute. */
531
532 /* If path is relative, join cwd and relative path. */
533 if (getcwd(cwd,sizeof(cwd)) == NULL) {
534 sdsfree(relpath);
535 return NULL;
536 }
537 abspath = sdsnew(cwd);
538 if (sdslen(abspath) && abspath[sdslen(abspath)-1] != '/')
539 abspath = sdscat(abspath,"/");
540
541 /* At this point we have the current path always ending with "/", and
542 * the trimmed relative path. Try to normalize the obvious case of
543 * trailing ../ elements at the start of the path.
544 *
545 * For every "../" we find in the filename, we remove it and also remove
546 * the last element of the cwd, unless the current cwd is "/". */
547 while (sdslen(relpath) >= 3 &&
548 relpath[0] == '.' && relpath[1] == '.' && relpath[2] == '/')
549 {
550 sdsrange(relpath,3,-1);
551 if (sdslen(abspath) > 1) {
552 char *p = abspath + sdslen(abspath)-2;
553 int trimlen = 1;
554
555 while(*p != '/') {
556 p--;
557 trimlen++;
558 }
559 sdsrange(abspath,0,-(trimlen+1));
560 }
561 }
562
563 /* Finally glue the two parts together. */
564 abspath = sdscatsds(abspath,relpath);
565 sdsfree(relpath);
566 return abspath;
567}

569/* Return true if the specified path is just a file basename without any
570 * relative or absolute path. This function just checks that no / or \
571 * character exists inside the specified path, that's enough in the
572 * environments where Redis runs. */
573int pathIsBaseName(char *path) {
574 return strchr(path,'/') == NULL && strchr(path,'\\') == NULL;
575}

606 assert(v == 0);
607
608 strcpy(buf,"1");
609 assert(string2ll(buf,strlen(buf),&v) == 1);

619
