55 * with three sequences. In order to be so cheap we may have some
56 * flickering effect with some slow terminal, but the lesser sequences
57 * the more compatible.
58 *
59 * EL (Erase Line)
60 * Sequence: ESC [ n K
61 * Effect: if n is 0 or missing, clear from cursor to end of line
62 * Effect: if n is 1, clear from beginning of line to cursor
63 * Effect: if n is 2, clear entire line
64 *

66 * Sequence: ESC [ n C
67 * Effect: moves cursor forward n chars
68 *

79 * where n is the row and m is the column
80 *
81 * When multi line mode is enabled, we also use an additional escape
82 * sequence. However multi line editing is disabled by default.
83 *
84 * CUU (Cursor Up)
85 * Sequence: ESC [ n A
86 * Effect: moves cursor up of n chars.
87 *
88 * CUD (Cursor Down)
89 * Sequence: ESC [ n B
90 * Effect: moves cursor down of n chars.
91 *
92 * When linenoiseClearScreen() is called, two additional escape sequences
93 * are used in order to clear the screen and position the cursor at home
94 * position.
95 *
96 * CUP (Cursor position)
97 * Sequence: ESC [ H
98 * Effect: moves the cursor to upper left corner
99 *
100 * ED (Erase display)
101 * Sequence: ESC [ 2 J
102 * Effect: clear the whole screen
103 *
104 */

132/* The linenoiseState structure represents the state during line editing.
133 * We pass this state to functions implementing specific editing
134 * functionalities. */
135struct linenoiseState {
136 int ifd; /* Terminal stdin file descriptor. */
137 int ofd; /* Terminal stdout file descriptor. */
138 char *buf; /* Edited line buffer. */
139 size_t buflen; /* Edited line buffer size. */
140 const char *prompt; /* Prompt to display. */
141 size_t plen; /* Prompt length. */
142 size_t pos; /* Current cursor position. */
143 size_t oldpos; /* Previous refresh cursor position. */
144 size_t len; /* Current edited line length. */
145 size_t cols; /* Number of columns in terminal. */
146 size_t maxrows; /* Maximum num of rows used so far (multiline mode) */
147 int history_index; /* The history index we are currently editing. */
148};

202/* Return true if the terminal name is in the list of terminals we know are
203 * not able to understand basic escape sequences. */
204static int isUnsupportedTerm(void) {
205 char *term = getenv("TERM");
206 int j;
207
208 if (term == NULL) return 0;
209 for (j = 0; unsupported_term[j]; j++)
210 if (!strcasecmp(term,unsupported_term[j])) return 1;
211 return 0;

214/* Raw mode: 1960 magic shit. */
215static int enableRawMode(int fd) {
216 struct termios raw;
217
218 if (!isatty(STDIN_FILENO)) goto fatal;
219 if (!atexit_registered) {
220 atexit(linenoiseAtExit);
221 atexit_registered = 1;
222 }
223 if (tcgetattr(fd,&orig_termios) == -1) goto fatal;
224
225 raw = orig_termios; /* modify the original mode */
226 /* input modes: no break, no CR to NL, no parity check, no strip char,
227 * no start/stop output control. */
228 raw.c_iflag &= ~(BRKINT | ICRNL | INPCK | ISTRIP | IXON);
229 /* output modes - disable post processing */
230 raw.c_oflag &= ~(OPOST);
231 /* control modes - set 8 bit chars */
232 raw.c_cflag |= (CS8);
233 /* local modes - choing off, canonical off, no extended functions,
234 * no signal chars (^Z,^C) */
235 raw.c_lflag &= ~(ECHO | ICANON | IEXTEN | ISIG);
236 /* control chars - set return condition: min number of bytes and timer.
237 * We want read to return every single byte, without timeout. */
238 raw.c_cc[VMIN] = 1; raw.c_cc[VTIME] = 0; /* 1 byte, no timer */
239
240 /* put terminal in raw mode after flushing */
241 if (tcsetattr(fd,TCSAFLUSH,&raw) < 0) goto fatal;
242 rawmode = 1;
243 return 0;
244
245fatal:
246 errno = ENOTTY;
247 return -1;
248}

250static void disableRawMode(int fd) {
251 /* Don't even check the return value as it's too late. */
252 if (rawmode && tcsetattr(fd,TCSAFLUSH,&orig_termios) != -1)
253 rawmode = 0;

256/* Use the ESC [6n escape sequence to query the horizontal cursor position
257 * and return it. On error -1 is returned, on success the position of the
258 * cursor. */
259static int getCursorPosition(int ifd, int ofd) {
260 char buf[32];
261 int cols, rows;
262 unsigned int i = 0;
263
264 /* Report cursor location */
265 if (write(ofd, "\x1b[6n", 4) != 4) return -1;
266
267 /* Read the response: ESC [ rows ; cols R */
268 while (i < sizeof(buf)-1) {
269 if (read(ifd,buf+i,1) != 1) break;
270 if (buf[i] == 'R') break;
271 i++;
272 }
273 buf[i] = '\0';
274
275 /* Parse it. */
276 if (buf[0] != ESC || buf[1] != '[') return -1;
277 if (sscanf(buf+2,"%d;%d",&rows,&cols) != 2) return -1;
278 return cols;
279}

281/* Try to get the number of columns in the current terminal, or assume 80
282 * if it fails. */
283static int getColumns(int ifd, int ofd) {
284 struct winsize ws;
285
286 if (ioctl(1, TIOCGWINSZ, &ws) == -1 || ws.ws_col == 0) {
287 /* ioctl() failed. Try to query the terminal itself. */
288 int start, cols;
289
290 /* Get the initial position so we can restore it later. */
291 start = getCursorPosition(ifd,ofd);
292 if (start == -1) goto failed;
293
294 /* Go to right margin and get position. */
295 if (write(ofd,"\x1b[999C",6) != 6) goto failed;
296 cols = getCursorPosition(ifd,ofd);
297 if (cols == -1) goto failed;
298
299 /* Restore position. */
300 if (cols > start) {
301 char seq[32];
302 snprintf(seq,32,"\x1b[%dD",cols-start);
303 if (write(ofd,seq,strlen(seq)) == -1) {
304 /* Can't recover... */
305 }
306 }
307 return cols;
308 } else {
309 return ws.ws_col;
310 }
311
312failed:
313 return 80;
314}

341/* This is an helper function for linenoiseEdit() and is called when the
342 * user types the <tab> key in order to complete the string currently in the
343 * input.
344 *
345 * The state of the editing is encapsulated into the pointed linenoiseState
346 * structure as described in the structure definition. */
347static int completeLine(struct linenoiseState *ls) {
348 linenoiseCompletions lc = { 0, NULL };
349 int nread, nwritten;
350 char c = 0;
351
352 completionCallback(ls->buf,&lc);
353 if (lc.len == 0) {
354 linenoiseBeep();
355 } else {
356 size_t stop = 0, i = 0;
357
358 while(!stop) {
359 /* Show completion or original buffer */
360 if (i < lc.len) {
361 struct linenoiseState saved = *ls;
362
363 ls->len = ls->pos = strlen(lc.cvec[i]);
364 ls->buf = lc.cvec[i];
365 refreshLine(ls);
366 ls->len = saved.len;
367 ls->pos = saved.pos;
368 ls->buf = saved.buf;
369 } else {
370 refreshLine(ls);
371 }
372
373 nread = read(ls->ifd,&c,1);
374 if (nread <= 0) {
375 freeCompletions(&lc);
376 return -1;
377 }
378
379 switch(c) {
380 case 9: /* tab */
381 i = (i+1) % (lc.len+1);
382 if (i == lc.len) linenoiseBeep();
383 break;
384 case 27: /* escape */
385 /* Re-show original buffer */
386 if (i < lc.len) refreshLine(ls);
387 stop = 1;
388 break;
389 default:
390 /* Update buffer and return */
391 if (i < lc.len) {
392 nwritten = snprintf(ls->buf,ls->buflen,"%s",lc.cvec[i]);
393 ls->len = ls->pos = nwritten;
394 }
395 stop = 1;
396 break;
397 }
398 }
399 }
400
401 freeCompletions(&lc);
402 return c; /* Return last read character */
403}

410/* This function is used by the callback function registered by the user
411 * in order to add completion options given the input string when the
412 * user typed <tab>. See the example.c source code for a very easy to
413 * understand example. */
414void linenoiseAddCompletion(linenoiseCompletions *lc, const char *str) {
415 size_t len = strlen(str);
416 char *copy, **cvec;
417
418 copy = malloc(len+1);
419 if (copy == NULL) return;
420 memcpy(copy,str,len+1);
421 cvec = realloc(lc->cvec,sizeof(char*)*(lc->len+1));
422 if (cvec == NULL) {
423 free(copy);
424 return;
425 }
426 lc->cvec = cvec;
427 lc->cvec[lc->len++] = copy;
428}

432/* We define a very simple "append buffer" structure, that is an heap
433 * allocated string where we can append to. This is useful in order to
434 * write all the escape sequences in a buffer and flush them to the standard
435 * output in a single call, to avoid flickering effects. */
436struct abuf {
437 char *b;
438 int len;
439};
440
441static void abInit(struct abuf *ab) {
442 ab->b = NULL;
443 ab->len = 0;
444}
445
446static void abAppend(struct abuf *ab, const char *s, int len) {
447 char *new = realloc(ab->b,ab->len+len);
448
449 if (new == NULL) return;
450 memcpy(new+ab->len,s,len);
451 ab->b = new;
452 ab->len += len;
453}
454
455static void abFree(struct abuf *ab) {
456 free(ab->b);
457}

459/* Single line low level line refresh.
460 *
461 * Rewrite the currently edited line accordingly to the buffer content,
462 * cursor position, and number of columns of the terminal. */
463static void refreshSingleLine(struct linenoiseState *l) {
464 char seq[64];
465 size_t plen = strlen(l->prompt);
466 int fd = l->ofd;
467 char *buf = l->buf;
468 size_t len = l->len;
469 size_t pos = l->pos;
470 struct abuf ab;
471
472 while((plen+pos) >= l->cols) {
473 buf++;
474 len--;
475 pos--;
476 }
477 while (plen+len > l->cols) {
478 len--;
479 }
480
481 abInit(&ab);
482 /* Cursor to left edge */
483 snprintf(seq,64,"\r");
484 abAppend(&ab,seq,strlen(seq));
485 /* Write the prompt and the current buffer content */
486 abAppend(&ab,l->prompt,strlen(l->prompt));
487 abAppend(&ab,buf,len);
488 /* Erase to right */
489 snprintf(seq,64,"\x1b[0K");
490 abAppend(&ab,seq,strlen(seq));
491 /* Move cursor to original position. */
492 snprintf(seq,64,"\r\x1b[%dC", (int)(pos+plen));
493 abAppend(&ab,seq,strlen(seq));
494 if (write(fd,ab.b,ab.len) == -1) {} /* Can't recover from write error. */
495 abFree(&ab);
496}

498/* Multi line low level line refresh.
499 *
500 * Rewrite the currently edited line accordingly to the buffer content,
501 * cursor position, and number of columns of the terminal. */
502static void refreshMultiLine(struct linenoiseState *l) {
503 char seq[64];
504 int plen = strlen(l->prompt);
505 int rows = (plen+l->len+l->cols-1)/l->cols; /* rows used by current buf. */
506 int rpos = (plen+l->oldpos+l->cols)/l->cols; /* cursor relative row. */
507 int rpos2; /* rpos after refresh. */
508 int col; /* colum position, zero-based. */
509 int old_rows = l->maxrows;
510 int fd = l->ofd, j;
511 struct abuf ab;
512
513 /* Update maxrows if needed. */
514 if (rows > (int)l->maxrows) l->maxrows = rows;
515
516 /* First step: clear all the lines used before. To do so start by
517 * going to the last row. */
518 abInit(&ab);
519 if (old_rows-rpos > 0) {
520 lndebug("go down %d", old_rows-rpos);
521 snprintf(seq,64,"\x1b[%dB", old_rows-rpos);
522 abAppend(&ab,seq,strlen(seq));
523 }
524
525 /* Now for every row clear it, go up. */
526 for (j = 0; j < old_rows-1; j++) {
527 lndebug("clear+up");
528 snprintf(seq,64,"\r\x1b[0K\x1b[1A");
529 abAppend(&ab,seq,strlen(seq));
530 }
531
532 /* Clean the top line. */
533 lndebug("clear");
534 snprintf(seq,64,"\r\x1b[0K");
535 abAppend(&ab,seq,strlen(seq));
536
537 /* Write the prompt and the current buffer content */
538 abAppend(&ab,l->prompt,strlen(l->prompt));
539 abAppend(&ab,l->buf,l->len);
540
541 /* If we are at the very end of the screen with our prompt, we need to
542 * emit a newline and move the prompt to the first column. */
543 if (l->pos &&
544 l->pos == l->len &&
545 (l->pos+plen) % l->cols == 0)
546 {
547 lndebug("<newline>");
548 abAppend(&ab,"\n",1);
549 snprintf(seq,64,"\r");
550 abAppend(&ab,seq,strlen(seq));
551 rows++;
552 if (rows > (int)l->maxrows) l->maxrows = rows;
553 }
554
555 /* Move cursor to right position. */
556 rpos2 = (plen+l->pos+l->cols)/l->cols; /* current cursor relative row. */
557 lndebug("rpos2 %d", rpos2);
558
559 /* Go up till we reach the expected positon. */
560 if (rows-rpos2 > 0) {
561 lndebug("go-up %d", rows-rpos2);
562 snprintf(seq,64,"\x1b[%dA", rows-rpos2);
563 abAppend(&ab,seq,strlen(seq));
564 }
565
566 /* Set column. */
567 col = (plen+(int)l->pos) % (int)l->cols;
568 lndebug("set col %d", 1+col);
569 if (col)
570 snprintf(seq,64,"\r\x1b[%dC", col);
571 else
572 snprintf(seq,64,"\r");
573 abAppend(&ab,seq,strlen(seq));
574
575 lndebug("\n");
576 l->oldpos = l->pos;
577
578 if (write(fd,ab.b,ab.len) == -1) {} /* Can't recover from write error. */
579 abFree(&ab);
580}

582/* Calls the two low level functions refreshSingleLine() or
583 * refreshMultiLine() according to the selected mode. */
584static void refreshLine(struct linenoiseState *l) {
585 if (mlmode)
586 refreshMultiLine(l);
587 else
588 refreshSingleLine(l);
589}

591/* Insert the character 'c' at cursor current position.
592 *
593 * On error writing to the terminal -1 is returned, otherwise 0. */
594int linenoiseEditInsert(struct linenoiseState *l, char c) {
595 if (l->len < l->buflen) {
596 if (l->len == l->pos) {
597 l->buf[l->pos] = c;
598 l->pos++;
599 l->len++;
600 l->buf[l->len] = '\0';
601 if ((!mlmode && l->plen+l->len < l->cols) /* || mlmode */) {
602 /* Avoid a full update of the line in the
603 * trivial case. */
604 if (write(l->ofd,&c,1) == -1) return -1;
605 } else {
606 refreshLine(l);
607 }
608 } else {
609 memmove(l->buf+l->pos+1,l->buf+l->pos,l->len-l->pos);
610 l->buf[l->pos] = c;
611 l->len++;
612 l->pos++;
613 l->buf[l->len] = '\0';
614 refreshLine(l);
615 }
616 }
617 return 0;
618}

652/* Substitute the currently edited line with the next or previous history
653 * entry as specified by 'dir'. */
654#define LINENOISE_HISTORY_NEXT 0
655#define LINENOISE_HISTORY_PREV 1
656void linenoiseEditHistoryNext(struct linenoiseState *l, int dir) {
657 if (history_len > 1) {
658 /* Update the current history entry before to
659 * overwrite it with the next one. */
660 free(history[history_len - 1 - l->history_index]);
661 history[history_len - 1 - l->history_index] = strdup(l->buf);
662 /* Show the new entry */
663 l->history_index += (dir == LINENOISE_HISTORY_PREV) ? 1 : -1;
664 if (l->history_index < 0) {
665 l->history_index = 0;
666 return;
667 } else if (l->history_index >= history_len) {
668 l->history_index = history_len-1;
669 return;
670 }
671 strncpy(l->buf,history[history_len - 1 - l->history_index],l->buflen);
672 l->buf[l->buflen-1] = '\0';
673 l->len = l->pos = strlen(l->buf);
674 refreshLine(l);
675 }
676}

678/* Delete the character at the right of the cursor without altering the cursor
679 * position. Basically this is what happens with the "Delete" keyboard key. */
680void linenoiseEditDelete(struct linenoiseState *l) {
681 if (l->len > 0 && l->pos < l->len) {
682 memmove(l->buf+l->pos,l->buf+l->pos+1,l->len-l->pos-1);
683 l->len--;
684 l->buf[l->len] = '\0';
685 refreshLine(l);
686 }
687}

692 memmove(l->buf+l->pos-1,l->buf+l->pos,l->len-l->pos);
693 l->pos--;
694 l->len--;
695 l->buf[l->len] = '\0';
696 refreshLine(l);

700/* Delete the previosu word, maintaining the cursor at the start of the
701 * current word. */
702void linenoiseEditDeletePrevWord(struct linenoiseState *l) {
703 size_t old_pos = l->pos;
704 size_t diff;
705
706 while (l->pos > 0 && l->buf[l->pos-1] == ' ')
707 l->pos--;
708 while (l->pos > 0 && l->buf[l->pos-1] != ' ')
709 l->pos--;
710 diff = old_pos - l->pos;
711 memmove(l->buf+l->pos,l->buf+old_pos,l->len-old_pos+1);
712 l->len -= diff;
713 refreshLine(l);
714}

716/* This function is the core of the line editing capability of linenoise.
717 * It expects 'fd' to be already in "raw mode" so that every key pressed
718 * will be returned ASAP to read().
719 *
720 * The resulting string is put into 'buf' when the user type enter, or
721 * when ctrl+d is typed.
722 *
723 * The function returns the length of the current buffer. */
724static int linenoiseEdit(int stdin_fd, int stdout_fd, char *buf, size_t buflen, const char *prompt)
725{
726 struct linenoiseState l;
727
728 /* Populate the linenoise state that we pass to functions implementing
729 * specific editing functionalities. */
730 l.ifd = stdin_fd;
731 l.ofd = stdout_fd;
732 l.buf = buf;
733 l.buflen = buflen;
734 l.prompt = prompt;
735 l.plen = strlen(prompt);
736 l.oldpos = l.pos = 0;
737 l.len = 0;
738 l.cols = getColumns(stdin_fd, stdout_fd);
739 l.maxrows = 0;
740 l.history_index = 0;
741
742 /* Buffer starts empty. */
743 l.buf[0] = '\0';
744 l.buflen--; /* Make sure there is always space for the nulterm */
745
746 /* The latest history entry is always our current buffer, that
747 * initially is just an empty string. */
748 linenoiseHistoryAdd("");
749
750 if (write(l.ofd,prompt,l.plen) == -1) return -1;
751 while(1) {
752 char c;
753 int nread;
754 char seq[3];
755
756 nread = read(l.ifd,&c,1);
757 if (nread <= 0) return l.len;
758
759 /* Only autocomplete when the callback is set. It returns < 0 when
760 * there was an error reading from fd. Otherwise it will return the
761 * character that should be handled next. */
762 if (c == 9 && completionCallback != NULL) {
763 c = completeLine(&l);
764 /* Return on errors */
765 if (c < 0) return l.len;
766 /* Read next character when 0 */
767 if (c == 0) continue;
768 }
769
770 switch(c) {
771 case ENTER: /* enter */
772 history_len--;
773 free(history[history_len]);
774 if (mlmode) linenoiseEditMoveEnd(&l);
775 return (int)l.len;
776 case CTRL_C: /* ctrl-c */
777 errno = EAGAIN;
778 return -1;
779 case BACKSPACE: /* backspace */
780 case 8: /* ctrl-h */
781 linenoiseEditBackspace(&l);
782 break;
783 case CTRL_D: /* ctrl-d, remove char at right of cursor, or if the
784 line is empty, act as end-of-file. */
785 if (l.len > 0) {
786 linenoiseEditDelete(&l);
787 } else {
788 history_len--;
789 free(history[history_len]);
790 return -1;
791 }
792 break;
793 case CTRL_T: /* ctrl-t, swaps current character with previous. */
794 if (l.pos > 0 && l.pos < l.len) {
795 int aux = buf[l.pos-1];
796 buf[l.pos-1] = buf[l.pos];
797 buf[l.pos] = aux;
798 if (l.pos != l.len-1) l.pos++;
799 refreshLine(&l);
800 }
801 break;
802 case CTRL_B: /* ctrl-b */
803 linenoiseEditMoveLeft(&l);
804 break;
805 case CTRL_F: /* ctrl-f */
806 linenoiseEditMoveRight(&l);
807 break;
808 case CTRL_P: /* ctrl-p */
809 linenoiseEditHistoryNext(&l, LINENOISE_HISTORY_PREV);
810 break;
811 case CTRL_N: /* ctrl-n */
812 linenoiseEditHistoryNext(&l, LINENOISE_HISTORY_NEXT);
813 break;
814 case ESC: /* escape sequence */
815 /* Read the next two bytes representing the escape sequence.
816 * Use two calls to handle slow terminals returning the two
817 * chars at different times. */
818 if (read(l.ifd,seq,1) == -1) break;
819 if (read(l.ifd,seq+1,1) == -1) break;
820
821 /* ESC [ sequences. */
822 if (seq[0] == '[') {
823 if (seq[1] >= '0' && seq[1] <= '9') {
824 /* Extended escape, read additional byte. */
825 if (read(l.ifd,seq+2,1) == -1) break;
826 if (seq[2] == '~') {
827 switch(seq[1]) {
828 case '3': /* Delete key. */
829 linenoiseEditDelete(&l);
830 break;
831 }
832 }
833 } else {
834 switch(seq[1]) {
835 case 'A': /* Up */
836 linenoiseEditHistoryNext(&l, LINENOISE_HISTORY_PREV);
837 break;
838 case 'B': /* Down */
839 linenoiseEditHistoryNext(&l, LINENOISE_HISTORY_NEXT);
840 break;
841 case 'C': /* Right */
842 linenoiseEditMoveRight(&l);
843 break;
844 case 'D': /* Left */
845 linenoiseEditMoveLeft(&l);
846 break;
847 case 'H': /* Home */
848 linenoiseEditMoveHome(&l);
849 break;
850 case 'F': /* End*/
851 linenoiseEditMoveEnd(&l);
852 break;
853 }
854 }
855 }
856
857 /* ESC O sequences. */
858 else if (seq[0] == 'O') {
859 switch(seq[1]) {
860 case 'H': /* Home */
861 linenoiseEditMoveHome(&l);
862 break;
863 case 'F': /* End*/
864 linenoiseEditMoveEnd(&l);
865 break;
866 }
867 }
868 break;
869 default:
870 if (linenoiseEditInsert(&l,c)) return -1;
871 break;
872 case CTRL_U: /* Ctrl+u, delete the whole line. */
873 buf[0] = '\0';
874 l.pos = l.len = 0;
875 refreshLine(&l);
876 break;
877 case CTRL_K: /* Ctrl+k, delete from current to end of line. */
878 buf[l.pos] = '\0';
879 l.len = l.pos;
880 refreshLine(&l);
881 break;
882 case CTRL_A: /* Ctrl+a, go to the start of the line */
883 linenoiseEditMoveHome(&l);
884 break;
885 case CTRL_E: /* ctrl+e, go to the end of the line */
886 linenoiseEditMoveEnd(&l);
887 break;
888 case CTRL_L: /* ctrl+l, clear screen */
889 linenoiseClearScreen();
890 refreshLine(&l);
891 break;
892 case CTRL_W: /* ctrl+w, delete previous word */
893 linenoiseEditDeletePrevWord(&l);
894 break;
895 }
896 }
897 return l.len;
898}

900/* This special mode is used by linenoise in order to print scan codes
901 * on screen for debugging / development purposes. It is implemented
902 * by the linenoise_example program using the --keycodes option. */
903void linenoisePrintKeyCodes(void) {
904 char quit[4];
905
906 printf("Linenoise key codes debugging mode.\n"
907 "Press keys to see scan codes. Type 'quit' at any time to exit.\n");
908 if (enableRawMode(STDIN_FILENO) == -1) return;
909 memset(quit,' ',4);
910 while(1) {
911 char c;
912 int nread;
913
914 nread = read(STDIN_FILENO,&c,1);
915 if (nread <= 0) continue;
916 memmove(quit,quit+1,sizeof(quit)-1); /* shift string to left. */
917 quit[sizeof(quit)-1] = c; /* Insert current char on the right. */
918 if (memcmp(quit,"quit",sizeof(quit)) == 0) break;
919
920 printf("'%c' %02x (%d) (type quit to exit)\n",
921 isprint(c) ? c : '?', (int)c, (int)c);
922 printf("\r"); /* Go left edge manually, we are in raw mode. */
923 fflush(stdout);
924 }
925 disableRawMode(STDIN_FILENO);
926}

928/* This function calls the line editing function linenoiseEdit() using
929 * the STDIN file descriptor set in raw mode. */
930static int linenoiseRaw(char *buf, size_t buflen, const char *prompt) {
931 int count;
932
933 if (buflen == 0) {
934 errno = EINVAL;
935 return -1;
936 }
937 if (!isatty(STDIN_FILENO)) {
938 /* Not a tty: read from file / pipe. */
939 if (fgets(buf, buflen, stdin) == NULL) return -1;
940 count = strlen(buf);
941 if (count && buf[count-1] == '\n') {
942 count--;
943 buf[count] = '\0';
944 }
945 } else {
946 /* Interactive editing. */
947 if (enableRawMode(STDIN_FILENO) == -1) return -1;
948 count = linenoiseEdit(STDIN_FILENO, STDOUT_FILENO, buf, buflen, prompt);
949 disableRawMode(STDIN_FILENO);
950 printf("\n");
951 }
952 return count;
953}

955/* The high level function that is the main API of the linenoise library.
956 * This function checks if the terminal has basic capabilities, just checking
957 * for a blacklist of stupid terminals, and later either calls the line
958 * editing function or uses dummy fgets() so that you will be able to type
959 * something even in the most desperate of the conditions. */
960char *linenoise(const char *prompt) {
961 char buf[LINENOISE_MAX_LINE];
962 int count;
963
964 if (isUnsupportedTerm()) {
965 size_t len;
966
967 printf("%s",prompt);
968 fflush(stdout);
969 if (fgets(buf,LINENOISE_MAX_LINE,stdin) == NULL) return NULL;
970 len = strlen(buf);
971 while(len && (buf[len-1] == '\n' || buf[len-1] == '\r')) {
972 len--;
973 buf[len] = '\0';
974 }
975 return strdup(buf);
976 } else {
977 count = linenoiseRaw(buf,LINENOISE_MAX_LINE,prompt);
978 if (count == -1) return NULL;
979 return strdup(buf);
980 }
981}

986 * exit() to avoid memory leaks are reported by valgrind & co. */

988 if (history) {
989 int j;
990

1003/* This is the API call to add a new entry in the linenoise history.
1004 * It uses a fixed array of char pointers that are shifted (memmoved)
1005 * when the history max length is reached in order to remove the older
1006 * entry and make room for the new one, so it is not exactly suitable for huge
1007 * histories, but will work well for a few hundred of entries.
1008 *
1009 * Using a circular buffer is smarter, but a bit more complex to handle. */
1010int linenoiseHistoryAdd(const char *line) {
1011 char *linecopy;
1012
1013 if (history_max_len == 0) return 0;
1014
1015 /* Initialization on first call. */
1016 if (history == NULL) {
1017 history = malloc(sizeof(char*)*history_max_len);
1018 if (history == NULL) return 0;
1019 memset(history,0,(sizeof(char*)*history_max_len));
1020 }
1021
1022 /* Don't add duplicated lines. */
1023 if (history_len && !strcmp(history[history_len-1], line)) return 0;
1024
1025 /* Add an heap allocated copy of the line in the history.
1026 * If we reached the max length, remove the older line. */
1027 linecopy = strdup(line);
1028 if (!linecopy) return 0;
1029 if (history_len == history_max_len) {
1030 free(history[0]);
1031 memmove(history,history+1,sizeof(char*)*(history_max_len-1));
1032 history_len--;
1033 }
1034 history[history_len] = linecopy;
1035 history_len++;
1036 return 1;
1037}

1039/* Set the maximum length for the history. This function can be called even
1040 * if there is already some history, the function will make sure to retain
1041 * just the latest 'len' elements if the new history length value is smaller
1042 * than the amount of items already inside the history. */
1043int linenoiseHistorySetMaxLen(int len) {
1044 char **new;
1045
1046 if (len < 1) return 0;
1047 if (history) {
1048 int tocopy = history_len;
1049
1050 new = malloc(sizeof(char*)*len);
1051 if (new == NULL) return 0;
1052
1053 /* If we can't copy everything, free the elements we'll not use. */
1054 if (len < tocopy) {
1055 int j;
1056
1057 for (j = 0; j < tocopy-len; j++) free(history[j]);
1058 tocopy = len;
1059 }
1060 memset(new,0,sizeof(char*)*len);
1061 memcpy(new,history+(history_len-tocopy), sizeof(char*)*tocopy);
1062 free(history);
1063 history = new;
1064 }
1065 history_max_len = len;
1066 if (history_len > history_max_len)
1067 history_len = history_max_len;
1068 return 1;
1069}

1071/* Save the history in the specified file. On success 0 is returned
1072 * otherwise -1 is returned. */
1073int linenoiseHistorySave(const char *filename) {
1074 FILE *fp = fopen(filename,"w");
1075 int j;
1076
1077 if (fp == NULL) return -1;
1078 for (j = 0; j < history_len; j++)
1079 fprintf(fp,"%s\n",history[j]);
1080 fclose(fp);
1081 return 0;
1082}

1084/* Load the history from the specified file. If the file does not exist
1085 * zero is returned and no operation is performed.
1086 *
1087 * If the file exists and the operation succeeded 0 is returned, otherwise
1088 * on error -1 is returned. */
1089int linenoiseHistoryLoad(const char *filename) {
1090 FILE *fp = fopen(filename,"r");
1091 char buf[LINENOISE_MAX_LINE];
1092
1093 if (fp == NULL) return -1;
1094
1095 while (fgets(buf,LINENOISE_MAX_LINE,fp) != NULL) {
1096 char *p;
1097
1098 p = strchr(buf,'\r');
1099 if (!p) p = strchr(buf,'\n');
1100 if (p) *p = '\0';
1101 linenoiseHistoryAdd(buf);
1102 }
1103 fclose(fp);
1104 return 0;
1105}
