31#ifndef __SDS_H
32#define __SDS_H
33
34#define SDS_MAX_PREALLOC (1024*1024)
35
36#include <sys/types.h>
37#include <stdarg.h>
38
39typedef char *sds;
40
41struct sdshdr {
42 unsigned int len;
43 unsigned int free;
44 char buf[];
45};
46
47static inline size_t sdslen(const sds s) {
48 struct sdshdr *sh = (void*)(s-(sizeof(struct sdshdr)));
49 return sh->len;
50}
51
52static inline size_t sdsavail(const sds s) {
53 struct sdshdr *sh = (void*)(s-(sizeof(struct sdshdr)));
54 return sh->free;
55}
