50/* Using dictEnableResize() / dictDisableResize() we make possible to
51 * enable/disable resizing of the hash table as needed. This is very important
52 * for Redis, as we use copy-on-write and don't want to move too much memory
53 * around when there is a child performing saving operations.
54 *
55 * Note that even when dict_can_resize is set to 0, not all resizes are
56 * prevented: a hash table is still allowed to grow if the ratio between
57 * the number of elements and the buckets > dict_force_resize_ratio. */
58static int dict_can_resize = 1;
59static unsigned int dict_force_resize_ratio = 5;

70/* Thomas Wang's 32 bit Mix Function */
71unsigned int dictIntHashFunction(unsigned int key)
72{
73 key += ~(key << 15);
74 key ^= (key >> 10);
75 key += (key << 3);
76 key ^= (key >> 6);
77 key += ~(key << 11);
78 key ^= (key >> 16);
79 return key;
80}

92/* MurmurHash2, by Austin Appleby
93 * Note - This code makes a few assumptions about how your machine behaves -
94 * 1. We can read a 4-byte value from any address without crashing
95 * 2. sizeof(int) == 4
96 *
97 * And it has a few limitations -
98 *
99 * 1. It will not work incrementally.
100 * 2. It will not produce the same results on little-endian and big-endian
101 * machines.
102 */
103unsigned int dictGenHashFunction(const void *key, int len) {
104 /* 'm' and 'r' are mixing constants generated offline.
105 They're not really 'magic', they just happen to work well. */
106 uint32_t seed = dict_hash_function_seed;
107 const uint32_t m = 0x5bd1e995;
108 const int r = 24;
109
110 /* Initialize the hash to a 'random' value */
111 uint32_t h = seed ^ len;
112
113 /* Mix 4 bytes at a time into the hash */
114 const unsigned char *data = (const unsigned char *)key;
115
116 while(len >= 4) {
117 uint32_t k = *(uint32_t*)data;
118
119 k *= m;
120 k ^= k >> r;
121 k *= m;
122
123 h *= m;
124 h ^= k;
125
126 data += 4;
127 len -= 4;
128 }
129
130 /* Handle the last few bytes of the input array */
131 switch(len) {
132 case 3: h ^= data[2] << 16;
133 case 2: h ^= data[1] << 8;
134 case 1: h ^= data[0]; h *= m;
135 };
136
137 /* Do a few final mixes of the hash to ensure the last few
138 * bytes are well-incorporated. */
139 h ^= h >> 13;
140 h *= m;
141 h ^= h >> 15;
142
143 return (unsigned int)h;
144}

146/* And a case insensitive hash function (based on djb hash) */
147unsigned int dictGenCaseHashFunction(const unsigned char *buf, int len) {
148 unsigned int hash = (unsigned int)dict_hash_function_seed;
149
150 while (len--)
151 hash = ((hash << 5) + hash) + (tolower(*buf++)); /* hash * 33 + c */
152 return hash;
153}

190/* Resize the table to the minimal size that contains all the elements,
191 * but with the invariant of a USED/BUCKETS ratio near to <= 1 */
192int dictResize(dict *d)
193{
194 int minimal;
195
196 if (!dict_can_resize || dictIsRehashing(d)) return DICT_ERR;
197 minimal = d->ht[0].used;
198 if (minimal < DICT_HT_INITIAL_SIZE)
199 minimal = DICT_HT_INITIAL_SIZE;
200 return dictExpand(d, minimal);
201}

203/* Expand or create the hash table */
204int dictExpand(dict *d, unsigned long size)
205{
206 dictht n; /* the new hash table */
207 unsigned long realsize = _dictNextPower(size);
208
209 /* the size is invalid if it is smaller than the number of
210 * elements already inside the hash table */
211 if (dictIsRehashing(d) || d->ht[0].used > size)
212 return DICT_ERR;
213
214 /* Rehashing to the same table size is not useful. */
215 if (realsize == d->ht[0].size) return DICT_ERR;
216
217 /* Allocate the new hash table and initialize all pointers to NULL */
218 n.size = realsize;
219 n.sizemask = realsize-1;
220 n.table = zcalloc(realsize*sizeof(dictEntry*));
221 n.used = 0;
222
223 /* Is this the first initialization? If so it's not really a rehashing
224 * we just set the first hash table so that it can accept keys. */
225 if (d->ht[0].table == NULL) {
226 d->ht[0] = n;
227 return DICT_OK;
228 }
229
230 /* Prepare a second hash table for incremental rehashing */
231 d->ht[1] = n;
232 d->rehashidx = 0;
233 return DICT_OK;
234}
235
236/* Performs N steps of incremental rehashing. Returns 1 if there are still
237 * keys to move from the old to the new hash table, otherwise 0 is returned.
238 *
239 * Note that a rehashing step consists in moving a bucket (that may have more
240 * than one key as we use chaining) from the old to the new hash table, however
241 * since part of the hash table may be composed of empty spaces, it is not
242 * guaranteed that this function will rehash even a single bucket, since it
243 * will visit at max N*10 empty buckets in total, otherwise the amount of
244 * work it does would be unbound and the function may block for a long time. */
245int dictRehash(dict *d, int n) {
246 int empty_visits = n*10; /* Max number of empty buckets to visit. */
247 if (!dictIsRehashing(d)) return 0;
248
249 while(n-- && d->ht[0].used != 0) {
250 dictEntry *de, *nextde;
251
252 /* Note that rehashidx can't overflow as we are sure there are more
253 * elements because ht[0].used != 0 */
254 assert(d->ht[0].size > (unsigned long)d->rehashidx);
255 while(d->ht[0].table[d->rehashidx] == NULL) {
256 d->rehashidx++;
257 if (--empty_visits == 0) return 1;
258 }
259 de = d->ht[0].table[d->rehashidx];
260 /* Move all the keys in this bucket from the old to the new hash HT */
261 while(de) {
262 unsigned int h;
263
264 nextde = de->next;
265 /* Get the index in the new hash table */
266 h = dictHashKey(d, de->key) & d->ht[1].sizemask;
267 de->next = d->ht[1].table[h];
268 d->ht[1].table[h] = de;
269 d->ht[0].used--;
270 d->ht[1].used++;
271 de = nextde;
272 }
273 d->ht[0].table[d->rehashidx] = NULL;
274 d->rehashidx++;
275 }
276
277 /* Check if we already rehashed the whole table... */
278 if (d->ht[0].used == 0) {
279 zfree(d->ht[0].table);
280 d->ht[0] = d->ht[1];
281 _dictReset(&d->ht[1]);
282 d->rehashidx = -1;
283 return 0;
284 }
285
286 /* More to rehash... */
287 return 1;
288}

290long long timeInMilliseconds(void) {
291 struct timeval tv;
292
293 gettimeofday(&tv,NULL);
294 return (((long long)tv.tv_sec)*1000)+(tv.tv_usec/1000);
295}

297/* Rehash for an amount of time between ms milliseconds and ms+1 milliseconds */
298int dictRehashMilliseconds(dict *d, int ms) {
299 long long start = timeInMilliseconds();
300 int rehashes = 0;
301
302 while(dictRehash(d,100)) {
303 rehashes += 100;
304 if (timeInMilliseconds()-start > ms) break;
305 }
306 return rehashes;
307}

309/* This function performs just a step of rehashing, and only if there are
310 * no safe iterators bound to our hash table. When we have iterators in the
311 * middle of a rehashing we can't mess with the two hash tables otherwise
312 * some element can be missed or duplicated.
313 *
314 * This function is called by common lookup or update operations in the
315 * dictionary so that the hash table automatically migrates from H1 to H2
316 * while it is actively used. */
317static void _dictRehashStep(dict *d) {
318 if (d->iterators == 0) dictRehash(d,1);
319}

321/* Add an element to the target hash table */
322int dictAdd(dict *d, void *key, void *val)
323{
324 dictEntry *entry = dictAddRaw(d,key);
325
326 if (!entry) return DICT_ERR;
327 dictSetVal(d, entry, val);
328 return DICT_OK;
329}

331/* Low level add. This function adds the entry but instead of setting
332 * a value returns the dictEntry structure to the user, that will make
333 * sure to fill the value field as he wishes.
334 *
335 * This function is also directly exposed to the user API to be called
336 * mainly in order to store non-pointers inside the hash value, example:
337 *
338 * entry = dictAddRaw(dict,mykey);
339 * if (entry != NULL) dictSetSignedIntegerVal(entry,1000);
340 *
341 * Return values:
342 *
343 * If key already exists NULL is returned.
344 * If key was added, the hash entry is returned to be manipulated by the caller.
345 */
346dictEntry *dictAddRaw(dict *d, void *key)
347{
348 int index;
349 dictEntry *entry;
350 dictht *ht;
351
352 if (dictIsRehashing(d)) _dictRehashStep(d);
353
354 /* Get the index of the new element, or -1 if
355 * the element already exists. */
356 if ((index = _dictKeyIndex(d, key)) == -1)
357 return NULL;
358
359 /* Allocate the memory and store the new entry */
360 ht = dictIsRehashing(d) ? &d->ht[1] : &d->ht[0];
361 entry = zmalloc(sizeof(*entry));
362 entry->next = ht->table[index];
363 ht->table[index] = entry;
364 ht->used++;
365
366 /* Set the hash entry fields. */
367 dictSetKey(d, entry, key);
368 return entry;
369}

371/* Add an element, discarding the old if the key already exists.
372 * Return 1 if the key was added from scratch, 0 if there was already an
373 * element with such key and dictReplace() just performed a value update
374 * operation. */
375int dictReplace(dict *d, void *key, void *val)
376{
377 dictEntry *entry, auxentry;
378
379 /* Try to add the element. If the key
380 * does not exists dictAdd will suceed. */
381 if (dictAdd(d, key, val) == DICT_OK)
382 return 1;
383 /* It already exists, get the entry */
384 entry = dictFind(d, key);
385 /* Set the new value and free the old one. Note that it is important
386 * to do that in this order, as the value may just be exactly the same
387 * as the previous one. In this context, think to reference counting,
388 * you want to increment (set), and then decrement (free), and not the
389 * reverse. */
390 auxentry = *entry;
391 dictSetVal(d, entry, val);
392 dictFreeVal(d, &auxentry);
393 return 0;
394}
395
396/* dictReplaceRaw() is simply a version of dictAddRaw() that always
397 * returns the hash entry of the specified key, even if the key already
398 * exists and can't be added (in that case the entry of the already
399 * existing key is returned.)
400 *
401 * See dictAddRaw() for more information. */
402dictEntry *dictReplaceRaw(dict *d, void *key) {
403 dictEntry *entry = dictFind(d,key);
404
405 return entry ? entry : dictAddRaw(d,key);
406}

408/* Search and remove an element */
409static int dictGenericDelete(dict *d, const void *key, int nofree)
410{
411 unsigned int h, idx;
412 dictEntry *he, *prevHe;
413 int table;
414
415 if (d->ht[0].size == 0) return DICT_ERR; /* d->ht[0].table is NULL */
416 if (dictIsRehashing(d)) _dictRehashStep(d);
417 h = dictHashKey(d, key);
418
419 for (table = 0; table <= 1; table++) {
420 idx = h & d->ht[table].sizemask;
421 he = d->ht[table].table[idx];
422 prevHe = NULL;
423 while(he) {
424 if (dictCompareKeys(d, key, he->key)) {
425 /* Unlink the element from the list */
426 if (prevHe)
427 prevHe->next = he->next;
428 else
429 d->ht[table].table[idx] = he->next;
430 if (!nofree) {
431 dictFreeKey(d, he);
432 dictFreeVal(d, he);
433 }
434 zfree(he);
435 d->ht[table].used--;
436 return DICT_OK;
437 }
438 prevHe = he;
439 he = he->next;
440 }
441 if (!dictIsRehashing(d)) break;
442 }
443 return DICT_ERR; /* not found */
444}

454/* Destroy an entire dictionary */
455int _dictClear(dict *d, dictht *ht, void(callback)(void *)) {
456 unsigned long i;
457
458 /* Free all the elements */
459 for (i = 0; i < ht->size && ht->used > 0; i++) {
460 dictEntry *he, *nextHe;
461
462 if (callback && (i & 65535) == 0) callback(d->privdata);
463
464 if ((he = ht->table[i]) == NULL) continue;
465 while(he) {
466 nextHe = he->next;
467 dictFreeKey(d, he);
468 dictFreeVal(d, he);
469 zfree(he);
470 ht->used--;
471 he = nextHe;
472 }
473 }
474 /* Free the table and the allocated cache structure */
475 zfree(ht->table);
476 /* Re-initialize the table */
477 _dictReset(ht);
478 return DICT_OK; /* never fails */
479}

489dictEntry *dictFind(dict *d, const void *key)
490{
491 dictEntry *he;
492 unsigned int h, idx, table;
493
494 if (d->ht[0].size == 0) return NULL; /* We don't have a table at all */
495 if (dictIsRehashing(d)) _dictRehashStep(d);
496 h = dictHashKey(d, key);
497 for (table = 0; table <= 1; table++) {
498 idx = h & d->ht[table].sizemask;
499 he = d->ht[table].table[idx];
500 while(he) {
501 if (dictCompareKeys(d, key, he->key))
502 return he;
503 he = he->next;
504 }
505 if (!dictIsRehashing(d)) return NULL;
506 }
507 return NULL;
508}

517/* A fingerprint is a 64 bit number that represents the state of the dictionary
518 * at a given time, it's just a few dict properties xored together.
519 * When an unsafe iterator is initialized, we get the dict fingerprint, and check
520 * the fingerprint again when the iterator is released.
521 * If the two fingerprints are different it means that the user of the iterator
522 * performed forbidden operations against the dictionary while iterating. */
523long long dictFingerprint(dict *d) {
524 long long integers[6], hash = 0;
525 int j;
526
527 integers[0] = (long) d->ht[0].table;
528 integers[1] = d->ht[0].size;
529 integers[2] = d->ht[0].used;
530 integers[3] = (long) d->ht[1].table;
531 integers[4] = d->ht[1].size;
532 integers[5] = d->ht[1].used;
533
534 /* We hash N integers by summing every successive integer with the integer
535 * hashing of the previous sum. Basically:
536 *
537 * Result = hash(hash(hash(int1)+int2)+int3) ...
538 *
539 * This way the same set of integers in a different order will (likely) hash
540 * to a different number. */
541 for (j = 0; j < 6; j++) {
542 hash += integers[j];
543 /* For the hashing step we use Tomas Wang's 64 bit integer hash. */
544 hash = (~hash) + (hash << 21); // hash = (hash << 21) - hash - 1;
545 hash = hash ^ (hash >> 24);
546 hash = (hash + (hash << 3)) + (hash << 8); // hash * 265
547 hash = hash ^ (hash >> 14);
548 hash = (hash + (hash << 2)) + (hash << 4); // hash * 21
549 hash = hash ^ (hash >> 28);
550 hash = hash + (hash << 31);
551 }
552 return hash;
553}

575dictEntry *dictNext(dictIterator *iter)
576{
577 while (1) {
578 if (iter->entry == NULL) {
579 dictht *ht = &iter->d->ht[iter->table];
580 if (iter->index == -1 && iter->table == 0) {
581 if (iter->safe)
582 iter->d->iterators++;
583 else
584 iter->fingerprint = dictFingerprint(iter->d);
585 }
586 iter->index++;
587 if (iter->index >= (long) ht->size) {
588 if (dictIsRehashing(iter->d) && iter->table == 0) {
589 iter->table++;
590 iter->index = 0;
591 ht = &iter->d->ht[1];
592 } else {
593 break;
594 }
595 }
596 iter->entry = ht->table[iter->index];
597 } else {
598 iter->entry = iter->nextEntry;
599 }
600 if (iter->entry) {
601 /* We need to save the 'next' here, the iterator user
602 * may delete the entry we are returning. */
603 iter->nextEntry = iter->entry->next;
604 return iter->entry;
605 }
606 }
607 return NULL;
608}

610void dictReleaseIterator(dictIterator *iter)
611{
612 if (!(iter->index == -1 && iter->table == 0)) {
613 if (iter->safe)
614 iter->d->iterators--;
615 else
616 assert(iter->fingerprint == dictFingerprint(iter->d));
617 }
618 zfree(iter);
619}

621/* Return a random entry from the hash table. Useful to
622 * implement randomized algorithms */
623dictEntry *dictGetRandomKey(dict *d)
624{
625 dictEntry *he, *orighe;
626 unsigned int h;
627 int listlen, listele;
628
629 if (dictSize(d) == 0) return NULL;
630 if (dictIsRehashing(d)) _dictRehashStep(d);
631 if (dictIsRehashing(d)) {
632 do {
633 /* We are sure there are no elements in indexes from 0
634 * to rehashidx-1 */
635 h = d->rehashidx + (random() % (d->ht[0].size +
636 d->ht[1].size -
637 d->rehashidx));
638 he = (h >= d->ht[0].size) ? d->ht[1].table[h - d->ht[0].size] :
639 d->ht[0].table[h];
640 } while(he == NULL);
641 } else {
642 do {
643 h = random() & d->ht[0].sizemask;
644 he = d->ht[0].table[h];
645 } while(he == NULL);
646 }
647
648 /* Now we found a non empty bucket, but it is a linked
649 * list and we need to get a random element from the list.
650 * The only sane way to do so is counting the elements and
651 * select a random index. */
652 listlen = 0;
653 orighe = he;
654 while(he) {
655 he = he->next;
656 listlen++;
657 }
658 listele = random() % listlen;
659 he = orighe;
660 while(listele--) he = he->next;
661 return he;
662}

664/* This function samples the dictionary to return a few keys from random
665 * locations.
666 *
667 * It does not guarantee to return all the keys specified in 'count', nor
668 * it does guarantee to return non-duplicated elements, however it will make
669 * some effort to do both things.
670 *
671 * Returned pointers to hash table entries are stored into 'des' that
672 * points to an array of dictEntry pointers. The array must have room for
673 * at least 'count' elements, that is the argument we pass to the function
674 * to tell how many random elements we need.
675 *
676 * The function returns the number of items stored into 'des', that may
677 * be less than 'count' if the hash table has less than 'count' elements
678 * inside, or if not enough elements were found in a reasonable amount of
679 * steps.
680 *
681 * Note that this function is not suitable when you need a good distribution
682 * of the returned items, but only when you need to "sample" a given number
683 * of continuous elements to run some kind of algorithm or to produce
684 * statistics. However the function is much faster than dictGetRandomKey()
685 * at producing N elements. */
686unsigned int dictGetSomeKeys(dict *d, dictEntry **des, unsigned int count) {
687 unsigned int j; /* internal hash table id, 0 or 1. */
688 unsigned int tables; /* 1 or 2 tables? */
689 unsigned int stored = 0, maxsizemask;
690 unsigned int maxsteps;
691
692 if (dictSize(d) < count) count = dictSize(d);
693 maxsteps = count*10;
694
695 /* Try to do a rehashing work proportional to 'count'. */
696 for (j = 0; j < count; j++) {
697 if (dictIsRehashing(d))
698 _dictRehashStep(d);
699 else
700 break;
701 }
702
703 tables = dictIsRehashing(d) ? 2 : 1;
704 maxsizemask = d->ht[0].sizemask;
705 if (tables > 1 && maxsizemask < d->ht[1].sizemask)
706 maxsizemask = d->ht[1].sizemask;
707
708 /* Pick a random point inside the larger table. */
709 unsigned int i = random() & maxsizemask;
710 unsigned int emptylen = 0; /* Continuous empty entries so far. */
711 while(stored < count && maxsteps--) {
712 for (j = 0; j < tables; j++) {
713 /* Invariant of the dict.c rehashing: up to the indexes already
714 * visited in ht[0] during the rehashing, there are no populated
715 * buckets, so we can skip ht[0] for indexes between 0 and idx-1. */
716 if (tables == 2 && j == 0 && i < (unsigned int) d->rehashidx) {
717 /* Moreover, if we are currently out of range in the second
718 * table, there will be no elements in both tables up to
719 * the current rehashing index, so we jump if possible.
720 * (this happens when going from big to small table). */
721 if (i >= d->ht[1].size) i = d->rehashidx;
722 continue;
723 }
724 if (i >= d->ht[j].size) continue; /* Out of range for this table. */
725 dictEntry *he = d->ht[j].table[i];
726
727 /* Count contiguous empty buckets, and jump to other
728 * locations if they reach 'count' (with a minimum of 5). */
729 if (he == NULL) {
730 emptylen++;
731 if (emptylen >= 5 && emptylen > count) {
732 i = random() & maxsizemask;
733 emptylen = 0;
734 }
735 } else {
736 emptylen = 0;
737 while (he) {
738 /* Collect all the elements of the buckets found non
739 * empty while iterating. */
740 *des = he;
741 des++;
742 he = he->next;
743 stored++;
744 if (stored == count) return stored;
745 }
746 }
747 }
748 i = (i+1) & maxsizemask;
749 }
750 return stored;
751}

753/* Function to reverse bits. Algorithm from:
754 * http://graphics.stanford.edu/~seander/bithacks.html#ReverseParallel */
755static unsigned long rev(unsigned long v) {
756 unsigned long s = 8 * sizeof(v); // bit size; must be power of 2
757 unsigned long mask = ~0;
758 while ((s >>= 1) > 0) {
759 mask ^= (mask << s);
760 v = ((v >> s) & mask) | ((v << s) & ~mask);
761 }
762 return v;
763}

765/* dictScan() is used to iterate over the elements of a dictionary.
766 *
767 * Iterating works the following way:
768 *
769 * 1) Initially you call the function using a cursor (v) value of 0.
770 * 2) The function performs one step of the iteration, and returns the
771 * new cursor value you must use in the next call.
772 * 3) When the returned cursor is 0, the iteration is complete.
773 *
774 * The function guarantees all elements present in the
775 * dictionary get returned between the start and end of the iteration.
776 * However it is possible some elements get returned multiple times.
777 *
778 * For every element returned, the callback argument 'fn' is
779 * called with 'privdata' as first argument and the dictionary entry
780 * 'de' as second argument.
781 *
782 * HOW IT WORKS.
783 *
784 * The iteration algorithm was designed by Pieter Noordhuis.
785 * The main idea is to increment a cursor starting from the higher order
786 * bits. That is, instead of incrementing the cursor normally, the bits
787 * of the cursor are reversed, then the cursor is incremented, and finally
788 * the bits are reversed again.
789 *
790 * This strategy is needed because the hash table may be resized between
791 * iteration calls.
792 *
793 * dict.c hash tables are always power of two in size, and they
794 * use chaining, so the position of an element in a given table is given
795 * by computing the bitwise AND between Hash(key) and SIZE-1
796 * (where SIZE-1 is always the mask that is equivalent to taking the rest
797 * of the division between the Hash of the key and SIZE).
798 *
799 * For example if the current hash table size is 16, the mask is
800 * (in binary) 1111. The position of a key in the hash table will always be
801 * the last four bits of the hash output, and so forth.
802 *
803 * WHAT HAPPENS IF THE TABLE CHANGES IN SIZE?
804 *
805 * If the hash table grows, elements can go anywhere in one multiple of
806 * the old bucket: for example let's say we already iterated with
807 * a 4 bit cursor 1100 (the mask is 1111 because hash table size = 16).
808 *
809 * If the hash table will be resized to 64 elements, then the new mask will
810 * be 111111. The new buckets you obtain by substituting in ??1100
811 * with either 0 or 1 can be targeted only by keys we already visited
812 * when scanning the bucket 1100 in the smaller hash table.
813 *
814 * By iterating the higher bits first, because of the inverted counter, the
815 * cursor does not need to restart if the table size gets bigger. It will
816 * continue iterating using cursors without '1100' at the end, and also
817 * without any other combination of the final 4 bits already explored.
818 *
819 * Similarly when the table size shrinks over time, for example going from
820 * 16 to 8, if a combination of the lower three bits (the mask for size 8
821 * is 111) were already completely explored, it would not be visited again
822 * because we are sure we tried, for example, both 0111 and 1111 (all the
823 * variations of the higher bit) so we don't need to test it again.
824 *
825 * WAIT... YOU HAVE *TWO* TABLES DURING REHASHING!
826 *
827 * Yes, this is true, but we always iterate the smaller table first, then
828 * we test all the expansions of the current cursor into the larger
829 * table. For example if the current cursor is 101 and we also have a
830 * larger table of size 16, we also test (0)101 and (1)101 inside the larger
831 * table. This reduces the problem back to having only one table, where
832 * the larger one, if it exists, is just an expansion of the smaller one.
833 *
834 * LIMITATIONS
835 *
836 * This iterator is completely stateless, and this is a huge advantage,
837 * including no additional memory used.
838 *
839 * The disadvantages resulting from this design are:
840 *
841 * 1) It is possible we return elements more than once. However this is usually
842 * easy to deal with in the application level.
843 * 2) The iterator must return multiple elements per call, as it needs to always
844 * return all the keys chained in a given bucket, and all the expansions, so
845 * we are sure we don't miss keys moving during rehashing.
846 * 3) The reverse cursor is somewhat hard to understand at first, but this
847 * comment is supposed to help.
848 */
849unsigned long dictScan(dict *d,
850 unsigned long v,
851 dictScanFunction *fn,
852 void *privdata)
853{
854 dictht *t0, *t1;
855 const dictEntry *de;
856 unsigned long m0, m1;
857
858 if (dictSize(d) == 0) return 0;
859
860 if (!dictIsRehashing(d)) {
861 t0 = &(d->ht[0]);
862 m0 = t0->sizemask;
863
864 /* Emit entries at cursor */
865 de = t0->table[v & m0];
866 while (de) {
867 fn(privdata, de);
868 de = de->next;
869 }
870
871 } else {
872 t0 = &d->ht[0];
873 t1 = &d->ht[1];
874
875 /* Make sure t0 is the smaller and t1 is the bigger table */
876 if (t0->size > t1->size) {
877 t0 = &d->ht[1];
878 t1 = &d->ht[0];
879 }
880
881 m0 = t0->sizemask;
882 m1 = t1->sizemask;
883
884 /* Emit entries at cursor */
885 de = t0->table[v & m0];
886 while (de) {
887 fn(privdata, de);
888 de = de->next;
889 }
890
891 /* Iterate over indices in larger table that are the expansion
892 * of the index pointed to by the cursor in the smaller table */
893 do {
894 /* Emit entries at cursor */
895 de = t1->table[v & m1];
896 while (de) {
897 fn(privdata, de);
898 de = de->next;
899 }
900
901 /* Increment bits not covered by the smaller mask */
902 v = (((v | m0) + 1) & ~m0) | (v & m0);
903
904 /* Continue while bits covered by mask difference is non-zero */
905 } while (v & (m0 ^ m1));
906 }
907
908 /* Set unmasked bits so incrementing the reversed cursor
909 * operates on the masked bits of the smaller table */
910 v |= ~m0;
911
912 /* Increment the reverse cursor */
913 v = rev(v);
914 v++;
915 v = rev(v);
916
917 return v;
918}

922/* Expand the hash table if needed */
923static int _dictExpandIfNeeded(dict *d)
924{
925 /* Incremental rehashing already in progress. Return. */
926 if (dictIsRehashing(d)) return DICT_OK;
927
928 /* If the hash table is empty expand it to the initial size. */
929 if (d->ht[0].size == 0) return dictExpand(d, DICT_HT_INITIAL_SIZE);
930
931 /* If we reached the 1:1 ratio, and we are allowed to resize the hash
932 * table (global setting) or we should avoid it but the ratio between
933 * elements/buckets is over the "safe" threshold, we resize doubling
934 * the number of buckets. */
935 if (d->ht[0].used >= d->ht[0].size &&
936 (dict_can_resize ||
937 d->ht[0].used/d->ht[0].size > dict_force_resize_ratio))
938 {
939 return dictExpand(d, d->ht[0].used*2);
940 }
941 return DICT_OK;
942}

944/* Our hash table capability is a power of two */
945static unsigned long _dictNextPower(unsigned long size)
946{
947 unsigned long i = DICT_HT_INITIAL_SIZE;
948
949 if (size >= LONG_MAX) return LONG_MAX;
950 while(1) {
951 if (i >= size)
952 return i;
953 i *= 2;
954 }

957/* Returns the index of a free slot that can be populated with
958 * a hash entry for the given 'key'.
959 * If the key already exists, -1 is returned.
960 *
961 * Note that if we are in the process of rehashing the hash table, the
962 * index is always returned in the context of the second (new) hash table. */
963static int _dictKeyIndex(dict *d, const void *key)
964{
965 unsigned int h, idx, table;
966 dictEntry *he;
967
968 /* Expand the hash table if needed */
969 if (_dictExpandIfNeeded(d) == DICT_ERR)
970 return -1;
971 /* Compute the key hash value */
972 h = dictHashKey(d, key);
973 for (table = 0; table <= 1; table++) {
974 idx = h & d->ht[table].sizemask;
975 /* Search if this slot does not already contain the given key */
976 he = d->ht[table].table[idx];
977 while(he) {
978 if (dictCompareKeys(d, key, he->key))
979 return -1;
980 he = he->next;
981 }
982 if (!dictIsRehashing(d)) break;
983 }
984 return idx;
985}
