39robj *createObject(int type, void *ptr) {
40 robj *o = zmalloc(sizeof(*o));
41 o->type = type;
42 o->encoding = REDIS_ENCODING_RAW;
43 o->ptr = ptr;
44 o->refcount = 1;
45
46 /* Set the LRU to the current lruclock (minutes resolution). */
47 o->lru = LRU_CLOCK();
48 return o;
49}

51/* Create a string object with encoding REDIS_ENCODING_RAW, that is a plain
52 * string object where o->ptr points to a proper sds string. */
53robj *createRawStringObject(char *ptr, size_t len) {
54 return createObject(REDIS_STRING,sdsnewlen(ptr,len));
55}
56
57/* Create a string object with encoding REDIS_ENCODING_EMBSTR, that is
58 * an object where the sds string is actually an unmodifiable string
59 * allocated in the same chunk as the object itself. */
60robj *createEmbeddedStringObject(char *ptr, size_t len) {
61 robj *o = zmalloc(sizeof(robj)+sizeof(struct sdshdr)+len+1);
62 struct sdshdr *sh = (void*)(o+1);
63
64 o->type = REDIS_STRING;
65 o->encoding = REDIS_ENCODING_EMBSTR;
66 o->ptr = sh+1;
67 o->refcount = 1;
68 o->lru = LRU_CLOCK();
69
70 sh->len = len;
71 sh->free = 0;
72 if (ptr) {
73 memcpy(sh->buf,ptr,len);
74 sh->buf[len] = '\0';
75 } else {
76 memset(sh->buf,0,len+1);
77 }
78 return o;
79}

81/* Create a string object with EMBSTR encoding if it is smaller than
82 * REIDS_ENCODING_EMBSTR_SIZE_LIMIT, otherwise the RAW encoding is
83 * used.
84 *
85 * The current limit of 39 is chosen so that the biggest string object
86 * we allocate as EMBSTR will still fit into the 64 byte arena of jemalloc. */
87#define REDIS_ENCODING_EMBSTR_SIZE_LIMIT 39
88robj *createStringObject(char *ptr, size_t len) {
89 if (len <= REDIS_ENCODING_EMBSTR_SIZE_LIMIT)
90 return createEmbeddedStringObject(ptr,len);
91 else
92 return createRawStringObject(ptr,len);
93}

95robj *createStringObjectFromLongLong(long long value) {
96 robj *o;
97 if (value >= 0 && value < REDIS_SHARED_INTEGERS) {
98 incrRefCount(shared.integers[value]);
99 o = shared.integers[value];
100 } else {
101 if (value >= LONG_MIN && value <= LONG_MAX) {
102 o = createObject(REDIS_STRING, NULL);
103 o->encoding = REDIS_ENCODING_INT;
104 o->ptr = (void*)((long)value);
105 } else {
106 o = createObject(REDIS_STRING,sdsfromlonglong(value));
107 }
108 }
109 return o;
110}

112/* Create a string object from a long double. If humanfriendly is non-zero
113 * it does not use exponential format and trims trailing zeroes at the end,
114 * however this results in loss of precision. Otherwise exp format is used
115 * and the output of snprintf() is not modified.
116 *
117 * The 'humanfriendly' option is used for INCRBYFLOAT and HINCRBYFLOAT. */
118robj *createStringObjectFromLongDouble(long double value, int humanfriendly) {
119 char buf[256];
120 int len;
121
122 if (isinf(value)) {
123 /* Libc in odd systems (Hi Solaris!) will format infinite in a
124 * different way, so better to handle it in an explicit way. */
125 if (value > 0) {
126 memcpy(buf,"inf",3);
127 len = 3;
128 } else {
129 memcpy(buf,"-inf",4);
130 len = 4;
131 }
132 } else if (humanfriendly) {
133 /* We use 17 digits precision since with 128 bit floats that precision
134 * after rounding is able to represent most small decimal numbers in a
135 * way that is "non surprising" for the user (that is, most small
136 * decimal numbers will be represented in a way that when converted
137 * back into a string are exactly the same as what the user typed.) */
138 len = snprintf(buf,sizeof(buf),"%.17Lf", value);
139 /* Now remove trailing zeroes after the '.' */
140 if (strchr(buf,'.') != NULL) {
141 char *p = buf+len-1;
142 while(*p == '0') {
143 p--;
144 len--;
145 }
146 if (*p == '.') len--;
147 }
148 } else {
149 len = snprintf(buf,sizeof(buf),"%.17Lg", value);
150 }
151 return createStringObject(buf,len);
152}

154/* Duplicate a string object, with the guarantee that the returned object
155 * has the same encoding as the original one.
156 *
157 * This function also guarantees that duplicating a small integere object
158 * (or a string object that contains a representation of a small integer)
159 * will always result in a fresh object that is unshared (refcount == 1).
160 *
161 * The resulting object always has refcount set to 1. */
162robj *dupStringObject(robj *o) {
163 robj *d;
164
165 redisAssert(o->type == REDIS_STRING);
166
167 switch(o->encoding) {
168 case REDIS_ENCODING_RAW:
169 return createRawStringObject(o->ptr,sdslen(o->ptr));
170 case REDIS_ENCODING_EMBSTR:
171 return createEmbeddedStringObject(o->ptr,sdslen(o->ptr));
172 case REDIS_ENCODING_INT:
173 d = createObject(REDIS_STRING, NULL);
174 d->encoding = REDIS_ENCODING_INT;
175 d->ptr = o->ptr;
176 return d;
177 default:
178 redisPanic("Wrong encoding.");
179 break;
180 }
181}

300void incrRefCount(robj *o) {
301 o->refcount++;
302}
303
304void decrRefCount(robj *o) {
305 if (o->refcount <= 0) redisPanic("decrRefCount against refcount <= 0");
306 if (o->refcount == 1) {
307 switch(o->type) {
308 case REDIS_STRING: freeStringObject(o); break;
309 case REDIS_LIST: freeListObject(o); break;
310 case REDIS_SET: freeSetObject(o); break;
311 case REDIS_ZSET: freeZsetObject(o); break;
312 case REDIS_HASH: freeHashObject(o); break;
313 default: redisPanic("Unknown object type"); break;
314 }
315 zfree(o);
316 } else {
317 o->refcount--;
318 }
319}

321/* This variant of decrRefCount() gets its argument as void, and is useful
322 * as free method in data structures that expect a 'void free_object(void*)'
323 * prototype for the free method. */
324void decrRefCountVoid(void *o) {
325 decrRefCount(o);

328/* This function set the ref count to zero without freeing the object.
329 * It is useful in order to pass a new object to functions incrementing
330 * the ref count of the received object. Example:
331 *
332 * functionThatWillIncrementRefCount(resetRefCount(CreateObject(...)));
333 *
334 * Otherwise you need to resort to the less elegant pattern:
335 *
336 * *obj = createObject(...);
337 * functionThatWillIncrementRefCount(obj);
338 * decrRefCount(obj);
339 */
340robj *resetRefCount(robj *obj) {
341 obj->refcount = 0;
342 return obj;
343}

363/* Try to encode a string object in order to save space */
364robj *tryObjectEncoding(robj *o) {
365 long value;
366 sds s = o->ptr;
367 size_t len;
368
369 /* Make sure this is a string object, the only type we encode
370 * in this function. Other types use encoded memory efficient
371 * representations but are handled by the commands implementing
372 * the type. */
373 redisAssertWithInfo(NULL,o,o->type == REDIS_STRING);
374
375 /* We try some specialized encoding only for objects that are
376 * RAW or EMBSTR encoded, in other words objects that are still
377 * in represented by an actually array of chars. */
378 if (!sdsEncodedObject(o)) return o;
379
380 /* It's not safe to encode shared objects: shared objects can be shared
381 * everywhere in the "object space" of Redis and may end in places where
382 * they are not handled. We handle them only as values in the keyspace. */
383 if (o->refcount > 1) return o;
384
385 /* Check if we can represent this string as a long integer.
386 * Note that we are sure that a string larger than 21 chars is not
387 * representable as a 32 nor 64 bit integer. */
388 len = sdslen(s);
389 if (len <= 21 && string2l(s,len,&value)) {
390 /* This object is encodable as a long. Try to use a shared object.
391 * Note that we avoid using shared integers when maxmemory is used
392 * because every object needs to have a private LRU field for the LRU
393 * algorithm to work well. */
394 if ((server.maxmemory == 0 ||
395 (server.maxmemory_policy != REDIS_MAXMEMORY_VOLATILE_LRU &&
396 server.maxmemory_policy != REDIS_MAXMEMORY_ALLKEYS_LRU)) &&
397 value >= 0 &&
398 value < REDIS_SHARED_INTEGERS)
399 {
400 decrRefCount(o);
401 incrRefCount(shared.integers[value]);
402 return shared.integers[value];
403 } else {
404 if (o->encoding == REDIS_ENCODING_RAW) sdsfree(o->ptr);
405 o->encoding = REDIS_ENCODING_INT;
406 o->ptr = (void*) value;
407 return o;
408 }
409 }
410
411 /* If the string is small and is still RAW encoded,
412 * try the EMBSTR encoding which is more efficient.
413 * In this representation the object and the SDS string are allocated
414 * in the same chunk of memory to save space and cache misses. */
415 if (len <= REDIS_ENCODING_EMBSTR_SIZE_LIMIT) {
416 robj *emb;
417
418 if (o->encoding == REDIS_ENCODING_EMBSTR) return o;
419 emb = createEmbeddedStringObject(s,sdslen(s));
420 decrRefCount(o);
421 return emb;
422 }
423
424 /* We can't encode the object...
425 *
426 * Do the last try, and at least optimize the SDS string inside
427 * the string object to require little space, in case there
428 * is more than 10% of free space at the end of the SDS string.
429 *
430 * We do that only for relatively large strings as this branch
431 * is only entered if the length of the string is greater than
432 * REDIS_ENCODING_EMBSTR_SIZE_LIMIT. */
433 if (o->encoding == REDIS_ENCODING_RAW &&
434 sdsavail(s) > len/10)
435 {
436 o->ptr = sdsRemoveFreeSpace(o->ptr);
437 }
438
439 /* Return the original object. */
440 return o;
441}

443/* Get a decoded version of an encoded object (returned as a new object).
444 * If the object is already raw-encoded just increment the ref count. */
445robj *getDecodedObject(robj *o) {
446 robj *dec;
447
448 if (sdsEncodedObject(o)) {
449 incrRefCount(o);
450 return o;
451 }
452 if (o->type == REDIS_STRING && o->encoding == REDIS_ENCODING_INT) {
453 char buf[32];
454
455 ll2string(buf,32,(long)o->ptr);
456 dec = createStringObject(buf,strlen(buf));
457 return dec;
458 } else {
459 redisPanic("Unknown encoding type");
460 }
461}

463/* Compare two string objects via strcmp() or strcoll() depending on flags.
464 * Note that the objects may be integer-encoded. In such a case we
465 * use ll2string() to get a string representation of the numbers on the stack
466 * and compare the strings, it's much faster than calling getDecodedObject().
467 *
468 * Important note: when REDIS_COMPARE_BINARY is used a binary-safe comparison
469 * is used. */
470
471#define REDIS_COMPARE_BINARY (1<<0)
472#define REDIS_COMPARE_COLL (1<<1)
473
474int compareStringObjectsWithFlags(robj *a, robj *b, int flags) {
475 redisAssertWithInfo(NULL,a,a->type == REDIS_STRING && b->type == REDIS_STRING);
476 char bufa[128], bufb[128], *astr, *bstr;
477 size_t alen, blen, minlen;
478
479 if (a == b) return 0;
480 if (sdsEncodedObject(a)) {
481 astr = a->ptr;
482 alen = sdslen(astr);
483 } else {
484 alen = ll2string(bufa,sizeof(bufa),(long) a->ptr);
485 astr = bufa;
486 }
487 if (sdsEncodedObject(b)) {
488 bstr = b->ptr;
489 blen = sdslen(bstr);
490 } else {
491 blen = ll2string(bufb,sizeof(bufb),(long) b->ptr);
492 bstr = bufb;
493 }
494 if (flags & REDIS_COMPARE_COLL) {
495 return strcoll(astr,bstr);
496 } else {
497 int cmp;
498
499 minlen = (alen < blen) ? alen : blen;
500 cmp = memcmp(astr,bstr,minlen);
501 if (cmp == 0) return alen-blen;
502 return cmp;
503 }
504}

506/* Wrapper for compareStringObjectsWithFlags() using binary comparison. */
507int compareStringObjects(robj *a, robj *b) {
508 return compareStringObjectsWithFlags(a,b,REDIS_COMPARE_BINARY);
509}
510
511/* Wrapper for compareStringObjectsWithFlags() using collation. */
512int collateStringObjects(robj *a, robj *b) {
513 return compareStringObjectsWithFlags(a,b,REDIS_COMPARE_COLL);
514}

516/* Equal string objects return 1 if the two objects are the same from the
517 * point of view of a string comparison, otherwise 0 is returned. Note that
518 * this function is faster then checking for (compareStringObject(a,b) == 0)
519 * because it can perform some more optimization. */
520int equalStringObjects(robj *a, robj *b) {
521 if (a->encoding == REDIS_ENCODING_INT &&
522 b->encoding == REDIS_ENCODING_INT){
523 /* If both strings are integer encoded just check if the stored
524 * long is the same. */
525 return a->ptr == b->ptr;
526 } else {
527 return compareStringObjects(a,b) == 0;
528 }
529}

542int getDoubleFromObject(robj *o, double *target) {
543 double value;
544 char *eptr;
545
546 if (o == NULL) {
547 value = 0;
548 } else {
549 redisAssertWithInfo(NULL,o,o->type == REDIS_STRING);
550 if (sdsEncodedObject(o)) {
551 errno = 0;
552 value = strtod(o->ptr, &eptr);
553 if (isspace(((char*)o->ptr)[0]) ||
554 eptr[0] != '\0' ||
555 (errno == ERANGE &&
556 (value == HUGE_VAL || value == -HUGE_VAL || value == 0)) ||
557 errno == EINVAL ||
558 isnan(value))
559 return REDIS_ERR;
560 } else if (o->encoding == REDIS_ENCODING_INT) {
561 value = (long)o->ptr;
562 } else {
563 redisPanic("Unknown string encoding");
564 }
565 }
566 *target = value;
567 return REDIS_OK;
568}

690/* Given an object returns the min number of milliseconds the object was never
691 * requested, using an approximated LRU algorithm. */
692unsigned long long estimateObjectIdleTime(robj *o) {
693 unsigned long long lruclock = LRU_CLOCK();
694 if (lruclock >= o->lru) {
695 return (lruclock - o->lru) * REDIS_LRU_CLOCK_RESOLUTION;
696 } else {
697 return (lruclock + (REDIS_LRU_CLOCK_MAX - o->lru)) *
698 REDIS_LRU_CLOCK_RESOLUTION;
699 }
700}

718/* Object command allows to inspect the internals of an Redis Object.
719 * Usage: OBJECT <refcount|encoding|idletime> <key> */
720void objectCommand(redisClient *c) {
721 robj *o;
722
723 if (!strcasecmp(c->argv[1]->ptr,"refcount") && c->argc == 3) {
724 if ((o = objectCommandLookupOrReply(c,c->argv[2],shared.nullbulk))
725 == NULL) return;
726 addReplyLongLong(c,o->refcount);
727 } else if (!strcasecmp(c->argv[1]->ptr,"encoding") && c->argc == 3) {
728 if ((o = objectCommandLookupOrReply(c,c->argv[2],shared.nullbulk))
729 == NULL) return;
730 addReplyBulkCString(c,strEncoding(o->encoding));
731 } else if (!strcasecmp(c->argv[1]->ptr,"idletime") && c->argc == 3) {
732 if ((o = objectCommandLookupOrReply(c,c->argv[2],shared.nullbulk))
733 == NULL) return;
734 addReplyLongLong(c,estimateObjectIdleTime(o)/1000);
735 } else {
736 addReplyError(c,"Syntax error. Try OBJECT (refcount|encoding|idletime)");
737 }
738}
