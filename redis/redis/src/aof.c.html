45/* ----------------------------------------------------------------------------
46 * AOF rewrite buffer implementation.
47 *
48 * The following code implement a simple buffer used in order to accumulate
49 * changes while the background process is rewriting the AOF file.
50 *
51 * We only need to append, but can't just use realloc with a large block
52 * because 'huge' reallocs are not always handled as one could expect
53 * (via remapping of pages at OS level) but may involve copying data.
54 *
55 * For this reason we use a list of blocks, every block is
56 * AOF_RW_BUF_BLOCK_SIZE bytes.
57 * ------------------------------------------------------------------------- */
58
59#define AOF_RW_BUF_BLOCK_SIZE (1024*1024*10) /* 10 MB per block */
60
61typedef struct aofrwblock {
62 unsigned long used, free;
63 char buf[AOF_RW_BUF_BLOCK_SIZE];
64} aofrwblock;
65
66/* This function free the old AOF rewrite buffer if needed, and initialize
67 * a fresh new one. It tests for server.aof_rewrite_buf_blocks equal to NULL
68 * so can be used for the first initialization as well. */
69void aofRewriteBufferReset(void) {
70 if (server.aof_rewrite_buf_blocks)
71 listRelease(server.aof_rewrite_buf_blocks);
72
73 server.aof_rewrite_buf_blocks = listCreate();
74 listSetFreeMethod(server.aof_rewrite_buf_blocks,zfree);
75}

77/* Return the current size of the AOF rewrite buffer. */
78unsigned long aofRewriteBufferSize(void) {
79 listNode *ln;
80 listIter li;
81 unsigned long size = 0;
82
83 listRewind(server.aof_rewrite_buf_blocks,&li);
84 while((ln = listNext(&li))) {
85 aofrwblock *block = listNodeValue(ln);
86 size += block->used;
87 }
88 return size;
89}

91/* Event handler used to send data to the child process doing the AOF
92 * rewrite. We send pieces of our AOF differences buffer so that the final
93 * write when the child finishes the rewrite will be small. */
94void aofChildWriteDiffData(aeEventLoop *el, int fd, void *privdata, int mask) {
95 listNode *ln;
96 aofrwblock *block;
97 ssize_t nwritten;
98 REDIS_NOTUSED(el);
99 REDIS_NOTUSED(fd);
100 REDIS_NOTUSED(privdata);
101 REDIS_NOTUSED(mask);
102
103 while(1) {
104 ln = listFirst(server.aof_rewrite_buf_blocks);
105 block = ln ? ln->value : NULL;
106 if (server.aof_stop_sending_diff || !block) {
107 aeDeleteFileEvent(server.el,server.aof_pipe_write_data_to_child,
108 AE_WRITABLE);
109 return;
110 }
111 if (block->used > 0) {
112 nwritten = write(server.aof_pipe_write_data_to_child,
113 block->buf,block->used);
114 if (nwritten <= 0) return;
115 memmove(block->buf,block->buf+nwritten,block->used-nwritten);
116 block->used -= nwritten;
117 }
118 if (block->used == 0) listDelNode(server.aof_rewrite_buf_blocks,ln);
119 }
120}

122/* Append data to the AOF rewrite buffer, allocating new blocks if needed. */
123void aofRewriteBufferAppend(unsigned char *s, unsigned long len) {
124 listNode *ln = listLast(server.aof_rewrite_buf_blocks);
125 aofrwblock *block = ln ? ln->value : NULL;
126
127 while(len) {
128 /* If we already got at least an allocated block, try appending
129 * at least some piece into it. */
130 if (block) {
131 unsigned long thislen = (block->free < len) ? block->free : len;
132 if (thislen) { /* The current block is not already full. */
133 memcpy(block->buf+block->used, s, thislen);
134 block->used += thislen;
135 block->free -= thislen;
136 s += thislen;
137 len -= thislen;
138 }
139 }
140
141 if (len) { /* First block to allocate, or need another block. */
142 int numblocks;
143
144 block = zmalloc(sizeof(*block));
145 block->free = AOF_RW_BUF_BLOCK_SIZE;
146 block->used = 0;
147 listAddNodeTail(server.aof_rewrite_buf_blocks,block);
148
149 /* Log every time we cross more 10 or 100 blocks, respectively
150 * as a notice or warning. */
151 numblocks = listLength(server.aof_rewrite_buf_blocks);
152 if (((numblocks+1) % 10) == 0) {
153 int level = ((numblocks+1) % 100) == 0 ? REDIS_WARNING :
154 REDIS_NOTICE;
155 redisLog(level,"Background AOF buffer size: %lu MB",
156 aofRewriteBufferSize()/(1024*1024));
157 }
158 }
159 }
160
161 /* Install a file event to send data to the rewrite child if there is
162 * not one already. */
163 if (aeGetFileEvents(server.el,server.aof_pipe_write_data_to_child) == 0) {
164 aeCreateFileEvent(server.el, server.aof_pipe_write_data_to_child,
165 AE_WRITABLE, aofChildWriteDiffData, NULL);
166 }
167}

169/* Write the buffer (possibly composed of multiple blocks) into the specified
170 * fd. If a short write or any other error happens -1 is returned,
171 * otherwise the number of bytes written is returned. */
172ssize_t aofRewriteBufferWrite(int fd) {
173 listNode *ln;
174 listIter li;
175 ssize_t count = 0;
176
177 listRewind(server.aof_rewrite_buf_blocks,&li);
178 while((ln = listNext(&li))) {
179 aofrwblock *block = listNodeValue(ln);
180 ssize_t nwritten;
181
182 if (block->used) {
183 nwritten = write(fd,block->buf,block->used);
184 if (nwritten != (ssize_t)block->used) {
185 if (nwritten == 0) errno = EIO;
186 return -1;
187 }
188 count += nwritten;
189 }
190 }
191 return count;
192}

204/* Called when the user switches from "appendonly yes" to "appendonly no"
205 * at runtime using the CONFIG command. */
206void stopAppendOnly(void) {
207 redisAssert(server.aof_state != REDIS_AOF_OFF);
208 flushAppendOnlyFile(1);
209 aof_fsync(server.aof_fd);
210 close(server.aof_fd);
211
212 server.aof_fd = -1;
213 server.aof_selected_db = -1;
214 server.aof_state = REDIS_AOF_OFF;
215 /* rewrite operation in progress? kill it, wait child exit */
216 if (server.aof_child_pid != -1) {
217 int statloc;
218
219 redisLog(REDIS_NOTICE,"Killing running AOF rewrite child: %ld",
220 (long) server.aof_child_pid);
221 if (kill(server.aof_child_pid,SIGUSR1) != -1)
222 wait3(&statloc,0,NULL);
223 /* reset the buffer accumulating changes while the child saves */
224 aofRewriteBufferReset();
225 aofRemoveTempFile(server.aof_child_pid);
226 server.aof_child_pid = -1;
227 server.aof_rewrite_time_start = -1;
228 /* close pipes used for IPC between the two processes. */
229 aofClosePipes();
230 }
231}

233/* Called when the user switches from "appendonly no" to "appendonly yes"
234 * at runtime using the CONFIG command. */
235int startAppendOnly(void) {
236 server.aof_last_fsync = server.unixtime;
237 server.aof_fd = open(server.aof_filename,O_WRONLY|O_APPEND|O_CREAT,0644);
238 redisAssert(server.aof_state == REDIS_AOF_OFF);
239 if (server.aof_fd == -1) {
240 redisLog(REDIS_WARNING,"Redis needs to enable the AOF but can't open the append only file: %s",strerror(errno));
241 return REDIS_ERR;
242 }
243 if (rewriteAppendOnlyFileBackground() == REDIS_ERR) {
244 close(server.aof_fd);
245 redisLog(REDIS_WARNING,"Redis needs to enable the AOF but can't trigger a background AOF rewrite operation. Check the above logs for more info about the error.");
246 return REDIS_ERR;
247 }
248 /* We correctly switched on AOF, now wait for the rewrite to be complete
249 * in order to append data on disk. */
250 server.aof_state = REDIS_AOF_WAIT_REWRITE;
251 return REDIS_OK;
252}

254/* Write the append only file buffer on disk.
255 *
256 * Since we are required to write the AOF before replying to the client,
257 * and the only way the client socket can get a write is entering when the
258 * the event loop, we accumulate all the AOF writes in a memory
259 * buffer and write it on disk using this function just before entering
260 * the event loop again.
261 *
262 * About the 'force' argument:
263 *
264 * When the fsync policy is set to 'everysec' we may delay the flush if there
265 * is still an fsync() going on in the background thread, since for instance
266 * on Linux write(2) will be blocked by the background fsync anyway.
267 * When this happens we remember that there is some aof buffer to be
268 * flushed ASAP, and will try to do that in the serverCron() function.
269 *
270 * However if force is set to 1 we'll write regardless of the background
271 * fsync. */
272#define AOF_WRITE_LOG_ERROR_RATE 30 /* Seconds between errors logging. */
273void flushAppendOnlyFile(int force) {
274 ssize_t nwritten;
275 int sync_in_progress = 0;
276 mstime_t latency;
277
278 if (sdslen(server.aof_buf) == 0) return;
279
280 if (server.aof_fsync == AOF_FSYNC_EVERYSEC)
281 sync_in_progress = bioPendingJobsOfType(REDIS_BIO_AOF_FSYNC) != 0;
282
283 if (server.aof_fsync == AOF_FSYNC_EVERYSEC && !force) {
284 /* With this append fsync policy we do background fsyncing.
285 * If the fsync is still in progress we can try to delay
286 * the write for a couple of seconds. */
287 if (sync_in_progress) {
288 if (server.aof_flush_postponed_start == 0) {
289 /* No previous write postponing, remember that we are
290 * postponing the flush and return. */
291 server.aof_flush_postponed_start = server.unixtime;
292 return;
293 } else if (server.unixtime - server.aof_flush_postponed_start < 2) {
294 /* We were already waiting for fsync to finish, but for less
295 * than two seconds this is still ok. Postpone again. */
296 return;
297 }
298 /* Otherwise fall trough, and go write since we can't wait
299 * over two seconds. */
300 server.aof_delayed_fsync++;
301 redisLog(REDIS_NOTICE,"Asynchronous AOF fsync is taking too long (disk is busy?). Writing the AOF buffer without waiting for fsync to complete, this may slow down Redis.");
302 }
303 }
304 /* We want to perform a single write. This should be guaranteed atomic
305 * at least if the filesystem we are writing is a real physical one.
306 * While this will save us against the server being killed I don't think
307 * there is much to do about the whole server stopping for power problems
308 * or alike */
309
310 latencyStartMonitor(latency);
311 nwritten = write(server.aof_fd,server.aof_buf,sdslen(server.aof_buf));
312 latencyEndMonitor(latency);
313 /* We want to capture different events for delayed writes:
314 * when the delay happens with a pending fsync, or with a saving child
315 * active, and when the above two conditions are missing.
316 * We also use an additional event name to save all samples which is
317 * useful for graphing / monitoring purposes. */
318 if (sync_in_progress) {
319 latencyAddSampleIfNeeded("aof-write-pending-fsync",latency);
320 } else if (server.aof_child_pid != -1 || server.rdb_child_pid != -1) {
321 latencyAddSampleIfNeeded("aof-write-active-child",latency);
322 } else {
323 latencyAddSampleIfNeeded("aof-write-alone",latency);
324 }
325 latencyAddSampleIfNeeded("aof-write",latency);
326
327 /* We performed the write so reset the postponed flush sentinel to zero. */
328 server.aof_flush_postponed_start = 0;
329
330 if (nwritten != (signed)sdslen(server.aof_buf)) {
331 static time_t last_write_error_log = 0;
332 int can_log = 0;
333
334 /* Limit logging rate to 1 line per AOF_WRITE_LOG_ERROR_RATE seconds. */
335 if ((server.unixtime - last_write_error_log) > AOF_WRITE_LOG_ERROR_RATE) {
336 can_log = 1;
337 last_write_error_log = server.unixtime;
338 }
339
340 /* Log the AOF write error and record the error code. */
341 if (nwritten == -1) {
342 if (can_log) {
343 redisLog(REDIS_WARNING,"Error writing to the AOF file: %s",
344 strerror(errno));
345 server.aof_last_write_errno = errno;
346 }
347 } else {
348 if (can_log) {
349 redisLog(REDIS_WARNING,"Short write while writing to "
350 "the AOF file: (nwritten=%lld, "
351 "expected=%lld)",
352 (long long)nwritten,
353 (long long)sdslen(server.aof_buf));
354 }
355
356 if (ftruncate(server.aof_fd, server.aof_current_size) == -1) {
357 if (can_log) {
358 redisLog(REDIS_WARNING, "Could not remove short write "
359 "from the append-only file. Redis may refuse "
360 "to load the AOF the next time it starts. "
361 "ftruncate: %s", strerror(errno));
362 }
363 } else {
364 /* If the ftruncate() succeeded we can set nwritten to
365 * -1 since there is no longer partial data into the AOF. */
366 nwritten = -1;
367 }
368 server.aof_last_write_errno = ENOSPC;
369 }
370
371 /* Handle the AOF write error. */
372 if (server.aof_fsync == AOF_FSYNC_ALWAYS) {
373 /* We can't recover when the fsync policy is ALWAYS since the
374 * reply for the client is already in the output buffers, and we
375 * have the contract with the user that on acknowledged write data
376 * is synced on disk. */
377 redisLog(REDIS_WARNING,"Can't recover from AOF write error when the AOF fsync policy is 'always'. Exiting...");
378 exit(1);
379 } else {
380 /* Recover from failed write leaving data into the buffer. However
381 * set an error to stop accepting writes as long as the error
382 * condition is not cleared. */
383 server.aof_last_write_status = REDIS_ERR;
384
385 /* Trim the sds buffer if there was a partial write, and there
386 * was no way to undo it with ftruncate(2). */
387 if (nwritten > 0) {
388 server.aof_current_size += nwritten;
389 sdsrange(server.aof_buf,nwritten,-1);
390 }
391 return; /* We'll try again on the next call... */
392 }
393 } else {
394 /* Successful write(2). If AOF was in error state, restore the
395 * OK state and log the event. */
396 if (server.aof_last_write_status == REDIS_ERR) {
397 redisLog(REDIS_WARNING,
398 "AOF write error looks solved, Redis can write again.");
399 server.aof_last_write_status = REDIS_OK;
400 }
401 }
402 server.aof_current_size += nwritten;
403
404 /* Re-use AOF buffer when it is small enough. The maximum comes from the
405 * arena size of 4k minus some overhead (but is otherwise arbitrary). */
406 if ((sdslen(server.aof_buf)+sdsavail(server.aof_buf)) < 4000) {
407 sdsclear(server.aof_buf);
408 } else {
409 sdsfree(server.aof_buf);
410 server.aof_buf = sdsempty();
411 }
412
413 /* Don't fsync if no-appendfsync-on-rewrite is set to yes and there are
414 * children doing I/O in the background. */
415 if (server.aof_no_fsync_on_rewrite &&
416 (server.aof_child_pid != -1 || server.rdb_child_pid != -1))
417 return;
418
419 /* Perform the fsync if needed. */
420 if (server.aof_fsync == AOF_FSYNC_ALWAYS) {
421 /* aof_fsync is defined as fdatasync() for Linux in order to avoid
422 * flushing metadata. */
423 latencyStartMonitor(latency);
424 aof_fsync(server.aof_fd); /* Let's try to get this data on the disk */
425 latencyEndMonitor(latency);
426 latencyAddSampleIfNeeded("aof-fsync-always",latency);
427 server.aof_last_fsync = server.unixtime;
428 } else if ((server.aof_fsync == AOF_FSYNC_EVERYSEC &&
429 server.unixtime > server.aof_last_fsync)) {
430 if (!sync_in_progress) aof_background_fsync(server.aof_fd);
431 server.aof_last_fsync = server.unixtime;
432 }
433}

435sds catAppendOnlyGenericCommand(sds dst, int argc, robj **argv) {
436 char buf[32];
437 int len, j;
438 robj *o;
439
440 buf[0] = '*';
441 len = 1+ll2string(buf+1,sizeof(buf)-1,argc);
442 buf[len++] = '\r';
443 buf[len++] = '\n';
444 dst = sdscatlen(dst,buf,len);
445
446 for (j = 0; j < argc; j++) {
447 o = getDecodedObject(argv[j]);
448 buf[0] = '$';
449 len = 1+ll2string(buf+1,sizeof(buf)-1,sdslen(o->ptr));
450 buf[len++] = '\r';
451 buf[len++] = '\n';
452 dst = sdscatlen(dst,buf,len);
453 dst = sdscatlen(dst,o->ptr,sdslen(o->ptr));
454 dst = sdscatlen(dst,"\r\n",2);
455 decrRefCount(o);
456 }
457 return dst;
458}

460/* Create the sds representation of an PEXPIREAT command, using
461 * 'seconds' as time to live and 'cmd' to understand what command
462 * we are translating into a PEXPIREAT.
463 *
464 * This command is used in order to translate EXPIRE and PEXPIRE commands
465 * into PEXPIREAT command so that we retain precision in the append only
466 * file, and the time is always absolute and not relative. */
467sds catAppendOnlyExpireAtCommand(sds buf, struct redisCommand *cmd, robj *key, robj *seconds) {
468 long long when;
469 robj *argv[3];
470
471 /* Make sure we can use strtoll */
472 seconds = getDecodedObject(seconds);
473 when = strtoll(seconds->ptr,NULL,10);
474 /* Convert argument into milliseconds for EXPIRE, SETEX, EXPIREAT */
475 if (cmd->proc == expireCommand || cmd->proc == setexCommand ||
476 cmd->proc == expireatCommand)
477 {
478 when *= 1000;
479 }
480 /* Convert into absolute time for EXPIRE, PEXPIRE, SETEX, PSETEX */
481 if (cmd->proc == expireCommand || cmd->proc == pexpireCommand ||
482 cmd->proc == setexCommand || cmd->proc == psetexCommand)
483 {
484 when += mstime();
485 }
486 decrRefCount(seconds);
487
488 argv[0] = createStringObject("PEXPIREAT",9);
489 argv[1] = key;
490 argv[2] = createStringObjectFromLongLong(when);
491 buf = catAppendOnlyGenericCommand(buf, 3, argv);
492 decrRefCount(argv[0]);
493 decrRefCount(argv[2]);
494 return buf;
495}

497void feedAppendOnlyFile(struct redisCommand *cmd, int dictid, robj **argv, int argc) {
498 sds buf = sdsempty();
499 robj *tmpargv[3];
500
501 /* The DB this command was targeting is not the same as the last command
502 * we appended. To issue a SELECT command is needed. */
503 if (dictid != server.aof_selected_db) {
504 char seldb[64];
505
506 snprintf(seldb,sizeof(seldb),"%d",dictid);
507 buf = sdscatprintf(buf,"*2\r\n$6\r\nSELECT\r\n$%lu\r\n%s\r\n",
508 (unsigned long)strlen(seldb),seldb);
509 server.aof_selected_db = dictid;
510 }
511
512 if (cmd->proc == expireCommand || cmd->proc == pexpireCommand ||
513 cmd->proc == expireatCommand) {
514 /* Translate EXPIRE/PEXPIRE/EXPIREAT into PEXPIREAT */
515 buf = catAppendOnlyExpireAtCommand(buf,cmd,argv[1],argv[2]);
516 } else if (cmd->proc == setexCommand || cmd->proc == psetexCommand) {
517 /* Translate SETEX/PSETEX to SET and PEXPIREAT */
518 tmpargv[0] = createStringObject("SET",3);
519 tmpargv[1] = argv[1];
520 tmpargv[2] = argv[3];
521 buf = catAppendOnlyGenericCommand(buf,3,tmpargv);
522 decrRefCount(tmpargv[0]);
523 buf = catAppendOnlyExpireAtCommand(buf,cmd,argv[1],argv[2]);
524 } else {
525 /* All the other commands don't need translation or need the
526 * same translation already operated in the command vector
527 * for the replication itself. */
528 buf = catAppendOnlyGenericCommand(buf,argc,argv);
529 }
530
531 /* Append to the AOF buffer. This will be flushed on disk just before
532 * of re-entering the event loop, so before the client will get a
533 * positive reply about the operation performed. */
534 if (server.aof_state == REDIS_AOF_ON)
535 server.aof_buf = sdscatlen(server.aof_buf,buf,sdslen(buf));
536
537 /* If a background append only file rewriting is in progress we want to
538 * accumulate the differences between the child DB and the current one
539 * in a buffer, so that when the child process will do its work we
540 * can append the differences to the new append only file. */
541 if (server.aof_child_pid != -1)
542 aofRewriteBufferAppend((unsigned char*)buf,sdslen(buf));
543
544 sdsfree(buf);
545}

551/* In Redis commands are always executed in the context of a client, so in
552 * order to load the append only file we need to create a fake client. */
553struct redisClient *createFakeClient(void) {
554 struct redisClient *c = zmalloc(sizeof(*c));
555
556 selectDb(c,0);
557 c->fd = -1;
558 c->name = NULL;
559 c->querybuf = sdsempty();
560 c->querybuf_peak = 0;
561 c->argc = 0;
562 c->argv = NULL;
563 c->bufpos = 0;
564 c->flags = 0;
565 c->btype = REDIS_BLOCKED_NONE;
566 /* We set the fake client as a slave waiting for the synchronization
567 * so that Redis will not try to send replies to this client. */
568 c->replstate = REDIS_REPL_WAIT_BGSAVE_START;
569 c->reply = listCreate();
570 c->reply_bytes = 0;
571 c->obuf_soft_limit_reached_time = 0;
572 c->watched_keys = listCreate();
573 c->peerid = NULL;
574 listSetFreeMethod(c->reply,decrRefCountVoid);
575 listSetDupMethod(c->reply,dupClientReplyValue);
576 initClientMultiState(c);
577 return c;
578}

596/* Replay the append log file. On success REDIS_OK is returned. On non fatal
597 * error (the append only file is zero-length) REDIS_ERR is returned. On
598 * fatal error an error message is logged and the program exists. */
599int loadAppendOnlyFile(char *filename) {
600 struct redisClient *fakeClient;
601 FILE *fp = fopen(filename,"r");
602 struct redis_stat sb;
603 int old_aof_state = server.aof_state;
604 long loops = 0;
605 off_t valid_up_to = 0; /* Offset of the latest well-formed command loaded. */
606
607 if (fp && redis_fstat(fileno(fp),&sb) != -1 && sb.st_size == 0) {
608 server.aof_current_size = 0;
609 fclose(fp);
610 return REDIS_ERR;
611 }
612
613 if (fp == NULL) {
614 redisLog(REDIS_WARNING,"Fatal error: can't open the append log file for reading: %s",strerror(errno));
615 exit(1);
616 }
617
618 /* Temporarily disable AOF, to prevent EXEC from feeding a MULTI
619 * to the same file we're about to read. */
620 server.aof_state = REDIS_AOF_OFF;
621
622 fakeClient = createFakeClient();
623 startLoading(fp);
624
625 while(1) {
626 int argc, j;
627 unsigned long len;
628 robj **argv;
629 char buf[128];
630 sds argsds;
631 struct redisCommand *cmd;
632
633 /* Serve the clients from time to time */
634 if (!(loops++ % 1000)) {
635 loadingProgress(ftello(fp));
636 processEventsWhileBlocked();
637 }
638
639 if (fgets(buf,sizeof(buf),fp) == NULL) {
640 if (feof(fp))
641 break;
642 else
643 goto readerr;
644 }
645 if (buf[0] != '*') goto fmterr;
646 if (buf[1] == '\0') goto readerr;
647 argc = atoi(buf+1);
648 if (argc < 1) goto fmterr;
649
650 argv = zmalloc(sizeof(robj*)*argc);
651 fakeClient->argc = argc;
652 fakeClient->argv = argv;
653
654 for (j = 0; j < argc; j++) {
655 if (fgets(buf,sizeof(buf),fp) == NULL) {
656 fakeClient->argc = j; /* Free up to j-1. */
657 freeFakeClientArgv(fakeClient);
658 goto readerr;
659 }
660 if (buf[0] != '$') goto fmterr;
661 len = strtol(buf+1,NULL,10);
662 argsds = sdsnewlen(NULL,len);
663 if (len && fread(argsds,len,1,fp) == 0) {
664 sdsfree(argsds);
665 fakeClient->argc = j; /* Free up to j-1. */
666 freeFakeClientArgv(fakeClient);
667 goto readerr;
668 }
669 argv[j] = createObject(REDIS_STRING,argsds);
670 if (fread(buf,2,1,fp) == 0) {
671 fakeClient->argc = j+1; /* Free up to j. */
672 freeFakeClientArgv(fakeClient);
673 goto readerr; /* discard CRLF */
674 }
675 }
676
677 /* Command lookup */
678 cmd = lookupCommand(argv[0]->ptr);
679 if (!cmd) {
680 redisLog(REDIS_WARNING,"Unknown command '%s' reading the append only file", (char*)argv[0]->ptr);
681 exit(1);
682 }
683
684 /* Run the command in the context of a fake client */
685 cmd->proc(fakeClient);
686
687 /* The fake client should not have a reply */
688 redisAssert(fakeClient->bufpos == 0 && listLength(fakeClient->reply) == 0);
689 /* The fake client should never get blocked */
690 redisAssert((fakeClient->flags & REDIS_BLOCKED) == 0);
691
692 /* Clean up. Command code may have changed argv/argc so we use the
693 * argv/argc of the client instead of the local variables. */
694 freeFakeClientArgv(fakeClient);
695 if (server.aof_load_truncated) valid_up_to = ftello(fp);
696 }
697
698 /* This point can only be reached when EOF is reached without errors.
699 * If the client is in the middle of a MULTI/EXEC, log error and quit. */
700 if (fakeClient->flags & REDIS_MULTI) goto uxeof;
701
702loaded_ok: /* DB loaded, cleanup and return REDIS_OK to the caller. */
703 fclose(fp);
704 freeFakeClient(fakeClient);
705 server.aof_state = old_aof_state;
706 stopLoading();
707 aofUpdateCurrentSize();
708 server.aof_rewrite_base_size = server.aof_current_size;
709 return REDIS_OK;
710
711readerr: /* Read error. If feof(fp) is true, fall through to unexpected EOF. */
712 if (!feof(fp)) {
713 redisLog(REDIS_WARNING,"Unrecoverable error reading the append only file: %s", strerror(errno));
714 exit(1);
715 }
716
717uxeof: /* Unexpected AOF end of file. */
718 if (server.aof_load_truncated) {
719 redisLog(REDIS_WARNING,"!!! Warning: short read while loading the AOF file !!!");
720 redisLog(REDIS_WARNING,"!!! Truncating the AOF at offset %llu !!!",
721 (unsigned long long) valid_up_to);
722 if (valid_up_to == -1 || truncate(filename,valid_up_to) == -1) {
723 if (valid_up_to == -1) {
724 redisLog(REDIS_WARNING,"Last valid command offset is invalid");
725 } else {
726 redisLog(REDIS_WARNING,"Error truncating the AOF file: %s",
727 strerror(errno));
728 }
729 } else {
730 /* Make sure the AOF file descriptor points to the end of the
731 * file after the truncate call. */
732 if (server.aof_fd != -1 && lseek(server.aof_fd,0,SEEK_END) == -1) {
733 redisLog(REDIS_WARNING,"Can't seek the end of the AOF file: %s",
734 strerror(errno));
735 } else {
736 redisLog(REDIS_WARNING,
737 "AOF loaded anyway because aof-load-truncated is enabled");
738 goto loaded_ok;
739 }
740 }
741 }
742 redisLog(REDIS_WARNING,"Unexpected end of file reading the append only file. You can: 1) Make a backup of your AOF file, then use ./redis-check-aof --fix <filename>. 2) Alternatively you can set the 'aof-load-truncated' configuration option to yes and restart the server.");
743 exit(1);
744
745fmterr: /* Format error. */
746 redisLog(REDIS_WARNING,"Bad file format reading the append only file: make a backup of your AOF file, then use ./redis-check-aof --fix <filename>");
747 exit(1);
748}

768/* Emit the commands needed to rebuild a list object.
769 * The function returns 0 on error, 1 on success. */
770int rewriteListObject(rio *r, robj *key, robj *o) {
771 long long count = 0, items = listTypeLength(o);
772
773 if (o->encoding == REDIS_ENCODING_ZIPLIST) {
774 unsigned char *zl = o->ptr;
775 unsigned char *p = ziplistIndex(zl,0);
776 unsigned char *vstr;
777 unsigned int vlen;
778 long long vlong;
779
780 while(ziplistGet(p,&vstr,&vlen,&vlong)) {
781 if (count == 0) {
782 int cmd_items = (items > REDIS_AOF_REWRITE_ITEMS_PER_CMD) ?
783 REDIS_AOF_REWRITE_ITEMS_PER_CMD : items;
784
785 if (rioWriteBulkCount(r,'*',2+cmd_items) == 0) return 0;
786 if (rioWriteBulkString(r,"RPUSH",5) == 0) return 0;
787 if (rioWriteBulkObject(r,key) == 0) return 0;
788 }
789 if (vstr) {
790 if (rioWriteBulkString(r,(char*)vstr,vlen) == 0) return 0;
791 } else {
792 if (rioWriteBulkLongLong(r,vlong) == 0) return 0;
793 }
794 p = ziplistNext(zl,p);
795 if (++count == REDIS_AOF_REWRITE_ITEMS_PER_CMD) count = 0;
796 items--;
797 }
798 } else if (o->encoding == REDIS_ENCODING_LINKEDLIST) {
799 list *list = o->ptr;
800 listNode *ln;
801 listIter li;
802
803 listRewind(list,&li);
804 while((ln = listNext(&li))) {
805 robj *eleobj = listNodeValue(ln);
806
807 if (count == 0) {
808 int cmd_items = (items > REDIS_AOF_REWRITE_ITEMS_PER_CMD) ?
809 REDIS_AOF_REWRITE_ITEMS_PER_CMD : items;
810
811 if (rioWriteBulkCount(r,'*',2+cmd_items) == 0) return 0;
812 if (rioWriteBulkString(r,"RPUSH",5) == 0) return 0;
813 if (rioWriteBulkObject(r,key) == 0) return 0;
814 }
815 if (rioWriteBulkObject(r,eleobj) == 0) return 0;
816 if (++count == REDIS_AOF_REWRITE_ITEMS_PER_CMD) count = 0;
817 items--;
818 }
819 } else {
820 redisPanic("Unknown list encoding");
821 }
822 return 1;
823}

999/* This function is called by the child rewriting the AOF file to read
1000 * the difference accumulated from the parent into a buffer, that is
1001 * concatenated at the end of the rewrite. */
1002ssize_t aofReadDiffFromParent(void) {
1003 char buf[65536]; /* Default pipe buffer size on most Linux systems. */
1004 ssize_t nread, total = 0;
1005
1006 while ((nread =
1007 read(server.aof_pipe_read_data_from_parent,buf,sizeof(buf))) > 0) {
1008 server.aof_child_diff = sdscatlen(server.aof_child_diff,buf,nread);
1009 total += nread;
1010 }
1011 return total;
1012}

1014/* Write a sequence of commands able to fully rebuild the dataset into
1015 * "filename". Used both by REWRITEAOF and BGREWRITEAOF.
1016 *
1017 * In order to minimize the number of commands needed in the rewritten
1018 * log Redis uses variadic commands when possible, such as RPUSH, SADD
1019 * and ZADD. However at max REDIS_AOF_REWRITE_ITEMS_PER_CMD items per time
1020 * are inserted using a single command. */
1021int rewriteAppendOnlyFile(char *filename) {
1022 dictIterator *di = NULL;
1023 dictEntry *de;
1024 rio aof;
1025 FILE *fp;
1026 char tmpfile[256];
1027 int j;
1028 long long now = mstime();
1029 char byte;
1030 size_t processed = 0;
1031
1032 /* Note that we have to use a different temp name here compared to the
1033 * one used by rewriteAppendOnlyFileBackground() function. */
1034 snprintf(tmpfile,256,"temp-rewriteaof-%d.aof", (int) getpid());
1035 fp = fopen(tmpfile,"w");
1036 if (!fp) {
1037 redisLog(REDIS_WARNING, "Opening the temp file for AOF rewrite in rewriteAppendOnlyFile(): %s", strerror(errno));
1038 return REDIS_ERR;
1039 }
1040
1041 server.aof_child_diff = sdsempty();
1042 rioInitWithFile(&aof,fp);
1043 if (server.aof_rewrite_incremental_fsync)
1044 rioSetAutoSync(&aof,REDIS_AOF_AUTOSYNC_BYTES);
1045 for (j = 0; j < server.dbnum; j++) {
1046 char selectcmd[] = "*2\r\n$6\r\nSELECT\r\n";
1047 redisDb *db = server.db+j;
1048 dict *d = db->dict;
1049 if (dictSize(d) == 0) continue;
1050 di = dictGetSafeIterator(d);
1051 if (!di) {
1052 fclose(fp);
1053 return REDIS_ERR;
1054 }
1055
1056 /* SELECT the new DB */
1057 if (rioWrite(&aof,selectcmd,sizeof(selectcmd)-1) == 0) goto werr;
1058 if (rioWriteBulkLongLong(&aof,j) == 0) goto werr;
1059
1060 /* Iterate this DB writing every entry */
1061 while((de = dictNext(di)) != NULL) {
1062 sds keystr;
1063 robj key, *o;
1064 long long expiretime;
1065
1066 keystr = dictGetKey(de);
1067 o = dictGetVal(de);
1068 initStaticStringObject(key,keystr);
1069
1070 expiretime = getExpire(db,&key);
1071
1072 /* If this key is already expired skip it */
1073 if (expiretime != -1 && expiretime < now) continue;
1074
1075 /* Save the key and associated value */
1076 if (o->type == REDIS_STRING) {
1077 /* Emit a SET command */
1078 char cmd[]="*3\r\n$3\r\nSET\r\n";
1079 if (rioWrite(&aof,cmd,sizeof(cmd)-1) == 0) goto werr;
1080 /* Key and value */
1081 if (rioWriteBulkObject(&aof,&key) == 0) goto werr;
1082 if (rioWriteBulkObject(&aof,o) == 0) goto werr;
1083 } else if (o->type == REDIS_LIST) {
1084 if (rewriteListObject(&aof,&key,o) == 0) goto werr;
1085 } else if (o->type == REDIS_SET) {
1086 if (rewriteSetObject(&aof,&key,o) == 0) goto werr;
1087 } else if (o->type == REDIS_ZSET) {
1088 if (rewriteSortedSetObject(&aof,&key,o) == 0) goto werr;
1089 } else if (o->type == REDIS_HASH) {
1090 if (rewriteHashObject(&aof,&key,o) == 0) goto werr;
1091 } else {
1092 redisPanic("Unknown object type");
1093 }
1094 /* Save the expire time */
1095 if (expiretime != -1) {
1096 char cmd[]="*3\r\n$9\r\nPEXPIREAT\r\n";
1097 if (rioWrite(&aof,cmd,sizeof(cmd)-1) == 0) goto werr;
1098 if (rioWriteBulkObject(&aof,&key) == 0) goto werr;
1099 if (rioWriteBulkLongLong(&aof,expiretime) == 0) goto werr;
1100 }
1101 /* Read some diff from the parent process from time to time. */
1102 if (aof.processed_bytes > processed+1024*10) {
1103 processed = aof.processed_bytes;
1104 aofReadDiffFromParent();
1105 }
1106 }
1107 dictReleaseIterator(di);
1108 di = NULL;
1109 }
1110
1111 /* Do an initial slow fsync here while the parent is still sending
1112 * data, in order to make the next final fsync faster. */
1113 if (fflush(fp) == EOF) goto werr;
1114 if (fsync(fileno(fp)) == -1) goto werr;
1115
1116 /* Read again a few times to get more data from the parent.
1117 * We can't read forever (the server may receive data from clients
1118 * faster than it is able to send data to the child), so we try to read
1119 * some more data in a loop as soon as there is a good chance more data
1120 * will come. If it looks like we are wasting time, we abort (this
1121 * happens after 20 ms without new data). */
1122 int nodata = 0;
1123 mstime_t start = mstime();
1124 while(mstime()-start < 1000 && nodata < 20) {
1125 if (aeWait(server.aof_pipe_read_data_from_parent, AE_READABLE, 1) <= 0)
1126 {
1127 nodata++;
1128 continue;
1129 }
1130 nodata = 0; /* Start counting from zero, we stop on N *contiguous*
1131 timeouts. */
1132 aofReadDiffFromParent();
1133 }
1134
1135 /* Ask the master to stop sending diffs. */
1136 if (write(server.aof_pipe_write_ack_to_parent,"!",1) != 1) goto werr;
1137 if (anetNonBlock(NULL,server.aof_pipe_read_ack_from_parent) != ANET_OK)
1138 goto werr;
1139 /* We read the ACK from the server using a 10 seconds timeout. Normally
1140 * it should reply ASAP, but just in case we lose its reply, we are sure
1141 * the child will eventually get terminated. */
1142 if (syncRead(server.aof_pipe_read_ack_from_parent,&byte,1,5000) != 1 ||
1143 byte != '!') goto werr;
1144 redisLog(REDIS_NOTICE,"Parent agreed to stop sending diffs. Finalizing AOF...");
1145
1146 /* Read the final diff if any. */
1147 aofReadDiffFromParent();
1148
1149 /* Write the received diff to the file. */
1150 redisLog(REDIS_NOTICE,
1151 "Concatenating %.2f MB of AOF diff received from parent.",
1152 (double) sdslen(server.aof_child_diff) / (1024*1024));
1153 if (rioWrite(&aof,server.aof_child_diff,sdslen(server.aof_child_diff)) == 0)
1154 goto werr;
1155
1156 /* Make sure data will not remain on the OS's output buffers */
1157 if (fflush(fp) == EOF) goto werr;
1158 if (fsync(fileno(fp)) == -1) goto werr;
1159 if (fclose(fp) == EOF) goto werr;
1160
1161 /* Use RENAME to make sure the DB file is changed atomically only
1162 * if the generate DB file is ok. */
1163 if (rename(tmpfile,filename) == -1) {
1164 redisLog(REDIS_WARNING,"Error moving temp append only file on the final destination: %s", strerror(errno));
1165 unlink(tmpfile);
1166 return REDIS_ERR;
1167 }
1168 redisLog(REDIS_NOTICE,"SYNC append only file rewrite performed");
1169 return REDIS_OK;
1170
1171werr:
1172 fclose(fp);
1173 unlink(tmpfile);
1174 redisLog(REDIS_WARNING,"Write error writing append only file on disk: %s", strerror(errno));
1175 if (di) dictReleaseIterator(di);
1176 return REDIS_ERR;
1177}

1183/* This event handler is called when the AOF rewriting child sends us a
1184 * single '!' char to signal we should stop sending buffer diffs. The
1185 * parent sends a '!' as well to acknowledge. */
1186void aofChildPipeReadable(aeEventLoop *el, int fd, void *privdata, int mask) {
1187 char byte;
1188 REDIS_NOTUSED(el);
1189 REDIS_NOTUSED(privdata);
1190 REDIS_NOTUSED(mask);
1191
1192 if (read(fd,&byte,1) == 1 && byte == '!') {
1193 redisLog(REDIS_NOTICE,"AOF rewrite child asks to stop sending diffs.");
1194 server.aof_stop_sending_diff = 1;
1195 if (write(server.aof_pipe_write_ack_to_child,"!",1) != 1) {
1196 /* If we can't send the ack, inform the user, but don't try again
1197 * since in the other side the children will use a timeout if the
1198 * kernel can't buffer our write, or, the children was
1199 * terminated. */
1200 redisLog(REDIS_WARNING,"Can't send ACK to AOF child: %s",
1201 strerror(errno));
1202 }
1203 }
1204 /* Remove the handler since this can be called only one time during a
1205 * rewrite. */
1206 aeDeleteFileEvent(server.el,server.aof_pipe_read_ack_from_child,AE_READABLE);
1207}

1209/* Create the pipes used for parent - child process IPC during rewrite.
1210 * We have a data pipe used to send AOF incremental diffs to the child,
1211 * and two other pipes used by the children to signal it finished with
1212 * the rewrite so no more data should be written, and another for the
1213 * parent to acknowledge it understood this new condition. */
1214int aofCreatePipes(void) {
1215 int fds[6] = {-1, -1, -1, -1, -1, -1};
1216 int j;
1217
1218 if (pipe(fds) == -1) goto error; /* parent -> children data. */
1219 if (pipe(fds+2) == -1) goto error; /* children -> parent ack. */
1220 if (pipe(fds+4) == -1) goto error; /* children -> parent ack. */
1221 /* Parent -> children data is non blocking. */
1222 if (anetNonBlock(NULL,fds[0]) != ANET_OK) goto error;
1223 if (anetNonBlock(NULL,fds[1]) != ANET_OK) goto error;
1224 if (aeCreateFileEvent(server.el, fds[2], AE_READABLE, aofChildPipeReadable, NULL) == AE_ERR) goto error;
1225
1226 server.aof_pipe_write_data_to_child = fds[1];
1227 server.aof_pipe_read_data_from_parent = fds[0];
1228 server.aof_pipe_write_ack_to_parent = fds[3];
1229 server.aof_pipe_read_ack_from_child = fds[2];
1230 server.aof_pipe_write_ack_to_child = fds[5];
1231 server.aof_pipe_read_ack_from_parent = fds[4];
1232 server.aof_stop_sending_diff = 0;
1233 return REDIS_OK;
1234
1235error:
1236 redisLog(REDIS_WARNING,"Error opening /setting AOF rewrite IPC pipes: %s",
1237 strerror(errno));
1238 for (j = 0; j < 6; j++) if(fds[j] != -1) close(fds[j]);
1239 return REDIS_ERR;
1240}

1244 aeDeleteFileEvent(server.el,server.aof_pipe_write_data_to_child,AE_WRITABLE);
1245 close(server.aof_pipe_write_data_to_child);
1246 close(server.aof_pipe_read_data_from_parent);
1247 close(server.aof_pipe_write_ack_to_parent);
1248 close(server.aof_pipe_read_ack_from_child);
1249 close(server.aof_pipe_write_ack_to_child);
1250 close(server.aof_pipe_read_ack_from_parent);

1257/* This is how rewriting of the append only file in background works:
1258 *
1259 * 1) The user calls BGREWRITEAOF
1260 * 2) Redis calls this function, that forks():
1261 * 2a) the child rewrite the append only file in a temp file.
1262 * 2b) the parent accumulates differences in server.aof_rewrite_buf.
1263 * 3) When the child finished '2a' exists.
1264 * 4) The parent will trap the exit code, if it's OK, will append the
1265 * data accumulated into server.aof_rewrite_buf into the temp file, and
1266 * finally will rename(2) the temp file in the actual file name.
1267 * The the new file is reopened as the new append only file. Profit!
1268 */
1269int rewriteAppendOnlyFileBackground(void) {
1270 pid_t childpid;
1271 long long start;
1272
1273 if (server.aof_child_pid != -1) return REDIS_ERR;
1274 if (aofCreatePipes() != REDIS_OK) return REDIS_ERR;
1275 start = ustime();
1276 if ((childpid = fork()) == 0) {
1277 char tmpfile[256];
1278
1279 /* Child */
1280 closeListeningSockets(0);
1281 redisSetProcTitle("redis-aof-rewrite");
1282 snprintf(tmpfile,256,"temp-rewriteaof-bg-%d.aof", (int) getpid());
1283 if (rewriteAppendOnlyFile(tmpfile) == REDIS_OK) {
1284 size_t private_dirty = zmalloc_get_private_dirty();
1285
1286 if (private_dirty) {
1287 redisLog(REDIS_NOTICE,
1288 "AOF rewrite: %zu MB of memory used by copy-on-write",
1289 private_dirty/(1024*1024));
1290 }
1291 exitFromChild(0);
1292 } else {
1293 exitFromChild(1);
1294 }
1295 } else {
1296 /* Parent */
1297 server.stat_fork_time = ustime()-start;
1298 server.stat_fork_rate = (double) zmalloc_used_memory() * 1000000 / server.stat_fork_time / (1024*1024*1024); /* GB per second. */
1299 latencyAddSampleIfNeeded("fork",server.stat_fork_time/1000);
1300 if (childpid == -1) {
1301 redisLog(REDIS_WARNING,
1302 "Can't rewrite append only file in background: fork: %s",
1303 strerror(errno));
1304 return REDIS_ERR;
1305 }
1306 redisLog(REDIS_NOTICE,
1307 "Background append only file rewriting started by pid %d",childpid);
1308 server.aof_rewrite_scheduled = 0;
1309 server.aof_rewrite_time_start = time(NULL);
1310 server.aof_child_pid = childpid;
1311 updateDictResizePolicy();
1312 /* We set appendseldb to -1 in order to force the next call to the
1313 * feedAppendOnlyFile() to issue a SELECT command, so the differences
1314 * accumulated by the parent into server.aof_rewrite_buf will start
1315 * with a SELECT statement and it will be safe to merge. */
1316 server.aof_selected_db = -1;
1317 replicationScriptCacheFlush();
1318 return REDIS_OK;
1319 }
1320 return REDIS_OK; /* unreached */
1321}

1362/* A background append only file rewriting (BGREWRITEAOF) terminated its work.
1363 * Handle this. */
1364void backgroundRewriteDoneHandler(int exitcode, int bysignal) {
1365 if (!bysignal && exitcode == 0) {
1366 int newfd, oldfd;
1367 char tmpfile[256];
1368 long long now = ustime();
1369 mstime_t latency;
1370
1371 redisLog(REDIS_NOTICE,
1372 "Background AOF rewrite terminated with success");
1373
1374 /* Flush the differences accumulated by the parent to the
1375 * rewritten AOF. */
1376 latencyStartMonitor(latency);
1377 snprintf(tmpfile,256,"temp-rewriteaof-bg-%d.aof",
1378 (int)server.aof_child_pid);
1379 newfd = open(tmpfile,O_WRONLY|O_APPEND);
1380 if (newfd == -1) {
1381 redisLog(REDIS_WARNING,
1382 "Unable to open the temporary AOF produced by the child: %s", strerror(errno));
1383 goto cleanup;
1384 }
1385
1386 if (aofRewriteBufferWrite(newfd) == -1) {
1387 redisLog(REDIS_WARNING,
1388 "Error trying to flush the parent diff to the rewritten AOF: %s", strerror(errno));
1389 close(newfd);
1390 goto cleanup;
1391 }
1392 latencyEndMonitor(latency);
1393 latencyAddSampleIfNeeded("aof-rewrite-diff-write",latency);
1394
1395 redisLog(REDIS_NOTICE,
1396 "Residual parent diff successfully flushed to the rewritten AOF (%.2f MB)", (double) aofRewriteBufferSize() / (1024*1024));
1397
1398 /* The only remaining thing to do is to rename the temporary file to
1399 * the configured file and switch the file descriptor used to do AOF
1400 * writes. We don't want close(2) or rename(2) calls to block the
1401 * server on old file deletion.
1402 *
1403 * There are two possible scenarios:
1404 *
1405 * 1) AOF is DISABLED and this was a one time rewrite. The temporary
1406 * file will be renamed to the configured file. When this file already
1407 * exists, it will be unlinked, which may block the server.
1408 *
1409 * 2) AOF is ENABLED and the rewritten AOF will immediately start
1410 * receiving writes. After the temporary file is renamed to the
1411 * configured file, the original AOF file descriptor will be closed.
1412 * Since this will be the last reference to that file, closing it
1413 * causes the underlying file to be unlinked, which may block the
1414 * server.
1415 *
1416 * To mitigate the blocking effect of the unlink operation (either
1417 * caused by rename(2) in scenario 1, or by close(2) in scenario 2), we
1418 * use a background thread to take care of this. First, we
1419 * make scenario 1 identical to scenario 2 by opening the target file
1420 * when it exists. The unlink operation after the rename(2) will then
1421 * be executed upon calling close(2) for its descriptor. Everything to
1422 * guarantee atomicity for this switch has already happened by then, so
1423 * we don't care what the outcome or duration of that close operation
1424 * is, as long as the file descriptor is released again. */
1425 if (server.aof_fd == -1) {
1426 /* AOF disabled */
1427
1428 /* Don't care if this fails: oldfd will be -1 and we handle that.
1429 * One notable case of -1 return is if the old file does
1430 * not exist. */
1431 oldfd = open(server.aof_filename,O_RDONLY|O_NONBLOCK);
1432 } else {
1433 /* AOF enabled */
1434 oldfd = -1; /* We'll set this to the current AOF filedes later. */
1435 }
1436
1437 /* Rename the temporary file. This will not unlink the target file if
1438 * it exists, because we reference it with "oldfd". */
1439 latencyStartMonitor(latency);
1440 if (rename(tmpfile,server.aof_filename) == -1) {
1441 redisLog(REDIS_WARNING,
1442 "Error trying to rename the temporary AOF file: %s", strerror(errno));
1443 close(newfd);
1444 if (oldfd != -1) close(oldfd);
1445 goto cleanup;
1446 }
1447 latencyEndMonitor(latency);
1448 latencyAddSampleIfNeeded("aof-rename",latency);
1449
1450 if (server.aof_fd == -1) {
1451 /* AOF disabled, we don't need to set the AOF file descriptor
1452 * to this new file, so we can close it. */
1453 close(newfd);
1454 } else {
1455 /* AOF enabled, replace the old fd with the new one. */
1456 oldfd = server.aof_fd;
1457 server.aof_fd = newfd;
1458 if (server.aof_fsync == AOF_FSYNC_ALWAYS)
1459 aof_fsync(newfd);
1460 else if (server.aof_fsync == AOF_FSYNC_EVERYSEC)
1461 aof_background_fsync(newfd);
1462 server.aof_selected_db = -1; /* Make sure SELECT is re-issued */
1463 aofUpdateCurrentSize();
1464 server.aof_rewrite_base_size = server.aof_current_size;
1465
1466 /* Clear regular AOF buffer since its contents was just written to
1467 * the new AOF from the background rewrite buffer. */
1468 sdsfree(server.aof_buf);
1469 server.aof_buf = sdsempty();
1470 }
1471
1472 server.aof_lastbgrewrite_status = REDIS_OK;
1473
1474 redisLog(REDIS_NOTICE, "Background AOF rewrite finished successfully");
1475 /* Change state from WAIT_REWRITE to ON if needed */
1476 if (server.aof_state == REDIS_AOF_WAIT_REWRITE)
1477 server.aof_state = REDIS_AOF_ON;
1478
1479 /* Asynchronously close the overwritten AOF. */
1480 if (oldfd != -1) bioCreateBackgroundJob(REDIS_BIO_CLOSE_FILE,(void*)(long)oldfd,NULL,NULL);
1481
1482 redisLog(REDIS_VERBOSE,
1483 "Background AOF rewrite signal handler took %lldus", ustime()-now);
1484 } else if (!bysignal && exitcode != 0) {
1485 server.aof_lastbgrewrite_status = REDIS_ERR;
1486
1487 redisLog(REDIS_WARNING,
1488 "Background AOF rewrite terminated with error");
1489 } else {
1490 server.aof_lastbgrewrite_status = REDIS_ERR;
1491
1492 redisLog(REDIS_WARNING,
1493 "Background AOF rewrite terminated by signal %d", bysignal);
1494 }
1495
1496cleanup:
1497 aofClosePipes();
1498 aofRewriteBufferReset();
1499 aofRemoveTempFile(server.aof_child_pid);
1500 server.aof_child_pid = -1;
1501 server.aof_rewrite_time_last = time(NULL)-server.aof_rewrite_time_start;
1502 server.aof_rewrite_time_start = -1;
1503 /* Schedule a new rewrite if we are waiting for it to switch the AOF ON. */
1504 if (server.aof_state == REDIS_AOF_WAIT_REWRITE)
1505 server.aof_rewrite_scheduled = 1;
1506}
