47/* Return the pointer to a string representing the slave ip:listening_port
48 * pair. Mostly useful for logging, since we want to log a slave using its
49 * IP address and it's listening port which is more clear for the user, for
50 * example: "Closing connection with slave 10.1.2.3:6380". */
51char *replicationGetSlaveName(redisClient *c) {
52 static char buf[REDIS_PEER_ID_LEN];
53 char ip[REDIS_IP_STR_LEN];
54
55 ip[0] = '\0';
56 buf[0] = '\0';
57 if (anetPeerToString(c->fd,ip,sizeof(ip),NULL) != -1) {
58 if (c->slave_listening_port)
59 snprintf(buf,sizeof(buf),"%s:%d",ip,c->slave_listening_port);
60 else
61 snprintf(buf,sizeof(buf),"%s:<unknown-slave-port>",ip);
62 } else {
63 snprintf(buf,sizeof(buf),"client id #%llu",
64 (unsigned long long) c->id);
65 }
66 return buf;
67}

71void createReplicationBacklog(void) {
72 redisAssert(server.repl_backlog == NULL);
73 server.repl_backlog = zmalloc(server.repl_backlog_size);
74 server.repl_backlog_histlen = 0;
75 server.repl_backlog_idx = 0;
76 /* When a new backlog buffer is created, we increment the replication
77 * offset by one to make sure we'll not be able to PSYNC with any
78 * previous slave. This is needed because we avoid incrementing the
79 * master_repl_offset if no backlog exists nor slaves are attached. */
80 server.master_repl_offset++;
81
82 /* We don't have any data inside our buffer, but virtually the first
83 * byte we have is the next byte that will be generated for the
84 * replication stream. */
85 server.repl_backlog_off = server.master_repl_offset+1;
86}
87
88/* This function is called when the user modifies the replication backlog
89 * size at runtime. It is up to the function to both update the
90 * server.repl_backlog_size and to resize the buffer and setup it so that
91 * it contains the same data as the previous one (possibly less data, but
92 * the most recent bytes, or the same data and more free space in case the
93 * buffer is enlarged). */
94void resizeReplicationBacklog(long long newsize) {
95 if (newsize < REDIS_REPL_BACKLOG_MIN_SIZE)
96 newsize = REDIS_REPL_BACKLOG_MIN_SIZE;
97 if (server.repl_backlog_size == newsize) return;
98
99 server.repl_backlog_size = newsize;
100 if (server.repl_backlog != NULL) {
101 /* What we actually do is to flush the old buffer and realloc a new
102 * empty one. It will refill with new data incrementally.
103 * The reason is that copying a few gigabytes adds latency and even
104 * worse often we need to alloc additional space before freeing the
105 * old buffer. */
106 zfree(server.repl_backlog);
107 server.repl_backlog = zmalloc(server.repl_backlog_size);
108 server.repl_backlog_histlen = 0;
109 server.repl_backlog_idx = 0;
110 /* Next byte we have is... the next since the buffer is empty. */
111 server.repl_backlog_off = server.master_repl_offset+1;
112 }
113}

115void freeReplicationBacklog(void) {
116 redisAssert(listLength(server.slaves) == 0);
117 zfree(server.repl_backlog);
118 server.repl_backlog = NULL;
119}

121/* Add data to the replication backlog.
122 * This function also increments the global replication offset stored at
123 * server.master_repl_offset, because there is no case where we want to feed
124 * the backlog without incrementing the buffer. */
125void feedReplicationBacklog(void *ptr, size_t len) {
126 unsigned char *p = ptr;
127
128 server.master_repl_offset += len;
129
130 /* This is a circular buffer, so write as much data we can at every
131 * iteration and rewind the "idx" index if we reach the limit. */
132 while(len) {
133 size_t thislen = server.repl_backlog_size - server.repl_backlog_idx;
134 if (thislen > len) thislen = len;
135 memcpy(server.repl_backlog+server.repl_backlog_idx,p,thislen);
136 server.repl_backlog_idx += thislen;
137 if (server.repl_backlog_idx == server.repl_backlog_size)
138 server.repl_backlog_idx = 0;
139 len -= thislen;
140 p += thislen;
141 server.repl_backlog_histlen += thislen;
142 }
143 if (server.repl_backlog_histlen > server.repl_backlog_size)
144 server.repl_backlog_histlen = server.repl_backlog_size;
145 /* Set the offset of the first byte we have in the backlog. */
146 server.repl_backlog_off = server.master_repl_offset -
147 server.repl_backlog_histlen + 1;
148}

150/* Wrapper for feedReplicationBacklog() that takes Redis string objects
151 * as input. */
152void feedReplicationBacklogWithObject(robj *o) {
153 char llstr[REDIS_LONGSTR_SIZE];
154 void *p;
155 size_t len;
156
157 if (o->encoding == REDIS_ENCODING_INT) {
158 len = ll2string(llstr,sizeof(llstr),(long)o->ptr);
159 p = llstr;
160 } else {
161 len = sdslen(o->ptr);
162 p = o->ptr;
163 }
164 feedReplicationBacklog(p,len);
165}

167void replicationFeedSlaves(list *slaves, int dictid, robj **argv, int argc) {
168 listNode *ln;
169 listIter li;
170 int j, len;
171 char llstr[REDIS_LONGSTR_SIZE];
172
173 /* If there aren't slaves, and there is no backlog buffer to populate,
174 * we can return ASAP. */
175 if (server.repl_backlog == NULL && listLength(slaves) == 0) return;
176
177 /* We can't have slaves attached and no backlog. */
178 redisAssert(!(listLength(slaves) != 0 && server.repl_backlog == NULL));
179
180 /* Send SELECT command to every slave if needed. */
181 if (server.slaveseldb != dictid) {
182 robj *selectcmd;
183
184 /* For a few DBs we have pre-computed SELECT command. */
185 if (dictid >= 0 && dictid < REDIS_SHARED_SELECT_CMDS) {
186 selectcmd = shared.select[dictid];
187 } else {
188 int dictid_len;
189
190 dictid_len = ll2string(llstr,sizeof(llstr),dictid);
191 selectcmd = createObject(REDIS_STRING,
192 sdscatprintf(sdsempty(),
193 "*2\r\n$6\r\nSELECT\r\n$%d\r\n%s\r\n",
194 dictid_len, llstr));
195 }
196
197 /* Add the SELECT command into the backlog. */
198 if (server.repl_backlog) feedReplicationBacklogWithObject(selectcmd);
199
200 /* Send it to slaves. */
201 listRewind(slaves,&li);
202 while((ln = listNext(&li))) {
203 redisClient *slave = ln->value;
204 addReply(slave,selectcmd);
205 }
206
207 if (dictid < 0 || dictid >= REDIS_SHARED_SELECT_CMDS)
208 decrRefCount(selectcmd);
209 }
210 server.slaveseldb = dictid;
211
212 /* Write the command to the replication backlog if any. */
213 if (server.repl_backlog) {
214 char aux[REDIS_LONGSTR_SIZE+3];
215
216 /* Add the multi bulk reply length. */
217 aux[0] = '*';
218 len = ll2string(aux+1,sizeof(aux)-1,argc);
219 aux[len+1] = '\r';
220 aux[len+2] = '\n';
221 feedReplicationBacklog(aux,len+3);
222
223 for (j = 0; j < argc; j++) {
224 long objlen = stringObjectLen(argv[j]);
225
226 /* We need to feed the buffer with the object as a bulk reply
227 * not just as a plain string, so create the $..CRLF payload len
228 * and add the final CRLF */
229 aux[0] = '$';
230 len = ll2string(aux+1,sizeof(aux)-1,objlen);
231 aux[len+1] = '\r';
232 aux[len+2] = '\n';
233 feedReplicationBacklog(aux,len+3);
234 feedReplicationBacklogWithObject(argv[j]);
235 feedReplicationBacklog(aux+len+1,2);
236 }
237 }
238
239 /* Write the command to every slave. */
240 listRewind(server.slaves,&li);
241 while((ln = listNext(&li))) {
242 redisClient *slave = ln->value;
243
244 /* Don't feed slaves that are still waiting for BGSAVE to start */
245 if (slave->replstate == REDIS_REPL_WAIT_BGSAVE_START) continue;
246
247 /* Feed slaves that are waiting for the initial SYNC (so these commands
248 * are queued in the output buffer until the initial SYNC completes),
249 * or are already in sync with the master. */
250
251 /* Add the multi bulk length. */
252 addReplyMultiBulkLen(slave,argc);
253
254 /* Finally any additional argument that was not stored inside the
255 * static buffer if any (from j to argc). */
256 for (j = 0; j < argc; j++)
257 addReplyBulk(slave,argv[j]);
258 }
259}

261void replicationFeedMonitors(redisClient *c, list *monitors, int dictid, robj **argv, int argc) {
262 listNode *ln;
263 listIter li;
264 int j;
265 sds cmdrepr = sdsnew("+");
266 robj *cmdobj;
267 struct timeval tv;
268
269 gettimeofday(&tv,NULL);
270 cmdrepr = sdscatprintf(cmdrepr,"%ld.%06ld ",(long)tv.tv_sec,(long)tv.tv_usec);
271 if (c->flags & REDIS_LUA_CLIENT) {
272 cmdrepr = sdscatprintf(cmdrepr,"[%d lua] ",dictid);
273 } else if (c->flags & REDIS_UNIX_SOCKET) {
274 cmdrepr = sdscatprintf(cmdrepr,"[%d unix:%s] ",dictid,server.unixsocket);
275 } else {
276 cmdrepr = sdscatprintf(cmdrepr,"[%d %s] ",dictid,getClientPeerId(c));
277 }
278
279 for (j = 0; j < argc; j++) {
280 if (argv[j]->encoding == REDIS_ENCODING_INT) {
281 cmdrepr = sdscatprintf(cmdrepr, "\"%ld\"", (long)argv[j]->ptr);
282 } else {
283 cmdrepr = sdscatrepr(cmdrepr,(char*)argv[j]->ptr,
284 sdslen(argv[j]->ptr));
285 }
286 if (j != argc-1)
287 cmdrepr = sdscatlen(cmdrepr," ",1);
288 }
289 cmdrepr = sdscatlen(cmdrepr,"\r\n",2);
290 cmdobj = createObject(REDIS_STRING,cmdrepr);
291
292 listRewind(monitors,&li);
293 while((ln = listNext(&li))) {
294 redisClient *monitor = ln->value;
295 addReply(monitor,cmdobj);
296 }
297 decrRefCount(cmdobj);
298}

300/* Feed the slave 'c' with the replication backlog starting from the
301 * specified 'offset' up to the end of the backlog. */
302long long addReplyReplicationBacklog(redisClient *c, long long offset) {
303 long long j, skip, len;
304
305 redisLog(REDIS_DEBUG, "[PSYNC] Slave request offset: %lld", offset);
306
307 if (server.repl_backlog_histlen == 0) {
308 redisLog(REDIS_DEBUG, "[PSYNC] Backlog history len is zero");
309 return 0;
310 }
311
312 redisLog(REDIS_DEBUG, "[PSYNC] Backlog size: %lld",
313 server.repl_backlog_size);
314 redisLog(REDIS_DEBUG, "[PSYNC] First byte: %lld",
315 server.repl_backlog_off);
316 redisLog(REDIS_DEBUG, "[PSYNC] History len: %lld",
317 server.repl_backlog_histlen);
318 redisLog(REDIS_DEBUG, "[PSYNC] Current index: %lld",
319 server.repl_backlog_idx);
320
321 /* Compute the amount of bytes we need to discard. */
322 skip = offset - server.repl_backlog_off;
323 redisLog(REDIS_DEBUG, "[PSYNC] Skipping: %lld", skip);
324
325 /* Point j to the oldest byte, that is actaully our
326 * server.repl_backlog_off byte. */
327 j = (server.repl_backlog_idx +
328 (server.repl_backlog_size-server.repl_backlog_histlen)) %
329 server.repl_backlog_size;
330 redisLog(REDIS_DEBUG, "[PSYNC] Index of first byte: %lld", j);
331
332 /* Discard the amount of data to seek to the specified 'offset'. */
333 j = (j + skip) % server.repl_backlog_size;
334
335 /* Feed slave with data. Since it is a circular buffer we have to
336 * split the reply in two parts if we are cross-boundary. */
337 len = server.repl_backlog_histlen - skip;
338 redisLog(REDIS_DEBUG, "[PSYNC] Reply total length: %lld", len);
339 while(len) {
340 long long thislen =
341 ((server.repl_backlog_size - j) < len) ?
342 (server.repl_backlog_size - j) : len;
343
344 redisLog(REDIS_DEBUG, "[PSYNC] addReply() length: %lld", thislen);
345 addReplySds(c,sdsnewlen(server.repl_backlog + j, thislen));
346 len -= thislen;
347 j = 0;
348 }
349 return server.repl_backlog_histlen - skip;
350}

352/* This function handles the PSYNC command from the point of view of a
353 * master receiving a request for partial resynchronization.
354 *
355 * On success return REDIS_OK, otherwise REDIS_ERR is returned and we proceed
356 * with the usual full resync. */
357int masterTryPartialResynchronization(redisClient *c) {
358 long long psync_offset, psync_len;
359 char *master_runid = c->argv[1]->ptr;
360 char buf[128];
361 int buflen;
362
363 /* Is the runid of this master the same advertised by the wannabe slave
364 * via PSYNC? If runid changed this master is a different instance and
365 * there is no way to continue. */
366 if (strcasecmp(master_runid, server.runid)) {
367 /* Run id "?" is used by slaves that want to force a full resync. */
368 if (master_runid[0] != '?') {
369 redisLog(REDIS_NOTICE,"Partial resynchronization not accepted: "
370 "Runid mismatch (Client asked for runid '%s', my runid is '%s')",
371 master_runid, server.runid);
372 } else {
373 redisLog(REDIS_NOTICE,"Full resync requested by slave %s",
374 replicationGetSlaveName(c));
375 }
376 goto need_full_resync;
377 }
378
379 /* We still have the data our slave is asking for? */
380 if (getLongLongFromObjectOrReply(c,c->argv[2],&psync_offset,NULL) !=
381 REDIS_OK) goto need_full_resync;
382 if (!server.repl_backlog ||
383 psync_offset < server.repl_backlog_off ||
384 psync_offset > (server.repl_backlog_off + server.repl_backlog_histlen))
385 {
386 redisLog(REDIS_NOTICE,
387 "Unable to partial resync with slave %s for lack of backlog (Slave request was: %lld).", replicationGetSlaveName(c), psync_offset);
388 if (psync_offset > server.master_repl_offset) {
389 redisLog(REDIS_WARNING,
390 "Warning: slave %s tried to PSYNC with an offset that is greater than the master replication offset.", replicationGetSlaveName(c));
391 }
392 goto need_full_resync;
393 }
394
395 /* If we reached this point, we are able to perform a partial resync:
396 * 1) Set client state to make it a slave.
397 * 2) Inform the client we can continue with +CONTINUE
398 * 3) Send the backlog data (from the offset to the end) to the slave. */
399 c->flags |= REDIS_SLAVE;
400 c->replstate = REDIS_REPL_ONLINE;
401 c->repl_ack_time = server.unixtime;
402 c->repl_put_online_on_ack = 0;
403 listAddNodeTail(server.slaves,c);
404 /* We can't use the connection buffers since they are used to accumulate
405 * new commands at this stage. But we are sure the socket send buffer is
406 * empty so this write will never fail actually. */
407 buflen = snprintf(buf,sizeof(buf),"+CONTINUE\r\n");
408 if (write(c->fd,buf,buflen) != buflen) {
409 freeClientAsync(c);
410 return REDIS_OK;
411 }
412 psync_len = addReplyReplicationBacklog(c,psync_offset);
413 redisLog(REDIS_NOTICE,
414 "Partial resynchronization request from %s accepted. Sending %lld bytes of backlog starting from offset %lld.",
415 replicationGetSlaveName(c),
416 psync_len, psync_offset);
417 /* Note that we don't need to set the selected DB at server.slaveseldb
418 * to -1 to force the master to emit SELECT, since the slave already
419 * has this state from the previous connection with the master. */
420
421 refreshGoodSlavesCount();
422 return REDIS_OK; /* The caller can return, no full resync needed. */
423
424need_full_resync:
425 /* We need a full resync for some reason... notify the client. */
426 psync_offset = server.master_repl_offset;
427 /* Add 1 to psync_offset if it the replication backlog does not exists
428 * as when it will be created later we'll increment the offset by one. */
429 if (server.repl_backlog == NULL) psync_offset++;
430 /* Again, we can't use the connection buffers (see above). */
431 buflen = snprintf(buf,sizeof(buf),"+FULLRESYNC %s %lld\r\n",
432 server.runid,psync_offset);
433 if (write(c->fd,buf,buflen) != buflen) {
434 freeClientAsync(c);
435 return REDIS_OK;
436 }
437 return REDIS_ERR;
438}

440/* Start a BGSAVE for replication goals, which is, selecting the disk or
441 * socket target depending on the configuration, and making sure that
442 * the script cache is flushed before to start.
443 *
444 * Returns REDIS_OK on success or REDIS_ERR otherwise. */
445int startBgsaveForReplication(void) {
446 int retval;
447
448 redisLog(REDIS_NOTICE,"Starting BGSAVE for SYNC with target: %s",
449 server.repl_diskless_sync ? "slaves sockets" : "disk");
450
451 if (server.repl_diskless_sync)
452 retval = rdbSaveToSlavesSockets();
453 else
454 retval = rdbSaveBackground(server.rdb_filename);
455
456 /* Flush the script cache, since we need that slave differences are
457 * accumulated without requiring slaves to match our cached scripts. */
458 if (retval == REDIS_OK) replicationScriptCacheFlush();
459 return retval;
460}

462/* SYNC and PSYNC command implemenation. */
463void syncCommand(redisClient *c) {
464 /* ignore SYNC if already slave or in monitor mode */
465 if (c->flags & REDIS_SLAVE) return;
466
467 /* Refuse SYNC requests if we are a slave but the link with our master
468 * is not ok... */
469 if (server.masterhost && server.repl_state != REDIS_REPL_CONNECTED) {
470 addReplyError(c,"Can't SYNC while not connected with my master");
471 return;
472 }
473
474 /* SYNC can't be issued when the server has pending data to send to
475 * the client about already issued commands. We need a fresh reply
476 * buffer registering the differences between the BGSAVE and the current
477 * dataset, so that we can copy to other slaves if needed. */
478 if (listLength(c->reply) != 0 || c->bufpos != 0) {
479 addReplyError(c,"SYNC and PSYNC are invalid with pending output");
480 return;
481 }
482
483 redisLog(REDIS_NOTICE,"Slave %s asks for synchronization",
484 replicationGetSlaveName(c));
485
486 /* Try a partial resynchronization if this is a PSYNC command.
487 * If it fails, we continue with usual full resynchronization, however
488 * when this happens masterTryPartialResynchronization() already
489 * replied with:
490 *
491 * +FULLRESYNC <runid> <offset>
492 *
493 * So the slave knows the new runid and offset to try a PSYNC later
494 * if the connection with the master is lost. */
495 if (!strcasecmp(c->argv[0]->ptr,"psync")) {
496 if (masterTryPartialResynchronization(c) == REDIS_OK) {
497 server.stat_sync_partial_ok++;
498 return; /* No full resync needed, return. */
499 } else {
500 char *master_runid = c->argv[1]->ptr;
501
502 /* Increment stats for failed PSYNCs, but only if the
503 * runid is not "?", as this is used by slaves to force a full
504 * resync on purpose when they are not albe to partially
505 * resync. */
506 if (master_runid[0] != '?') server.stat_sync_partial_err++;
507 }
508 } else {
509 /* If a slave uses SYNC, we are dealing with an old implementation
510 * of the replication protocol (like redis-cli --slave). Flag the client
511 * so that we don't expect to receive REPLCONF ACK feedbacks. */
512 c->flags |= REDIS_PRE_PSYNC;
513 }
514
515 /* Full resynchronization. */
516 server.stat_sync_full++;
517
518 /* Here we need to check if there is a background saving operation
519 * in progress, or if it is required to start one */
520 if (server.rdb_child_pid != -1 &&
521 server.rdb_child_type == REDIS_RDB_CHILD_TYPE_DISK)
522 {
523 /* Ok a background save is in progress. Let's check if it is a good
524 * one for replication, i.e. if there is another slave that is
525 * registering differences since the server forked to save. */
526 redisClient *slave;
527 listNode *ln;
528 listIter li;
529
530 listRewind(server.slaves,&li);
531 while((ln = listNext(&li))) {
532 slave = ln->value;
533 if (slave->replstate == REDIS_REPL_WAIT_BGSAVE_END) break;
534 }
535 if (ln) {
536 /* Perfect, the server is already registering differences for
537 * another slave. Set the right state, and copy the buffer. */
538 copyClientOutputBuffer(c,slave);
539 c->replstate = REDIS_REPL_WAIT_BGSAVE_END;
540 redisLog(REDIS_NOTICE,"Waiting for end of BGSAVE for SYNC");
541 } else {
542 /* No way, we need to wait for the next BGSAVE in order to
543 * register differences. */
544 c->replstate = REDIS_REPL_WAIT_BGSAVE_START;
545 redisLog(REDIS_NOTICE,"Waiting for next BGSAVE for SYNC");
546 }
547 } else if (server.rdb_child_pid != -1 &&
548 server.rdb_child_type == REDIS_RDB_CHILD_TYPE_SOCKET)
549 {
550 /* There is an RDB child process but it is writing directly to
551 * children sockets. We need to wait for the next BGSAVE
552 * in order to synchronize. */
553 c->replstate = REDIS_REPL_WAIT_BGSAVE_START;
554 redisLog(REDIS_NOTICE,"Waiting for next BGSAVE for SYNC");
555 } else {
556 if (server.repl_diskless_sync) {
557 /* Diskless replication RDB child is created inside
558 * replicationCron() since we want to delay its start a
559 * few seconds to wait for more slaves to arrive. */
560 c->replstate = REDIS_REPL_WAIT_BGSAVE_START;
561 if (server.repl_diskless_sync_delay)
562 redisLog(REDIS_NOTICE,"Delay next BGSAVE for SYNC");
563 } else {
564 /* Ok we don't have a BGSAVE in progress, let's start one. */
565 if (startBgsaveForReplication() != REDIS_OK) {
566 redisLog(REDIS_NOTICE,"Replication failed, can't BGSAVE");
567 addReplyError(c,"Unable to perform background save");
568 return;
569 }
570 c->replstate = REDIS_REPL_WAIT_BGSAVE_END;
571 }
572 }
573
574 if (server.repl_disable_tcp_nodelay)
575 anetDisableTcpNoDelay(NULL, c->fd); /* Non critical if it fails. */
576 c->repldbfd = -1;
577 c->flags |= REDIS_SLAVE;
578 server.slaveseldb = -1; /* Force to re-emit the SELECT command. */
579 listAddNodeTail(server.slaves,c);
580 if (listLength(server.slaves) == 1 && server.repl_backlog == NULL)
581 createReplicationBacklog();
582 return;
583}

585/* REPLCONF <option> <value> <option> <value> ...
586 * This command is used by a slave in order to configure the replication
587 * process before starting it with the SYNC command.
588 *
589 * Currently the only use of this command is to communicate to the master
590 * what is the listening port of the Slave redis instance, so that the
591 * master can accurately list slaves and their listening ports in
592 * the INFO output.
593 *
594 * In the future the same command can be used in order to configure
595 * the replication to initiate an incremental replication instead of a
596 * full resync. */
597void replconfCommand(redisClient *c) {
598 int j;
599
600 if ((c->argc % 2) == 0) {
601 /* Number of arguments must be odd to make sure that every
602 * option has a corresponding value. */
603 addReply(c,shared.syntaxerr);
604 return;
605 }
606
607 /* Process every option-value pair. */
608 for (j = 1; j < c->argc; j+=2) {
609 if (!strcasecmp(c->argv[j]->ptr,"listening-port")) {
610 long port;
611
612 if ((getLongFromObjectOrReply(c,c->argv[j+1],
613 &port,NULL) != REDIS_OK))
614 return;
615 c->slave_listening_port = port;
616 } else if (!strcasecmp(c->argv[j]->ptr,"ack")) {
617 /* REPLCONF ACK is used by slave to inform the master the amount
618 * of replication stream that it processed so far. It is an
619 * internal only command that normal clients should never use. */
620 long long offset;
621
622 if (!(c->flags & REDIS_SLAVE)) return;
623 if ((getLongLongFromObject(c->argv[j+1], &offset) != REDIS_OK))
624 return;
625 if (offset > c->repl_ack_off)
626 c->repl_ack_off = offset;
627 c->repl_ack_time = server.unixtime;
628 /* If this was a diskless replication, we need to really put
629 * the slave online when the first ACK is received (which
630 * confirms slave is online and ready to get more data). */
631 if (c->repl_put_online_on_ack && c->replstate == REDIS_REPL_ONLINE)
632 putSlaveOnline(c);
633 /* Note: this command does not reply anything! */
634 return;
635 } else if (!strcasecmp(c->argv[j]->ptr,"getack")) {
636 /* REPLCONF GETACK is used in order to request an ACK ASAP
637 * to the slave. */
638 if (server.masterhost && server.master) replicationSendAck();
639 /* Note: this command does not reply anything! */
640 } else {
641 addReplyErrorFormat(c,"Unrecognized REPLCONF option: %s",
642 (char*)c->argv[j]->ptr);
643 return;
644 }
645 }
646 addReply(c,shared.ok);
647}

649/* This function puts a slave in the online state, and should be called just
650 * after a slave received the RDB file for the initial synchronization, and
651 * we are finally ready to send the incremental stream of commands.
652 *
653 * It does a few things:
654 *
655 * 1) Put the slave in ONLINE state (useless when the function is called
656 * because state is already ONLINE but repl_put_online_on_ack is true).
657 * 2) Make sure the writable event is re-installed, since calling the SYNC
658 * command disables it, so that we can accumulate output buffer without
659 * sending it to the slave.
660 * 3) Update the count of good slaves. */
661void putSlaveOnline(redisClient *slave) {
662 slave->replstate = REDIS_REPL_ONLINE;
663 slave->repl_put_online_on_ack = 0;
664 slave->repl_ack_time = server.unixtime; /* Prevent false timeout. */
665 if (aeCreateFileEvent(server.el, slave->fd, AE_WRITABLE,
666 sendReplyToClient, slave) == AE_ERR) {
667 redisLog(REDIS_WARNING,"Unable to register writable event for slave bulk transfer: %s", strerror(errno));
668 freeClient(slave);
669 return;
670 }
671 refreshGoodSlavesCount();
672 redisLog(REDIS_NOTICE,"Synchronization with slave %s succeeded",
673 replicationGetSlaveName(slave));
674}

676void sendBulkToSlave(aeEventLoop *el, int fd, void *privdata, int mask) {
677 redisClient *slave = privdata;
678 REDIS_NOTUSED(el);
679 REDIS_NOTUSED(mask);
680 char buf[REDIS_IOBUF_LEN];
681 ssize_t nwritten, buflen;
682
683 /* Before sending the RDB file, we send the preamble as configured by the
684 * replication process. Currently the preamble is just the bulk count of
685 * the file in the form "$<length>\r\n". */
686 if (slave->replpreamble) {
687 nwritten = write(fd,slave->replpreamble,sdslen(slave->replpreamble));
688 if (nwritten == -1) {
689 redisLog(REDIS_VERBOSE,"Write error sending RDB preamble to slave: %s",
690 strerror(errno));
691 freeClient(slave);
692 return;
693 }
694 server.stat_net_output_bytes += nwritten;
695 sdsrange(slave->replpreamble,nwritten,-1);
696 if (sdslen(slave->replpreamble) == 0) {
697 sdsfree(slave->replpreamble);
698 slave->replpreamble = NULL;
699 /* fall through sending data. */
700 } else {
701 return;
702 }
703 }
704
705 /* If the preamble was already transfered, send the RDB bulk data. */
706 lseek(slave->repldbfd,slave->repldboff,SEEK_SET);
707 buflen = read(slave->repldbfd,buf,REDIS_IOBUF_LEN);
708 if (buflen <= 0) {
709 redisLog(REDIS_WARNING,"Read error sending DB to slave: %s",
710 (buflen == 0) ? "premature EOF" : strerror(errno));
711 freeClient(slave);
712 return;
713 }
714 if ((nwritten = write(fd,buf,buflen)) == -1) {
715 if (errno != EAGAIN) {
716 redisLog(REDIS_WARNING,"Write error sending DB to slave: %s",
717 strerror(errno));
718 freeClient(slave);
719 }
720 return;
721 }
722 slave->repldboff += nwritten;
723 server.stat_net_output_bytes += nwritten;
724 if (slave->repldboff == slave->repldbsize) {
725 close(slave->repldbfd);
726 slave->repldbfd = -1;
727 aeDeleteFileEvent(server.el,slave->fd,AE_WRITABLE);
728 putSlaveOnline(slave);
729 }
730}

732/* This function is called at the end of every background saving,
733 * or when the replication RDB transfer strategy is modified from
734 * disk to socket or the other way around.
735 *
736 * The goal of this function is to handle slaves waiting for a successful
737 * background saving in order to perform non-blocking synchronization, and
738 * to schedule a new BGSAVE if there are slaves that attached while a
739 * BGSAVE was in progress, but it was not a good one for replication (no
740 * other slave was accumulating differences).
741 *
742 * The argument bgsaveerr is REDIS_OK if the background saving succeeded
743 * otherwise REDIS_ERR is passed to the function.
744 * The 'type' argument is the type of the child that terminated
745 * (if it had a disk or socket target). */
746void updateSlavesWaitingBgsave(int bgsaveerr, int type) {
747 listNode *ln;
748 int startbgsave = 0;
749 listIter li;
750
751 listRewind(server.slaves,&li);
752 while((ln = listNext(&li))) {
753 redisClient *slave = ln->value;
754
755 if (slave->replstate == REDIS_REPL_WAIT_BGSAVE_START) {
756 startbgsave = 1;
757 slave->replstate = REDIS_REPL_WAIT_BGSAVE_END;
758 } else if (slave->replstate == REDIS_REPL_WAIT_BGSAVE_END) {
759 struct redis_stat buf;
760
761 /* If this was an RDB on disk save, we have to prepare to send
762 * the RDB from disk to the slave socket. Otherwise if this was
763 * already an RDB -> Slaves socket transfer, used in the case of
764 * diskless replication, our work is trivial, we can just put
765 * the slave online. */
766 if (type == REDIS_RDB_CHILD_TYPE_SOCKET) {
767 redisLog(REDIS_NOTICE,
768 "Streamed RDB transfer with slave %s succeeded (socket). Waiting for REPLCONF ACK from slave to enable streaming",
769 replicationGetSlaveName(slave));
770 /* Note: we wait for a REPLCONF ACK message from slave in
771 * order to really put it online (install the write handler
772 * so that the accumulated data can be transfered). However
773 * we change the replication state ASAP, since our slave
774 * is technically online now. */
775 slave->replstate = REDIS_REPL_ONLINE;
776 slave->repl_put_online_on_ack = 1;
777 slave->repl_ack_time = server.unixtime; /* Timeout otherwise. */
778 } else {
779 if (bgsaveerr != REDIS_OK) {
780 freeClient(slave);
781 redisLog(REDIS_WARNING,"SYNC failed. BGSAVE child returned an error");
782 continue;
783 }
784 if ((slave->repldbfd = open(server.rdb_filename,O_RDONLY)) == -1 ||
785 redis_fstat(slave->repldbfd,&buf) == -1) {
786 freeClient(slave);
787 redisLog(REDIS_WARNING,"SYNC failed. Can't open/stat DB after BGSAVE: %s", strerror(errno));
788 continue;
789 }
790 slave->repldboff = 0;
791 slave->repldbsize = buf.st_size;
792 slave->replstate = REDIS_REPL_SEND_BULK;
793 slave->replpreamble = sdscatprintf(sdsempty(),"$%lld\r\n",
794 (unsigned long long) slave->repldbsize);
795
796 aeDeleteFileEvent(server.el,slave->fd,AE_WRITABLE);
797 if (aeCreateFileEvent(server.el, slave->fd, AE_WRITABLE, sendBulkToSlave, slave) == AE_ERR) {
798 freeClient(slave);
799 continue;
800 }
801 }
802 }
803 }
804 if (startbgsave) {
805 if (startBgsaveForReplication() != REDIS_OK) {
806 listIter li;
807
808 listRewind(server.slaves,&li);
809 redisLog(REDIS_WARNING,"SYNC failed. BGSAVE failed");
810 while((ln = listNext(&li))) {
811 redisClient *slave = ln->value;
812
813 if (slave->replstate == REDIS_REPL_WAIT_BGSAVE_START)
814 freeClient(slave);
815 }
816 }
817 }
818}

859/* Asynchronously read the SYNC payload we receive from a master */
860#define REPL_MAX_WRITTEN_BEFORE_FSYNC (1024*1024*8) /* 8 MB */
861void readSyncBulkPayload(aeEventLoop *el, int fd, void *privdata, int mask) {
862 char buf[4096];
863 ssize_t nread, readlen;
864 off_t left;
865 REDIS_NOTUSED(el);
866 REDIS_NOTUSED(privdata);
867 REDIS_NOTUSED(mask);
868
869 /* Static vars used to hold the EOF mark, and the last bytes received
870 * form the server: when they match, we reached the end of the transfer. */
871 static char eofmark[REDIS_RUN_ID_SIZE];
872 static char lastbytes[REDIS_RUN_ID_SIZE];
873 static int usemark = 0;
874
875 /* If repl_transfer_size == -1 we still have to read the bulk length
876 * from the master reply. */
877 if (server.repl_transfer_size == -1) {
878 if (syncReadLine(fd,buf,1024,server.repl_syncio_timeout*1000) == -1) {
879 redisLog(REDIS_WARNING,
880 "I/O error reading bulk count from MASTER: %s",
881 strerror(errno));
882 goto error;
883 }
884
885 if (buf[0] == '-') {
886 redisLog(REDIS_WARNING,
887 "MASTER aborted replication with an error: %s",
888 buf+1);
889 goto error;
890 } else if (buf[0] == '\0') {
891 /* At this stage just a newline works as a PING in order to take
892 * the connection live. So we refresh our last interaction
893 * timestamp. */
894 server.repl_transfer_lastio = server.unixtime;
895 return;
896 } else if (buf[0] != '$') {
897 redisLog(REDIS_WARNING,"Bad protocol from MASTER, the first byte is not '$' (we received '%s'), are you sure the host and port are right?", buf);
898 goto error;
899 }
900
901 /* There are two possible forms for the bulk payload. One is the
902 * usual $<count> bulk format. The other is used for diskless transfers
903 * when the master does not know beforehand the size of the file to
904 * transfer. In the latter case, the following format is used:
905 *
906 * $EOF:<40 bytes delimiter>
907 *
908 * At the end of the file the announced delimiter is transmitted. The
909 * delimiter is long and random enough that the probability of a
910 * collision with the actual file content can be ignored. */
911 if (strncmp(buf+1,"EOF:",4) == 0 && strlen(buf+5) >= REDIS_RUN_ID_SIZE) {
912 usemark = 1;
913 memcpy(eofmark,buf+5,REDIS_RUN_ID_SIZE);
914 memset(lastbytes,0,REDIS_RUN_ID_SIZE);
915 /* Set any repl_transfer_size to avoid entering this code path
916 * at the next call. */
917 server.repl_transfer_size = 0;
918 redisLog(REDIS_NOTICE,
919 "MASTER <-> SLAVE sync: receiving streamed RDB from master");
920 } else {
921 usemark = 0;
922 server.repl_transfer_size = strtol(buf+1,NULL,10);
923 redisLog(REDIS_NOTICE,
924 "MASTER <-> SLAVE sync: receiving %lld bytes from master",
925 (long long) server.repl_transfer_size);
926 }
927 return;
928 }
929
930 /* Read bulk data */
931 if (usemark) {
932 readlen = sizeof(buf);
933 } else {
934 left = server.repl_transfer_size - server.repl_transfer_read;
935 readlen = (left < (signed)sizeof(buf)) ? left : (signed)sizeof(buf);
936 }
937
938 nread = read(fd,buf,readlen);
939 if (nread <= 0) {
940 redisLog(REDIS_WARNING,"I/O error trying to sync with MASTER: %s",
941 (nread == -1) ? strerror(errno) : "connection lost");
942 replicationAbortSyncTransfer();
943 return;
944 }
945 server.stat_net_input_bytes += nread;
946
947 /* When a mark is used, we want to detect EOF asap in order to avoid
948 * writing the EOF mark into the file... */
949 int eof_reached = 0;
950
951 if (usemark) {
952 /* Update the last bytes array, and check if it matches our delimiter.*/
953 if (nread >= REDIS_RUN_ID_SIZE) {
954 memcpy(lastbytes,buf+nread-REDIS_RUN_ID_SIZE,REDIS_RUN_ID_SIZE);
955 } else {
956 int rem = REDIS_RUN_ID_SIZE-nread;
957 memmove(lastbytes,lastbytes+nread,rem);
958 memcpy(lastbytes+rem,buf,nread);
959 }
960 if (memcmp(lastbytes,eofmark,REDIS_RUN_ID_SIZE) == 0) eof_reached = 1;
961 }
962
963 server.repl_transfer_lastio = server.unixtime;
964 if (write(server.repl_transfer_fd,buf,nread) != nread) {
965 redisLog(REDIS_WARNING,"Write error or short write writing to the DB dump file needed for MASTER <-> SLAVE synchronization: %s", strerror(errno));
966 goto error;
967 }
968 server.repl_transfer_read += nread;
969
970 /* Delete the last 40 bytes from the file if we reached EOF. */
971 if (usemark && eof_reached) {
972 if (ftruncate(server.repl_transfer_fd,
973 server.repl_transfer_read - REDIS_RUN_ID_SIZE) == -1)
974 {
975 redisLog(REDIS_WARNING,"Error truncating the RDB file received from the master for SYNC: %s", strerror(errno));
976 goto error;
977 }
978 }
979
980 /* Sync data on disk from time to time, otherwise at the end of the transfer
981 * we may suffer a big delay as the memory buffers are copied into the
982 * actual disk. */
983 if (server.repl_transfer_read >=
984 server.repl_transfer_last_fsync_off + REPL_MAX_WRITTEN_BEFORE_FSYNC)
985 {
986 off_t sync_size = server.repl_transfer_read -
987 server.repl_transfer_last_fsync_off;
988 rdb_fsync_range(server.repl_transfer_fd,
989 server.repl_transfer_last_fsync_off, sync_size);
990 server.repl_transfer_last_fsync_off += sync_size;
991 }
992
993 /* Check if the transfer is now complete */
994 if (!usemark) {
995 if (server.repl_transfer_read == server.repl_transfer_size)
996 eof_reached = 1;
997 }
998
999 if (eof_reached) {
1000 if (rename(server.repl_transfer_tmpfile,server.rdb_filename) == -1) {
1001 redisLog(REDIS_WARNING,"Failed trying to rename the temp DB into dump.rdb in MASTER <-> SLAVE synchronization: %s", strerror(errno));
1002 replicationAbortSyncTransfer();
1003 return;
1004 }
1005 redisLog(REDIS_NOTICE, "MASTER <-> SLAVE sync: Flushing old data");
1006 signalFlushedDb(-1);
1007 emptyDb(replicationEmptyDbCallback);
1008 /* Before loading the DB into memory we need to delete the readable
1009 * handler, otherwise it will get called recursively since
1010 * rdbLoad() will call the event loop to process events from time to
1011 * time for non blocking loading. */
1012 aeDeleteFileEvent(server.el,server.repl_transfer_s,AE_READABLE);
1013 redisLog(REDIS_NOTICE, "MASTER <-> SLAVE sync: Loading DB in memory");
1014 if (rdbLoad(server.rdb_filename) != REDIS_OK) {
1015 redisLog(REDIS_WARNING,"Failed trying to load the MASTER synchronization DB from disk");
1016 replicationAbortSyncTransfer();
1017 return;
1018 }
1019 /* Final setup of the connected slave <- master link */
1020 zfree(server.repl_transfer_tmpfile);
1021 close(server.repl_transfer_fd);
1022 server.master = createClient(server.repl_transfer_s);
1023 server.master->flags |= REDIS_MASTER;
1024 server.master->authenticated = 1;
1025 server.repl_state = REDIS_REPL_CONNECTED;
1026 server.master->reploff = server.repl_master_initial_offset;
1027 memcpy(server.master->replrunid, server.repl_master_runid,
1028 sizeof(server.repl_master_runid));
1029 /* If master offset is set to -1, this master is old and is not
1030 * PSYNC capable, so we flag it accordingly. */
1031 if (server.master->reploff == -1)
1032 server.master->flags |= REDIS_PRE_PSYNC;
1033 redisLog(REDIS_NOTICE, "MASTER <-> SLAVE sync: Finished with success");
1034 /* Restart the AOF subsystem now that we finished the sync. This
1035 * will trigger an AOF rewrite, and when done will start appending
1036 * to the new file. */
1037 if (server.aof_state != REDIS_AOF_OFF) {
1038 int retry = 10;
1039
1040 stopAppendOnly();
1041 while (retry-- && startAppendOnly() == REDIS_ERR) {
1042 redisLog(REDIS_WARNING,"Failed enabling the AOF after successful master synchronization! Trying it again in one second.");
1043 sleep(1);
1044 }
1045 if (!retry) {
1046 redisLog(REDIS_WARNING,"FATAL: this slave instance finished the synchronization with its master, but the AOF can't be turned on. Exiting now.");
1047 exit(1);
1048 }
1049 }
1050 }
1051
1052 return;
1053
1054error:
1055 replicationAbortSyncTransfer();
1056 return;
1057}

1059/* Send a synchronous command to the master. Used to send AUTH and
1060 * REPLCONF commands before starting the replication with SYNC.
1061 *
1062 * The command returns an sds string representing the result of the
1063 * operation. On error the first byte is a "-".
1064 */
1065char *sendSynchronousCommand(int fd, ...) {
1066 va_list ap;
1067 sds cmd = sdsempty();
1068 char *arg, buf[256];
1069
1070 /* Create the command to send to the master, we use simple inline
1071 * protocol for simplicity as currently we only send simple strings. */
1072 va_start(ap,fd);
1073 while(1) {
1074 arg = va_arg(ap, char*);
1075 if (arg == NULL) break;
1076
1077 if (sdslen(cmd) != 0) cmd = sdscatlen(cmd," ",1);
1078 cmd = sdscat(cmd,arg);
1079 }
1080 cmd = sdscatlen(cmd,"\r\n",2);
1081
1082 /* Transfer command to the server. */
1083 if (syncWrite(fd,cmd,sdslen(cmd),server.repl_syncio_timeout*1000) == -1) {
1084 sdsfree(cmd);
1085 return sdscatprintf(sdsempty(),"-Writing to master: %s",
1086 strerror(errno));
1087 }
1088 sdsfree(cmd);
1089
1090 /* Read the reply from the server. */
1091 if (syncReadLine(fd,buf,sizeof(buf),server.repl_syncio_timeout*1000) == -1)
1092 {
1093 return sdscatprintf(sdsempty(),"-Reading from master: %s",
1094 strerror(errno));
1095 }
1096 return sdsnew(buf);
1097}

1099/* Try a partial resynchronization with the master if we are about to reconnect.
1100 * If there is no cached master structure, at least try to issue a
1101 * "PSYNC ? -1" command in order to trigger a full resync using the PSYNC
1102 * command in order to obtain the master run id and the master replication
1103 * global offset.
1104 *
1105 * This function is designed to be called from syncWithMaster(), so the
1106 * following assumptions are made:
1107 *
1108 * 1) We pass the function an already connected socket "fd".
1109 * 2) This function does not close the file descriptor "fd". However in case
1110 * of successful partial resynchronization, the function will reuse
1111 * 'fd' as file descriptor of the server.master client structure.
1112 *
1113 * The function returns:
1114 *
1115 * PSYNC_CONTINUE: If the PSYNC command succeded and we can continue.
1116 * PSYNC_FULLRESYNC: If PSYNC is supported but a full resync is needed.
1117 * In this case the master run_id and global replication
1118 * offset is saved.
1119 * PSYNC_NOT_SUPPORTED: If the server does not understand PSYNC at all and
1120 * the caller should fall back to SYNC.
1121 */
1122
1123#define PSYNC_CONTINUE 0
1124#define PSYNC_FULLRESYNC 1
1125#define PSYNC_NOT_SUPPORTED 2
1126int slaveTryPartialResynchronization(int fd) {
1127 char *psync_runid;
1128 char psync_offset[32];
1129 sds reply;
1130
1131 /* Initially set repl_master_initial_offset to -1 to mark the current
1132 * master run_id and offset as not valid. Later if we'll be able to do
1133 * a FULL resync using the PSYNC command we'll set the offset at the
1134 * right value, so that this information will be propagated to the
1135 * client structure representing the master into server.master. */
1136 server.repl_master_initial_offset = -1;
1137
1138 if (server.cached_master) {
1139 psync_runid = server.cached_master->replrunid;
1140 snprintf(psync_offset,sizeof(psync_offset),"%lld", server.cached_master->reploff+1);
1141 redisLog(REDIS_NOTICE,"Trying a partial resynchronization (request %s:%s).", psync_runid, psync_offset);
1142 } else {
1143 redisLog(REDIS_NOTICE,"Partial resynchronization not possible (no cached master)");
1144 psync_runid = "?";
1145 memcpy(psync_offset,"-1",3);
1146 }
1147
1148 /* Issue the PSYNC command */
1149 reply = sendSynchronousCommand(fd,"PSYNC",psync_runid,psync_offset,NULL);
1150
1151 if (!strncmp(reply,"+FULLRESYNC",11)) {
1152 char *runid = NULL, *offset = NULL;
1153
1154 /* FULL RESYNC, parse the reply in order to extract the run id
1155 * and the replication offset. */
1156 runid = strchr(reply,' ');
1157 if (runid) {
1158 runid++;
1159 offset = strchr(runid,' ');
1160 if (offset) offset++;
1161 }
1162 if (!runid || !offset || (offset-runid-1) != REDIS_RUN_ID_SIZE) {
1163 redisLog(REDIS_WARNING,
1164 "Master replied with wrong +FULLRESYNC syntax.");
1165 /* This is an unexpected condition, actually the +FULLRESYNC
1166 * reply means that the master supports PSYNC, but the reply
1167 * format seems wrong. To stay safe we blank the master
1168 * runid to make sure next PSYNCs will fail. */
1169 memset(server.repl_master_runid,0,REDIS_RUN_ID_SIZE+1);
1170 } else {
1171 memcpy(server.repl_master_runid, runid, offset-runid-1);
1172 server.repl_master_runid[REDIS_RUN_ID_SIZE] = '\0';
1173 server.repl_master_initial_offset = strtoll(offset,NULL,10);
1174 redisLog(REDIS_NOTICE,"Full resync from master: %s:%lld",
1175 server.repl_master_runid,
1176 server.repl_master_initial_offset);
1177 }
1178 /* We are going to full resync, discard the cached master structure. */
1179 replicationDiscardCachedMaster();
1180 sdsfree(reply);
1181 return PSYNC_FULLRESYNC;
1182 }
1183
1184 if (!strncmp(reply,"+CONTINUE",9)) {
1185 /* Partial resync was accepted, set the replication state accordingly */
1186 redisLog(REDIS_NOTICE,
1187 "Successful partial resynchronization with master.");
1188 sdsfree(reply);
1189 replicationResurrectCachedMaster(fd);
1190 return PSYNC_CONTINUE;
1191 }
1192
1193 /* If we reach this point we receied either an error since the master does
1194 * not understand PSYNC, or an unexpected reply from the master.
1195 * Return PSYNC_NOT_SUPPORTED to the caller in both cases. */
1196
1197 if (strncmp(reply,"-ERR",4)) {
1198 /* If it's not an error, log the unexpected event. */
1199 redisLog(REDIS_WARNING,
1200 "Unexpected reply to PSYNC from master: %s", reply);
1201 } else {
1202 redisLog(REDIS_NOTICE,
1203 "Master does not support PSYNC or is in "
1204 "error state (reply: %s)", reply);
1205 }
1206 sdsfree(reply);
1207 replicationDiscardCachedMaster();
1208 return PSYNC_NOT_SUPPORTED;
1209}

1211void syncWithMaster(aeEventLoop *el, int fd, void *privdata, int mask) {
1212 char tmpfile[256], *err;
1213 int dfd, maxtries = 5;
1214 int sockerr = 0, psync_result;
1215 socklen_t errlen = sizeof(sockerr);
1216 REDIS_NOTUSED(el);
1217 REDIS_NOTUSED(privdata);
1218 REDIS_NOTUSED(mask);
1219
1220 /* If this event fired after the user turned the instance into a master
1221 * with SLAVEOF NO ONE we must just return ASAP. */
1222 if (server.repl_state == REDIS_REPL_NONE) {
1223 close(fd);
1224 return;
1225 }
1226
1227 /* Check for errors in the socket. */
1228 if (getsockopt(fd, SOL_SOCKET, SO_ERROR, &sockerr, &errlen) == -1)
1229 sockerr = errno;
1230 if (sockerr) {
1231 aeDeleteFileEvent(server.el,fd,AE_READABLE|AE_WRITABLE);
1232 redisLog(REDIS_WARNING,"Error condition on socket for SYNC: %s",
1233 strerror(sockerr));
1234 goto error;
1235 }
1236
1237 /* If we were connecting, it's time to send a non blocking PING, we want to
1238 * make sure the master is able to reply before going into the actual
1239 * replication process where we have long timeouts in the order of
1240 * seconds (in the meantime the slave would block). */
1241 if (server.repl_state == REDIS_REPL_CONNECTING) {
1242 redisLog(REDIS_NOTICE,"Non blocking connect for SYNC fired the event.");
1243 /* Delete the writable event so that the readable event remains
1244 * registered and we can wait for the PONG reply. */
1245 aeDeleteFileEvent(server.el,fd,AE_WRITABLE);
1246 server.repl_state = REDIS_REPL_RECEIVE_PONG;
1247 /* Send the PING, don't check for errors at all, we have the timeout
1248 * that will take care about this. */
1249 syncWrite(fd,"PING\r\n",6,100);
1250 return;
1251 }
1252
1253 /* Receive the PONG command. */
1254 if (server.repl_state == REDIS_REPL_RECEIVE_PONG) {
1255 char buf[1024];
1256
1257 /* Delete the readable event, we no longer need it now that there is
1258 * the PING reply to read. */
1259 aeDeleteFileEvent(server.el,fd,AE_READABLE);
1260
1261 /* Read the reply with explicit timeout. */
1262 buf[0] = '\0';
1263 if (syncReadLine(fd,buf,sizeof(buf),
1264 server.repl_syncio_timeout*1000) == -1)
1265 {
1266 redisLog(REDIS_WARNING,
1267 "I/O error reading PING reply from master: %s",
1268 strerror(errno));
1269 goto error;
1270 }
1271
1272 /* We accept only two replies as valid, a positive +PONG reply
1273 * (we just check for "+") or an authentication error.
1274 * Note that older versions of Redis replied with "operation not
1275 * permitted" instead of using a proper error code, so we test
1276 * both. */
1277 if (buf[0] != '+' &&
1278 strncmp(buf,"-NOAUTH",7) != 0 &&
1279 strncmp(buf,"-ERR operation not permitted",28) != 0)
1280 {
1281 redisLog(REDIS_WARNING,"Error reply to PING from master: '%s'",buf);
1282 goto error;
1283 } else {
1284 redisLog(REDIS_NOTICE,
1285 "Master replied to PING, replication can continue...");
1286 }
1287 }
1288
1289 /* AUTH with the master if required. */
1290 if(server.masterauth) {
1291 err = sendSynchronousCommand(fd,"AUTH",server.masterauth,NULL);
1292 if (err[0] == '-') {
1293 redisLog(REDIS_WARNING,"Unable to AUTH to MASTER: %s",err);
1294 sdsfree(err);
1295 goto error;
1296 }
1297 sdsfree(err);
1298 }
1299
1300 /* Set the slave port, so that Master's INFO command can list the
1301 * slave listening port correctly. */
1302 {
1303 sds port = sdsfromlonglong(server.port);
1304 err = sendSynchronousCommand(fd,"REPLCONF","listening-port",port,
1305 NULL);
1306 sdsfree(port);
1307 /* Ignore the error if any, not all the Redis versions support
1308 * REPLCONF listening-port. */
1309 if (err[0] == '-') {
1310 redisLog(REDIS_NOTICE,"(Non critical) Master does not understand REPLCONF listening-port: %s", err);
1311 }
1312 sdsfree(err);
1313 }
1314
1315 /* Try a partial resynchonization. If we don't have a cached master
1316 * slaveTryPartialResynchronization() will at least try to use PSYNC
1317 * to start a full resynchronization so that we get the master run id
1318 * and the global offset, to try a partial resync at the next
1319 * reconnection attempt. */
1320 psync_result = slaveTryPartialResynchronization(fd);
1321 if (psync_result == PSYNC_CONTINUE) {
1322 redisLog(REDIS_NOTICE, "MASTER <-> SLAVE sync: Master accepted a Partial Resynchronization.");
1323 return;
1324 }
1325
1326 /* Fall back to SYNC if needed. Otherwise psync_result == PSYNC_FULLRESYNC
1327 * and the server.repl_master_runid and repl_master_initial_offset are
1328 * already populated. */
1329 if (psync_result == PSYNC_NOT_SUPPORTED) {
1330 redisLog(REDIS_NOTICE,"Retrying with SYNC...");
1331 if (syncWrite(fd,"SYNC\r\n",6,server.repl_syncio_timeout*1000) == -1) {
1332 redisLog(REDIS_WARNING,"I/O error writing to MASTER: %s",
1333 strerror(errno));
1334 goto error;
1335 }
1336 }
1337
1338 /* Prepare a suitable temp file for bulk transfer */
1339 while(maxtries--) {
1340 snprintf(tmpfile,256,
1341 "temp-%d.%ld.rdb",(int)server.unixtime,(long int)getpid());
1342 dfd = open(tmpfile,O_CREAT|O_WRONLY|O_EXCL,0644);
1343 if (dfd != -1) break;
1344 sleep(1);
1345 }
1346 if (dfd == -1) {
1347 redisLog(REDIS_WARNING,"Opening the temp file needed for MASTER <-> SLAVE synchronization: %s",strerror(errno));
1348 goto error;
1349 }
1350
1351 /* Setup the non blocking download of the bulk file. */
1352 if (aeCreateFileEvent(server.el,fd, AE_READABLE,readSyncBulkPayload,NULL)
1353 == AE_ERR)
1354 {
1355 redisLog(REDIS_WARNING,
1356 "Can't create readable event for SYNC: %s (fd=%d)",
1357 strerror(errno),fd);
1358 goto error;
1359 }
1360
1361 server.repl_state = REDIS_REPL_TRANSFER;
1362 server.repl_transfer_size = -1;
1363 server.repl_transfer_read = 0;
1364 server.repl_transfer_last_fsync_off = 0;
1365 server.repl_transfer_fd = dfd;
1366 server.repl_transfer_lastio = server.unixtime;
1367 server.repl_transfer_tmpfile = zstrdup(tmpfile);
1368 return;
1369
1370error:
1371 close(fd);
1372 server.repl_transfer_s = -1;
1373 server.repl_state = REDIS_REPL_CONNECT;
1374 return;
1375}

1378 int fd;
1379
1380 fd = anetTcpNonBlockBindConnect(NULL,
1381 server.masterhost,server.masterport,REDIS_BIND_ADDR);

1415/* This function aborts a non blocking replication attempt if there is one
1416 * in progress, by canceling the non-blocking connect attempt or
1417 * the initial bulk transfer.
1418 *
1419 * If there was a replication handshake in progress 1 is returned and
1420 * the replication state (server.repl_state) set to REDIS_REPL_CONNECT.
1421 *
1422 * Otherwise zero is returned and no operation is perforemd at all. */
1423int cancelReplicationHandshake(void) {
1424 if (server.repl_state == REDIS_REPL_TRANSFER) {
1425 replicationAbortSyncTransfer();
1426 } else if (server.repl_state == REDIS_REPL_CONNECTING ||
1427 server.repl_state == REDIS_REPL_RECEIVE_PONG)
1428 {
1429 undoConnectWithMaster();
1430 } else {
1431 return 0;
1432 }
1433 return 1;
1434}

1436/* Set replication to the specified master address and port. */
1437void replicationSetMaster(char *ip, int port) {
1438 sdsfree(server.masterhost);
1439 server.masterhost = sdsnew(ip);
1440 server.masterport = port;

1443 disconnectSlaves(); /* Force our slaves to resync with us as well. */
1444 replicationDiscardCachedMaster(); /* Don't try a PSYNC. */
1445 freeReplicationBacklog(); /* Don't allow our chained slaves to PSYNC. */

1452/* Cancel replication, setting the instance as a master itself. */
1453void replicationUnsetMaster(void) {
1454 if (server.masterhost == NULL) return; /* Nothing to do. */
1455 sdsfree(server.masterhost);
1456 server.masterhost = NULL;
1457 if (server.master) {
1458 if (listLength(server.slaves) == 0) {
1459 /* If this instance is turned into a master and there are no
1460 * slaves, it inherits the replication offset from the master.
1461 * Under certain conditions this makes replicas comparable by
1462 * replication offset to understand what is the most updated. */
1463 server.master_repl_offset = server.master->reploff;
1464 freeReplicationBacklog();
1465 }
1466 freeClient(server.master);
1467 }
1468 replicationDiscardCachedMaster();
1469 cancelReplicationHandshake();
1470 server.repl_state = REDIS_REPL_NONE;
1471}

1473void slaveofCommand(redisClient *c) {
1474 /* SLAVEOF is not allowed in cluster mode as replication is automatically
1475 * configured using the current address of the master node. */
1476 if (server.cluster_enabled) {
1477 addReplyError(c,"SLAVEOF not allowed in cluster mode.");
1478 return;
1479 }
1480
1481 /* The special host/port combination "NO" "ONE" turns the instance
1482 * into a master. Otherwise the new master address is set. */
1483 if (!strcasecmp(c->argv[1]->ptr,"no") &&
1484 !strcasecmp(c->argv[2]->ptr,"one")) {
1485 if (server.masterhost) {
1486 replicationUnsetMaster();
1487 redisLog(REDIS_NOTICE,"MASTER MODE enabled (user request)");
1488 }
1489 } else {
1490 long port;
1491
1492 if ((getLongFromObjectOrReply(c, c->argv[2], &port, NULL) != REDIS_OK))
1493 return;
1494
1495 /* Check if we are already attached to the specified slave */
1496 if (server.masterhost && !strcasecmp(server.masterhost,c->argv[1]->ptr)
1497 && server.masterport == port) {
1498 redisLog(REDIS_NOTICE,"SLAVE OF would result into synchronization with the master we are already connected with. No operation performed.");
1499 addReplySds(c,sdsnew("+OK Already connected to specified master\r\n"));
1500 return;
1501 }
1502 /* There was no previous master or the user specified a different one,
1503 * we can continue. */
1504 replicationSetMaster(c->argv[1]->ptr, port);
1505 redisLog(REDIS_NOTICE,"SLAVE OF %s:%d enabled (user request)",
1506 server.masterhost, server.masterport);
1507 }
1508 addReply(c,shared.ok);
1509}

1576/* ---------------------- MASTER CACHING FOR PSYNC -------------------------- */
1577
1578/* In order to implement partial synchronization we need to be able to cache
1579 * our master's client structure after a transient disconnection.
1580 * It is cached into server.cached_master and flushed away using the following
1581 * functions. */
1582
1583/* This function is called by freeClient() in order to cache the master
1584 * client structure instead of destryoing it. freeClient() will return
1585 * ASAP after this function returns, so every action needed to avoid problems
1586 * with a client that is really "suspended" has to be done by this function.
1587 *
1588 * The other functions that will deal with the cached master are:
1589 *
1590 * replicationDiscardCachedMaster() that will make sure to kill the client
1591 * as for some reason we don't want to use it in the future.
1592 *
1593 * replicationResurrectCachedMaster() that is used after a successful PSYNC
1594 * handshake in order to reactivate the cached master.
1595 */
1596void replicationCacheMaster(redisClient *c) {
1597 listNode *ln;
1598
1599 redisAssert(server.master != NULL && server.cached_master == NULL);
1600 redisLog(REDIS_NOTICE,"Caching the disconnected master state.");
1601
1602 /* Remove from the list of clients, we don't want this client to be
1603 * listed by CLIENT LIST or processed in any way by batch operations. */
1604 ln = listSearchKey(server.clients,c);
1605 redisAssert(ln != NULL);
1606 listDelNode(server.clients,ln);
1607
1608 /* Save the master. Server.master will be set to null later by
1609 * replicationHandleMasterDisconnection(). */
1610 server.cached_master = server.master;
1611
1612 /* Remove the event handlers and close the socket. We'll later reuse
1613 * the socket of the new connection with the master during PSYNC. */
1614 aeDeleteFileEvent(server.el,c->fd,AE_READABLE);
1615 aeDeleteFileEvent(server.el,c->fd,AE_WRITABLE);
1616 close(c->fd);
1617
1618 /* Set fd to -1 so that we can safely call freeClient(c) later. */
1619 c->fd = -1;
1620
1621 /* Invalidate the Peer ID cache. */
1622 if (c->peerid) {
1623 sdsfree(c->peerid);
1624 c->peerid = NULL;
1625 }
1626
1627 /* Caching the master happens instead of the actual freeClient() call,
1628 * so make sure to adjust the replication state. This function will
1629 * also set server.master to NULL. */
1630 replicationHandleMasterDisconnection();
1631}

1633/* Free a cached master, called when there are no longer the conditions for
1634 * a partial resync on reconnection. */
1635void replicationDiscardCachedMaster(void) {
1636 if (server.cached_master == NULL) return;
1637
1638 redisLog(REDIS_NOTICE,"Discarding previously cached master state.");
1639 server.cached_master->flags &= ~REDIS_MASTER;
1640 freeClient(server.cached_master);
1641 server.cached_master = NULL;
1642}

1644/* Turn the cached master into the current master, using the file descriptor
1645 * passed as argument as the socket for the new master.
1646 *
1647 * This function is called when successfully setup a partial resynchronization
1648 * so the stream of data that we'll receive will start from were this
1649 * master left. */
1650void replicationResurrectCachedMaster(int newfd) {
1651 server.master = server.cached_master;
1652 server.cached_master = NULL;
1653 server.master->fd = newfd;
1654 server.master->flags &= ~(REDIS_CLOSE_AFTER_REPLY|REDIS_CLOSE_ASAP);
1655 server.master->authenticated = 1;
1656 server.master->lastinteraction = server.unixtime;
1657 server.repl_state = REDIS_REPL_CONNECTED;
1658
1659 /* Re-add to the list of clients. */
1660 listAddNodeTail(server.clients,server.master);
1661 if (aeCreateFileEvent(server.el, newfd, AE_READABLE,
1662 readQueryFromClient, server.master)) {
1663 redisLog(REDIS_WARNING,"Error resurrecting the cached master, impossible to add the readable handler: %s", strerror(errno));
1664 freeClientAsync(server.master); /* Close ASAP. */
1665 }
1666
1667 /* We may also need to install the write handler as well if there is
1668 * pending data in the write buffers. */
1669 if (server.master->bufpos || listLength(server.master->reply)) {
1670 if (aeCreateFileEvent(server.el, newfd, AE_WRITABLE,
1671 sendReplyToClient, server.master)) {
1672 redisLog(REDIS_WARNING,"Error resurrecting the cached master, impossible to add the writable handler: %s", strerror(errno));
1673 freeClientAsync(server.master); /* Close ASAP. */
1674 }
1675 }
1676}

1702/* ----------------------- REPLICATION SCRIPT CACHE --------------------------
1703 * The goal of this code is to keep track of scripts already sent to every
1704 * connected slave, in order to be able to replicate EVALSHA as it is without
1705 * translating it to EVAL every time it is possible.
1706 *
1707 * We use a capped collection implemented by a hash table for fast lookup
1708 * of scripts we can send as EVALSHA, plus a linked list that is used for
1709 * eviction of the oldest entry when the max number of items is reached.
1710 *
1711 * We don't care about taking a different cache for every different slave
1712 * since to fill the cache again is not very costly, the goal of this code
1713 * is to avoid that the same big script is trasmitted a big number of times
1714 * per second wasting bandwidth and processor speed, but it is not a problem
1715 * if we need to rebuild the cache from scratch from time to time, every used
1716 * script will need to be transmitted a single time to reappear in the cache.
1717 *
1718 * This is how the system works:
1719 *
1720 * 1) Every time a new slave connects, we flush the whole script cache.
1721 * 2) We only send as EVALSHA what was sent to the master as EVALSHA, without
1722 * trying to convert EVAL into EVALSHA specifically for slaves.
1723 * 3) Every time we trasmit a script as EVAL to the slaves, we also add the
1724 * corresponding SHA1 of the script into the cache as we are sure every
1725 * slave knows about the script starting from now.
1726 * 4) On SCRIPT FLUSH command, we replicate the command to all the slaves
1727 * and at the same time flush the script cache.
1728 * 5) When the last slave disconnects, flush the cache.
1729 * 6) We handle SCRIPT LOAD as well since that's how scripts are loaded
1730 * in the master sometimes.
1731 */
1732
1733/* Initialize the script cache, only called at startup. */
1734void replicationScriptCacheInit(void) {
1735 server.repl_scriptcache_size = 10000;
1736 server.repl_scriptcache_dict = dictCreate(&replScriptCacheDictType,NULL);
1737 server.repl_scriptcache_fifo = listCreate();
1738}
1739
1740/* Empty the script cache. Should be called every time we are no longer sure
1741 * that every slave knows about all the scripts in our set, or when the
1742 * current AOF "context" is no longer aware of the script. In general we
1743 * should flush the cache:
1744 *
1745 * 1) Every time a new slave reconnects to this master and performs a
1746 * full SYNC (PSYNC does not require flushing).
1747 * 2) Every time an AOF rewrite is performed.
1748 * 3) Every time we are left without slaves at all, and AOF is off, in order
1749 * to reclaim otherwise unused memory.
1750 */
1751void replicationScriptCacheFlush(void) {
1752 dictEmpty(server.repl_scriptcache_dict,NULL);
1753 listRelease(server.repl_scriptcache_fifo);
1754 server.repl_scriptcache_fifo = listCreate();
1755}

1757/* Add an entry into the script cache, if we reach max number of entries the
1758 * oldest is removed from the list. */
1759void replicationScriptCacheAdd(sds sha1) {
1760 int retval;
1761 sds key = sdsdup(sha1);
1762
1763 /* Evict oldest. */
1764 if (listLength(server.repl_scriptcache_fifo) == server.repl_scriptcache_size)
1765 {
1766 listNode *ln = listLast(server.repl_scriptcache_fifo);
1767 sds oldest = listNodeValue(ln);
1768
1769 retval = dictDelete(server.repl_scriptcache_dict,oldest);
1770 redisAssert(retval == DICT_OK);
1771 listDelNode(server.repl_scriptcache_fifo,ln);
1772 }
1773
1774 /* Add current. */
1775 retval = dictAdd(server.repl_scriptcache_dict,key,NULL);
1776 listAddNodeHead(server.repl_scriptcache_fifo,key);
1777 redisAssert(retval == DICT_OK);
1778}

1781 * if all the slaves are aware of this script SHA1. */

1786/* ----------------------- SYNCHRONOUS REPLICATION --------------------------
1787 * Redis synchronous replication design can be summarized in points:
1788 *
1789 * - Redis masters have a global replication offset, used by PSYNC.
1790 * - Master increment the offset every time new commands are sent to slaves.
1791 * - Slaves ping back masters with the offset processed so far.
1792 *
1793 * So synchronous replication adds a new WAIT command in the form:
1794 *
1795 * WAIT <num_replicas> <milliseconds_timeout>
1796 *
1797 * That returns the number of replicas that processed the query when
1798 * we finally have at least num_replicas, or when the timeout was
1799 * reached.
1800 *
1801 * The command is implemented in this way:
1802 *
1803 * - Every time a client processes a command, we remember the replication
1804 * offset after sending that command to the slaves.
1805 * - When WAIT is called, we ask slaves to send an acknowledgement ASAP.
1806 * The client is blocked at the same time (see blocked.c).
1807 * - Once we receive enough ACKs for a given offset or when the timeout
1808 * is reached, the WAIT command is unblocked and the reply sent to the
1809 * client.
1810 */
1811
1812/* This just set a flag so that we broadcast a REPLCONF GETACK command
1813 * to all the slaves in the beforeSleep() function. Note that this way
1814 * we "group" all the clients that want to wait for synchronouns replication
1815 * in a given event loop iteration, and send a single GETACK for them all. */
1816void replicationRequestAckFromSlaves(void) {
1817 server.get_ack_from_slaves = 1;
1818}
1819
1820/* Return the number of slaves that already acknowledged the specified
1821 * replication offset. */
1822int replicationCountAcksByOffset(long long offset) {
1823 listIter li;
1824 listNode *ln;
1825 int count = 0;
1826
1827 listRewind(server.slaves,&li);
1828 while((ln = listNext(&li))) {
1829 redisClient *slave = ln->value;
1830
1831 if (slave->replstate != REDIS_REPL_ONLINE) continue;
1832 if (slave->repl_ack_off >= offset) count++;
1833 }
1834 return count;
1835}

1837/* WAIT for N replicas to acknowledge the processing of our latest
1838 * write command (and all the previous commands). */
1839void waitCommand(redisClient *c) {
1840 mstime_t timeout;
1841 long numreplicas, ackreplicas;
1842 long long offset = c->woff;
1843
1844 /* Argument parsing. */
1845 if (getLongFromObjectOrReply(c,c->argv[1],&numreplicas,NULL) != REDIS_OK)
1846 return;
1847 if (getTimeoutFromObjectOrReply(c,c->argv[2],&timeout,UNIT_MILLISECONDS)
1848 != REDIS_OK) return;
1849
1850 /* First try without blocking at all. */
1851 ackreplicas = replicationCountAcksByOffset(c->woff);
1852 if (ackreplicas >= numreplicas || c->flags & REDIS_MULTI) {
1853 addReplyLongLong(c,ackreplicas);
1854 return;
1855 }
1856
1857 /* Otherwise block the client and put it into our list of clients
1858 * waiting for ack from slaves. */
1859 c->bpop.timeout = timeout;
1860 c->bpop.reploffset = offset;
1861 c->bpop.numreplicas = numreplicas;
1862 listAddNodeTail(server.clients_waiting_acks,c);
1863 blockClient(c,REDIS_BLOCKED_WAIT);
1864
1865 /* Make sure that the server will send an ACK request to all the slaves
1866 * before returning to the event loop. */
1867 replicationRequestAckFromSlaves();
1868}
1869
1870/* This is called by unblockClient() to perform the blocking op type
1871 * specific cleanup. We just remove the client from the list of clients
1872 * waiting for replica acks. Never call it directly, call unblockClient()
1873 * instead. */
1874void unblockClientWaitingReplicas(redisClient *c) {
1875 listNode *ln = listSearchKey(server.clients_waiting_acks,c);
1876 redisAssert(ln != NULL);
1877 listDelNode(server.clients_waiting_acks,ln);
1878}
1879
1880/* Check if there are clients blocked in WAIT that can be unblocked since
1881 * we received enough ACKs from slaves. */
1882void processClientsWaitingReplicas(void) {
1883 long long last_offset = 0;
1884 int last_numreplicas = 0;
1885
1886 listIter li;
1887 listNode *ln;
1888
1889 listRewind(server.clients_waiting_acks,&li);
1890 while((ln = listNext(&li))) {
1891 redisClient *c = ln->value;
1892
1893 /* Every time we find a client that is satisfied for a given
1894 * offset and number of replicas, we remember it so the next client
1895 * may be unblocked without calling replicationCountAcksByOffset()
1896 * if the requested offset / replicas were equal or less. */
1897 if (last_offset && last_offset > c->bpop.reploffset &&
1898 last_numreplicas > c->bpop.numreplicas)
1899 {
1900 unblockClient(c);
1901 addReplyLongLong(c,last_numreplicas);
1902 } else {
1903 int numreplicas = replicationCountAcksByOffset(c->bpop.reploffset);
1904
1905 if (numreplicas >= c->bpop.numreplicas) {
1906 last_offset = c->bpop.reploffset;
1907 last_numreplicas = numreplicas;
1908 unblockClient(c);
1909 addReplyLongLong(c,numreplicas);
1910 }
1911 }
1912 }
1913}

1915/* Return the slave replication offset for this instance, that is
1916 * the offset for which we already processed the master replication stream. */
1917long long replicationGetSlaveOffset(void) {
1918 long long offset = 0;
1919
1920 if (server.masterhost != NULL) {
1921 if (server.master) {
1922 offset = server.master->reploff;
1923 } else if (server.cached_master) {
1924 offset = server.cached_master->reploff;
1925 }
1926 }
1927 /* offset may be -1 when the master does not support it at all, however
1928 * this function is designed to return an offset that can express the
1929 * amount of data processed by the master, so we return a positive
1930 * integer. */
1931 if (offset < 0) offset = 0;
1932 return offset;
1933}

1937/* Replication cron function, called 1 time per second. */
1938void replicationCron(void) {
1939 /* Non blocking connection timeout? */
1940 if (server.masterhost &&
1941 (server.repl_state == REDIS_REPL_CONNECTING ||
1942 server.repl_state == REDIS_REPL_RECEIVE_PONG) &&
1943 (time(NULL)-server.repl_transfer_lastio) > server.repl_timeout)
1944 {
1945 redisLog(REDIS_WARNING,"Timeout connecting to the MASTER...");
1946 undoConnectWithMaster();
1947 }
1948
1949 /* Bulk transfer I/O timeout? */
1950 if (server.masterhost && server.repl_state == REDIS_REPL_TRANSFER &&
1951 (time(NULL)-server.repl_transfer_lastio) > server.repl_timeout)
1952 {
1953 redisLog(REDIS_WARNING,"Timeout receiving bulk data from MASTER... If the problem persists try to set the 'repl-timeout' parameter in redis.conf to a larger value.");
1954 replicationAbortSyncTransfer();
1955 }
1956
1957 /* Timed out master when we are an already connected slave? */
1958 if (server.masterhost && server.repl_state == REDIS_REPL_CONNECTED &&
1959 (time(NULL)-server.master->lastinteraction) > server.repl_timeout)
1960 {
1961 redisLog(REDIS_WARNING,"MASTER timeout: no data nor PING received...");
1962 freeClient(server.master);
1963 }
1964
1965 /* Check if we should connect to a MASTER */
1966 if (server.repl_state == REDIS_REPL_CONNECT) {
1967 redisLog(REDIS_NOTICE,"Connecting to MASTER %s:%d",
1968 server.masterhost, server.masterport);
1969 if (connectWithMaster() == REDIS_OK) {
1970 redisLog(REDIS_NOTICE,"MASTER <-> SLAVE sync started");
1971 }
1972 }
1973
1974 /* Send ACK to master from time to time.
1975 * Note that we do not send periodic acks to masters that don't
1976 * support PSYNC and replication offsets. */
1977 if (server.masterhost && server.master &&
1978 !(server.master->flags & REDIS_PRE_PSYNC))
1979 replicationSendAck();
1980
1981 /* If we have attached slaves, PING them from time to time.
1982 * So slaves can implement an explicit timeout to masters, and will
1983 * be able to detect a link disconnection even if the TCP connection
1984 * will not actually go down. */
1985 if (!(server.cronloops % (server.repl_ping_slave_period * server.hz))) {
1986 listIter li;
1987 listNode *ln;
1988 robj *ping_argv[1];
1989
1990 /* First, send PING */
1991 ping_argv[0] = createStringObject("PING",4);
1992 replicationFeedSlaves(server.slaves, server.slaveseldb, ping_argv, 1);
1993 decrRefCount(ping_argv[0]);
1994
1995 /* Second, send a newline to all the slaves in pre-synchronization
1996 * stage, that is, slaves waiting for the master to create the RDB file.
1997 * The newline will be ignored by the slave but will refresh the
1998 * last-io timer preventing a timeout. */
1999 listRewind(server.slaves,&li);
2000 while((ln = listNext(&li))) {
2001 redisClient *slave = ln->value;
2002
2003 if (slave->replstate == REDIS_REPL_WAIT_BGSAVE_START ||
2004 (slave->replstate == REDIS_REPL_WAIT_BGSAVE_END &&
2005 server.rdb_child_type != REDIS_RDB_CHILD_TYPE_SOCKET))
2006 {
2007 if (write(slave->fd, "\n", 1) == -1) {
2008 /* Don't worry, it's just a ping. */
2009 }
2010 }
2011 }
2012 }
2013
2014 /* Disconnect timedout slaves. */
2015 if (listLength(server.slaves)) {
2016 listIter li;
2017 listNode *ln;
2018
2019 listRewind(server.slaves,&li);
2020 while((ln = listNext(&li))) {
2021 redisClient *slave = ln->value;
2022
2023 if (slave->replstate != REDIS_REPL_ONLINE) continue;
2024 if (slave->flags & REDIS_PRE_PSYNC) continue;
2025 if ((server.unixtime - slave->repl_ack_time) > server.repl_timeout)
2026 {
2027 redisLog(REDIS_WARNING, "Disconnecting timedout slave: %s",
2028 replicationGetSlaveName(slave));
2029 freeClient(slave);
2030 }
2031 }
2032 }
2033
2034 /* If we have no attached slaves and there is a replication backlog
2035 * using memory, free it after some (configured) time. */
2036 if (listLength(server.slaves) == 0 && server.repl_backlog_time_limit &&
2037 server.repl_backlog)
2038 {
2039 time_t idle = server.unixtime - server.repl_no_slaves_since;
2040
2041 if (idle > server.repl_backlog_time_limit) {
2042 freeReplicationBacklog();
2043 redisLog(REDIS_NOTICE,
2044 "Replication backlog freed after %d seconds "
2045 "without connected slaves.",
2046 (int) server.repl_backlog_time_limit);
2047 }
2048 }
2049
2050 /* If AOF is disabled and we no longer have attached slaves, we can
2051 * free our Replication Script Cache as there is no need to propagate
2052 * EVALSHA at all. */
2053 if (listLength(server.slaves) == 0 &&
2054 server.aof_state == REDIS_AOF_OFF &&
2055 listLength(server.repl_scriptcache_fifo) != 0)
2056 {
2057 replicationScriptCacheFlush();
2058 }
2059
2060 /* If we are using diskless replication and there are slaves waiting
2061 * in WAIT_BGSAVE_START state, check if enough seconds elapsed and
2062 * start a BGSAVE.
2063 *
2064 * This code is also useful to trigger a BGSAVE if the diskless
2065 * replication was turned off with CONFIG SET, while there were already
2066 * slaves in WAIT_BGSAVE_START state. */
2067 if (server.rdb_child_pid == -1 && server.aof_child_pid == -1) {
2068 time_t idle, max_idle = 0;
2069 int slaves_waiting = 0;
2070 listNode *ln;
2071 listIter li;
2072
2073 listRewind(server.slaves,&li);
2074 while((ln = listNext(&li))) {
2075 redisClient *slave = ln->value;
2076 if (slave->replstate == REDIS_REPL_WAIT_BGSAVE_START) {
2077 idle = server.unixtime - slave->lastinteraction;
2078 if (idle > max_idle) max_idle = idle;
2079 slaves_waiting++;
2080 }
2081 }
2082
2083 if (slaves_waiting && max_idle > server.repl_diskless_sync_delay) {
2084 /* Start a BGSAVE. Usually with socket target, or with disk target
2085 * if there was a recent socket -> disk config change. */
2086 if (startBgsaveForReplication() == REDIS_OK) {
2087 /* It started! We need to change the state of slaves
2088 * from WAIT_BGSAVE_START to WAIT_BGSAVE_END in case
2089 * the current target is disk. Otherwise it was already done
2090 * by rdbSaveToSlavesSockets() which is called by
2091 * startBgsaveForReplication(). */
2092 listRewind(server.slaves,&li);
2093 while((ln = listNext(&li))) {
2094 redisClient *slave = ln->value;
2095 if (slave->replstate == REDIS_REPL_WAIT_BGSAVE_START)
2096 slave->replstate = REDIS_REPL_WAIT_BGSAVE_END;
2097 }
2098 }
2099 }
2100 }
2101
2102 /* Refresh the number of slaves with lag <= min-slaves-max-lag. */
2103 refreshGoodSlavesCount();
2104}
