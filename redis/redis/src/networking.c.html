36/* To evaluate the output buffer size of a client we need to get size of
37 * allocated objects, however we can't used zmalloc_size() directly on sds
38 * strings because of the trick they use to work (the header is before the
39 * returned pointer), so we use this helper function. */
40size_t zmalloc_size_sds(sds s) {
41 return zmalloc_size(s-sizeof(struct sdshdr));
42}

44/* Return the amount of memory used by the sds string at object->ptr
45 * for a string object. */
46size_t getStringObjectSdsUsedMemory(robj *o) {
47 redisAssertWithInfo(NULL,o,o->type == REDIS_STRING);
48 switch(o->encoding) {
49 case REDIS_ENCODING_RAW: return zmalloc_size_sds(o->ptr);
50 case REDIS_ENCODING_EMBSTR: return sdslen(o->ptr);
51 default: return 0; /* Just integer encoding for now. */
52 }
53}

64redisClient *createClient(int fd) {
65 redisClient *c = zmalloc(sizeof(redisClient));
66
67 /* passing -1 as fd it is possible to create a non connected client.
68 * This is useful since all the Redis commands needs to be executed
69 * in the context of a client. When commands are executed in other
70 * contexts (for instance a Lua script) we need a non connected client. */
71 if (fd != -1) {
72 anetNonBlock(NULL,fd);
73 anetEnableTcpNoDelay(NULL,fd);
74 if (server.tcpkeepalive)
75 anetKeepAlive(NULL,fd,server.tcpkeepalive);
76 if (aeCreateFileEvent(server.el,fd,AE_READABLE,
77 readQueryFromClient, c) == AE_ERR)
78 {
79 close(fd);
80 zfree(c);
81 return NULL;
82 }
83 }
84
85 selectDb(c,0);
86 c->id = server.next_client_id++;
87 c->fd = fd;
88 c->name = NULL;
89 c->bufpos = 0;
90 c->querybuf = sdsempty();
91 c->querybuf_peak = 0;
92 c->reqtype = 0;
93 c->argc = 0;
94 c->argv = NULL;
95 c->cmd = c->lastcmd = NULL;
96 c->multibulklen = 0;
97 c->bulklen = -1;
98 c->sentlen = 0;
99 c->flags = 0;
100 c->ctime = c->lastinteraction = server.unixtime;
101 c->authenticated = 0;
102 c->replstate = REDIS_REPL_NONE;
103 c->repl_put_online_on_ack = 0;
104 c->reploff = 0;
105 c->repl_ack_off = 0;
106 c->repl_ack_time = 0;
107 c->slave_listening_port = 0;
108 c->reply = listCreate();
109 c->reply_bytes = 0;
110 c->obuf_soft_limit_reached_time = 0;
111 listSetFreeMethod(c->reply,decrRefCountVoid);
112 listSetDupMethod(c->reply,dupClientReplyValue);
113 c->btype = REDIS_BLOCKED_NONE;
114 c->bpop.timeout = 0;
115 c->bpop.keys = dictCreate(&setDictType,NULL);
116 c->bpop.target = NULL;
117 c->bpop.numreplicas = 0;
118 c->bpop.reploffset = 0;
119 c->woff = 0;
120 c->watched_keys = listCreate();
121 c->pubsub_channels = dictCreate(&setDictType,NULL);
122 c->pubsub_patterns = listCreate();
123 c->peerid = NULL;
124 listSetFreeMethod(c->pubsub_patterns,decrRefCountVoid);
125 listSetMatchMethod(c->pubsub_patterns,listMatchObjects);
126 if (fd != -1) listAddNodeTail(server.clients,c);
127 initClientMultiState(c);
128 return c;
129}

131/* This function is called every time we are going to transmit new data
132 * to the client. The behavior is the following:
133 *
134 * If the client should receive new data (normal clients will) the function
135 * returns REDIS_OK, and make sure to install the write handler in our event
136 * loop so that when the socket is writable new data gets written.
137 *
138 * If the client should not receive new data, because it is a fake client
139 * (used to load AOF in memory), a master or because the setup of the write
140 * handler failed, the function returns REDIS_ERR.
141 *
142 * The function may return REDIS_OK without actually installing the write
143 * event handler in the following cases:
144 *
145 * 1) The event handler should already be installed since the output buffer
146 * already contained something.
147 * 2) The client is a slave but not yet online, so we want to just accumulate
148 * writes in the buffer but not actually sending them yet.
149 *
150 * Typically gets called every time a reply is built, before adding more
151 * data to the clients output buffers. If the function returns REDIS_ERR no
152 * data should be appended to the output buffers. */
153int prepareClientToWrite(redisClient *c) {
154 /* If it's the Lua client we always return ok without installing any
155 * handler since there is no socket at all. */
156 if (c->flags & REDIS_LUA_CLIENT) return REDIS_OK;
157
158 /* Masters don't receive replies, unless REDIS_MASTER_FORCE_REPLY flag
159 * is set. */
160 if ((c->flags & REDIS_MASTER) &&
161 !(c->flags & REDIS_MASTER_FORCE_REPLY)) return REDIS_ERR;
162
163 if (c->fd <= 0) return REDIS_ERR; /* Fake client for AOF loading. */
164
165 /* Only install the handler if not already installed and, in case of
166 * slaves, if the client can actually receive writes. */
167 if (c->bufpos == 0 && listLength(c->reply) == 0 &&
168 (c->replstate == REDIS_REPL_NONE ||
169 (c->replstate == REDIS_REPL_ONLINE && !c->repl_put_online_on_ack)))
170 {
171 /* Try to install the write handler. */
172 if (aeCreateFileEvent(server.el, c->fd, AE_WRITABLE,
173 sendReplyToClient, c) == AE_ERR)
174 {
175 freeClientAsync(c);
176 return REDIS_ERR;
177 }
178 }
179
180 /* Authorize the caller to queue in the output buffer of this client. */
181 return REDIS_OK;
182}

184/* Create a duplicate of the last object in the reply list when
185 * it is not exclusively owned by the reply list. */
186robj *dupLastObjectIfNeeded(list *reply) {
187 robj *new, *cur;
188 listNode *ln;
189 redisAssert(listLength(reply) > 0);
190 ln = listLast(reply);
191 cur = listNodeValue(ln);
192 if (cur->refcount > 1) {
193 new = dupStringObject(cur);
194 decrRefCount(cur);
195 listNodeValue(ln) = new;
196 }
197 return listNodeValue(ln);
198}

204int _addReplyToBuffer(redisClient *c, char *s, size_t len) {
205 size_t available = sizeof(c->buf)-c->bufpos;
206
207 if (c->flags & REDIS_CLOSE_AFTER_REPLY) return REDIS_OK;
208
209 /* If there already are entries in the reply list, we cannot
210 * add anything more to the static buffer. */
211 if (listLength(c->reply) > 0) return REDIS_ERR;
212
213 /* Check that the buffer has enough space available for this string. */
214 if (len > available) return REDIS_ERR;
215
216 memcpy(c->buf+c->bufpos,s,len);
217 c->bufpos+=len;
218 return REDIS_OK;
219}

221void _addReplyObjectToList(redisClient *c, robj *o) {
222 robj *tail;
223
224 if (c->flags & REDIS_CLOSE_AFTER_REPLY) return;
225
226 if (listLength(c->reply) == 0) {
227 incrRefCount(o);
228 listAddNodeTail(c->reply,o);
229 c->reply_bytes += getStringObjectSdsUsedMemory(o);
230 } else {
231 tail = listNodeValue(listLast(c->reply));
232
233 /* Append to this object when possible. */
234 if (tail->ptr != NULL &&
235 tail->encoding == REDIS_ENCODING_RAW &&
236 sdslen(tail->ptr)+sdslen(o->ptr) <= REDIS_REPLY_CHUNK_BYTES)
237 {
238 c->reply_bytes -= zmalloc_size_sds(tail->ptr);
239 tail = dupLastObjectIfNeeded(c->reply);
240 tail->ptr = sdscatlen(tail->ptr,o->ptr,sdslen(o->ptr));
241 c->reply_bytes += zmalloc_size_sds(tail->ptr);
242 } else {
243 incrRefCount(o);
244 listAddNodeTail(c->reply,o);
245 c->reply_bytes += getStringObjectSdsUsedMemory(o);
246 }
247 }
248 asyncCloseClientOnOutputBufferLimitReached(c);
249}

251/* This method takes responsibility over the sds. When it is no longer
252 * needed it will be free'd, otherwise it ends up in a robj. */
253void _addReplySdsToList(redisClient *c, sds s) {
254 robj *tail;
255
256 if (c->flags & REDIS_CLOSE_AFTER_REPLY) {
257 sdsfree(s);
258 return;
259 }
260
261 if (listLength(c->reply) == 0) {
262 listAddNodeTail(c->reply,createObject(REDIS_STRING,s));
263 c->reply_bytes += zmalloc_size_sds(s);
264 } else {
265 tail = listNodeValue(listLast(c->reply));
266
267 /* Append to this object when possible. */
268 if (tail->ptr != NULL && tail->encoding == REDIS_ENCODING_RAW &&
269 sdslen(tail->ptr)+sdslen(s) <= REDIS_REPLY_CHUNK_BYTES)
270 {
271 c->reply_bytes -= zmalloc_size_sds(tail->ptr);
272 tail = dupLastObjectIfNeeded(c->reply);
273 tail->ptr = sdscatlen(tail->ptr,s,sdslen(s));
274 c->reply_bytes += zmalloc_size_sds(tail->ptr);
275 sdsfree(s);
276 } else {
277 listAddNodeTail(c->reply,createObject(REDIS_STRING,s));
278 c->reply_bytes += zmalloc_size_sds(s);
279 }
280 }
281 asyncCloseClientOnOutputBufferLimitReached(c);
282}

284void _addReplyStringToList(redisClient *c, char *s, size_t len) {
285 robj *tail;
286
287 if (c->flags & REDIS_CLOSE_AFTER_REPLY) return;
288
289 if (listLength(c->reply) == 0) {
290 robj *o = createStringObject(s,len);
291
292 listAddNodeTail(c->reply,o);
293 c->reply_bytes += getStringObjectSdsUsedMemory(o);
294 } else {
295 tail = listNodeValue(listLast(c->reply));
296
297 /* Append to this object when possible. */
298 if (tail->ptr != NULL && tail->encoding == REDIS_ENCODING_RAW &&
299 sdslen(tail->ptr)+len <= REDIS_REPLY_CHUNK_BYTES)
300 {
301 c->reply_bytes -= zmalloc_size_sds(tail->ptr);
302 tail = dupLastObjectIfNeeded(c->reply);
303 tail->ptr = sdscatlen(tail->ptr,s,len);
304 c->reply_bytes += zmalloc_size_sds(tail->ptr);
305 } else {
306 robj *o = createStringObject(s,len);
307
308 listAddNodeTail(c->reply,o);
309 c->reply_bytes += getStringObjectSdsUsedMemory(o);
310 }
311 }
312 asyncCloseClientOnOutputBufferLimitReached(c);
313}

320void addReply(redisClient *c, robj *obj) {
321 if (prepareClientToWrite(c) != REDIS_OK) return;
322
323 /* This is an important place where we can avoid copy-on-write
324 * when there is a saving child running, avoiding touching the
325 * refcount field of the object if it's not needed.
326 *
327 * If the encoding is RAW and there is room in the static buffer
328 * we'll be able to send the object to the client without
329 * messing with its page. */
330 if (sdsEncodedObject(obj)) {
331 if (_addReplyToBuffer(c,obj->ptr,sdslen(obj->ptr)) != REDIS_OK)
332 _addReplyObjectToList(c,obj);
333 } else if (obj->encoding == REDIS_ENCODING_INT) {
334 /* Optimization: if there is room in the static buffer for 32 bytes
335 * (more than the max chars a 64 bit integer can take as string) we
336 * avoid decoding the object and go for the lower level approach. */
337 if (listLength(c->reply) == 0 && (sizeof(c->buf) - c->bufpos) >= 32) {
338 char buf[32];
339 int len;
340
341 len = ll2string(buf,sizeof(buf),(long)obj->ptr);
342 if (_addReplyToBuffer(c,buf,len) == REDIS_OK)
343 return;
344 /* else... continue with the normal code path, but should never
345 * happen actually since we verified there is room. */
346 }
347 obj = getDecodedObject(obj);
348 if (_addReplyToBuffer(c,obj->ptr,sdslen(obj->ptr)) != REDIS_OK)
349 _addReplyObjectToList(c,obj);
350 decrRefCount(obj);
351 } else {
352 redisPanic("Wrong obj->encoding in addReply()");
353 }
354}

356void addReplySds(redisClient *c, sds s) {
357 if (prepareClientToWrite(c) != REDIS_OK) {
358 /* The caller expects the sds to be free'd. */
359 sdsfree(s);
360 return;
361 }
362 if (_addReplyToBuffer(c,s,sdslen(s)) == REDIS_OK) {
363 sdsfree(s);
364 } else {
365 /* This method free's the sds when it is no longer needed. */
366 _addReplySdsToList(c,s);
367 }
368}

386void addReplyErrorFormat(redisClient *c, const char *fmt, ...) {
387 size_t l, j;
388 va_list ap;
389 va_start(ap,fmt);
390 sds s = sdscatvprintf(sdsempty(),fmt,ap);
391 va_end(ap);
392 /* Make sure there are no newlines in the string, otherwise invalid protocol
393 * is emitted. */
394 l = sdslen(s);
395 for (j = 0; j < l; j++) {
396 if (s[j] == '\r' || s[j] == '\n') s[j] = ' ';
397 }
398 addReplyErrorLength(c,s,sdslen(s));
399 sdsfree(s);
400}

421/* Adds an empty object to the reply list that will contain the multi bulk
422 * length, which is not known when this function is called. */
423void *addDeferredMultiBulkLength(redisClient *c) {
424 /* Note that we install the write event here even if the object is not
425 * ready to be sent, since we are sure that before returning to the
426 * event loop setDeferredMultiBulkLength() will be called. */
427 if (prepareClientToWrite(c) != REDIS_OK) return NULL;
428 listAddNodeTail(c->reply,createObject(REDIS_STRING,NULL));
429 return listLast(c->reply);

432/* Populate the length object and try gluing it to the next chunk. */
433void setDeferredMultiBulkLength(redisClient *c, void *node, long length) {
434 listNode *ln = (listNode*)node;
435 robj *len, *next;
436
437 /* Abort when *node is NULL (see addDeferredMultiBulkLength). */
438 if (node == NULL) return;
439
440 len = listNodeValue(ln);
441 len->ptr = sdscatprintf(sdsempty(),"*%ld\r\n",length);
442 len->encoding = REDIS_ENCODING_RAW; /* in case it was an EMBSTR. */
443 c->reply_bytes += zmalloc_size_sds(len->ptr);
444 if (ln->next != NULL) {
445 next = listNodeValue(ln->next);
446
447 /* Only glue when the next node is non-NULL (an sds in this case) */
448 if (next->ptr != NULL) {
449 c->reply_bytes -= zmalloc_size_sds(len->ptr);
450 c->reply_bytes -= getStringObjectSdsUsedMemory(next);
451 len->ptr = sdscatlen(len->ptr,next->ptr,sdslen(next->ptr));
452 c->reply_bytes += zmalloc_size_sds(len->ptr);
453 listDelNode(c->reply,ln->next);
454 }
455 }
456 asyncCloseClientOnOutputBufferLimitReached(c);
457}

459/* Add a double as a bulk reply */
460void addReplyDouble(redisClient *c, double d) {
461 char dbuf[128], sbuf[128];
462 int dlen, slen;
463 if (isinf(d)) {
464 /* Libc in odd systems (Hi Solaris!) will format infinite in a
465 * different way, so better to handle it in an explicit way. */
466 addReplyBulkCString(c, d > 0 ? "inf" : "-inf");
467 } else {
468 dlen = snprintf(dbuf,sizeof(dbuf),"%.17g",d);
469 slen = snprintf(sbuf,sizeof(sbuf),"$%d\r\n%s\r\n",dlen,dbuf);
470 addReplyString(c,sbuf,slen);
471 }
472}

474/* Add a long long as integer reply or bulk len / multi bulk count.
475 * Basically this is used to output <prefix><long long><crlf>. */
476void addReplyLongLongWithPrefix(redisClient *c, long long ll, char prefix) {
477 char buf[128];
478 int len;
479
480 /* Things like $3\r\n or *2\r\n are emitted very often by the protocol
481 * so we have a few shared objects to use if the integer is small
482 * like it is most of the times. */
483 if (prefix == '*' && ll < REDIS_SHARED_BULKHDR_LEN) {
484 addReply(c,shared.mbulkhdr[ll]);
485 return;
486 } else if (prefix == '$' && ll < REDIS_SHARED_BULKHDR_LEN) {
487 addReply(c,shared.bulkhdr[ll]);
488 return;
489 }
490
491 buf[0] = prefix;
492 len = ll2string(buf+1,sizeof(buf)-1,ll);
493 buf[len+1] = '\r';
494 buf[len+2] = '\n';
495 addReplyString(c,buf,len+3);
496}

514/* Create the length prefix of a bulk reply, example: $2234 */
515void addReplyBulkLen(redisClient *c, robj *obj) {
516 size_t len;
517
518 if (sdsEncodedObject(obj)) {
519 len = sdslen(obj->ptr);
520 } else {
521 long n = (long)obj->ptr;
522
523 /* Compute how many bytes will take this integer as a radix 10 string */
524 len = 1;
525 if (n < 0) {
526 len++;
527 n = -n;
528 }
529 while((n = n/10) != 0) {
530 len++;
531 }
532 }
533
534 if (len < REDIS_SHARED_BULKHDR_LEN)
535 addReply(c,shared.bulkhdr[len]);
536 else
537 addReplyLongLongWithPrefix(c,len,'$');
538}

572/* Copy 'src' client output buffers into 'dst' client output buffers.
573 * The function takes care of freeing the old output buffers of the
574 * destination client. */
575void copyClientOutputBuffer(redisClient *dst, redisClient *src) {
576 listRelease(dst->reply);
577 dst->reply = listDup(src->reply);
578 memcpy(dst->buf,src->buf,src->bufpos);
579 dst->bufpos = src->bufpos;
580 dst->reply_bytes = src->reply_bytes;
581}

583#define MAX_ACCEPTS_PER_CALL 1000
584static void acceptCommonHandler(int fd, int flags) {
585 redisClient *c;
586 if ((c = createClient(fd)) == NULL) {
587 redisLog(REDIS_WARNING,
588 "Error registering fd event for the new client: %s (fd=%d)",
589 strerror(errno),fd);
590 close(fd); /* May be already closed, just ignore errors */
591 return;
592 }
593 /* If maxclient directive is set and this is one client more... close the
594 * connection. Note that we create the client instead to check before
595 * for this condition, since now the socket is already set in non-blocking
596 * mode and we can send an error for free using the Kernel I/O */
597 if (listLength(server.clients) > server.maxclients) {
598 char *err = "-ERR max number of clients reached\r\n";
599
600 /* That's a best effort error message, don't check write errors */
601 if (write(c->fd,err,strlen(err)) == -1) {
602 /* Nothing to do, Just to avoid the warning... */
603 }
604 server.stat_rejected_conn++;
605 freeClient(c);
606 return;
607 }
608 server.stat_numconnections++;
609 c->flags |= flags;
610}

612void acceptTcpHandler(aeEventLoop *el, int fd, void *privdata, int mask) {
613 int cport, cfd, max = MAX_ACCEPTS_PER_CALL;
614 char cip[REDIS_IP_STR_LEN];
615 REDIS_NOTUSED(el);
616 REDIS_NOTUSED(mask);
617 REDIS_NOTUSED(privdata);
618
619 while(max--) {
620 cfd = anetTcpAccept(server.neterr, fd, cip, sizeof(cip), &cport);
621 if (cfd == ANET_ERR) {
622 if (errno != EWOULDBLOCK)
623 redisLog(REDIS_WARNING,
624 "Accepting client connection: %s", server.neterr);
625 return;
626 }
627 redisLog(REDIS_VERBOSE,"Accepted %s:%d", cip, cport);
628 acceptCommonHandler(cfd,0);
629 }
630}

632void acceptUnixHandler(aeEventLoop *el, int fd, void *privdata, int mask) {
633 int cfd, max = MAX_ACCEPTS_PER_CALL;
634 REDIS_NOTUSED(el);
635 REDIS_NOTUSED(mask);
636 REDIS_NOTUSED(privdata);
637
638 while(max--) {
639 cfd = anetUnixAccept(server.neterr, fd);
640 if (cfd == ANET_ERR) {
641 if (errno != EWOULDBLOCK)
642 redisLog(REDIS_WARNING,

683void freeClient(redisClient *c) {
684 listNode *ln;
685
686 /* If this is marked as current client unset it */
687 if (server.current_client == c) server.current_client = NULL;
688
689 /* If it is our master that's beging disconnected we should make sure
690 * to cache the state to try a partial resynchronization later.
691 *
692 * Note that before doing this we make sure that the client is not in
693 * some unexpected state, by checking its flags. */
694 if (server.master && c->flags & REDIS_MASTER) {
695 redisLog(REDIS_WARNING,"Connection with master lost.");
696 if (!(c->flags & (REDIS_CLOSE_AFTER_REPLY|
697 REDIS_CLOSE_ASAP|
698 REDIS_BLOCKED|
699 REDIS_UNBLOCKED)))
700 {
701 replicationCacheMaster(c);
702 return;
703 }
704 }
705
706 /* Log link disconnection with slave */
707 if ((c->flags & REDIS_SLAVE) && !(c->flags & REDIS_MONITOR)) {
708 redisLog(REDIS_WARNING,"Connection with slave %s lost.",
709 replicationGetSlaveName(c));
710 }
711
712 /* Free the query buffer */
713 sdsfree(c->querybuf);
714 c->querybuf = NULL;
715
716 /* Deallocate structures used to block on blocking ops. */
717 if (c->flags & REDIS_BLOCKED) unblockClient(c);
718 dictRelease(c->bpop.keys);
719
720 /* UNWATCH all the keys */
721 unwatchAllKeys(c);
722 listRelease(c->watched_keys);
723
724 /* Unsubscribe from all the pubsub channels */
725 pubsubUnsubscribeAllChannels(c,0);
726 pubsubUnsubscribeAllPatterns(c,0);
727 dictRelease(c->pubsub_channels);
728 listRelease(c->pubsub_patterns);
729
730 /* Close socket, unregister events, and remove list of replies and
731 * accumulated arguments. */
732 if (c->fd != -1) {
733 aeDeleteFileEvent(server.el,c->fd,AE_READABLE);
734 aeDeleteFileEvent(server.el,c->fd,AE_WRITABLE);
735 close(c->fd);
736 }
737 listRelease(c->reply);
738 freeClientArgv(c);
739
740 /* Remove from the list of clients */
741 if (c->fd != -1) {
742 ln = listSearchKey(server.clients,c);
743 redisAssert(ln != NULL);
744 listDelNode(server.clients,ln);
745 }
746
747 /* When client was just unblocked because of a blocking operation,
748 * remove it from the list of unblocked clients. */
749 if (c->flags & REDIS_UNBLOCKED) {
750 ln = listSearchKey(server.unblocked_clients,c);
751 redisAssert(ln != NULL);
752 listDelNode(server.unblocked_clients,ln);
753 }
754
755 /* Master/slave cleanup Case 1:
756 * we lost the connection with a slave. */
757 if (c->flags & REDIS_SLAVE) {
758 if (c->replstate == REDIS_REPL_SEND_BULK) {
759 if (c->repldbfd != -1) close(c->repldbfd);
760 if (c->replpreamble) sdsfree(c->replpreamble);
761 }
762 list *l = (c->flags & REDIS_MONITOR) ? server.monitors : server.slaves;
763 ln = listSearchKey(l,c);
764 redisAssert(ln != NULL);
765 listDelNode(l,ln);
766 /* We need to remember the time when we started to have zero
767 * attached slaves, as after some time we'll free the replication
768 * backlog. */
769 if (c->flags & REDIS_SLAVE && listLength(server.slaves) == 0)
770 server.repl_no_slaves_since = server.unixtime;
771 refreshGoodSlavesCount();
772 }
773
774 /* Master/slave cleanup Case 2:
775 * we lost the connection with the master. */
776 if (c->flags & REDIS_MASTER) replicationHandleMasterDisconnection();
777
778 /* If this client was scheduled for async freeing we need to remove it
779 * from the queue. */
780 if (c->flags & REDIS_CLOSE_ASAP) {
781 ln = listSearchKey(server.clients_to_close,c);
782 redisAssert(ln != NULL);
783 listDelNode(server.clients_to_close,ln);
784 }
785
786 /* Release other dynamically allocated client structure fields,
787 * and finally release the client structure itself. */
788 if (c->name) decrRefCount(c->name);
789 zfree(c->argv);
790 freeClientMultiState(c);
791 sdsfree(c->peerid);
792 zfree(c);
793}

816void sendReplyToClient(aeEventLoop *el, int fd, void *privdata, int mask) {
817 redisClient *c = privdata;
818 int nwritten = 0, totwritten = 0, objlen;
819 size_t objmem;
820 robj *o;
821 REDIS_NOTUSED(el);
822 REDIS_NOTUSED(mask);
823
824 while(c->bufpos > 0 || listLength(c->reply)) {
825 if (c->bufpos > 0) {
826 nwritten = write(fd,c->buf+c->sentlen,c->bufpos-c->sentlen);
827 if (nwritten <= 0) break;
828 c->sentlen += nwritten;
829 totwritten += nwritten;
830
831 /* If the buffer was sent, set bufpos to zero to continue with
832 * the remainder of the reply. */
833 if (c->sentlen == c->bufpos) {
834 c->bufpos = 0;
835 c->sentlen = 0;
836 }
837 } else {
838 o = listNodeValue(listFirst(c->reply));
839 objlen = sdslen(o->ptr);
840 objmem = getStringObjectSdsUsedMemory(o);
841
842 if (objlen == 0) {
843 listDelNode(c->reply,listFirst(c->reply));
844 c->reply_bytes -= objmem;
845 continue;
846 }
847
848 nwritten = write(fd, ((char*)o->ptr)+c->sentlen,objlen-c->sentlen);
849 if (nwritten <= 0) break;
850 c->sentlen += nwritten;
851 totwritten += nwritten;
852
853 /* If we fully sent the object on head go to the next one */
854 if (c->sentlen == objlen) {
855 listDelNode(c->reply,listFirst(c->reply));
856 c->sentlen = 0;
857 c->reply_bytes -= objmem;
858 }
859 }
860 /* Note that we avoid to send more than REDIS_MAX_WRITE_PER_EVENT
861 * bytes, in a single threaded server it's a good idea to serve
862 * other clients as well, even if a very large request comes from
863 * super fast link that is always able to accept data (in real world
864 * scenario think about 'KEYS *' against the loopback interface).
865 *
866 * However if we are over the maxmemory limit we ignore that and
867 * just deliver as much data as it is possible to deliver. */
868 server.stat_net_output_bytes += totwritten;
869 if (totwritten > REDIS_MAX_WRITE_PER_EVENT &&
870 (server.maxmemory == 0 ||
871 zmalloc_used_memory() < server.maxmemory)) break;
872 }
873 if (nwritten == -1) {
874 if (errno == EAGAIN) {
875 nwritten = 0;
876 } else {
877 redisLog(REDIS_VERBOSE,
878 "Error writing to client: %s", strerror(errno));
879 freeClient(c);
880 return;
881 }
882 }
883 if (totwritten > 0) {
884 /* For clients representing masters we don't count sending data
885 * as an interaction, since we always send REPLCONF ACK commands
886 * that take some time to just fill the socket output buffer.
887 * We just rely on data / pings received for timeout detection. */
888 if (!(c->flags & REDIS_MASTER)) c->lastinteraction = server.unixtime;
889 }
890 if (c->bufpos == 0 && listLength(c->reply) == 0) {
891 c->sentlen = 0;
892 aeDeleteFileEvent(server.el,c->fd,AE_WRITABLE);
893
894 /* Close connection after entire reply has been sent. */
895 if (c->flags & REDIS_CLOSE_AFTER_REPLY) freeClient(c);
896 }
897}

913int processInlineBuffer(redisClient *c) {
914 char *newline;
915 int argc, j;
916 sds *argv, aux;
917 size_t querylen;
918
919 /* Search for end of line */
920 newline = strchr(c->querybuf,'\n');
921
922 /* Nothing to do without a \r\n */
923 if (newline == NULL) {
924 if (sdslen(c->querybuf) > REDIS_INLINE_MAX_SIZE) {
925 addReplyError(c,"Protocol error: too big inline request");
926 setProtocolError(c,0);
927 }
928 return REDIS_ERR;
929 }
930
931 /* Handle the \r\n case. */
932 if (newline && newline != c->querybuf && *(newline-1) == '\r')
933 newline--;
934
935 /* Split the input buffer up to the \r\n */
936 querylen = newline-(c->querybuf);
937 aux = sdsnewlen(c->querybuf,querylen);
938 argv = sdssplitargs(aux,&argc);
939 sdsfree(aux);
940 if (argv == NULL) {
941 addReplyError(c,"Protocol error: unbalanced quotes in request");
942 setProtocolError(c,0);
943 return REDIS_ERR;
944 }
945
946 /* Newline from slaves can be used to refresh the last ACK time.
947 * This is useful for a slave to ping back while loading a big
948 * RDB file. */
949 if (querylen == 0 && c->flags & REDIS_SLAVE)
950 c->repl_ack_time = server.unixtime;
951
952 /* Leave data after the first line of the query in the buffer */
953 sdsrange(c->querybuf,querylen+2,-1);
954
955 /* Setup argv array on client structure */
956 if (argc) {
957 if (c->argv) zfree(c->argv);
958 c->argv = zmalloc(sizeof(robj*)*argc);
959 }
960
961 /* Create redis objects for all arguments. */
962 for (c->argc = 0, j = 0; j < argc; j++) {
963 if (sdslen(argv[j])) {
964 c->argv[c->argc] = createObject(REDIS_STRING,argv[j]);
965 c->argc++;
966 } else {
967 sdsfree(argv[j]);
968 }
969 }
970 zfree(argv);
971 return REDIS_OK;
972}

974/* Helper function. Trims query buffer to make the function that processes

976static void setProtocolError(redisClient *c, int pos) {
977 if (server.verbosity >= REDIS_VERBOSE) {
978 sds client = catClientInfoString(sdsempty(),c);
979 redisLog(REDIS_VERBOSE,
980 "Protocol error from client: %s", client);

987int processMultibulkBuffer(redisClient *c) {
988 char *newline = NULL;
989 int pos = 0, ok;
990 long long ll;
991
992 if (c->multibulklen == 0) {
993 /* The client should have been reset */
994 redisAssertWithInfo(c,NULL,c->argc == 0);
995
996 /* Multi bulk length cannot be read without a \r\n */
997 newline = strchr(c->querybuf,'\r');
998 if (newline == NULL) {
999 if (sdslen(c->querybuf) > REDIS_INLINE_MAX_SIZE) {
1000 addReplyError(c,"Protocol error: too big mbulk count string");
1001 setProtocolError(c,0);
1002 }
1003 return REDIS_ERR;
1004 }
1005
1006 /* Buffer should also contain \n */
1007 if (newline-(c->querybuf) > ((signed)sdslen(c->querybuf)-2))
1008 return REDIS_ERR;
1009
1010 /* We know for sure there is a whole line since newline != NULL,
1011 * so go ahead and find out the multi bulk length. */
1012 redisAssertWithInfo(c,NULL,c->querybuf[0] == '*');
1013 ok = string2ll(c->querybuf+1,newline-(c->querybuf+1),&ll);
1014 if (!ok || ll > 1024*1024) {
1015 addReplyError(c,"Protocol error: invalid multibulk length");
1016 setProtocolError(c,pos);
1017 return REDIS_ERR;
1018 }
1019
1020 pos = (newline-c->querybuf)+2;
1021 if (ll <= 0) {
1022 sdsrange(c->querybuf,pos,-1);
1023 return REDIS_OK;
1024 }
1025
1026 c->multibulklen = ll;
1027
1028 /* Setup argv array on client structure */
1029 if (c->argv) zfree(c->argv);
1030 c->argv = zmalloc(sizeof(robj*)*c->multibulklen);
1031 }
1032
1033 redisAssertWithInfo(c,NULL,c->multibulklen > 0);
1034 while(c->multibulklen) {
1035 /* Read bulk length if unknown */
1036 if (c->bulklen == -1) {
1037 newline = strchr(c->querybuf+pos,'\r');
1038 if (newline == NULL) {
1039 if (sdslen(c->querybuf) > REDIS_INLINE_MAX_SIZE) {
1040 addReplyError(c,
1041 "Protocol error: too big bulk count string");
1042 setProtocolError(c,0);
1043 return REDIS_ERR;
1044 }
1045 break;
1046 }
1047
1048 /* Buffer should also contain \n */
1049 if (newline-(c->querybuf) > ((signed)sdslen(c->querybuf)-2))
1050 break;
1051
1052 if (c->querybuf[pos] != '$') {
1053 addReplyErrorFormat(c,
1054 "Protocol error: expected '$', got '%c'",
1055 c->querybuf[pos]);
1056 setProtocolError(c,pos);
1057 return REDIS_ERR;
1058 }
1059
1060 ok = string2ll(c->querybuf+pos+1,newline-(c->querybuf+pos+1),&ll);
1061 if (!ok || ll < 0 || ll > 512*1024*1024) {
1062 addReplyError(c,"Protocol error: invalid bulk length");
1063 setProtocolError(c,pos);
1064 return REDIS_ERR;
1065 }
1066
1067 pos += newline-(c->querybuf+pos)+2;
1068 if (ll >= REDIS_MBULK_BIG_ARG) {
1069 size_t qblen;
1070
1071 /* If we are going to read a large object from network
1072 * try to make it likely that it will start at c->querybuf
1073 * boundary so that we can optimize object creation
1074 * avoiding a large copy of data. */
1075 sdsrange(c->querybuf,pos,-1);
1076 pos = 0;
1077 qblen = sdslen(c->querybuf);
1078 /* Hint the sds library about the amount of bytes this string is
1079 * going to contain. */
1080 if (qblen < (size_t)ll+2)
1081 c->querybuf = sdsMakeRoomFor(c->querybuf,ll+2-qblen);
1082 }
1083 c->bulklen = ll;
1084 }
1085
1086 /* Read bulk argument */
1087 if (sdslen(c->querybuf)-pos < (unsigned)(c->bulklen+2)) {
1088 /* Not enough data (+2 == trailing \r\n) */
1089 break;
1090 } else {
1091 /* Optimization: if the buffer contains JUST our bulk element
1092 * instead of creating a new object by *copying* the sds we
1093 * just use the current sds string. */
1094 if (pos == 0 &&
1095 c->bulklen >= REDIS_MBULK_BIG_ARG &&
1096 (signed) sdslen(c->querybuf) == c->bulklen+2)
1097 {
1098 c->argv[c->argc++] = createObject(REDIS_STRING,c->querybuf);
1099 sdsIncrLen(c->querybuf,-2); /* remove CRLF */
1100 c->querybuf = sdsempty();
1101 /* Assume that if we saw a fat argument we'll see another one
1102 * likely... */
1103 c->querybuf = sdsMakeRoomFor(c->querybuf,c->bulklen+2);
1104 pos = 0;
1105 } else {
1106 c->argv[c->argc++] =
1107 createStringObject(c->querybuf+pos,c->bulklen);
1108 pos += c->bulklen+2;
1109 }
1110 c->bulklen = -1;
1111 c->multibulklen--;
1112 }
1113 }
1114
1115 /* Trim to pos */
1116 if (pos) sdsrange(c->querybuf,pos,-1);
1117
1118 /* We're done when c->multibulk == 0 */
1119 if (c->multibulklen == 0) return REDIS_OK;
1120
1121 /* Still not read to process the command */
1122 return REDIS_ERR;
1123}

1125void processInputBuffer(redisClient *c) {
1126 /* Keep processing while there is something in the input buffer */
1127 while(sdslen(c->querybuf)) {
1128 /* Return if clients are paused. */
1129 if (!(c->flags & REDIS_SLAVE) && clientsArePaused()) return;
1130
1131 /* Immediately abort if the client is in the middle of something. */
1132 if (c->flags & REDIS_BLOCKED) return;
1133
1134 /* REDIS_CLOSE_AFTER_REPLY closes the connection once the reply is
1135 * written to the client. Make sure to not let the reply grow after
1136 * this flag has been set (i.e. don't process more commands). */
1137 if (c->flags & REDIS_CLOSE_AFTER_REPLY) return;
1138
1139 /* Determine request type when unknown. */
1140 if (!c->reqtype) {
1141 if (c->querybuf[0] == '*') {
1142 c->reqtype = REDIS_REQ_MULTIBULK;
1143 } else {
1144 c->reqtype = REDIS_REQ_INLINE;
1145 }
1146 }
1147
1148 if (c->reqtype == REDIS_REQ_INLINE) {
1149 if (processInlineBuffer(c) != REDIS_OK) break;
1150 } else if (c->reqtype == REDIS_REQ_MULTIBULK) {
1151 if (processMultibulkBuffer(c) != REDIS_OK) break;
1152 } else {
1153 redisPanic("Unknown request type");
1154 }
1155
1156 /* Multibulk processing could see a <= 0 length. */
1157 if (c->argc == 0) {
1158 resetClient(c);
1159 } else {
1160 /* Only reset the client when the command was executed. */
1161 if (processCommand(c) == REDIS_OK)
1162 resetClient(c);
1163 }
1164 }
1165}

1167void readQueryFromClient(aeEventLoop *el, int fd, void *privdata, int mask) {
1168 redisClient *c = (redisClient*) privdata;
1169 int nread, readlen;
1170 size_t qblen;
1171 REDIS_NOTUSED(el);
1172 REDIS_NOTUSED(mask);
1173
1174 server.current_client = c;
1175 readlen = REDIS_IOBUF_LEN;
1176 /* If this is a multi bulk request, and we are processing a bulk reply
1177 * that is large enough, try to maximize the probability that the query
1178 * buffer contains exactly the SDS string representing the object, even
1179 * at the risk of requiring more read(2) calls. This way the function
1180 * processMultiBulkBuffer() can avoid copying buffers to create the
1181 * Redis Object representing the argument. */
1182 if (c->reqtype == REDIS_REQ_MULTIBULK && c->multibulklen && c->bulklen != -1
1183 && c->bulklen >= REDIS_MBULK_BIG_ARG)
1184 {
1185 int remaining = (unsigned)(c->bulklen+2)-sdslen(c->querybuf);
1186
1187 if (remaining < readlen) readlen = remaining;
1188 }
1189
1190 qblen = sdslen(c->querybuf);
1191 if (c->querybuf_peak < qblen) c->querybuf_peak = qblen;
1192 c->querybuf = sdsMakeRoomFor(c->querybuf, readlen);
1193 nread = read(fd, c->querybuf+qblen, readlen);
1194 if (nread == -1) {
1195 if (errno == EAGAIN) {
1196 nread = 0;
1197 } else {
1198 redisLog(REDIS_VERBOSE, "Reading from client: %s",strerror(errno));
1199 freeClient(c);
1200 return;
1201 }
1202 } else if (nread == 0) {
1203 redisLog(REDIS_VERBOSE, "Client closed connection");
1204 freeClient(c);
1205 return;
1206 }
1207 if (nread) {
1208 sdsIncrLen(c->querybuf,nread);
1209 c->lastinteraction = server.unixtime;
1210 if (c->flags & REDIS_MASTER) c->reploff += nread;
1211 server.stat_net_input_bytes += nread;
1212 } else {
1213 server.current_client = NULL;
1214 return;
1215 }
1216 if (sdslen(c->querybuf) > server.client_max_querybuf_len) {
1217 sds ci = catClientInfoString(sdsempty(),c), bytes = sdsempty();
1218
1219 bytes = sdscatrepr(bytes,c->querybuf,64);
1220 redisLog(REDIS_WARNING,"Closing client that reached max query buffer length: %s (qbuf initial bytes: %s)", ci, bytes);
1221 sdsfree(ci);
1222 sdsfree(bytes);
1223 freeClient(c);
1224 return;
1225 }
1226 processInputBuffer(c);
1227 server.current_client = NULL;
1228}

1248/* This is a helper function for genClientPeerId().
1249 * It writes the specified ip/port to "peerid" as a null termiated string
1250 * in the form ip:port if ip does not contain ":" itself, otherwise
1251 * [ip]:port format is used (for IPv6 addresses basically). */
1252void formatPeerId(char *peerid, size_t peerid_len, char *ip, int port) {
1253 if (strchr(ip,':'))
1254 snprintf(peerid,peerid_len,"[%s]:%d",ip,port);
1255 else
1256 snprintf(peerid,peerid_len,"%s:%d",ip,port);

1259/* A Redis "Peer ID" is a colon separated ip:port pair.
1260 * For IPv4 it's in the form x.y.z.k:port, example: "127.0.0.1:1234".
1261 * For IPv6 addresses we use [] around the IP part, like in "[::1]:1234".
1262 * For Unix sockets we use path:0, like in "/tmp/redis:0".
1263 *
1264 * A Peer ID always fits inside a buffer of REDIS_PEER_ID_LEN bytes, including
1265 * the null term.
1266 *
1267 * The function returns REDIS_OK on succcess, and REDIS_ERR on failure.
1268 *
1269 * On failure the function still populates 'peerid' with the "?:0" string
1270 * in case you want to relax error checking or need to display something
1271 * anyway (see anetPeerToString implementation for more info). */
1272int genClientPeerId(redisClient *client, char *peerid, size_t peerid_len) {
1273 char ip[REDIS_IP_STR_LEN];
1274 int port;
1275

1370void clientCommand(redisClient *c) {
1371 listNode *ln;
1372 listIter li;
1373 redisClient *client;
1374
1375 if (!strcasecmp(c->argv[1]->ptr,"list") && c->argc == 2) {
1376 /* CLIENT LIST */
1377 sds o = getAllClientsInfoString();
1378 addReplyBulkCBuffer(c,o,sdslen(o));
1379 sdsfree(o);
1380 } else if (!strcasecmp(c->argv[1]->ptr,"kill")) {
1381 /* CLIENT KILL <ip:port>
1382 * CLIENT KILL <option> [value] ... <option> [value] */
1383 char *addr = NULL;
1384 int type = -1;
1385 uint64_t id = 0;
1386 int skipme = 1;
1387 int killed = 0, close_this_client = 0;
1388
1389 if (c->argc == 3) {
1390 /* Old style syntax: CLIENT KILL <addr> */
1391 addr = c->argv[2]->ptr;
1392 skipme = 0; /* With the old form, you can kill yourself. */
1393 } else if (c->argc > 3) {
1394 int i = 2; /* Next option index. */
1395
1396 /* New style syntax: parse options. */
1397 while(i < c->argc) {
1398 int moreargs = c->argc > i+1;
1399
1400 if (!strcasecmp(c->argv[i]->ptr,"id") && moreargs) {
1401 long long tmp;
1402
1403 if (getLongLongFromObjectOrReply(c,c->argv[i+1],&tmp,NULL)
1404 != REDIS_OK) return;
1405 id = tmp;
1406 } else if (!strcasecmp(c->argv[i]->ptr,"type") && moreargs) {
1407 type = getClientTypeByName(c->argv[i+1]->ptr);
1408 if (type == -1) {
1409 addReplyErrorFormat(c,"Unknown client type '%s'",
1410 (char*) c->argv[i+1]->ptr);
1411 return;
1412 }
1413 } else if (!strcasecmp(c->argv[i]->ptr,"addr") && moreargs) {
1414 addr = c->argv[i+1]->ptr;
1415 } else if (!strcasecmp(c->argv[i]->ptr,"skipme") && moreargs) {
1416 if (!strcasecmp(c->argv[i+1]->ptr,"yes")) {
1417 skipme = 1;
1418 } else if (!strcasecmp(c->argv[i+1]->ptr,"no")) {
1419 skipme = 0;
1420 } else {
1421 addReply(c,shared.syntaxerr);
1422 return;
1423 }
1424 } else {
1425 addReply(c,shared.syntaxerr);
1426 return;
1427 }
1428 i += 2;
1429 }
1430 } else {
1431 addReply(c,shared.syntaxerr);
1432 return;
1433 }
1434
1435 /* Iterate clients killing all the matching clients. */
1436 listRewind(server.clients,&li);
1437 while ((ln = listNext(&li)) != NULL) {
1438 client = listNodeValue(ln);
1439 if (addr && strcmp(getClientPeerId(client),addr) != 0) continue;
1440 if (type != -1 &&
1441 (client->flags & REDIS_MASTER ||
1442 getClientType(client) != type)) continue;
1443 if (id != 0 && client->id != id) continue;
1444 if (c == client && skipme) continue;
1445
1446 /* Kill it. */
1447 if (c == client) {
1448 close_this_client = 1;
1449 } else {
1450 freeClient(client);
1451 }
1452 killed++;
1453 }
1454
1455 /* Reply according to old/new format. */
1456 if (c->argc == 3) {
1457 if (killed == 0)
1458 addReplyError(c,"No such client");
1459 else
1460 addReply(c,shared.ok);
1461 } else {
1462 addReplyLongLong(c,killed);
1463 }
1464
1465 /* If this client has to be closed, flag it as CLOSE_AFTER_REPLY
1466 * only after we queued the reply to its output buffers. */
1467 if (close_this_client) c->flags |= REDIS_CLOSE_AFTER_REPLY;
1468 } else if (!strcasecmp(c->argv[1]->ptr,"setname") && c->argc == 3) {
1469 int j, len = sdslen(c->argv[2]->ptr);
1470 char *p = c->argv[2]->ptr;
1471
1472 /* Setting the client name to an empty string actually removes
1473 * the current name. */
1474 if (len == 0) {
1475 if (c->name) decrRefCount(c->name);
1476 c->name = NULL;
1477 addReply(c,shared.ok);
1478 return;
1479 }
1480
1481 /* Otherwise check if the charset is ok. We need to do this otherwise
1482 * CLIENT LIST format will break. You should always be able to
1483 * split by space to get the different fields. */
1484 for (j = 0; j < len; j++) {
1485 if (p[j] < '!' || p[j] > '~') { /* ASCII is assumed. */
1486 addReplyError(c,
1487 "Client names cannot contain spaces, "
1488 "newlines or special characters.");
1489 return;
1490 }
1491 }
1492 if (c->name) decrRefCount(c->name);
1493 c->name = c->argv[2];
1494 incrRefCount(c->name);
1495 addReply(c,shared.ok);
1496 } else if (!strcasecmp(c->argv[1]->ptr,"getname") && c->argc == 2) {
1497 if (c->name)
1498 addReplyBulk(c,c->name);
1499 else
1500 addReply(c,shared.nullbulk);
1501 } else if (!strcasecmp(c->argv[1]->ptr,"pause") && c->argc == 3) {
1502 long long duration;
1503
1504 if (getTimeoutFromObjectOrReply(c,c->argv[2],&duration,UNIT_MILLISECONDS)
1505 != REDIS_OK) return;
1506 pauseClients(duration);
1507 addReply(c,shared.ok);
1508 } else {
1509 addReplyError(c, "Syntax error, try CLIENT (LIST | KILL ip:port | GETNAME | SETNAME connection-name)");
1510 }
1511}

1513/* Rewrite the command vector of the client. All the new objects ref count
1514 * is incremented. The old command vector is freed, and the old objects
1515 * ref count is decremented. */
1516void rewriteClientCommandVector(redisClient *c, int argc, ...) {
1517 va_list ap;
1518 int j;
1519 robj **argv; /* The new argument vector */
1520
1521 argv = zmalloc(sizeof(robj*)*argc);
1522 va_start(ap,argc);
1523 for (j = 0; j < argc; j++) {
1524 robj *a;
1525
1526 a = va_arg(ap, robj*);
1527 argv[j] = a;
1528 incrRefCount(a);
1529 }
1530 /* We free the objects in the original vector at the end, so we are
1531 * sure that if the same objects are reused in the new vector the
1532 * refcount gets incremented before it gets decremented. */
1533 for (j = 0; j < c->argc; j++) decrRefCount(c->argv[j]);
1534 zfree(c->argv);
1535 /* Replace argv and argc with our new versions. */
1536 c->argv = argv;
1537 c->argc = argc;
1538 c->cmd = lookupCommandOrOriginal(c->argv[0]->ptr);
1539 redisAssertWithInfo(c,NULL,c->cmd != NULL);
1540 va_end(ap);
1541}

1544 * The new val ref count is incremented, and the old decremented. */
1545void rewriteClientCommandArgument(redisClient *c, int i, robj *newval) {

1561/* This function returns the number of bytes that Redis is virtually
1562 * using to store the reply still not read by the client.
1563 * It is "virtual" since the reply output list may contain objects that
1564 * are shared and are not really using additional memory.
1565 *
1566 * The function returns the total sum of the length of all the objects
1567 * stored in the output list, plus the memory used to allocate every
1568 * list node. The static reply buffer is not taken into account since it
1569 * is allocated anyway.
1570 *
1571 * Note: this function is very fast so can be called as many time as
1572 * the caller wishes. The main usage of this function currently is
1573 * enforcing the client output length limits. */
1574unsigned long getClientOutputBufferMemoryUsage(redisClient *c) {
1575 unsigned long list_item_size = sizeof(listNode)+sizeof(robj);
1576
1577 return c->reply_bytes + (list_item_size*listLength(c->reply));
1578}

1612/* The function checks if the client reached output buffer soft or hard
1613 * limit, and also update the state needed to check the soft limit as
1614 * a side effect.
1615 *
1616 * Return value: non-zero if the client reached the soft or the hard limit.
1617 * Otherwise zero is returned. */
1618int checkClientOutputBufferLimits(redisClient *c) {
1619 int soft = 0, hard = 0, class;
1620 unsigned long used_mem = getClientOutputBufferMemoryUsage(c);
1621
1622 class = getClientType(c);
1623 if (server.client_obuf_limits[class].hard_limit_bytes &&
1624 used_mem >= server.client_obuf_limits[class].hard_limit_bytes)
1625 hard = 1;
1626 if (server.client_obuf_limits[class].soft_limit_bytes &&
1627 used_mem >= server.client_obuf_limits[class].soft_limit_bytes)
1628 soft = 1;
1629
1630 /* We need to check if the soft limit is reached continuously for the
1631 * specified amount of seconds. */
1632 if (soft) {
1633 if (c->obuf_soft_limit_reached_time == 0) {
1634 c->obuf_soft_limit_reached_time = server.unixtime;
1635 soft = 0; /* First time we see the soft limit reached */
1636 } else {
1637 time_t elapsed = server.unixtime - c->obuf_soft_limit_reached_time;
1638
1639 if (elapsed <=
1640 server.client_obuf_limits[class].soft_limit_seconds) {
1641 soft = 0; /* The client still did not reached the max number of
1642 seconds for the soft limit to be considered
1643 reached. */
1644 }
1645 }
1646 } else {
1647 c->obuf_soft_limit_reached_time = 0;
1648 }
1649 return soft || hard;
1650}

1652/* Asynchronously close a client if soft or hard limit is reached on the
1653 * output buffer size. The caller can check if the client will be closed
1654 * checking if the client REDIS_CLOSE_ASAP flag is set.
1655 *
1656 * Note: we need to close the client asynchronously because this function is
1657 * called from contexts where the client can't be freed safely, i.e. from the
1658 * lower level functions pushing data inside the client output buffers. */
1659void asyncCloseClientOnOutputBufferLimitReached(redisClient *c) {
1660 redisAssert(c->reply_bytes < ULONG_MAX-(1024*64));
1661 if (c->reply_bytes == 0 || c->flags & REDIS_CLOSE_ASAP) return;
1662 if (checkClientOutputBufferLimits(c)) {
1663 sds client = catClientInfoString(sdsempty(),c);
1664
1665 freeClientAsync(c);
1666 redisLog(REDIS_WARNING,"Client %s scheduled to be closed ASAP for overcoming of output buffer limits.", client);
1667 sdsfree(client);
1668 }

1692/* Pause clients up to the specified unixtime (in ms). While clients
1693 * are paused no command is processed from clients, so the data set can't
1694 * change during that time.
1695 *
1696 * However while this function pauses normal and Pub/Sub clients, slaves are
1697 * still served, so this function can be used on server upgrades where it is
1698 * required that slaves process the latest bytes from the replication stream
1699 * before being turned to masters.
1700 *
1701 * This function is also internally used by Redis Cluster for the manual
1702 * failover procedure implemented by CLUSTER FAILOVER.
1703 *
1704 * The function always succeed, even if there is already a pause in progress.
1705 * In such a case, the pause is extended if the duration is more than the
1706 * time left for the previous duration. However if the duration is smaller
1707 * than the time left for the previous pause, no change is made to the
1708 * left duration. */
1709void pauseClients(mstime_t end) {
1710 if (!server.clients_paused || end > server.clients_pause_end_time)
1711 server.clients_pause_end_time = end;
1712 server.clients_paused = 1;
1713}

1715/* Return non-zero if clients are currently paused. As a side effect the
1716 * function checks if the pause time was reached and clear it. */
1717int clientsArePaused(void) {
1718 if (server.clients_paused &&
1719 server.clients_pause_end_time < server.mstime)
1720 {
1721 listNode *ln;
1722 listIter li;
1723 redisClient *c;
1724
1725 server.clients_paused = 0;
1726
1727 /* Put all the clients in the unblocked clients queue in order to
1728 * force the re-processing of the input buffer if any. */
1729 listRewind(server.clients,&li);
1730 while ((ln = listNext(&li)) != NULL) {
1731 c = listNodeValue(ln);
1732
1733 /* Don't touch slaves and blocked clients. The latter pending
1734 * requests be processed when unblocked. */
1735 if (c->flags & (REDIS_SLAVE|REDIS_BLOCKED)) continue;
1736 c->flags |= REDIS_UNBLOCKED;
1737 listAddNodeTail(server.unblocked_clients,c);
1738 }
1739 }
1740 return server.clients_paused;
1741}

1743/* This function is called by Redis in order to process a few events from
1744 * time to time while blocked into some not interruptible operation.
1745 * This allows to reply to clients with the -LOADING error while loading the
1746 * data set at startup or after a full resynchronization with the master
1747 * and so forth.
1748 *
1749 * It calls the event loop in order to process a few events. Specifically we
1750 * try to call the event loop for times as long as we receive acknowledge that
1751 * some event was processed, in order to go forward with the accept, read,
1752 * write, close sequence needed to serve a client.
1753 *
1754 * The function returns the total number of events processed. */
1755int processEventsWhileBlocked(void) {
1756 int iterations = 4; /* See the function top-comment. */
1757 int count = 0;
1758 while (iterations--) {
1759 int events = aeProcessEvents(server.el, AE_FILE_EVENTS|AE_DONT_WAIT);
1760 if (!events) break;
1761 count += events;
1762 }
1763 return count;
1764}
