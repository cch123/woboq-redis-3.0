92/* Define rdb_fsync_range to sync_file_range() on Linux, otherwise we use
93 * the plain fsync() call. */
94#ifdef __linux__
95#if defined(__GLIBC__) && defined(__GLIBC_PREREQ)
96#if (LINUX_VERSION_CODE >= 0x020611 && __GLIBC_PREREQ(2, 6))
97#define HAVE_SYNC_FILE_RANGE 1
98#endif
99#else
100#if (LINUX_VERSION_CODE >= 0x020611)
101#define HAVE_SYNC_FILE_RANGE 1
102#endif
103#endif
104#endif
105
106#ifdef HAVE_SYNC_FILE_RANGE
107#define rdb_fsync_range(fd,off,size) sync_file_range(fd,off,size,SYNC_FILE_RANGE_WAIT_BEFORE|SYNC_FILE_RANGE_WRITE)
108#else
109#define rdb_fsync_range(fd,off,size) fsync(fd)
110#endif

126/* Byte ordering detection */
127#include <sys/types.h> /* This will likely define BYTE_ORDER */
128
129#ifndef BYTE_ORDER
130#if (BSD >= 199103)
131# include <machine/endian.h>
132#else
133#if defined(linux) || defined(__linux__)
134# include <endian.h>
135#else
136#define LITTLE_ENDIAN 1234 /* least-significant byte first (vax, pc) */
137#define BIG_ENDIAN 4321 /* most-significant byte first (IBM, net) */
138#define PDP_ENDIAN 3412 /* LSB first in word, MSW first in long (pdp)*/
139
140#if defined(__i386__) || defined(__x86_64__) || defined(__amd64__) || \
141 defined(vax) || defined(ns32000) || defined(sun386) || \
142 defined(MIPSEL) || defined(_MIPSEL) || defined(BIT_ZERO_ON_RIGHT) || \
143 defined(__alpha__) || defined(__alpha)
144#define BYTE_ORDER LITTLE_ENDIAN
145#endif
146
147#if defined(sel) || defined(pyr) || defined(mc68000) || defined(sparc) || \
148 defined(is68k) || defined(tahoe) || defined(ibm032) || defined(ibm370) || \
149 defined(MIPSEB) || defined(_MIPSEB) || defined(_IBMR2) || defined(DGUX) ||\
150 defined(apollo) || defined(__convex__) || defined(_CRAY) || \
151 defined(__hppa) || defined(__hp9000) || \
152 defined(__hp9000s300) || defined(__hp9000s700) || \
153 defined (BIT_ZERO_ON_LEFT) || defined(m68k) || defined(__sparc)
154#define BYTE_ORDER BIG_ENDIAN
155#endif
156#endif /* linux */
157#endif /* BSD */
158#endif /* BYTE_ORDER */
159
160/* Sometimes after including an OS-specific header that defines the
161 * endianess we end with __BYTE_ORDER but not with BYTE_ORDER that is what
162 * the Redis code uses. In this case let's define everything without the
163 * underscores. */
164#ifndef BYTE_ORDER
165#ifdef __BYTE_ORDER
166#if defined(__LITTLE_ENDIAN) && defined(__BIG_ENDIAN)
167#ifndef LITTLE_ENDIAN
168#define LITTLE_ENDIAN __LITTLE_ENDIAN
169#endif
170#ifndef BIG_ENDIAN
171#define BIG_ENDIAN __BIG_ENDIAN
172#endif
173#if (__BYTE_ORDER == __LITTLE_ENDIAN)
174#define BYTE_ORDER LITTLE_ENDIAN
175#else
176#define BYTE_ORDER BIG_ENDIAN
177#endif
178#endif
179#endif
180#endif
181
182#if !defined(BYTE_ORDER) || \
183 (BYTE_ORDER != BIG_ENDIAN && BYTE_ORDER != LITTLE_ENDIAN)
184 /* you must determine what the correct bit order is for
185 * your compiler - the next line is an intentional error
186 * which will force your compiles to bomb until you fix
187 * the above macros.
188 */
189#error "Undefined or invalid BYTE_ORDER"
190#endif
