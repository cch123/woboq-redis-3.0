31#ifndef __INTSET_H
32#define __INTSET_H
33#include <stdint.h>
34
35typedef struct intset {
36 uint32_t encoding;
37 uint32_t length;
38 int8_t contents[];
39} intset;
40
41intset *intsetNew(void);
42intset *intsetAdd(intset *is, int64_t value, uint8_t *success);
43intset *intsetRemove(intset *is, int64_t value, int *success);
44uint8_t intsetFind(intset *is, int64_t value);
45int64_t intsetRandom(intset *is);
46uint8_t intsetGet(intset *is, uint32_t pos, int64_t *value);
47uint32_t intsetLen(intset *is);
48size_t intsetBlobLen(intset *is);
49
50#endif // __INTSET_H
