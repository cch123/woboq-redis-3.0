76static int callbackKeyCompare(void *privdata, const void *key1, const void *key2) {
77 int l1, l2;
78 ((void) privdata);
79
80 l1 = sdslen((const sds)key1);
81 l2 = sdslen((const sds)key2);
82 if (l1 != l2) return 0;
83 return memcmp(key1,key2,l1) == 0;
84}

105static redisAsyncContext *redisAsyncInitialize(redisContext *c) {
106 redisAsyncContext *ac;
107
108 ac = realloc(c,sizeof(redisAsyncContext));
109 if (ac == NULL)
110 return NULL;
111
112 c = &(ac->c);
113
114 /* The regular connect functions will always set the flag REDIS_CONNECTED.
115 * For the async API, we want to wait until the first write event is
116 * received up before setting this flag, so reset it here. */
117 c->flags &= ~REDIS_CONNECTED;
118
119 ac->err = 0;
120 ac->errstr = NULL;
121 ac->data = NULL;
122
123 ac->ev.data = NULL;
124 ac->ev.addRead = NULL;
125 ac->ev.delRead = NULL;
126 ac->ev.addWrite = NULL;
127 ac->ev.delWrite = NULL;
128 ac->ev.cleanup = NULL;
129
130 ac->onConnect = NULL;
131 ac->onDisconnect = NULL;
132
133 ac->replies.head = NULL;
134 ac->replies.tail = NULL;
135 ac->sub.invalid.head = NULL;
136 ac->sub.invalid.tail = NULL;
137 ac->sub.channels = dictCreate(&callbackDict,NULL);
138 ac->sub.patterns = dictCreate(&callbackDict,NULL);
139 return ac;
140}

150redisAsyncContext *redisAsyncConnect(const char *ip, int port) {
151 redisContext *c;
152 redisAsyncContext *ac;
153
154 c = redisConnectNonBlock(ip,port);
155 if (c == NULL)
156 return NULL;
157
158 ac = redisAsyncInitialize(c);
159 if (ac == NULL) {
160 redisFree(c);
161 return NULL;
162 }
163
164 __redisAsyncCopyError(ac);
165 return ac;
166}

215/* Helper functions to push/shift callbacks */
216static int __redisPushCallback(redisCallbackList *list, redisCallback *source) {
217 redisCallback *cb;
218
219 /* Copy callback from stack to heap */
220 cb = malloc(sizeof(*cb));
221 if (cb == NULL)
222 return REDIS_ERR_OOM;
223
224 if (source != NULL) {
225 memcpy(cb,source,sizeof(*cb));
226 cb->next = NULL;
227 }
228
229 /* Store callback in list */
230 if (list->head == NULL)
231 list->head = cb;
232 if (list->tail != NULL)
233 list->tail->next = cb;
234 list->tail = cb;
235 return REDIS_OK;
236}
237
238static int __redisShiftCallback(redisCallbackList *list, redisCallback *target) {
239 redisCallback *cb = list->head;
240 if (cb != NULL) {
241 list->head = cb->next;
242 if (cb == list->tail)
243 list->tail = NULL;
244
245 /* Copy callback from heap to stack */
246 if (target != NULL)
247 memcpy(target,cb,sizeof(*cb));
248 free(cb);
249 return REDIS_OK;
250 }
251 return REDIS_ERR;
252}

254static void __redisRunCallback(redisAsyncContext *ac, redisCallback *cb, redisReply *reply) {
255 redisContext *c = &(ac->c);
256 if (cb->fn != NULL) {
257 c->flags |= REDIS_IN_CALLBACK;
258 cb->fn(ac,reply,cb->privdata);
259 c->flags &= ~REDIS_IN_CALLBACK;
260 }
261}

263/* Helper function to free the context. */
264static void __redisAsyncFree(redisAsyncContext *ac) {
265 redisContext *c = &(ac->c);
266 redisCallback cb;
267 dictIterator *it;
268 dictEntry *de;
269
270 /* Execute pending callbacks with NULL reply. */
271 while (__redisShiftCallback(&ac->replies,&cb) == REDIS_OK)
272 __redisRunCallback(ac,&cb,NULL);
273
274 /* Execute callbacks for invalid commands */
275 while (__redisShiftCallback(&ac->sub.invalid,&cb) == REDIS_OK)
276 __redisRunCallback(ac,&cb,NULL);
277
278 /* Run subscription callbacks callbacks with NULL reply */
279 it = dictGetIterator(ac->sub.channels);
280 while ((de = dictNext(it)) != NULL)
281 __redisRunCallback(ac,dictGetEntryVal(de),NULL);
282 dictReleaseIterator(it);
283 dictRelease(ac->sub.channels);
284
285 it = dictGetIterator(ac->sub.patterns);
286 while ((de = dictNext(it)) != NULL)
287 __redisRunCallback(ac,dictGetEntryVal(de),NULL);
288 dictReleaseIterator(it);
289 dictRelease(ac->sub.patterns);
290
291 /* Signal event lib to clean up */
292 _EL_CLEANUP(ac);
293
294 /* Execute disconnect callback. When redisAsyncFree() initiated destroying
295 * this context, the status will always be REDIS_OK. */
296 if (ac->onDisconnect && (c->flags & REDIS_CONNECTED)) {
297 if (c->flags & REDIS_FREEING) {
298 ac->onDisconnect(ac,REDIS_OK);
299 } else {
300 ac->onDisconnect(ac,(ac->err == 0) ? REDIS_OK : REDIS_ERR);
301 }
302 }
303
304 /* Cleanup self */
305 redisFree(c);
306}

308/* Free the async context. When this function is called from a callback,
309 * control needs to be returned to redisProcessCallbacks() before actual
310 * free'ing. To do so, a flag is set on the context which is picked up by
311 * redisProcessCallbacks(). Otherwise, the context is immediately free'd. */
312void redisAsyncFree(redisAsyncContext *ac) {
313 redisContext *c = &(ac->c);
314 c->flags |= REDIS_FREEING;
315 if (!(c->flags & REDIS_IN_CALLBACK))
316 __redisAsyncFree(ac);
317}

319/* Helper function to make the disconnect happen and clean up. */
320static void __redisAsyncDisconnect(redisAsyncContext *ac) {
321 redisContext *c = &(ac->c);
322
323 /* Make sure error is accessible if there is any */
324 __redisAsyncCopyError(ac);
325
326 if (ac->err == 0) {
327 /* For clean disconnects, there should be no pending callbacks. */
328 assert(__redisShiftCallback(&ac->replies,NULL) == REDIS_ERR);
329 } else {
330 /* Disconnection is caused by an error, make sure that pending
331 * callbacks cannot call new commands. */
332 c->flags |= REDIS_DISCONNECTING;
333 }
334
335 /* For non-clean disconnects, __redisAsyncFree() will execute pending
336 * callbacks with a NULL-reply. */
337 __redisAsyncFree(ac);
338}
339
340/* Tries to do a clean disconnect from Redis, meaning it stops new commands
341 * from being issued, but tries to flush the output buffer and execute
342 * callbacks for all remaining replies. When this function is called from a
343 * callback, there might be more replies and we can safely defer disconnecting
344 * to redisProcessCallbacks(). Otherwise, we can only disconnect immediately
345 * when there are no pending callbacks. */
346void redisAsyncDisconnect(redisAsyncContext *ac) {
347 redisContext *c = &(ac->c);
348 c->flags |= REDIS_DISCONNECTING;
349 if (!(c->flags & REDIS_IN_CALLBACK) && ac->replies.head == NULL)
350 __redisAsyncDisconnect(ac);
351}

353static int __redisGetSubscribeCallback(redisAsyncContext *ac, redisReply *reply, redisCallback *dstcb) {
354 redisContext *c = &(ac->c);
355 dict *callbacks;
356 dictEntry *de;
357 int pvariant;
358 char *stype;
359 sds sname;
360
361 /* Custom reply functions are not supported for pub/sub. This will fail
362 * very hard when they are used... */
363 if (reply->type == REDIS_REPLY_ARRAY) {
364 assert(reply->elements >= 2);
365 assert(reply->element[0]->type == REDIS_REPLY_STRING);
366 stype = reply->element[0]->str;
367 pvariant = (tolower(stype[0]) == 'p') ? 1 : 0;
368
369 if (pvariant)
370 callbacks = ac->sub.patterns;
371 else
372 callbacks = ac->sub.channels;
373
374 /* Locate the right callback */
375 assert(reply->element[1]->type == REDIS_REPLY_STRING);
376 sname = sdsnewlen(reply->element[1]->str,reply->element[1]->len);
377 de = dictFind(callbacks,sname);
378 if (de != NULL) {
379 memcpy(dstcb,dictGetEntryVal(de),sizeof(*dstcb));
380
381 /* If this is an unsubscribe message, remove it. */
382 if (strcasecmp(stype+pvariant,"unsubscribe") == 0) {
383 dictDelete(callbacks,sname);
384
385 /* If this was the last unsubscribe message, revert to
386 * non-subscribe mode. */
387 assert(reply->element[2]->type == REDIS_REPLY_INTEGER);
388 if (reply->element[2]->integer == 0)
389 c->flags &= ~REDIS_SUBSCRIBED;
390 }
391 }
392 sdsfree(sname);
393 } else {
394 /* Shift callback for invalid commands. */
395 __redisShiftCallback(&ac->sub.invalid,dstcb);
396 }
397 return REDIS_OK;
398}

400void redisProcessCallbacks(redisAsyncContext *ac) {
401 redisContext *c = &(ac->c);
402 redisCallback cb = {NULL, NULL, NULL};
403 void *reply = NULL;
404 int status;
405
406 while((status = redisGetReply(c,&reply)) == REDIS_OK) {
407 if (reply == NULL) {
408 /* When the connection is being disconnected and there are
409 * no more replies, this is the cue to really disconnect. */
410 if (c->flags & REDIS_DISCONNECTING && sdslen(c->obuf) == 0) {
411 __redisAsyncDisconnect(ac);
412 return;
413 }
414
415 /* If monitor mode, repush callback */
416 if(c->flags & REDIS_MONITORING) {
417 __redisPushCallback(&ac->replies,&cb);
418 }
419
420 /* When the connection is not being disconnected, simply stop
421 * trying to get replies and wait for the next loop tick. */
422 break;
423 }
424
425 /* Even if the context is subscribed, pending regular callbacks will
426 * get a reply before pub/sub messages arrive. */
427 if (__redisShiftCallback(&ac->replies,&cb) != REDIS_OK) {
428 /*
429 * A spontaneous reply in a not-subscribed context can be the error
430 * reply that is sent when a new connection exceeds the maximum
431 * number of allowed connections on the server side.
432 *
433 * This is seen as an error instead of a regular reply because the
434 * server closes the connection after sending it.
435 *
436 * To prevent the error from being overwritten by an EOF error the
437 * connection is closed here. See issue #43.
438 *
439 * Another possibility is that the server is loading its dataset.
440 * In this case we also want to close the connection, and have the
441 * user wait until the server is ready to take our request.
442 */
443 if (((redisReply*)reply)->type == REDIS_REPLY_ERROR) {
444 c->err = REDIS_ERR_OTHER;
445 snprintf(c->errstr,sizeof(c->errstr),"%s",((redisReply*)reply)->str);
446 __redisAsyncDisconnect(ac);
447 return;
448 }
449 /* No more regular callbacks and no errors, the context *must* be subscribed or monitoring. */
450 assert((c->flags & REDIS_SUBSCRIBED || c->flags & REDIS_MONITORING));
451 if(c->flags & REDIS_SUBSCRIBED)
452 __redisGetSubscribeCallback(ac,reply,&cb);
453 }
454
455 if (cb.fn != NULL) {
456 __redisRunCallback(ac,&cb,reply);
457 c->reader->fn->freeObject(reply);
458
459 /* Proceed with free'ing when redisAsyncFree() was called. */
460 if (c->flags & REDIS_FREEING) {
461 __redisAsyncFree(ac);
462 return;
463 }
464 } else {
465 /* No callback for this reply. This can either be a NULL callback,
466 * or there were no callbacks to begin with. Either way, don't
467 * abort with an error, but simply ignore it because the client
468 * doesn't know what the server will spit out over the wire. */
469 c->reader->fn->freeObject(reply);
470 }
471 }
472
473 /* Disconnect when there was an error reading the reply */
474 if (status != REDIS_OK)
475 __redisAsyncDisconnect(ac);
476}

478/* Internal helper function to detect socket status the first time a read or
479 * write event fires. When connecting was not succesful, the connect callback
480 * is called with a REDIS_ERR status and the context is free'd. */
481static int __redisAsyncHandleConnect(redisAsyncContext *ac) {
482 redisContext *c = &(ac->c);
483
484 if (redisCheckSocketError(c) == REDIS_ERR) {
485 /* Try again later when connect(2) is still in progress. */
486 if (errno == EINPROGRESS)
487 return REDIS_OK;
488
489 if (ac->onConnect) ac->onConnect(ac,REDIS_ERR);
490 __redisAsyncDisconnect(ac);
491 return REDIS_ERR;
492 }
493
494 /* Mark context as connected. */
495 c->flags |= REDIS_CONNECTED;
496 if (ac->onConnect) ac->onConnect(ac,REDIS_OK);
497 return REDIS_OK;
498}

500/* This function should be called when the socket is readable.
501 * It processes all replies that can be read and executes their callbacks.
502 */
503void redisAsyncHandleRead(redisAsyncContext *ac) {
504 redisContext *c = &(ac->c);
505
506 if (!(c->flags & REDIS_CONNECTED)) {
507 /* Abort connect was not successful. */
508 if (__redisAsyncHandleConnect(ac) != REDIS_OK)
509 return;
510 /* Try again later when the context is still not connected. */
511 if (!(c->flags & REDIS_CONNECTED))
512 return;
513 }
514
515 if (redisBufferRead(c) == REDIS_ERR) {
516 __redisAsyncDisconnect(ac);
517 } else {
518 /* Always re-schedule reads */
519 _EL_ADD_READ(ac);
520 redisProcessCallbacks(ac);
521 }
522}

524void redisAsyncHandleWrite(redisAsyncContext *ac) {
525 redisContext *c = &(ac->c);
526 int done = 0;
527
528 if (!(c->flags & REDIS_CONNECTED)) {
529 /* Abort connect was not successful. */
530 if (__redisAsyncHandleConnect(ac) != REDIS_OK)
531 return;
532 /* Try again later when the context is still not connected. */
533 if (!(c->flags & REDIS_CONNECTED))
534 return;
535 }
536
537 if (redisBufferWrite(c,&done) == REDIS_ERR) {
538 __redisAsyncDisconnect(ac);
539 } else {
540 /* Continue writing when not done, stop writing otherwise */
541 if (!done)
542 _EL_ADD_WRITE(ac);
543 else
544 _EL_DEL_WRITE(ac);
545
546 /* Always schedule reads after writes */
547 _EL_ADD_READ(ac);
548 }
549}

551/* Sets a pointer to the first argument and its length starting at p. Returns
552 * the number of bytes to skip to get to the following argument. */
553static char *nextArgument(char *start, char **str, size_t *len) {
554 char *p = start;
555 if (p[0] != '$') {
556 p = strchr(p,'$');
557 if (p == NULL) return NULL;
558 }
559
560 *len = (int)strtol(p+1,NULL,10);
561 p = strchr(p,'\r');
562 assert(p);
563 *str = p+2;
564 return p+2+(*len)+2;
565}

567/* Helper function for the redisAsyncCommand* family of functions. Writes a
568 * formatted command to the output buffer and registers the provided callback
569 * function with the context. */
570static int __redisAsyncCommand(redisAsyncContext *ac, redisCallbackFn *fn, void *privdata, char *cmd, size_t len) {
571 redisContext *c = &(ac->c);
572 redisCallback cb;
573 int pvariant, hasnext;
574 char *cstr, *astr;
575 size_t clen, alen;
576 char *p;
577 sds sname;
578
579 /* Don't accept new commands when the connection is about to be closed. */
580 if (c->flags & (REDIS_DISCONNECTING | REDIS_FREEING)) return REDIS_ERR;
581
582 /* Setup callback */
583 cb.fn = fn;
584 cb.privdata = privdata;
585
586 /* Find out which command will be appended. */
587 p = nextArgument(cmd,&cstr,&clen);
588 assert(p != NULL);
589 hasnext = (p[0] == '$');
590 pvariant = (tolower(cstr[0]) == 'p') ? 1 : 0;
591 cstr += pvariant;
592 clen -= pvariant;
593
594 if (hasnext && strncasecmp(cstr,"subscribe\r\n",11) == 0) {
595 c->flags |= REDIS_SUBSCRIBED;
596
597 /* Add every channel/pattern to the list of subscription callbacks. */
598 while ((p = nextArgument(p,&astr,&alen)) != NULL) {
599 sname = sdsnewlen(astr,alen);
600 if (pvariant)
601 dictReplace(ac->sub.patterns,sname,&cb);
602 else
603 dictReplace(ac->sub.channels,sname,&cb);
604 }
605 } else if (strncasecmp(cstr,"unsubscribe\r\n",13) == 0) {
606 /* It is only useful to call (P)UNSUBSCRIBE when the context is
607 * subscribed to one or more channels or patterns. */
608 if (!(c->flags & REDIS_SUBSCRIBED)) return REDIS_ERR;
609
610 /* (P)UNSUBSCRIBE does not have its own response: every channel or
611 * pattern that is unsubscribed will receive a message. This means we
612 * should not append a callback function for this command. */
613 } else if(strncasecmp(cstr,"monitor\r\n",9) == 0) {
614 /* Set monitor flag and push callback */
615 c->flags |= REDIS_MONITORING;
616 __redisPushCallback(&ac->replies,&cb);
617 } else {
618 if (c->flags & REDIS_SUBSCRIBED)
619 /* This will likely result in an error reply, but it needs to be
620 * received and passed to the callback. */
621 __redisPushCallback(&ac->sub.invalid,&cb);
622 else
623 __redisPushCallback(&ac->replies,&cb);
624 }
625
626 __redisAppendCommand(c,cmd,len);
627
628 /* Always schedule a write when the write buffer is non-empty */
629 _EL_ADD_WRITE(ac);
630
631 return REDIS_OK;
632}
